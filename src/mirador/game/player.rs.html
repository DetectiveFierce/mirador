1//! Player state and movement logic.
2//!
3//! This module defines the [`Player`] struct, which tracks the player's position, orientation,
4//! and movement parameters, and provides methods for movement and view matrix calculation.
5//!
6//! # Overview
7//!
8//! The player system handles:
9//! - **Position and Orientation**: 3D world position, pitch/yaw angles for camera control
10//! - **Movement**: Forward/backward/left/right movement with speed-based physics
11//! - **Stamina System**: Sprint mechanics with depletion and regeneration
12//! - **Maze Integration**: Cell-based positioning and spawn logic
13//! - **View Matrix**: Camera transformation calculations for rendering
14//!
15//! # Coordinate System
16//!
17//! The player uses a right-handed coordinate system:
18//! - X-axis: Left/Right movement
19//! - Y-axis: Up/Down movement (height)
20//! - Z-axis: Forward/Backward movement
21//!
22//! Angles are measured in degrees:
23//! - **Pitch**: Up/down look angle (-89° to +89°)
24//! - **Yaw**: Left/right look angle (0° to 360°)
25//!
26//! # Usage Example
27//!
28//! ```rust
29//! use crate::game::player::Player;
30//!
31//! let mut player = Player::new();
32//!
33//! // Handle mouse movement
34//! player.mouse_movement(10.0, 5.0);
35//!
36//! // Move player forward
37//! player.move_forward(0.016); // 60 FPS delta time
38//!
39//! // Update stamina
40//! player.update_stamina(true, true, 0.016);
41//!
42//! // Get view matrix for rendering
43//! let view_matrix = player.get_view_matrix();
44//! ```

50/// Represents the player character's state in the world.
51///
52/// The `Player` struct encapsulates all player-related state including position,
53/// orientation, movement parameters, and stamina system. It provides methods for
54/// movement, camera control, and integration with the maze system.
55///
56/// # Fields
57///
58/// ## Position and Orientation
59/// - `position`: 3D world coordinates `[x, y, z]` where y represents height
60/// - `pitch`: Vertical look angle in degrees (-89° to +89°)
61/// - `yaw`: Horizontal look angle in degrees (0° to 360°)
62/// - `fov`: Field of view in degrees for perspective projection
63///
64/// ## Movement Parameters
65/// - `base_speed`: Base movement speed in units per second
66/// - `speed`: Current movement speed (can be modified by upgrades/effects)
67/// - `mouse_sensitivity`: Multiplier for mouse movement sensitivity
68///
69/// ## Maze Integration
70/// - `current_cell`: The maze cell the player is currently occupying
71///
72/// ## Stamina System
73/// - `stamina`: Current stamina value (0.0 to max_stamina)
74/// - `max_stamina`: Maximum stamina capacity
75/// - `stamina_regen_cooldown`: Seconds to wait before stamina regeneration starts
76/// - `stamina_regen_rate`: Stamina points regenerated per second
77/// - `last_sprint_time`: Time accumulator for regeneration cooldown
78///
79/// # Examples
80///
81/// ```rust
82/// use crate::game::player::Player;
83///
84/// let player = Player::new();
85/// assert_eq!(player.position[1], crate::math::coordinates::constants::PLAYER_HEIGHT);
86/// assert_eq!(player.stamina, 1.0);
87/// assert_eq!(player.max_stamina, 2.0);

220    /// Computes the view matrix for the player's current position and orientation.
221    ///
222    /// The view matrix transforms world coordinates into camera/view space.
223    /// This matrix is used by the rendering system to position the camera
224    /// in the 3D world.
225    ///
226    /// # Returns
227    ///
228    /// A 4x4 transformation matrix that combines:
229    /// 1. Translation to move the world relative to the camera position
230    /// 2. Rotation to orient the camera based on pitch and yaw angles
231    ///
232    /// # Algorithm
233    ///
234    /// 1. Creates rotation matrices for pitch (X-axis) and yaw (Y-axis)
235    /// 2. Combines rotations: applies yaw first, then pitch
236    /// 3. Creates translation matrix (negative position to move world opposite to camera)
237    /// 4. Multiplies translation by rotation: `translation * rotation`
238    ///
239    /// # Examples
240    ///
241    /// ```rust
242    /// use crate::game::player::Player;
243    /// use crate::math::mat::Mat4;
244    ///
245    /// let player = Player::new();
246    /// let view_matrix = player.get_view_matrix();
247    ///
248    /// // The view matrix can be used in shaders for camera transformation
249    /// // view_matrix is a 4x4 matrix that transforms world coordinates to view space
250    /// ```
251    pub fn get_view_matrix(&self) -> Mat4 {
252        // Create rotation matrices for pitch and yaw
253        let pitch_matrix = Mat4::rotation_x(self.pitch);
254        let yaw_matrix = Mat4::rotation_y(self.yaw);
255
256        // Combine rotations: apply yaw first, then pitch
257        let rotation_matrix = yaw_matrix.multiply(&pitch_matrix);
258
259        // Create translation matrix (negative because we move the world opposite to camera)
260        let translation_matrix =
261            Mat4::translation(-self.position[0], -self.position[1], -self.position[2]);
262
263        // View matrix = rotation * translation
264        translation_matrix.multiply(&rotation_matrix)
265    }

267    /// Computes the combined view-projection matrix for rendering.
268    ///
269    /// This method combines the view matrix (camera transformation) with
270    /// the projection matrix (perspective transformation) into a single
271    /// matrix for efficient GPU usage.
272    ///
273    /// # Arguments
274    ///
275    /// * `aspect_ratio` - Width divided by height of the viewport
276    /// * `near` - Distance to the near clipping plane
277    /// * `far` - Distance to the far clipping plane
278    ///
279    /// # Returns
280    ///
281    /// A 4x4 transformation matrix that transforms world coordinates
282    /// directly to clip space: `projection * view`
283    ///
284    /// # Examples
285    ///
286    /// ```rust
287    /// use crate::game::player::Player;
288    ///
289    /// let player = Player::new();
290    /// let view_proj_matrix = player.get_view_proj_matrix(16.0/9.0, 0.1, 1000.0);
291    ///
292    /// // This matrix can be passed directly to shaders for efficient rendering
293    /// ```
294    pub fn get_view_proj_matrix(&self, aspect_ratio: f32, near: f32, far: f32) -> Mat4 {
295        let view_matrix = self.get_view_matrix();
296        let projection_matrix = Mat4::perspective(
297            self.fov, // Keep in degrees since your Mat4 likely expects degrees
298            aspect_ratio,
299            near,
300            far,
301        );
302
303        // Projection * View (note the order)
304        view_matrix.multiply(&projection_matrix)
305    }

307    /// Updates the player's orientation based on mouse movement.
308    ///
309    /// This method handles mouse input to control the camera orientation.
310    /// The mouse movement is converted to angular changes in pitch and yaw.
311    ///
312    /// # Arguments
313    ///
314    /// * `delta_x` - Mouse movement in the X direction (positive = right, negative = left)
315    /// * `delta_y` - Mouse movement in the Y direction (positive = down, negative = up)
316    ///
317    /// # Behavior
318    ///
319    /// - **Yaw**: Decreases with positive delta_x (mouse right = look left)
320    /// - **Pitch**: Decreases with positive delta_y (mouse down = look down)
321    /// - **Pitch Clamping**: Automatically clamped to [-89°, +89°] to prevent camera flipping
322    /// - **Sensitivity**: Movement is scaled by `mouse_sensitivity` multiplier
323    ///
324    /// # Examples
325    ///
326    /// ```rust
327    /// use crate::game::player::Player;
328    ///
329    /// let mut player = Player::new();
330    /// let initial_yaw = player.yaw;
331    /// let initial_pitch = player.pitch;
332    ///
333    /// // Move mouse right (positive delta_x)
334    /// player.mouse_movement(10.0, 0.0);
335    /// assert!(player.yaw < initial_yaw); // Looking more left
336    ///
337    /// // Move mouse down (positive delta_y)
338    /// player.mouse_movement(0.0, 5.0);
339    /// assert!(player.pitch < initial_pitch); // Looking more down
340    /// ```
341    pub fn mouse_movement(&mut self, delta_x: f64, delta_y: f64) {
342        self.yaw -= delta_x as f32 * self.mouse_sensitivity;
343        self.pitch -= delta_y as f32 * self.mouse_sensitivity;
344
345        // Clamp pitch to prevent flipping
346        self.pitch = self.pitch.clamp(-89.0, 89.0);
347    }

349    /// Moves the player forward based on current yaw and speed.
350    ///
351    /// Moves the player in the direction they are currently facing.
352    /// The movement is based on the player's yaw angle and current speed.
353    ///
354    /// # Arguments
355    ///
356    /// * `delta_time` - Time elapsed since last frame in seconds
357    ///
358    /// # Movement Calculation
359    ///
360    /// The forward direction is calculated as:
361    /// - `forward_x = sin(yaw_radians)`
362    /// - `forward_z = cos(yaw_radians)`
363    ///
364    /// Position is updated as:
365    /// - `position.x -= forward_x * speed * delta_time`
366    /// - `position.z -= forward_z * speed * delta_time`
367    ///
368    /// # Examples
369    ///
370    /// ```rust
371    /// use crate::game::player::Player;
372    ///
373    /// let mut player = Player::new();
374    /// let initial_pos = player.position;
375    ///
376    /// // Move forward for 1 second at 120 units/second
377    /// player.move_forward(1.0);
378    ///
379    /// // Position should have changed based on initial yaw
380    /// assert_ne!(player.position, initial_pos);
381    /// ```
382    pub fn move_forward(&mut self, delta_time: f32) {
383        let forward_x = self.yaw.to_radians().sin();
384        let forward_z = self.yaw.to_radians().cos();
385        self.position[0] -= forward_x * self.speed * delta_time;
386        self.position[2] -= forward_z * self.speed * delta_time;
387    }

596    /// Spawns the player at the bottom-left cell of the maze.

598    /// Sets the player's position to the maze entrance (bottom-left cell)

602    /// # Arguments
603    ///
604    /// * `maze_grid` - 2D grid representing the maze layout (true = wall, false = passage)
605    /// * `is_test_mode` - Whether test mode is enabled (affects floor size calculations)
606    ///
607    /// # Behavior
608    ///
609    /// 1. Calculates maze dimensions from the grid
610    /// 2. Gets the bottom-left cell coordinates
611    /// 3. Converts cell coordinates to world coordinates
612    /// 4. Sets player position to the entrance
613    /// 5. Sets yaw to face north (0°)
614    /// 6. Updates current_cell to match the entrance
615    ///
616    /// # Examples
617    ///
618    /// ```rust
619    /// use crate::game::player::Player;
620    ///
621    /// let mut player = Player::new();
622    /// let maze_grid = vec![vec![false, true], vec![false, false]]; // 2x2 maze
623    ///
624    /// // Spawn at maze entrance
625    /// player.spawn_at_maze_entrance(&maze_grid, false);

628    /// assert_eq!(player.yaw, 0.0); // North
629    /// ```
630    pub fn spawn_at_maze_entrance(&mut self, maze_grid: &[Vec<bool>], is_test_mode: bool) {
631        let maze_width = maze_grid[0].len();
632        let maze_height = maze_grid.len();
633        let maze_dimensions = (maze_width, maze_height);
634
635        // Set the player at the bottom-left cell of the maze
636        let entrance_cell = coordinates::get_bottom_left_cell(maze_dimensions);
637        self.position = coordinates::maze_to_world(
638            &entrance_cell,
639            maze_dimensions,
640            self.position[1],
641            is_test_mode,
642        );
643        self.current_cell = entrance_cell;
644
645        // Set the initial orientation to face north (into the maze)
646        self.yaw = coordinates::direction_to_yaw(coordinates::Direction::North);
647    }

649    /// Updates the player's stamina based on sprinting state and time.
650    ///
651    /// This method should be called every frame to manage the stamina system.
652    /// It handles stamina depletion during sprinting and regeneration during rest.

661    ///
662    /// ## Stamina Depletion
663    /// - Occurs when `is_sprinting && is_moving && stamina > 0.0`
664    /// - Depletes at rate of `0.7` points per second
665    /// - Clamps to minimum of `0.0`
666    /// - Resets regeneration cooldown timer
667    ///
668    /// ## Stamina Regeneration
669    /// - Starts after `stamina_regen_cooldown` seconds of not sprinting
670    /// - Regenerates at rate of `stamina_regen_rate` points per second
671    /// - Clamps to maximum of `max_stamina`
672    /// - Cooldown timer accumulates when not sprinting
673    ///
674    /// # Examples
675    ///
676    /// ```rust
677    /// use crate::game::player::Player;
678    ///
679    /// let mut player = Player::new();
680    /// let initial_stamina = player.stamina;
681    ///
682    /// // Sprint for 1 second
683    /// player.update_stamina(true, true, 1.0);
684    /// assert!(player.stamina < initial_stamina);
685    ///
686    /// // Rest for 1 second (should start regenerating after 0.7s cooldown)
687    /// player.update_stamina(false, false, 1.0);
688    /// assert!(player.stamina > initial_stamina - 0.7);
689    /// ```
690    pub fn update_stamina(&mut self, is_sprinting: bool, is_moving: bool, delta_time: f32) {
691        if is_sprinting && is_moving && self.stamina > 0.0 {
692            self.stamina -= 0.7 * delta_time; // Deplete stamina
693            if self.stamina < 0.0 {
694                self.stamina = 0.0;
695            }
696            self.last_sprint_time = 0.0;
697        } else {
698            self.last_sprint_time += delta_time;
699            if self.last_sprint_time > self.stamina_regen_cooldown {
700                self.stamina += self.stamina_regen_rate * delta_time;
701                if self.stamina > self.max_stamina {
702                    self.stamina = self.max_stamina;
703                }
704            }
705        }
706    }
707
708    /// Returns the current stamina as a ratio from 0.0 to 1.0.
709    ///
710    /// This is useful for UI elements like stamina bars that need
711    /// to display stamina as a percentage or normalized value.
712    ///
713    /// # Returns
714    ///
715    /// A value between 0.0 and 1.0 representing the stamina ratio:
716    /// - `0.0`: No stamina (empty)
717    /// - `1.0`: Full stamina
718    /// - Values in between: Partial stamina
719    ///
720    /// # Examples
721    ///
722    /// ```rust
723    /// use crate::game::player::Player;
724    ///
725    /// let mut player = Player::new();
726    ///
727    /// // Full stamina
728    /// assert_eq!(player.stamina_ratio(), 0.5); // 1.0 / 2.0 = 0.5
729    ///
730    /// // Deplete stamina
731    /// player.stamina = 0.0;
732    /// assert_eq!(player.stamina_ratio(), 0.0);
733    ///
734    /// // Full stamina
735    /// player.stamina = player.max_stamina;
736    /// assert_eq!(player.stamina_ratio(), 1.0);
737    /// ```
738    pub fn stamina_ratio(&self) -> f32 {
739        (self.stamina / self.max_stamina).clamp(0.0, 1.0)
740    }
