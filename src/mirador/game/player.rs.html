10/// Represents the player character's state in the world.
11///
12/// Includes position, orientation (pitch/yaw), field of view, movement speed, and mouse sensitivity.
13#[derive(Debug, Default, Clone)]
14pub struct Player {
15    /// Player's world position `[x, y, z]`.
16    pub position: [f32; 3],
17    /// Pitch (up/down look), in degrees.
18    pub pitch: f32,
19    /// Yaw (left/right look), in degrees.
20    pub yaw: f32,
21    /// Field of view (degrees).
22    pub fov: f32,
23    /// Base movement speed (units per second).
24    pub base_speed: f32,
25    /// Current movement speed (units per second).
26    pub speed: f32,
27    /// Mouse sensitivity multiplier.
28    pub mouse_sensitivity: f32,
29    /// Current Cell
30    pub current_cell: Cell,
31}

33impl Player {
34    /// Creates a new [`Player`] with default starting position and parameters.
35    pub fn new() -> Self {
36        Self {
37            position: [0.0, PLAYER_HEIGHT, 0.0], // Will be set correctly when spawning
38            pitch: 3.0,
39            yaw: 316.0,
40            fov: 100.0,
41            base_speed: 100.0,
42            speed: 100.0,
43            mouse_sensitivity: 1.0,
44            current_cell: Cell::default(),
45        }
46    }

48    /// Computes the view matrix for the player's current position and orientation.
49    ///
50    /// Combines pitch and yaw rotations, then applies translation.
51    pub fn get_view_matrix(&self) -> Mat4 {
52        // Create rotation matrices for pitch and yaw
53        let pitch_matrix = Mat4::rotation_x(self.pitch);
54        let yaw_matrix = Mat4::rotation_y(self.yaw);
55
56        // Combine rotations: apply yaw first, then pitch
57        let rotation_matrix = pitch_matrix.multiply(&yaw_matrix);
58
59        // Create translation matrix (negative because we move the world opposite to camera)
60        let translation_matrix =
61            Mat4::translation(-self.position[0], -self.position[1], -self.position[2]);
62
63        // View matrix = rotation * translation
64        rotation_matrix.multiply(&translation_matrix)
65    }
66
67    pub fn get_view_proj_matrix(&self, aspect_ratio: f32, near: f32, far: f32) -> Mat4 {
68        let view_matrix = self.get_view_matrix();
69        let projection_matrix = Mat4::perspective(
70            self.fov, // Keep in degrees since your Mat4 likely expects degrees
71            aspect_ratio,
72            near,
73            far,
74        );
75
76        // Projection * View (note the order)
77        projection_matrix.multiply(&view_matrix)
78    }

80    /// Updates the player's orientation based on mouse movement.
81    ///
82    /// # Arguments
83    /// * `delta_x` - Mouse movement in the X direction.
84    /// * `delta_y` - Mouse movement in the Y direction.
85    ///
86    /// Clamps pitch to prevent flipping.
87    pub fn mouse_movement(&mut self, delta_x: f64, delta_y: f64) {
88        self.yaw -= delta_x as f32 * self.mouse_sensitivity;
89        self.pitch -= delta_y as f32 * self.mouse_sensitivity;
90
91        // Clamp pitch to prevent flipping
92        self.pitch = self.pitch.clamp(-89.0, 89.0);
93    }

95    /// Moves the player forward based on current yaw and speed.
96    pub fn move_forward(&mut self, delta_time: f32) {
97        let forward_x = self.yaw.to_radians().sin();
98        let forward_z = self.yaw.to_radians().cos();
99        self.position[0] -= forward_x * self.speed * delta_time;
100        self.position[2] -= forward_z * self.speed * delta_time;
101    }
102
103    /// Moves the player backward based on current yaw and speed.
104    pub fn move_backward(&mut self, delta_time: f32) {
105        let forward_x = self.yaw.to_radians().sin();
106        let forward_z = self.yaw.to_radians().cos();
107        self.position[0] += forward_x * self.speed * delta_time;
108        self.position[2] += forward_z * self.speed * delta_time;
109    }
110
111    /// Moves the player left based on current yaw and speed.
112    pub fn move_left(&mut self, delta_time: f32) {
113        let right_x = self.yaw.to_radians().cos();
114        let right_z = self.yaw.to_radians().sin();
115        self.position[0] -= right_x * self.speed * delta_time;
116        self.position[2] += right_z * self.speed * delta_time;
117    }
118
119    /// Moves the player right based on current yaw and speed.
120    pub fn move_right(&mut self, delta_time: f32) {
121        let right_x = self.yaw.to_radians().cos();
122        let right_z = self.yaw.to_radians().sin();
123        self.position[0] += right_x * self.speed * delta_time;
124        self.position[2] -= right_z * self.speed * delta_time;
125    }

127    pub fn update_cell(&mut self, maze_grid: &[Vec<bool>]) {
128        let maze_width = maze_grid[0].len();
129        let maze_height = maze_grid.len();
130        let maze_dimensions = (maze_width, maze_height);
131
132        self.current_cell = coordinates::world_to_maze(self.position, maze_dimensions);
133    }
134
135    /// Spawns the player at the bottom-left cell of the maze.
136    ///
137    /// # Arguments
138    /// * `maze_grid` - The maze grid representing walls and passages
139    pub fn spawn_at_maze_entrance(&mut self, maze_grid: &[Vec<bool>]) {
140        let maze_width = maze_grid[0].len();
141        let maze_height = maze_grid.len();
142        let maze_dimensions = (maze_width, maze_height);
143
144        // Set the player at the bottom-left cell of the maze
145        let entrance_cell = coordinates::get_bottom_left_cell(maze_dimensions);
146        self.position = coordinates::maze_to_world(&entrance_cell, maze_dimensions, PLAYER_HEIGHT);
147        self.current_cell = entrance_cell;
148
149        // Set the initial orientation to face north (into the maze)
150        self.yaw = coordinates::direction_to_yaw(coordinates::Direction::North);
151    }
