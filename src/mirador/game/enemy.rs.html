1//! Enemy AI system for the Mirador game.
2//!
3//! This module provides a sophisticated enemy AI system with pathfinding, level-based scaling,
4//! and strategic placement. The enemy AI adapts its behavior based on the current game level,
5//! becoming more aggressive and intelligent as the player progresses.
6//!
7//! # Key Features
8//!
9//! - **Level-based scaling**: Enemy speed, aggression, and intelligence scale with game level
10//! - **Advanced pathfinding**: Uses rotation-based pathfinding with collision detection
11//! - **Strategic placement**: Enemies are placed intelligently relative to player and exit
12//! - **Stuck detection**: AI can detect when stuck and attempt escape maneuvers
13//! - **Pursuit behavior**: Enemies become more aggressive when player is within detection range
14//!
15//! # Usage
16//!
17//! ```rust
18//! use crate::game::enemy::{Enemy, place_enemy_standard};
19//!
20//! // Create an enemy at a specific position
21//! let mut enemy = Enemy::new([100.0, 30.0, 100.0], 150.0);
22//!
23//! // Update enemy AI each frame
24//! enemy.update(player_position, delta_time, current_level, collision_checker);
25//!
26//! // Place enemy strategically
27//! let enemy = place_enemy_standard(exit_pos, player_pos, level, collision_checker);
28//! ```

33/// Represents an enemy entity in the game with AI-driven behavior.
34///
35/// The enemy uses a pathfinding system to navigate toward the player while avoiding
36/// obstacles. Its behavior scales with the game level, becoming more aggressive
37/// and intelligent as the player progresses.
38///
39/// # Behavior Scaling
40///
41/// - **Speed**: Increases by 20% per level, capped at 500% of base speed
42/// - **Path radius**: Decreases with level (enemy gets closer before acting)
43/// - **Arrival threshold**: Decreases with level (enemy is more persistent)
44/// - **Rotation step**: Increases with level (enemy tries more directions faster)
45/// - **Pursuit distance**: Increases with level (enemy detects player from farther away)
46///
47/// # Example
48///
49/// ```rust
50/// let mut enemy = Enemy::new([100.0, 30.0, 100.0], 150.0);
51///
52/// // Update enemy behavior each frame
53/// enemy.update(
54///     player_position,
55///     delta_time,
56///     current_level,
57///     |start, end| collision_system.intersects(start, end)
58/// );
59/// ```
60#[derive(Debug, Clone)]
61pub struct Enemy {
62    /// The visual size of the enemy sprite in pixels
63    pub size: f32,
64    /// The pathfinding system that controls enemy movement
65    pub pathfinder: EnemyPathfinder,
66    /// Base movement speed in units per second (before level scaling)
67    pub base_speed: f32,
68    /// Current movement speed after level-based scaling
69    pub current_speed: f32,
70}

98    /// Updates the enemy's behavior and position based on the current game state.
99    ///
100    /// This method handles level-based scaling, pathfinding updates, and movement.
101    /// The enemy will attempt to move toward the player while avoiding obstacles.
102    ///
103    /// # Arguments
104    ///
105    /// * `player_position` - Current 3D position of the player `[x, y, z]`
106    /// * `delta_time` - Time elapsed since last frame in seconds
107    /// * `level` - Current game level (affects enemy aggression)
108    /// * `line_intersects_geometry` - Function to check if a line intersects with game geometry
109    ///
110    /// # Behavior
111    ///
112    /// - Scales enemy aggression based on level
113    /// - Updates pathfinding to find optimal path to player
114    /// - Moves enemy toward current target while respecting speed limits
115    /// - Handles collision detection and avoidance
116    ///
117    /// # Example
118    ///
119    /// ```rust
120    /// enemy.update(
121    ///     [50.0, 30.0, 50.0],  // player position
122    ///     0.016,               // delta time (60 FPS)
123    ///     3,                   // level 3
124    ///     |start, end| collision_system.line_intersects_wall(start, end)
125    /// );
126    /// ```
127    pub fn update<F>(
128        &mut self,
129        player_position: [f32; 3],
130        delta_time: f32,
131        level: u32,
132        line_intersects_geometry: F,
133    ) where
134        F: Fn([f32; 3], [f32; 3]) -> bool,
135    {
136        // Prevent movement if locked
137        if self.pathfinder.locked {
138            return;
139        }
140        // Scale aggression based on level
141        self.scale_aggression_by_level(level);
142
143        // Update pathfinding with level-aware parameters
144        if let Some(_target) =
145            self.pathfinder
146                .update(player_position, level, line_intersects_geometry)
147        {
148            // Move towards the target
149            if let Some(direction) = self.pathfinder.get_movement_direction() {
150                let direction_vec = Vec3(direction);
151                if let Some(target) = self.pathfinder.current_target {
152                    let position_vec = Vec3(self.pathfinder.position);
153                    let target_vec = Vec3(target);
154                    let distance_to_target = position_vec.distance_to(&target_vec);
155                    let max_step = self.current_speed * delta_time;
156                    let step = max_step.min(distance_to_target);
157                    let movement = direction_vec * step;
158                    let new_position = position_vec + movement;
159                    self.pathfinder.set_position(*new_position.as_array());
160                }
161            }
162        }
163    }

165    /// Scales enemy aggression parameters based on the current game level.
166    ///
167    /// This method adjusts various enemy attributes to make them more challenging
168    /// as the player progresses through levels.
169    ///
170    /// # Scaling Factors
171    ///
172    /// - **Speed**: Increases by 20% per level, capped at 500% of base speed
173    /// - **Path radius**: Decreases by 15% per level, minimum 30% of original
174    /// - **Arrival threshold**: Decreases by 10% per level, minimum 20% of original
175    /// - **Rotation step**: Increases by 25% per level, maximum 300% of original
176    /// - **Pursuit distance**: Increases by 30% per level, maximum 400% of original
177    ///
178    /// # Arguments
179    ///
180    /// * `level` - Current game level (1-based)
181    ///
182    /// # Example
183    ///
184    /// ```rust
185    /// enemy.scale_aggression_by_level(5); // Scale for level 5
186    /// ```
187    fn scale_aggression_by_level(&mut self, level: u32) {
188        let level_f = level as f32;
189
190        // Speed scaling: increases by 20% per level, capped at 500% of base speed
191        let speed_multiplier = (1.0 + (level_f * 0.2)).min(5.0);
192        self.current_speed = self.base_speed * speed_multiplier;
193
194        // Update pathfinder aggression parameters
195        self.pathfinder.update_aggression_for_level(level);
196    }

225/// Advanced pathfinding system for enemy movement and navigation.
226///
227/// The `EnemyPathfinder` handles all aspects of enemy movement including:
228/// - Target calculation and pathfinding
229/// - Collision detection and avoidance
230/// - Stuck detection and recovery
231/// - Level-based behavior scaling
232/// - Pursuit behavior when player is detected
233///
234/// # Pathfinding Strategy
235///
236/// The pathfinder uses a rotation-based approach where it:
237/// 1. Calculates an ideal path toward the player
238/// 2. If blocked, rotates around the ideal direction to find alternative paths
239/// 3. Uses level-based parameters to adjust aggressiveness
240/// 4. Implements stuck detection to prevent infinite loops
241///
242/// # Example
243///
244/// ```rust
245/// let mut pathfinder = EnemyPathfinder::new([100.0, 30.0, 100.0], 150.0);
246///
247/// // Update pathfinding
248/// if let Some(target) = pathfinder.update(player_pos, level, collision_checker) {
249///     // Enemy has a valid target to move toward
250/// }
251/// ```
252#[derive(Debug, Clone)]
253pub struct EnemyPathfinder {
254    /// Current 3D position of the enemy `[x, y, z]`
255    pub position: [f32; 3],
256    /// Current target position the enemy is moving toward
257    pub current_target: Option<[f32; 3]>,
258    /// Current path radius (scales with level)
259    pub path_radius: f32,
260    /// Base path radius (before level scaling)
261    pub base_path_radius: f32,
262    /// Current rotation step for pathfinding (scales with level)
263    pub rotation_step: f32,
264    /// Base rotation step (before level scaling)
265    pub base_rotation_step: f32,
266    /// Distance threshold for considering target reached (scales with level)
267    pub arrival_threshold: f32,
268    /// Base arrival threshold (before level scaling)
269    pub base_arrival_threshold: f32,
270    /// Counter for stuck detection
271    pub stuck_counter: i32,
272    /// Previous position for stuck detection
273    pub last_position: [f32; 3],
274    /// Whether the enemy has reached the player
275    pub reached_player: bool,
276    /// Whether the enemy is locked (cannot move)
277    pub locked: bool,
278    /// Current aggression level (matches game level)
279    pub aggression_level: u32,
280    /// Distance at which enemy starts pursuing more aggressively (scales with level)
281    pub pursuit_distance: f32,
282    /// Base pursuit distance (before level scaling)
283    pub base_pursuit_distance: f32,
284}

323    /// Updates aggression parameters based on the current game level.
324    ///
325    /// This method scales various pathfinding parameters to make the enemy
326    /// more challenging and intelligent at higher levels.
327    ///
328    /// # Scaling Details
329    ///
330    /// - **Path radius**: Decreases by 15% per level (enemy gets closer before acting)
331    /// - **Arrival threshold**: Decreases by 10% per level (enemy is more persistent)
332    /// - **Rotation step**: Increases by 25% per level (enemy tries more directions faster)
333    /// - **Pursuit distance**: Increases by 30% per level (enemy detects player from farther away)

335    /// # Arguments

337    /// * `level` - Current game level (1-based)

344    pub fn update_aggression_for_level(&mut self, level: u32) {
345        self.aggression_level = level;
346        let level_f = level as f32;
347
348        // Path radius decreases with level (enemy gets closer before acting)
349        // Reduces by 15% per level, minimum 30% of original
350        let radius_multiplier = (1.0 - (level_f * 0.15)).max(0.3);
351        self.path_radius = self.base_path_radius * radius_multiplier;
352
353        // Arrival threshold decreases (enemy is more persistent)
354        // Reduces by 10% per level, minimum 20% of original
355        let threshold_multiplier = (1.0 - (level_f * 0.1)).max(0.2);
356        self.arrival_threshold = self.base_arrival_threshold * threshold_multiplier;
357
358        // Rotation step increases (enemy tries more directions faster)
359        // Increases by 25% per level, maximum 300% of original
360        let rotation_multiplier = (1.0 + (level_f * 0.25)).min(3.0);
361        self.rotation_step = self.base_rotation_step * rotation_multiplier;
362
363        // Pursuit distance increases (enemy detects player from farther away)
364        // Increases by 30% per level, maximum 400% of original
365        let pursuit_multiplier = (1.0 + (level_f * 0.3)).min(4.0);
366        self.pursuit_distance = self.base_pursuit_distance * pursuit_multiplier;
367    }

414    /// Updates stuck detection with level-based sensitivity.
415    ///
416    /// This method tracks enemy movement and detects when the enemy is stuck.
417    /// Higher levels have lower tolerance for being stuck and reset faster.
418    ///
419    /// # Arguments
420    ///
421    /// * `level` - Current game level for tolerance scaling
422    ///
423    /// # Behavior
424    ///
425    /// - Tracks distance moved since last update
426    /// - Higher levels have lower stuck thresholds
427    /// - Resets stuck counter faster at higher levels
428    /// - Updates last position for next frame
429    ///
430    /// # Example
431    ///
432    /// ```rust
433    /// pathfinder.update_stuck_detection(5); // Update for level 5
434    /// ```
435    fn update_stuck_detection(&mut self, level: u32) {
436        let current_vec = Vec3(self.position);
437        let last_vec = Vec3(self.last_position);
438        let distance_moved = current_vec.distance_to(&last_vec);
439
440        // Higher levels have lower tolerance for being stuck
441        let stuck_threshold = (5.0 / (1.0 + level as f32 * 0.2)).max(1.0);
442
443        if distance_moved < stuck_threshold {
444            self.stuck_counter += 1;
445        } else {
446            self.stuck_counter = 0;
447        }
448
449        // Reset stuck counter faster at higher levels
450        let max_stuck_time = (120.0 / (1.0 + level as f32 * 0.1)) as i32;
451        if self.stuck_counter > max_stuck_time {
452            self.stuck_counter = 0;
453        }
454
455        self.last_position = self.position;
456    }

484    fn needs_new_target(&self, level: u32) -> bool {
485        match self.current_target {
486            None => true,
487            Some(target) => {
488                let enemy_vec = Vec3(self.position);
489                let target_vec = Vec3(target);
490                let distance = enemy_vec.distance_to(&target_vec);
491
492                // Higher levels recalculate targets more frequently
493                let stuck_threshold = (60.0 / (1.0 + level as f32 * 0.2)) as i32;
494
495                distance <= self.arrival_threshold || self.stuck_counter > stuck_threshold
496            }
497        }

500    /// Calculates a new target point with level-based aggression.
501    ///
502    /// This method implements the core pathfinding algorithm using a rotation-based
503    /// approach. It tries to find the optimal path toward the player while avoiding
504    /// obstacles and adapting behavior based on the current level.
505    ///
506    /// # Arguments
507    ///
508    /// * `player_position` - Current 3D position of the player `[x, y, z]`
509    /// * `level` - Current game level for behavior scaling
510    /// * `line_intersects_geometry` - Function to check for collision with geometry
511    ///
512    /// # Algorithm
513    ///
514    /// 1. **Direct pursuit**: If player is within close pursuit radius, move directly toward player
515    /// 2. **Ideal path**: Try to move toward ideal target point
516    /// 3. **Rotation search**: If blocked, rotate around ideal direction to find alternative paths
517    /// 4. **Escape behavior**: If stuck, try escape maneuvers with increased radius
518    ///
519    /// # Level Scaling
520    ///
521    /// - Higher levels have tighter pursuit radius
522    /// - More aggressive rotation patterns at higher levels
523    /// - Enhanced escape behavior with more directions
524    /// - Varied radius calculations based on stuck counter
525    ///
526    /// # Example
527    ///
528    /// ```rust
529    /// pathfinder.calculate_new_target(
530    ///     [50.0, 30.0, 50.0],  // player position
531    ///     3,                   // level 3
532    ///     |start, end| collision_system.intersects(start, end)
533    /// );
534    /// ```
535    fn calculate_new_target<F>(
536        &mut self,
537        player_position: [f32; 3],
538        level: u32,
539        line_intersects_geometry: F,
540    ) where
541        F: Fn([f32; 3], [f32; 3]) -> bool,
542    {
543        let enemy_vec = Vec3(self.position);
544        let player_vec = Vec3(player_position);
545        let enemy_2d = enemy_vec.to_2d();
546        let player_2d = player_vec.to_2d();
547
548        let direction_to_player = (player_2d - enemy_2d).normalize();
549        let distance_to_player = enemy_2d.distance_to(&player_2d);
550
551        if self.locked {
552            self.current_target = Some(self.position);
553            return;
554        }
555
556        // Check if player is within pursuit distance (scales with level)
557        if distance_to_player < self.pursuit_distance {
558            // Higher levels are more aggressive in close pursuit
559            let close_pursuit_radius = self.path_radius * (1.0 - (level as f32 * 0.1).min(0.5));
560
561            if distance_to_player < close_pursuit_radius {
562                self.current_target = Some(player_position);
563
564                // Tighter capture radius at higher levels
565                let capture_distance = (15.0 / (1.0 + level as f32 * 0.2)).max(10.0);
566                if distance_to_player < capture_distance {
567                    self.reached_player = true;
568                    self.position = [0.0, 30.0, 0.0];
569                    self.locked = true;
570                }
571                return;
572            }
573        }
574
575        // Calculate ideal target with level-based aggressiveness
576        let ideal_target_2d = enemy_2d + direction_to_player * self.path_radius;
577        let ideal_target = Vec3::from_2d(ideal_target_2d, self.position[1]);
578
579        if self.is_safe_path(
580            self.position,
581            *ideal_target.as_array(),
582            &line_intersects_geometry,
583        ) {
584            self.current_target = Some(*ideal_target.as_array());
585            return;
586        }
587
588        // Enhanced rotation strategy with level scaling
589        let base_bias = if self.stuck_counter > 30 { 1.8 } else { 1.25 };
590        let level_bias = 1.0 + (level as f32 * 0.15); // Higher levels try more rotations
591        let rotation_bias = base_bias * level_bias;
592        let adjusted_step = self.rotation_step * rotation_bias;
593        let max_rotations = ((PI / adjusted_step) as i32).min(16); // Cap to prevent infinite loops
594
595        // More aggressive movement patterns at higher levels
596        let base_direction = if self.stuck_counter > (30 / level.max(1)) as i32 {
597            (enemy_2d - player_2d).normalize() // Move away from player
598        } else {
599            direction_to_player // Move toward player
600        };
601
602        // Try both directions with level-based persistence
603        for direction_multiplier in [1.0, -1.0] {
604            let mut current_direction = base_direction;
605
606            for i in 1..=max_rotations {
607                current_direction = current_direction.rotate(adjusted_step * direction_multiplier);
608
609                // More varied radius at higher levels
610                let radius_variation = if self.stuck_counter > (30 / level.max(1)) as i32 {
611                    let variation_factor =
612                        1.0 + (0.5 + level as f32 * 0.1) * (i as f32 / max_rotations as f32);
613                    self.path_radius * variation_factor
614                } else {
615                    self.path_radius
616                };
617
618                let test_target_2d = enemy_2d + current_direction * radius_variation;
619                let test_target = Vec3::from_2d(test_target_2d, self.position[1]);
620
621                if self.is_safe_path(
622                    self.position,
623                    *test_target.as_array(),
624                    &line_intersects_geometry,
625                ) {
626                    self.current_target = Some(*test_target.as_array());
627                    return;
628                }
629            }
630        }
631
632        // Enhanced escape behavior for higher levels
633        if self.stuck_counter > (30 / level.max(1)) as i32 {
634            let escape_radius = self.path_radius * (2.0 + level as f32 * 0.3);
635            let num_test_directions = 6 + (level * 2) as usize; // More directions at higher levels
636
637            for i in 0..num_test_directions {
638                let angle = (i as f32 * 2.0 * PI) / num_test_directions as f32;
639                let escape_direction = base_direction.rotate(angle);
640                let test_target_2d = enemy_2d + escape_direction * escape_radius;
641                let test_target = Vec3::from_2d(test_target_2d, self.position[1]);
642
643                if self.is_safe_path(
644                    self.position,
645                    *test_target.as_array(),
646                    &line_intersects_geometry,
647                ) {
648                    self.current_target = Some(*test_target.as_array());
649                    return;
650                }
651            }
652        }
653    }

655    /// Enhanced path safety checking with level-based precision.
656    ///
657    /// This method validates that a path from start to end is safe by checking
658    /// for collisions at multiple points along the path. Higher levels use
659    /// more thorough checking with smaller collision buffers.
660    ///
661    /// # Arguments
662    ///
663    /// * `start` - Starting position `[x, y, z]`
664    /// * `end` - Ending position `[x, y, z]`
665    /// * `line_intersects_geometry` - Function to check for collision with geometry
666    ///
667    /// # Returns
668    ///
669    /// `true` if the path is safe, `false` if it intersects with geometry.
670    ///
671    /// # Algorithm
672    ///
673    /// 1. Divides path into multiple check points based on level
674    /// 2. Checks each point for collision with geometry
675    /// 3. Performs buffer checking perpendicular to path direction
676    /// 4. Higher levels use more check points and smaller buffers
677    ///
678    /// # Example
679    ///
680    /// ```rust
681    /// let is_safe = pathfinder.is_safe_path(
682    ///     [100.0, 30.0, 100.0],  // start
683    ///     [150.0, 30.0, 150.0],  // end
684    ///     |start, end| collision_system.intersects(start, end)
685    /// );
686    ///
687    /// if is_safe {
688    ///     // Path is clear, enemy can move
689    /// }
690    /// ```
691    fn is_safe_path<F>(&self, start: [f32; 3], end: [f32; 3], line_intersects_geometry: F) -> bool
692    where
693        F: Fn([f32; 3], [f32; 3]) -> bool,
694    {
695        let start_vec = Vec3(start);
696        let end_vec = Vec3(end);
697        let direction = (end_vec - start_vec).normalize();
698        let distance = start_vec.distance_to(&end_vec);
699
700        // More thorough checking at higher levels
701        let num_checks = 5 + (self.aggression_level * 2) as usize;
702        let step_size = distance / num_checks as f32;
703
704        // Smaller collision buffer at higher levels (more risk-taking)
705        let collision_buffer = (25.0 / (1.0 + self.aggression_level as f32 * 0.1)).max(10.0);
706
707        for i in 0..=num_checks {
708            let t = i as f32 * step_size;
709            let test_point = start_vec + direction * t;
710
711            if line_intersects_geometry(start, *test_point.as_array()) {
712                return false;
713            }
714
715            // Buffer checking with level-based precision
716            let perpendicular = Vec3([direction.as_array()[1], -direction.as_array()[0], 0.0]);
717
718            for &side in &[-1.0, 1.0] {
719                let buffer_point = test_point + perpendicular * collision_buffer * side;
720                if line_intersects_geometry(*test_point.as_array(), *buffer_point.as_array()) {
721                    return false;
722                }
723            }
724        }
725
726        true
727    }

732    ///
733    /// * `new_position` - New 3D position `[x, y, z]`
734    ///
735    /// # Example
736    ///
737    /// ```rust
738    /// pathfinder.set_position([120.0, 30.0, 120.0]);

749    /// The direction is a normalized 3D vector `[x, y, z]`.

815/// Places an enemy strategically with level-based positioning.
816///
817/// This function calculates an optimal position for an enemy based on the
818/// player's position, exit position, and current game level. Higher levels
819/// place enemies closer to the exit for increased challenge.
820///
821/// # Arguments
822///
823/// * `exit_position` - 3D position of the level exit `[x, y, z]`
824/// * `player_position` - 3D position of the player `[x, y, z]`
825/// * `level` - Current game level for positioning scaling
826/// * `placement_factor` - Factor between 0.0 and 1.0 for placement along player-exit line
827/// * `offset_distance` - Optional perpendicular offset from the main path
828/// * `line_intersects_geometry` - Function to check for collision with geometry
829///
830/// # Returns
831///
832/// A new `Enemy` instance positioned strategically relative to the player and exit.
833///
834/// # Placement Strategy
835///
836/// 1. **Base position**: Calculated along the line from player to exit
837/// 2. **Level scaling**: Higher levels place enemies closer to exit
838/// 3. **Offset application**: Applies perpendicular offset if specified
839/// 4. **Validation**: Ensures position doesn't intersect with geometry
840/// 5. **Path radius**: Scaled based on level and distance to exit
841///
842/// # Example
843///
844/// ```rust
845/// let enemy = place_enemy(
846///     [200.0, 30.0, 200.0],  // exit position
847///     [50.0, 30.0, 50.0],    // player position
848///     3,                     // level 3
849///     0.6,                   // 60% along player-exit line
850///     Some(50.0),            // 50 unit offset
851///     |start, end| collision_system.intersects(start, end)
852/// );
853/// ```
854pub fn place_enemy<F>(
855    exit_position: [f32; 3],
856    player_position: [f32; 3],
857    level: u32,
858    placement_factor: f32,
859    offset_distance: Option<f32>,
860    line_intersects_geometry: F,
861) -> Enemy
862where
863    F: Fn([f32; 3], [f32; 3]) -> bool,
864{
865    let player_vec = Vec3(player_position);
866    let exit_vec = Vec3(exit_position);
867
868    let direction_to_exit = (exit_vec - player_vec).normalize();
869    let distance_to_exit = player_vec.distance_to(&exit_vec);
870
871    // Higher levels place enemies closer to the exit (more challenging)
872    let level_factor = (level as f32 * 0.05).min(0.3);
873    let adjusted_factor = (placement_factor + level_factor).clamp(0.1, 0.95);
874
875    let base_position = player_vec + direction_to_exit * (distance_to_exit * adjusted_factor);
876
877    // Apply offset with level-based variation
878    let final_position = if let Some(offset) = offset_distance {
879        let level_offset_multiplier = 1.0 + (level as f32 * 0.1);
880        let adjusted_offset = offset * level_offset_multiplier;
881
882        let perpendicular = Vec3([
883            -direction_to_exit.as_array()[1],
884            direction_to_exit.as_array()[0],
885            0.0,
886        ])
887        .normalize();
888
889        let offset_positions = [
890            base_position + perpendicular * adjusted_offset,
891            base_position - perpendicular * adjusted_offset,
892        ];
893
894        let mut valid_position = base_position;
895        for &test_position in &offset_positions {
896            if !line_intersects_geometry(player_position, *test_position.as_array())
897                && !line_intersects_geometry(*test_position.as_array(), exit_position)
898            {
899                valid_position = test_position;
900                break;
901            }
902        }
903
904        valid_position
905    } else {
906        base_position
907    };
908
909    let validated_position = validate_enemy_position(
910        *final_position.as_array(),
911        player_position,
912        exit_position,
913        &line_intersects_geometry,
914    );
915
916    // Smaller path radius at higher levels (more aggressive)
917    let base_radius = (distance_to_exit * 0.3).clamp(50.0, 200.0);
918    let level_radius_reduction = (level as f32 * 0.05).min(0.4);
919    let path_radius = base_radius * (1.0 - level_radius_reduction);
920
921    let mut enemy = Enemy::new(validated_position, path_radius);
922    enemy.scale_aggression_by_level(level);
923    enemy
924}

926/// Validates and adjusts enemy position to ensure it doesn't intersect with geometry.
927///
928/// This function checks if the proposed enemy position creates valid paths to both
929/// the player and exit. If not, it searches for alternative positions in a circular
930/// pattern around the proposed position.
931///
932/// # Arguments
933///
934/// * `proposed_position` - The initial proposed position `[x, y, z]`
935/// * `player_position` - Current player position `[x, y, z]`
936/// * `exit_position` - Level exit position `[x, y, z]`
937/// * `line_intersects_geometry` - Function to check for collision with geometry
938///
939/// # Returns
940///
941/// A validated position that doesn't intersect with geometry, or a fallback position.
942///
943/// # Algorithm
944///
945/// 1. Check if proposed position creates valid paths to player and exit
946/// 2. If invalid, search in 8 directions around the proposed position
947/// 3. If still invalid, return midpoint between player and exit
948///
949/// # Example
950///
951/// ```rust
952/// let valid_position = validate_enemy_position(
953///     [100.0, 30.0, 100.0],  // proposed position
954///     [50.0, 30.0, 50.0],    // player position
955///     [200.0, 30.0, 200.0],  // exit position
956///     |start, end| collision_system.intersects(start, end)
957/// );
958/// ```
959fn validate_enemy_position<F>(
960    proposed_position: [f32; 3],
961    player_position: [f32; 3],
962    exit_position: [f32; 3],
963    line_intersects_geometry: F,
964) -> [f32; 3]
965where
966    F: Fn([f32; 3], [f32; 3]) -> bool,
967{
968    if line_intersects_geometry(player_position, proposed_position)
969        || line_intersects_geometry(proposed_position, exit_position)
970    {
971        let base_vec = Vec3(proposed_position);
972        let search_radius = 50.0;
973        let search_angles = [0.0, 45.0, 90.0, 135.0, 180.0, 225.0, 270.0, 315.0];
974
975        for &angle in &search_angles {
976            let angle_rad = (angle as f32).to_radians();
977            let offset = Vec3([
978                search_radius * angle_rad.cos(),
979                search_radius * angle_rad.sin(),
980                0.0,
981            ]);
982
983            let test_position = base_vec + offset;
984            let test_pos_array = *test_position.as_array();
985
986            if !line_intersects_geometry(player_position, test_pos_array)
987                && !line_intersects_geometry(test_pos_array, exit_position)
988            {
989                return test_pos_array;
990            }
991        }
992
993        let player_vec = Vec3(player_position);
994        let exit_vec = Vec3(exit_position);
995        let center_position = (player_vec + exit_vec) * 0.5;
996
997        *center_position.as_array()
998    } else {
999        proposed_position
1000    }
1001}
