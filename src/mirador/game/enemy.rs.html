6pub struct Enemy {

12impl Enemy {
13    pub fn new(position: [f32; 3], path_radius: f32) -> Self {
14        Self {
15            size: 100.0, // Default sprite size
16            pathfinder: EnemyPathfinder::new(position, path_radius),
17            speed: 100.0,
18        }
19    }
20
21    pub fn update<F>(
22        &mut self,
23        player_position: [f32; 3],
24        delta_time: f32,
25        line_intersects_geometry: F,
26    ) where
27        F: Fn([f32; 3], [f32; 3]) -> bool,
28    {
29        // Update pathfinding
30        if let Some(_target) = self
31            .pathfinder
32            .update(player_position, line_intersects_geometry)
33        {
34            // Move towards the target
35            if let Some(direction) = self.pathfinder.get_movement_direction() {
36                let direction_vec = Vec3(direction);
37                let movement = direction_vec * self.speed * delta_time;
38
39                let position_vec = Vec3(self.pathfinder.position);
40                let new_position = position_vec + movement;
41
42                self.pathfinder.set_position(*new_position.as_array());
43            }
44        }
45    }
46}

49pub struct EnemyPathfinder {
50    pub position: [f32; 3],
51    pub current_target: Option<[f32; 3]>,
52    pub path_radius: f32,
53    pub rotation_step: f32, // Radians per step when rotating around obstacles
54    pub arrival_threshold: f32, // Distance threshold to consider target reached
55}
56
57impl EnemyPathfinder {
58    pub fn new(position: [f32; 3], path_radius: f32) -> Self {
59        Self {
60            position,
61            current_target: None,
62            path_radius,
63            rotation_step: PI / 16.0, // 11.25 degrees per step
64            arrival_threshold: 0.5,
65        }
66    }
67
68    /// Main pathfinding update function
69    /// Should be called every frame or when the enemy needs to recalculate its path
70    pub fn update<F>(
71        &mut self,
72        player_position: [f32; 3],
73        line_intersects_geometry: F,
74    ) -> Option<[f32; 3]>
75    where
76        F: Fn([f32; 3], [f32; 3]) -> bool,
77    {
78        // Check if we need to calculate a new target
79        if self.needs_new_target() {
80            self.calculate_new_target(player_position, line_intersects_geometry);
81        }
82
83        self.current_target
84    }
85
86    /// Determines if the enemy needs a new target
87    fn needs_new_target(&self) -> bool {
88        match self.current_target {
89            None => true,
90            Some(target) => {
91                // Check if we've reached the current target
92                let enemy_vec = Vec3(self.position);
93                let target_vec = Vec3(target);
94                enemy_vec.distance_to(&target_vec) <= self.arrival_threshold
95            }
96        }
97    }
98
99    /// Calculates a new target point for the enemy
100    fn calculate_new_target<F>(&mut self, player_position: [f32; 3], line_intersects_geometry: F)
101    where
102        F: Fn([f32; 3], [f32; 3]) -> bool,
103    {
104        let enemy_vec = Vec3(self.position);
105        let player_vec = Vec3(player_position);
106        let enemy_2d = enemy_vec.to_2d();
107        let player_2d = player_vec.to_2d();
108
109        let direction_to_player = (player_2d - enemy_2d).normalize();
110
111        if enemy_2d.distance_to(&player_2d) < self.path_radius {
112            self.current_target = Some(self.position);
113            return;
114        }
115
116        let ideal_target_2d = enemy_2d + direction_to_player * self.path_radius;
117        let ideal_target = Vec3::from_2d(ideal_target_2d, self.position[1]);
118
119        if !line_intersects_geometry(self.position, *ideal_target.as_array()) {
120            self.current_target = Some(*ideal_target.as_array());
121            return;
122        }
123
124        // Rotation bias: rotate slightly more per step
125        let rotation_bias = 1.25; // 25% more than base
126        let adjusted_step = self.rotation_step * rotation_bias;
127        let max_rotations = (2.0 * PI / adjusted_step) as i32;
128
129        let mut current_direction = direction_to_player;
130
131        for _ in 0..max_rotations {
132            current_direction = current_direction.rotate(adjusted_step);
133            let test_target_2d = enemy_2d + current_direction * self.path_radius;
134            let test_target = Vec3::from_2d(test_target_2d, self.position[1]);
135
136            if !line_intersects_geometry(self.position, *test_target.as_array()) {
137                self.current_target = Some(*test_target.as_array());
138                return;
139            }
140        }
141
142        self.current_target = Some(self.position);
143    }
144
145    /// Updates the enemy's position (call this when the enemy moves)
146    pub fn set_position(&mut self, new_position: [f32; 3]) {
147        self.position = new_position;
148    }
149
150    /// Gets the current movement direction
151    pub fn get_movement_direction(&self) -> Option<[f32; 3]> {
152        self.current_target.map(|target| {
153            let position_vec = Vec3(self.position);
154            let target_vec = Vec3(target);
155            let direction = (target_vec - position_vec).normalize();
156            *direction.as_array()
157        })
158    }
159
160    /// Gets the distance to the current target
161    pub fn distance_to_target(&self) -> Option<f32> {
162        self.current_target.map(|target| {
163            let position_vec = Vec3(self.position);
164            let target_vec = Vec3(target);
165            position_vec.distance_to(&target_vec)
166        })
167    }
168
169    /// Check if the enemy has reached its current target
170    pub fn has_reached_target(&self) -> bool {
171        match self.current_target {
172            Some(target) => {
173                let position_vec = Vec3(self.position);
174                let target_vec = Vec3(target);
175                position_vec.distance_to(&target_vec) <= self.arrival_threshold
176            }
177            None => true,
178        }
179    }
