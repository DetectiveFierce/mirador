1//! Upgrade system for the Mirador game.
2//!
3//! This module provides a comprehensive upgrade system that allows players to enhance their
4//! abilities throughout the game. Upgrades are categorized by rarity and can be randomly
5//! selected based on weighted probabilities.
6//!
7//! # Overview
8//!
9//! The upgrade system consists of:
10//! - `Upgrade`: Represents a single upgrade with name, tooltip, and rarity
11//! - `UpgradeRarity`: Defines the rarity levels and their associated weights
12//! - `AvailableUpgrade`: Enum of all possible upgrades in the game
13//! - `UpgradeManager`: Manages player upgrades and provides selection logic
14//!
15//! # Usage
16//!
17//! ```rust
18//! use mirador::game::upgrades::{UpgradeManager, AvailableUpgrade};
19//!
20//! let mut manager = UpgradeManager::new();
21//!
22//! // Select 3 random upgrades for the player to choose from
23//! let options = manager.select_random_upgrades(3);
24//!
25//! // Apply a chosen upgrade
26//! manager.apply_upgrade(&AvailableUpgrade::SpeedUp);
27//!
28//! // Check how many of a specific upgrade the player has
29//! let speed_count = manager.get_upgrade_count(&AvailableUpgrade::SpeedUp);
30//! ```

35/// Represents a single upgrade that can be applied to the player.
36///
37/// Each upgrade has a name, descriptive tooltip, and rarity level that determines
38/// how likely it is to appear in random selections.
39#[derive(Debug, Clone)]
40pub struct Upgrade {
41    /// The display name of the upgrade
42    pub name: String,
43    /// A description of what the upgrade does
44    pub tooltip: String,
45    /// The rarity level of the upgrade, affecting selection probability
46    pub rarity: UpgradeRarity,
47}
48
49/// Defines the rarity levels for upgrades and their associated weights.
50///
51/// Rarer upgrades have lower weights, making them less likely to appear
52/// in random selections. The weights are used for weighted random selection.
53#[derive(Debug, Clone, PartialEq)]
54pub enum UpgradeRarity {
55    /// Common upgrades (40% chance)
56    Common,
57    /// Uncommon upgrades (30% chance)
58    Uncommon,
59    /// Rare upgrades (20% chance)
60    Rare,
61    /// Epic upgrades (8% chance)
62    Epic,
63    /// Legendary upgrades (2% chance)
64    Legendary,
65}
66
67impl UpgradeRarity {
68    /// Returns the weight value for this rarity level.
69    ///
70    /// Weights are used in weighted random selection. Higher weights
71    /// mean the upgrade is more likely to be selected.
72    ///
73    /// # Returns
74    ///
75    /// A `f32` representing the weight of this rarity level.
76    ///
77    /// # Example
78    ///
79    /// ```rust
80    /// use mirador::game::upgrades::UpgradeRarity;
81    ///
82    /// assert_eq!(UpgradeRarity::Common.weight(), 40.0);
83    /// assert_eq!(UpgradeRarity::Legendary.weight(), 2.0);
84    /// ```
85    pub fn weight(&self) -> f32 {
86        match self {
87            UpgradeRarity::Common => 40.0,   // 40% chance
88            UpgradeRarity::Uncommon => 30.0, // 30% chance
89            UpgradeRarity::Rare => 20.0,     // 20% chance
90            UpgradeRarity::Epic => 8.0,      // 8% chance
91            UpgradeRarity::Legendary => 2.0, // 2% chance
92        }
93    }
94}

96/// Enum representing all available upgrades in the game.
97///
98/// Each variant corresponds to a specific upgrade that can be applied
99/// to the player. This enum is used as a key in the upgrade manager
100/// to track how many of each upgrade the player has collected.
101#[derive(Debug, Clone, PartialEq, Eq, Hash)]
102pub enum AvailableUpgrade {
103    /// Increases player movement speed
104    SpeedUp,
105    /// Makes time pass slower, giving more time to navigate
106    SlowTime,
107    /// Reduces noise made while moving
108    SilentStep,
109    /// Makes the player taller to see over walls
110    TallBoots,
111    /// Prevents enemy movement at level start
112    HeadStart,
113    /// Increases maximum stamina for longer sprinting
114    Dash,
115    /// A mysterious upgrade with unknown effects
116    Unknown,
117}
118
119impl AvailableUpgrade {
120    /// Converts this upgrade variant into a full `Upgrade` struct.
121    ///
122    /// This method provides the name, tooltip, and rarity for each upgrade type.
123    ///
124    /// # Returns
125    ///
126    /// A complete `Upgrade` struct with all information populated.
127    ///
128    /// # Example
129    ///
130    /// ```rust
131    /// use mirador::game::upgrades::{AvailableUpgrade, UpgradeRarity};
132    ///
133    /// let upgrade = AvailableUpgrade::SpeedUp.to_upgrade();
134    /// assert_eq!(upgrade.name, "Speed Up");
135    /// assert_eq!(upgrade.rarity, UpgradeRarity::Common);
136    /// ```
137    pub fn to_upgrade(&self) -> Upgrade {
138        match self {
139            AvailableUpgrade::SpeedUp => Upgrade {
140                name: "Speed Up".to_string(),
141                tooltip: "Increases your movement speed, making you faster and more agile."
142                    .to_string(),
143                rarity: UpgradeRarity::Common,
144            },
145            AvailableUpgrade::SlowTime => Upgrade {
146                name: "Slow Time".to_string(),
147                tooltip: "Each second lasts longer, giving you more time to navigate the maze."
148                    .to_string(),
149                rarity: UpgradeRarity::Uncommon,
150            },
151            AvailableUpgrade::SilentStep => Upgrade {
152                name: "Silent Step".to_string(),
153                tooltip: "Reduces the noise you make while moving, making you harder to detect."
154                    .to_string(),
155                rarity: UpgradeRarity::Rare,
156            },
157            AvailableUpgrade::TallBoots => Upgrade {
158                name: "Tall Boots".to_string(),
159                tooltip: "Makes you taller, allowing you to better see over the walls of the maze."
160                    .to_string(),
161                rarity: UpgradeRarity::Uncommon,
162            },
163            AvailableUpgrade::HeadStart => Upgrade {
164                name: "Head Start".to_string(),
165                tooltip:
166                    "Prevents the enemy from moving for a short time at the start of each level."
167                        .to_string(),
168                rarity: UpgradeRarity::Rare,
169            },
170            AvailableUpgrade::Dash => Upgrade {
171                name: "Dash".to_string(),
172                tooltip: "Increase your maximum stamina, allowing you to sprint for longer."
173                    .to_string(),
174                rarity: UpgradeRarity::Epic,
175            },
176            AvailableUpgrade::Unknown => Upgrade {
177                name: "Unknown".to_string(),
178                tooltip: "A mysterious upgrade with unpredictable effects. What could it do?"
179                    .to_string(),
180                rarity: UpgradeRarity::Legendary,
181            },
182        }
183    }

186/// Manages the player's upgrades and provides functionality for upgrade selection.
187///
188/// The `UpgradeManager` tracks how many of each upgrade the player has collected
189/// and provides methods for selecting random upgrades and applying new ones.
190///
191/// # Examples
192///
193/// ```rust
194/// use mirador::game::upgrades::{UpgradeManager, AvailableUpgrade};
195///
196/// let mut manager = UpgradeManager::new();
197///
198/// // Select 3 random upgrades for the player to choose from
199/// let options = manager.select_random_upgrades(3);
200///
201/// // Apply a chosen upgrade
202/// manager.apply_upgrade(&AvailableUpgrade::SpeedUp);
203///
204/// // Check upgrade count
205/// assert_eq!(manager.get_upgrade_count(&AvailableUpgrade::SpeedUp), 1);
206/// ```
207pub struct UpgradeManager {
208    /// Maps each upgrade type to the number of times the player has collected it
209    pub player_upgrades: HashMap<AvailableUpgrade, u32>, // upgrade -> count
210}
211
212impl UpgradeManager {
213    /// Creates a new `UpgradeManager` with no upgrades collected.
214    ///
215    /// # Returns
216    ///
217    /// A new `UpgradeManager` instance with an empty upgrade collection.
218    pub fn new() -> Self {
219        Self {
220            player_upgrades: HashMap::new(),
221        }
222    }
223
224    /// Selects a specified number of random upgrades for the player to choose from.
225    ///
226    /// This method uses weighted random selection based on upgrade rarity to ensure
227    /// that rarer upgrades appear less frequently. No duplicate upgrades are returned
228    /// in a single selection.
229    ///
230    /// # Arguments
231    ///
232    /// * `count` - The number of upgrades to select
233    ///
234    /// # Returns
235    ///
236    /// A vector of `Upgrade` structs. If `count` is greater than the number of
237    /// available upgrades, fewer upgrades than requested may be returned.
238    ///
239    /// # Examples
240    ///
241    /// ```rust
242    /// use mirador::game::upgrades::UpgradeManager;
243    ///
244    /// let manager = UpgradeManager::new();
245    /// let selected = manager.select_random_upgrades(3);
246    ///
247    /// // Should get exactly 3 unique upgrades
248    /// assert_eq!(selected.len(), 3);
249    ///
250    /// // All upgrades should be unique
251    /// let names: Vec<String> = selected.iter().map(|u| u.name.clone()).collect();
252    /// let unique_names: Vec<String> = names.iter().cloned().collect::<std::collections::HashSet<_>>().into_iter().collect();
253    /// assert_eq!(names.len(), unique_names.len());
254    /// ```
255    pub fn select_random_upgrades(&self, count: usize) -> Vec<Upgrade> {
256        let mut rng = rand::thread_rng();
257        let mut selected_upgrades = Vec::new();
258
259        // Get all available upgrades
260        let mut available_upgrades = vec![
261            AvailableUpgrade::SpeedUp,
262            AvailableUpgrade::SlowTime,
263            AvailableUpgrade::SilentStep,
264            AvailableUpgrade::TallBoots,
265            AvailableUpgrade::HeadStart,
266            AvailableUpgrade::Dash,
267        ];
268
269        // Weighted random selection based on rarity, ensuring no duplicates
270        for _ in 0..count {
271            // If we've run out of upgrades, break to avoid infinite loop
272            // This can happen if count > available_upgrades.len()
273            if available_upgrades.is_empty() {
274                break;
275            }
276
277            // Calculate total weight of remaining upgrades
278            let total_weight: f32 = available_upgrades
279                .iter()
280                .map(|upgrade| upgrade.to_upgrade().rarity.weight())
281                .sum();
282
283            // Generate random value
284            let random_value = rng.gen_range(0.0..total_weight);
285
286            // Find the upgrade based on weight
287            let mut cumulative_weight = 0.0;
288            let mut selected_index = 0; // Default to first upgrade
289
290            for (index, upgrade) in available_upgrades.iter().enumerate() {
291                cumulative_weight += upgrade.to_upgrade().rarity.weight();
292                if cumulative_weight >= random_value {
293                    selected_index = index;
294                    break;
295                }
296            }
297
298            // Get the selected upgrade and remove it from available pool
299            let selected_upgrade = available_upgrades.remove(selected_index);
300
301            // Convert to Upgrade and add to selection
302            selected_upgrades.push(selected_upgrade.to_upgrade());
303        }
304
305        // Note: If count > available_upgrades.len(), this will return fewer upgrades than requested
306        // This is the expected behavior to ensure no duplicates
307        selected_upgrades
308    }
309
310    /// Applies an upgrade to the player, incrementing its count.
311    ///
312    /// This method increases the count of the specified upgrade by 1.
313    /// If the upgrade hasn't been collected before, it will be added
314    /// to the collection with a count of 1.
315    ///
316    /// # Arguments
317    ///
318    /// * `upgrade` - The upgrade to apply
319    ///
320    /// # Examples
321    ///
322    /// ```rust
323    /// use mirador::game::upgrades::{UpgradeManager, AvailableUpgrade};
324    ///
325    /// let mut manager = UpgradeManager::new();
326    ///
327    /// // Apply an upgrade for the first time
328    /// manager.apply_upgrade(&AvailableUpgrade::SpeedUp);
329    /// assert_eq!(manager.get_upgrade_count(&AvailableUpgrade::SpeedUp), 1);
330    ///
331    /// // Apply the same upgrade again
332    /// manager.apply_upgrade(&AvailableUpgrade::SpeedUp);
333    /// assert_eq!(manager.get_upgrade_count(&AvailableUpgrade::SpeedUp), 2);
334    /// ```
335    pub fn apply_upgrade(&mut self, upgrade: &AvailableUpgrade) {
336        let current_count = *self.player_upgrades.get(upgrade).unwrap_or(&0);
337        self.player_upgrades
338            .insert(upgrade.clone(), current_count + 1);
339
340        let upgrade_info = upgrade.to_upgrade();
341        println!(
342            "Applied upgrade: {} (Total: {})",
343            upgrade_info.name,
344            current_count + 1
345        );
346    }

348    /// Returns the number of times the player has collected a specific upgrade.
349    ///
350    /// # Arguments
351    ///
352    /// * `upgrade` - The upgrade to check
353    ///
354    /// # Returns
355    ///
356    /// The number of times the upgrade has been collected, or 0 if never collected.
357    ///
358    /// # Examples
359    ///
360    /// ```rust
361    /// use mirador::game::upgrades::{UpgradeManager, AvailableUpgrade};
362    ///
363    /// let mut manager = UpgradeManager::new();
364    ///
365    /// // Initially no upgrades collected
366    /// assert_eq!(manager.get_upgrade_count(&AvailableUpgrade::SpeedUp), 0);
367    ///
368    /// // After applying an upgrade
369    /// manager.apply_upgrade(&AvailableUpgrade::SpeedUp);
370    /// assert_eq!(manager.get_upgrade_count(&AvailableUpgrade::SpeedUp), 1);
371    /// ```
372    pub fn get_upgrade_count(&self, upgrade: &AvailableUpgrade) -> u32 {
373        *self.player_upgrades.get(upgrade).unwrap_or(&0)
374    }

376    /// Returns display information for an upgrade, including level and tooltip.
377    ///
378    /// This method is useful for UI display, providing both the current level
379    /// of the upgrade (if collected) and its tooltip description.
380    ///
381    /// # Arguments
382    ///
383    /// * `upgrade` - The upgrade to get display info for
384    ///
385    /// # Returns
386    ///
387    /// A tuple containing:
388    /// - `String`: The level text (e.g., "Level 2" or "New Upgrade")
389    /// - `String`: The tooltip description
390    ///
391    /// # Examples
392    ///
393    /// ```rust
394    /// use mirador::game::upgrades::{UpgradeManager, AvailableUpgrade};
395    ///
396    /// let mut manager = UpgradeManager::new();
397    /// let upgrade = AvailableUpgrade::SpeedUp.to_upgrade();
398    ///
399    /// // For a new upgrade
400    /// let (level, tooltip) = manager.get_upgrade_display_info(&upgrade);
401    /// assert_eq!(level, "New Upgrade");
402    /// assert!(tooltip.contains("movement speed"));
403    ///
404    /// // After applying the upgrade
405    /// manager.apply_upgrade(&AvailableUpgrade::SpeedUp);
406    /// let (level, tooltip) = manager.get_upgrade_display_info(&upgrade);
407    /// assert_eq!(level, "Level 1");
408    /// ```
409    pub fn get_upgrade_display_info(&self, upgrade: &Upgrade) -> (String, String) {
410        // Find the corresponding AvailableUpgrade
411        let available_upgrade = self.find_available_upgrade(upgrade);
412        let current_count = self.get_upgrade_count(&available_upgrade);
413
414        let level_text = if current_count > 0 {
415            format!("Level {}", current_count)
416        } else {
417            "New Upgrade".to_string()
418        };
419
420        let tooltip_text = upgrade.tooltip.clone();
421
422        (level_text, tooltip_text)
423    }

425    /// Finds the corresponding `AvailableUpgrade` for a given `Upgrade` struct.
426    ///
427    /// This is a helper method that maps from the display `Upgrade` back to
428    /// the enum variant used internally for tracking.
429    ///
430    /// # Arguments
431    ///
432    /// * `upgrade` - The upgrade to find the enum variant for
433    ///
434    /// # Returns
435    ///
436    /// The corresponding `AvailableUpgrade` enum variant, or `SpeedUp` as fallback.
437    fn find_available_upgrade(&self, upgrade: &Upgrade) -> AvailableUpgrade {
438        // Find the AvailableUpgrade that matches this Upgrade
439        let all_upgrades = vec![
440            AvailableUpgrade::SpeedUp,
441            AvailableUpgrade::SlowTime,
442            AvailableUpgrade::SilentStep,
443            AvailableUpgrade::TallBoots,
444            AvailableUpgrade::HeadStart,
445            AvailableUpgrade::Dash,
446        ];
447
448        for available in all_upgrades {
449            if available.to_upgrade().name == upgrade.name {
450                return available;
451            }
452        }
453
454        // Fallback
455        AvailableUpgrade::SpeedUp
456    }
457}

463    /// Tests that no duplicate upgrades are returned in a single selection.
464    #[test]
465    fn test_no_duplicates_in_selection() {
466        let upgrade_manager = UpgradeManager::new();
467
468        // Test multiple selections to ensure no duplicates
469        for _ in 0..10 {
470            let selected = upgrade_manager.select_random_upgrades(3);
471
472            // Check that we got exactly 3 upgrades
473            assert_eq!(selected.len(), 3);
474
475            // Check that all upgrades are unique
476            let mut names: Vec<String> = selected.iter().map(|u| u.name.clone()).collect();
477            names.sort();
478            names.dedup(); // Remove duplicates
479            assert_eq!(names.len(), 3, "Found duplicate upgrades: {:?}", selected);
480        }
481    }
