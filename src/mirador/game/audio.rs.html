83    /// Start or restart the background music track
84    pub fn start_background_music(&mut self) -> Result<(), Box<dyn Error>> {
85        // Stop existing background music if playing
86        if let Some(mut handle) = self.background_music_handle.take() {
87            handle.stop(Tween::default());
88        }
89
90        // Create settings for background music with low volume and looping
91        let settings = StaticSoundSettings::new()
92            .volume(Decibels::from(-20.0)) // Low volume (-20dB)
93            .loop_region(..); // Loop the entire track
94
95        // Play the background music
96        let handle = self
97            .audio_manager
98            .play(self.background_music_data.clone().with_settings(settings))?;
99        self.background_music_handle = Some(handle);
100
101        Ok(())
102    }

109    /// Adjust audio volumes for title screen (louder background music, quieter enemies)
110    pub fn set_title_screen_volumes(&mut self) -> Result<(), Box<dyn Error>> {
111        // Make background music louder on title screen
112        if self.background_music_handle.is_some() {
113            let handle = self.background_music_handle.as_mut().unwrap();
114            let tween = Tween {
115                start_time: StartTime::Immediate,
116                duration: Duration::from_millis(500), // Smooth transition
117                easing: Easing::Linear,
118            };
119            handle.set_volume(Decibels::from(-5.0), tween);
120        }
121
122        // Make enemy sounds quieter on title screen by adjusting the spatial track volume
123        let enemy_ids: Vec<String> = self.enemy_sounds.keys().cloned().collect();
124        for enemy_id in enemy_ids {
125            if let Some(track) = self.spatial_tracks.get_mut(&enemy_id) {
126                let tween = Tween {
127                    start_time: StartTime::Immediate,
128                    duration: Duration::from_millis(500), // Smooth transition
129                    easing: Easing::Linear,
130                };
131                // Adjust the spatial track's volume directly
132                track.set_volume(Decibels::from(-10.0), tween);
133            }
134        }
135
136        Ok(())
137    }

197    fn set_movement_state(&mut self, new_state: MovementState) -> Result<(), Box<dyn Error>> {
198        if self.movement_state != new_state {
199            // Stop current footstep sound if any
200            if let Some(mut handle) = self.footstep_sound.take() {
201                handle.stop(Tween::default());
202            }
203
204            self.movement_state = new_state;
205
206            // Start new footstep sound based on state
207            match new_state {
208                MovementState::Idle => {
209                    // No footstep sound for idle
210                }
211                MovementState::Walking => {
212                    let mut sound_handle = self.audio_manager.play(self.footstep_data.clone())?;
213                    sound_handle.set_loop_region(0.0..0.5); // Normal walking speed
214                    self.footstep_sound = Some(sound_handle);
215                }
216                MovementState::Sprinting => {
217                    let mut sound_handle = self.audio_manager.play(self.footstep_data.clone())?;
218                    sound_handle.set_loop_region(0.0..0.25); // Faster loop for sprinting
219                    self.footstep_sound = Some(sound_handle);
220                }
221            }
222        }
223        Ok(())
224    }

245    pub fn spawn_enemy(
246        &mut self,
247        enemy_id: String,
248        position: [f32; 3],
249    ) -> Result<(), Box<dyn Error>> {
250        let mut spatial_track = self.audio_manager.add_spatial_sub_track(
251            &self.listener,
252            position,
253            SpatialTrackBuilder::new()
254                .spatialization_strength(1.0)
255                .distances(SpatialTrackDistances {
256                    min_distance: 1.0,
257                    max_distance: 3200.0,
258                })
259                .with_effect(ReverbBuilder::new().mix(Value::Fixed(0.3.into())))
260                .with_effect(VolumeControlBuilder::new(Value::FromListenerDistance(
261                    Mapping {
262                        input_range: (5.0, 3200.0),
263                        output_range: ((20.0).into(), (-50.0).into()),
264                        easing: Easing::OutPowi(3),
265                    },
266                ))),
267        )?;
268
269        let sound_handle = spatial_track.play(self.enemy_data.clone().loop_region(..))?;
270
271        self.spatial_tracks.insert(enemy_id.clone(), spatial_track);
272        self.enemy_sounds.insert(enemy_id, sound_handle);
273        Ok(())
274    }

308        &mut self,
309        audio_data: StaticSoundData,
310        volume: f32,
311    ) -> Result<(), Box<dyn Error>> {
312        // Convert linear volume (0.0-1.0) to decibels
313        let volume_db = if volume <= 0.0 {
314            Decibels::SILENCE
315        } else {
316            volume.into()
317        };
318
319        // Create sound settings with the specified volume
320        let settings = StaticSoundSettings::new().volume(volume_db);
321
322        // Play the sound with the volume setting
323        self.audio_manager
324            .play(audio_data.with_settings(settings))?;
325
326        Ok(())
327    }

385    /// Play the wall hit sound effect
386    pub fn wall_hit(&mut self) -> Result<(), Box<dyn Error>> {
387        let now = Instant::now();
388
389        // Check if enough time has passed since last hit
390        if let Some(last_hit) = self.last_wall_hit {
391            if now.duration_since(last_hit) < self.wall_hit_cooldown {
392                return Ok(()); // Skip playing sound
393            }
394        }
395
396        // Play sound at 1/3 volume (0.33)
397        self.play_with_volume(self.wall_hit_data.clone(), 0.0001)?;
398        self.last_wall_hit = Some(now);
399        Ok(())
400    }
