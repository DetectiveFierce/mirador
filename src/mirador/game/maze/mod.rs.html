5

8use self::generator::Cell;
9use std::fs::File;
10use std::io::{BufRead, BufReader};
11/// Parses a maze file into a 2D vector of wall booleans and detects the exit cell if marked.
12///
13/// Each line of the file is read as a row of the maze. Each character is mapped as follows:
14/// - `#` becomes `true` (wall)
15/// - Any other character becomes `false` (open space)
16///
17/// The function also detects an optional exit cell, marked with `*`.
18/// The coordinates of the `*` character are interpreted in the wall grid, which includes both walls
19/// and spaces. To convert this to the maze cell coordinates (excluding walls), the function checks:
20/// - The row and column indices of `*` must both be **odd** (i.e., part of a valid cell, not a wall)
21/// - The maze cell coordinate is then computed by halving both indices:
22///   `maze_row = (maze_height - 1) - (row_idx / 2)`, `maze_col = col_idx / 2`
23///
24/// # Arguments
25/// * `path` - Path to the maze file to parse.
26///
27/// # Returns
28/// A tuple:
29/// - A 2D vector (`Vec<Vec<bool>>`) where `true` indicates a wall and `false` an open cell
30/// - An optional `Cell` representing the location of the exit cell, if found (with bottom-left origin)
31///
32/// # Panics
33/// - If the file cannot be opened.
34/// - If any line cannot be read.
35///
36/// # Example
37/// ```text
38/// # # #
39/// # * #
40/// # # #
41/// ```
42/// becomes:
43/// ```ignore
44/// vec![
45///     vec![true, false, true],
46///     vec![true, false, true],
47///     vec![true, false, true],
48/// ]
49/// exit_cell: Some(Cell { row: 0, col: 0 }) // bottom-left origin
50/// ```
51pub fn parse_maze_file(path: &str) -> (Vec<Vec<bool>>, Option<Cell>) {
52    let file = File::open(path).expect("Failed to open maze file");
53    let reader = BufReader::new(file);
54
55    let lines: Vec<String> = reader
56        .lines()
57        .map(|line| line.expect("Failed to read line"))
58        .collect();
59
60    let mut maze_grid = Vec::with_capacity(lines.len());
61    let mut exit_cell = None;
62
63    for (wall_row_idx, line) in lines.into_iter().enumerate() {
64        let mut row = Vec::new();
65        for (wall_col_idx, c) in line.chars().enumerate() {
66            let is_wall = c == '#';
67            row.push(is_wall);
68
69            // Check for exit marker
70            if c == '*' && wall_row_idx % 2 == 1 && wall_col_idx % 2 == 1 {
71                let maze_row = wall_row_idx;
72                let maze_col = wall_col_idx;
73                exit_cell = Some(Cell::new(maze_row, maze_col));
74            }
75        }
76        maze_grid.push(row);
77    }
78
79    (maze_grid, exit_cell)
80}
