38fn custom_slider(
39    ui: &mut egui::Ui,
40    label: &str,
41    value: &mut f32,
42    range: std::ops::RangeInclusive<f32>,
43    speed: f64,
44    decimals: usize,
45) {
46    ui.horizontal(|ui| {
47        ui.label(label);
48        ui.add(
49            egui::DragValue::new(value)
50                .speed(speed)
51                .fixed_decimals(decimals)
52                .range(range.clone())
53                .suffix(""),
54        );
55        ui.add(egui::Slider::new(value, range).show_value(false));
56    });
57}
