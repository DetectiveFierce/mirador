1//! # Rectangle Renderer for Menu Creation
2//!
3//! This module provides the core primitives for creating UI menus using WebGPU.
4//! It defines a flexible rectangle rendering system that can be used to build
5//! various menu components like buttons, panels, dropdowns, and other UI elements.
6//!
7//! ## Core Components
8//!
9//! - `Vertex`: Defines the data structure for each vertex in a rectangle
10//! - `Rectangle`: Represents a menu primitive with position, size, color, and styling
11//! - `RectangleRenderer`: Manages the rendering pipeline and batch rendering of rectangles
12//!
13//! ## Usage for Menu Creation
14//!
15//! This system is designed to be the foundation for menu systems where:
16//! - Each menu item can be represented as a `Rectangle`
17//! - Multiple rectangles can be batched and rendered efficiently
18//! - Rounded corners provide modern UI aesthetics
19//! - Alpha blending enables overlays and transparency effects
20

28/// Vertex data structure for rectangle rendering in menu systems.
29///
30/// Each vertex contains all the information needed to render a portion of a rectangle
31/// in the GPU shader. This design allows for efficient batch rendering of multiple
32/// menu elements while supporting advanced features like rounded corners.
33///
34/// ## Memory Layout
35///
36/// The struct uses `#[repr(C)]` to ensure consistent memory layout across platforms,
37/// which is crucial for GPU buffer compatibility. The total size is 48 bytes per vertex
38/// with proper 16-byte alignment.
39///
40/// ## Usage in Menu Creation
41///
42/// For a typical menu button, four vertices are created (one for each corner)
43/// with the same rectangle data but different UV coordinates to enable
44/// proper texture mapping and corner radius calculations.
45#[repr(C)]
46#[derive(Copy, Clone, Debug, bytemuck::Pod, bytemuck::Zeroable)]
47struct Vertex {
48    /// Screen position in normalized device coordinates (-1.0 to 1.0)
49    /// Used by the vertex shader to position the rectangle on screen
50    position: [f32; 2],
51
52    /// RGBA color values (0.0 to 1.0) for the rectangle
53    /// Enables different colored menu items (e.g., hover states, active buttons)
54    color: [f32; 4],
55
56    /// UV coordinates for texture mapping and distance calculations
57    /// Used in the fragment shader to determine pixel position within the rectangle
58    /// Essential for rounded corner calculations
59    uv: [f32; 2],
60
61    /// Original rectangle dimensions in screen pixels
62    /// Passed to fragment shader for accurate corner radius calculations
63    /// regardless of the rectangle's screen position
64    rect_size: [f32; 2],
65
66    /// Corner radius in pixels for rounded rectangles
67    /// Enables modern UI aesthetics for menu buttons and panels
68    corner_radius: f32,
69
70    /// Padding to ensure 16-byte alignment required by GPU buffers
71    _padding: f32,
72}
73
74impl Vertex {
75    /// Defines the vertex buffer layout for the GPU pipeline.
76    ///
77    /// This describes how vertex data is organized in memory and maps
78    /// each field to a shader input location. The GPU uses this information
79    /// to correctly interpret the vertex buffer data.
80    ///
81    /// ## Shader Locations
82    ///
83    /// - Location 0: Position (vec2)
84    /// - Location 1: Color (vec4)  
85    /// - Location 2: UV coordinates (vec2)
86    /// - Location 3: Rectangle size (vec2)
87    /// - Location 4: Corner radius (float)
88    fn desc<'a>() -> VertexBufferLayout<'a> {
89        VertexBufferLayout {
90            array_stride: mem::size_of::<Vertex>() as wgpu::BufferAddress,
91            step_mode: wgpu::VertexStepMode::Vertex,
92            attributes: &[
93                // Position attribute - where the vertex appears on screen
94                VertexAttribute {
95                    offset: 0,
96                    shader_location: 0,
97                    format: VertexFormat::Float32x2,
98                },
99                // Color attribute - RGBA color for this menu element
100                VertexAttribute {
101                    offset: mem::size_of::<[f32; 2]>() as wgpu::BufferAddress,
102                    shader_location: 1,
103                    format: VertexFormat::Float32x4,
104                },
105                // UV attribute - texture coordinates for corner radius calculations
106                VertexAttribute {
107                    offset: (mem::size_of::<[f32; 2]>() + mem::size_of::<[f32; 4]>())
108                        as wgpu::BufferAddress,
109                    shader_location: 2,
110                    format: VertexFormat::Float32x2,
111                },
112                // Rectangle size - original dimensions for distance calculations
113                VertexAttribute {
114                    offset: (mem::size_of::<[f32; 2]>()
115                        + mem::size_of::<[f32; 4]>()
116                        + mem::size_of::<[f32; 2]>())
117                        as wgpu::BufferAddress,
118                    shader_location: 3,
119                    format: VertexFormat::Float32x2,
120                },
121                // Corner radius - enables rounded rectangle rendering
122                VertexAttribute {
123                    offset: (mem::size_of::<[f32; 2]>()
124                        + mem::size_of::<[f32; 4]>()
125                        + mem::size_of::<[f32; 2]>()
126                        + mem::size_of::<[f32; 2]>())
127                        as wgpu::BufferAddress,
128                    shader_location: 4,
129                    format: VertexFormat::Float32,
130                },
131            ],
132        }
133    }
134}

136/// Represents a rectangular primitive for menu creation.
137///
138/// This is the fundamental building block for menu systems. Each `Rectangle`
139/// can represent various menu components:
140///
141/// - Menu buttons and items
142/// - Background panels and containers
143/// - Separators and dividers
144/// - Hover and selection highlights
145/// - Dropdown menus and tooltips
146///
147/// ## Coordinate System
148///
149/// Uses screen-space coordinates where (0,0) is the top-left corner,
150/// which is natural for UI layout systems.
151///
152/// ## Examples
153///
154/// ```rust
155/// // Create a menu button
156/// let button = Rectangle::new(100.0, 50.0, 200.0, 40.0, [0.2, 0.4, 0.8, 1.0])
157///     .with_corner_radius(8.0);
158///
159/// // Create a menu panel background
160/// let panel = Rectangle::new(0.0, 0.0, 300.0, 400.0, [0.1, 0.1, 0.1, 0.9])
161///     .with_corner_radius(12.0);
162/// ```
163#[derive(Debug, Clone)]
164pub struct Rectangle {
165    /// X coordinate of the rectangle's top-left corner in screen pixels
166    pub x: f32,
167
168    /// Y coordinate of the rectangle's top-left corner in screen pixels
169    pub y: f32,
170
171    /// Width of the rectangle in screen pixels
172    pub width: f32,
173
174    /// Height of the rectangle in screen pixels  
175    pub height: f32,
176
177    /// RGBA color values (0.0 to 1.0) for the rectangle
178    /// Alpha channel enables transparency for overlays and hover effects
179    pub color: [f32; 4],
180
181    /// Corner radius in pixels for rounded rectangles
182    /// Set to 0.0 for sharp corners, or positive values for rounded corners
183    pub corner_radius: f32,
184}
185
186impl Rectangle {
187    /// Creates a new rectangle with sharp corners.
188    ///
189    /// This is the primary constructor for menu primitives. The rectangle
190    /// is positioned using screen coordinates where (0,0) is top-left.
191    ///
192    /// ## Parameters
193    ///
194    /// - `x`, `y`: Top-left corner position in screen pixels
195    /// - `width`, `height`: Dimensions in screen pixels
196    /// - `color`: RGBA color array with values from 0.0 to 1.0
197    ///
198    /// ## Returns
199    ///
200    /// A new `Rectangle` with sharp corners (corner_radius = 0.0)
201    pub fn new(x: f32, y: f32, width: f32, height: f32, color: [f32; 4]) -> Self {
202        Self {
203            x,
204            y,
205            width,
206            height,
207            color,
208            corner_radius: 0.0,
209        }
210    }
211
212    /// Sets the corner radius for rounded rectangles.
213    ///
214    /// This builder method enables the creation of modern-looking menu
215    /// elements with rounded corners. The radius is applied to all four
216    /// corners equally.
217    ///
218    /// ## Parameters
219    ///
220    /// - `radius`: Corner radius in pixels. Should be <= min(width, height) / 2
221    ///
222    /// ## Returns
223    ///
224    /// The modified rectangle with the specified corner radius
225    ///
226    /// ## Usage
227    ///
228    /// ```rust
229    /// let rounded_button = Rectangle::new(x, y, w, h, color)
230    ///     .with_corner_radius(8.0);  // 8-pixel corner radius
231    /// ```
232    pub fn with_corner_radius(mut self, radius: f32) -> Self {
233        self.corner_radius = radius;
234        self
235    }
236}

238/// High-performance rectangle renderer for menu systems.
239///
240/// This renderer is optimized for UI applications where multiple rectangles
241/// (menu items, buttons, panels) need to be drawn efficiently. It uses
242/// batch rendering to minimize GPU draw calls and supports advanced
243/// features like transparency and rounded corners.
244///
245/// ## Key Features
246///
247/// - **Batch Rendering**: All rectangles are rendered in a single draw call
248/// - **Alpha Blending**: Supports transparent and semi-transparent elements
249/// - **Rounded Corners**: Hardware-accelerated rounded rectangle rendering
250/// - **Efficient Memory Usage**: Vertices and indices are created dynamically
251///
252/// ## Usage Pattern
253///
254/// 1. Create the renderer with GPU device and surface format
255/// 2. Add rectangles representing menu elements
256/// 3. Call render() to draw all rectangles in a batch
257/// 4. Clear rectangles for the next frame if needed
258///
259/// ## Performance Characteristics
260///
261/// - O(n) vertex generation where n is the number of rectangles
262/// - Single GPU draw call regardless of rectangle count
263/// - Memory allocation only occurs during rendering
264pub struct RectangleRenderer {
265    /// WebGPU render pipeline configured for rectangle rendering
266    /// Includes vertex and fragment shaders, blending state, and vertex layout
267    render_pipeline: RenderPipeline,
268
269    /// Collection of rectangles to be rendered
270    /// Represents all menu elements that will be drawn in the next render call
271    rectangles: Vec<Rectangle>,
272
273    /// Current window width in pixels
274    /// Used for coordinate transformation from screen space to NDC
275    window_width: f32,
276
277    /// Current window height in pixels  
278    /// Used for coordinate transformation from screen space to NDC
279    window_height: f32,
280}
281
282impl RectangleRenderer {
283    /// Creates a new rectangle renderer for menu systems.
284    ///
285    /// Initializes the WebGPU rendering pipeline with shaders optimized
286    /// for rectangle rendering. The pipeline is configured with:
287    ///
288    /// - Alpha blending for transparency effects
289    /// - Counter-clockwise front face orientation
290    /// - Triangle list primitive topology
291    /// - Front face culling for performance
292    ///
293    /// ## Parameters
294    ///
295    /// - `device`: WebGPU device for creating GPU resources
296    /// - `surface_format`: The texture format of the render target
297    ///
298    /// ## Returns
299    ///
300    /// A new `RectangleRenderer` ready to render menu elements
301    ///
302    /// ## Shader Requirements
303    ///
304    /// Expects a WGSL shader file at "shaders/rectangle.wgsl" with:
305    /// - `vs_main` vertex shader entry point
306    /// - `fs_main` fragment shader entry point
307    /// - Support for the vertex attributes defined in `Vertex::desc()`
308    pub fn new(device: &Device, surface_format: wgpu::TextureFormat) -> Self {
309        // Load the rectangle shader that handles rounded corner rendering
310        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
311            label: Some("Rectangle Shader"),
312            source: wgpu::ShaderSource::Wgsl(include_str!("shaders/rectangle.wgsl").into()),
313        });
314
315        // Create pipeline layout (no bind groups needed for basic rectangles)
316        let render_pipeline_layout =
317            device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
318                label: Some("Rectangle Pipeline Layout"),
319                bind_group_layouts: &[],
320                push_constant_ranges: &[],
321            });
322
323        // Configure the rendering pipeline for efficient rectangle rendering
324        let render_pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
325            label: Some("Rectangle Pipeline"),
326            layout: Some(&render_pipeline_layout),
327            vertex: VertexState {
328                module: &shader,
329                entry_point: Some("vs_main"),
330                buffers: &[Vertex::desc()],
331                compilation_options: Default::default(),
332            },
333            fragment: Some(FragmentState {
334                module: &shader,
335                entry_point: Some("fs_main"),
336                targets: &[Some(ColorTargetState {
337                    format: surface_format,
338                    // Enable alpha blending for transparent menu elements
339                    blend: Some(BlendState::ALPHA_BLENDING),
340                    write_mask: ColorWrites::ALL,
341                })],
342                compilation_options: Default::default(),
343            }),
344            primitive: PrimitiveState {
345                topology: wgpu::PrimitiveTopology::TriangleList,
346                strip_index_format: None,
347                front_face: wgpu::FrontFace::Ccw,
348                // Cull front faces for performance (we render rectangles from inside)
349                cull_mode: Some(wgpu::Face::Front),
350                polygon_mode: wgpu::PolygonMode::Fill,
351                unclipped_depth: false,
352                conservative: false,
353            },
354            depth_stencil: None,
355            multisample: MultisampleState {
356                count: 1,
357                mask: !0,
358                alpha_to_coverage_enabled: false,
359            },
360            multiview: None,
361            cache: None,
362        });
363
364        Self {
365            render_pipeline,
366            rectangles: Vec::new(),
367            // Default window size - should be updated via resize()
368            window_width: 1360.0,
369            window_height: 768.0,
370        }
371    }

373    /// Adds a rectangle to be rendered in the next draw call.
374    ///
375    /// This method is used to build up a collection of menu elements
376    /// that will all be rendered together for optimal performance.
377    ///
378    /// ## Parameters
379    ///
380    /// - `rectangle`: The rectangle primitive to add to the render queue
381    ///
382    /// ## Usage Examples
383    ///
384    /// ```rust
385    /// // Add a menu button
386    /// renderer.add_rectangle(
387    ///     Rectangle::new(100.0, 50.0, 200.0, 40.0, [0.2, 0.4, 0.8, 1.0])
388    ///         .with_corner_radius(8.0)
389    /// );
390    ///
391    /// // Add a hover highlight
392    /// renderer.add_rectangle(
393    ///     Rectangle::new(95.0, 45.0, 210.0, 50.0, [1.0, 1.0, 1.0, 0.1])
394    ///         .with_corner_radius(10.0)
395    /// );
396    /// ```
397    pub fn add_rectangle(&mut self, rectangle: Rectangle) {
398        self.rectangles.push(rectangle);
399    }

401    /// Clears all rectangles from the render queue.
402    ///
403    /// This is typically called at the beginning of each frame to
404    /// remove the previous frame's menu elements before adding
405    /// new ones.
406    ///
407    /// ## Usage
408    ///
409    /// ```rust
410    /// // Start of frame
411    /// renderer.clear_rectangles();
412    ///
413    /// // Add current frame's menu elements
414    /// for menu_item in menu.items() {
415    ///     renderer.add_rectangle(menu_item.to_rectangle());
416    /// }
417    ///
418    /// // Render all elements
419    /// renderer.render(&device, &mut render_pass);
420    /// ```
421    pub fn clear_rectangles(&mut self) {
422        self.rectangles.clear();
423    }
424
425    /// Updates the window dimensions for coordinate transformation.
426    ///
427    /// This method must be called whenever the window is resized to
428    /// ensure rectangles are positioned correctly. The renderer uses
429    /// these dimensions to convert from screen coordinates to
430    /// normalized device coordinates (NDC).
431    ///
432    /// ## Parameters
433    ///
434    /// - `width`: New window width in pixels
435    /// - `height`: New window height in pixels
436    ///
437    /// ## Important
438    ///
439    /// Call this method in your window resize handler to maintain
440    /// correct menu positioning and proportions.
441    pub fn resize(&mut self, width: f32, height: f32) {
442        self.window_width = width;
443        self.window_height = height;
444    }
445
446    /// Renders all queued rectangles in a single optimized draw call.
447    ///
448    /// This method performs batch rendering of all rectangles that have
449    /// been added since the last clear. It:
450    ///
451    /// 1. Converts screen coordinates to normalized device coordinates
452    /// 2. Generates vertices for each rectangle (4 vertices per rectangle)
453    /// 3. Creates indices for triangle rendering (2 triangles per rectangle)
454    /// 4. Uploads vertex and index data to the GPU
455    /// 5. Executes a single indexed draw call
456    ///
457    /// ## Performance Notes
458    ///
459    /// - All rectangles are rendered in one draw call for maximum performance
460    /// - Vertex and index buffers are created dynamically each frame
461    /// - Memory allocation is proportional to the number of rectangles
462    /// - GPU memory is automatically reclaimed after rendering
463    ///
464    /// ## Coordinate Transformation
465    ///
466    /// The method handles the conversion from screen space (0,0 at top-left)
467    /// to NDC space (-1,-1 to 1,1 with origin at center), including proper
468    /// Y-axis flipping for correct rendering.
469    ///
470    /// ## Parameters
471    ///
472    /// - `device`: WebGPU device for creating GPU buffers
473    /// - `render_pass`: Active render pass to submit draw commands to
474    pub fn render(&mut self, device: &Device, render_pass: &mut RenderPass) {
475        // Early return if no rectangles to render
476        if self.rectangles.is_empty() {
477            return;
478        }
479
480        // Set the rendering pipeline for rectangle rendering
481        render_pass.set_pipeline(&self.render_pipeline);
482
483        // Batch all rectangle data for efficient GPU upload
484        let mut all_vertices = Vec::new();
485        let mut all_indices = Vec::new();
486
487        // Process each rectangle in the render queue
488        for (rect_index, rectangle) in self.rectangles.iter().enumerate() {
489            // Transform screen coordinates to normalized device coordinates (NDC)
490            // Screen space: (0,0) = top-left, positive Y = down
491            // NDC space: (-1,-1) = bottom-left, (1,1) = top-right
492            let x = (rectangle.x / self.window_width) * 2.0 - 1.0;
493            let y = 1.0 - (rectangle.y / self.window_height) * 2.0; // Flip Y-axis
494            let width = (rectangle.width / self.window_width) * 2.0;
495            let height = -(rectangle.height / self.window_height) * 2.0; // Negative due to Y-flip
496
497            // Create the four vertices for this rectangle
498            // Each vertex contains position, color, UV coords, size, and corner radius
499            let vertices = [
500                // Top-left vertex
501                Vertex {
502                    position: [x, y],
503                    color: rectangle.color,
504                    uv: [0.0, 0.0], // UV coordinates for fragment shader distance calculations
505                    rect_size: [rectangle.width, rectangle.height],
506                    corner_radius: rectangle.corner_radius,
507                    _padding: 0.0,
508                },
509                // Top-right vertex
510                Vertex {
511                    position: [x + width, y],
512                    color: rectangle.color,
513                    uv: [rectangle.width, 0.0],
514                    rect_size: [rectangle.width, rectangle.height],
515                    corner_radius: rectangle.corner_radius,
516                    _padding: 0.0,
517                },
518                // Bottom-right vertex
519                Vertex {
520                    position: [x + width, y + height],
521                    color: rectangle.color,
522                    uv: [rectangle.width, rectangle.height],
523                    rect_size: [rectangle.width, rectangle.height],
524                    corner_radius: rectangle.corner_radius,
525                    _padding: 0.0,
526                },
527                // Bottom-left vertex
528                Vertex {
529                    position: [x, y + height],
530                    color: rectangle.color,
531                    uv: [0.0, rectangle.height],
532                    rect_size: [rectangle.width, rectangle.height],
533                    corner_radius: rectangle.corner_radius,
534                    _padding: 0.0,
535                },
536            ];
537
538            // Add vertices to the batch buffer
539            all_vertices.extend_from_slice(&vertices);
540
541            // Create triangle indices for this rectangle
542            // Two triangles per rectangle: (0,1,2) and (0,2,3)
543            let base_index = (rect_index * 4) as u16;
544            let indices = [
545                base_index,
546                base_index + 1,
547                base_index + 2, // First triangle
548                base_index,
549                base_index + 2,
550                base_index + 3, // Second triangle
551            ];
552            all_indices.extend_from_slice(&indices);
553        }
554
555        // Create GPU vertex buffer with all rectangle vertices
556        let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
557            label: Some("Rectangle Vertex Buffer"),
558            contents: bytemuck::cast_slice(&all_vertices),
559            usage: BufferUsages::VERTEX,
560        });
561
562        // Create GPU index buffer with all triangle indices
563        let index_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
564            label: Some("Rectangle Index Buffer"),
565            contents: bytemuck::cast_slice(&all_indices),
566            usage: BufferUsages::INDEX,
567        });
568
569        // Bind buffers and execute the draw call
570        render_pass.set_vertex_buffer(0, vertex_buffer.slice(..));
571        render_pass.set_index_buffer(index_buffer.slice(..), wgpu::IndexFormat::Uint16);
572
573        // Draw all rectangles in a single indexed draw call
574        // This renders all menu elements with optimal GPU performance
575        render_pass.draw_indexed(0..all_indices.len() as u32, 0, 0..1);
576    }
