1//! # Maze Generation Animation Rendering System
2//!
3//! This module provides specialized renderers for different aspects of a maze visualization
4//! application. It includes renderers for the maze itself, loading progress indicators,
5//! and animated exit effects. As well as the main 3D maze renderer.
6//!
7//! ## Architecture Overview
8//!
9//! The rendering system is built on top of the pipeline builder utilities and provides
10//! three main renderer types:
11//!
12//! - [`MazeRenderer`] - Renders the maze texture to screen
13//! - [`LoadingBarRenderer`] - Shows loading progress with a progress bar
14//! - [`ExitShaderRenderer`] - Creates animated effects for the maze exit
15//! - [`GameRenderer`] - Renders the maze in 3D with the background defined in stars.rs
16//!
17//! ## Coordinate System
18//!
19//! The maze uses a grid-based coordinate system where (0,0) is typically the top-left
20//! corner. The renderers handle conversion between grid coordinates and screen space.
21//!
22//! ## Usage Pattern
23//!
24//! 1. Create renderers during initialization with device and surface configuration
25//! 2. Update renderer state each frame (progress, time, etc.)
26//! 3. Render during the render pass
27//!
28//! ## Example Setup
29//!
30//! ```rust,no_run
31//! use crate::renderer::maze_renderer::{MazeRenderer, LoadingBarRenderer, MazeRenderConfig};
32//! # let device: egui_wgpu::wgpu::Device = unimplemented!();
33//! # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
34//!
35//! // Create maze texture
36//! let config = MazeRenderConfig::new(50, 50); // 50x50 maze
37//! let (texture, texture_view, sampler) = config.create_maze_texture(&device);
38//!
39//! // Create renderers
40//! let maze_renderer = MazeRenderer::new(&device, &surface_config, &texture_view, &sampler);
41//! let loading_renderer = LoadingBarRenderer::new(&device, &surface_config);
42//! ```

69/// Main renderer for the 3D maze game.
70///
71/// The `GameRenderer` is responsible for rendering the complete 3D maze environment,
72/// including floors, walls, starfield backgrounds, and debug overlays. It manages
73/// all rendering pipelines, vertex data, and GPU resources required for the game's
74/// visual presentation.
75///
76/// ## Rendering Pipeline
77///
78/// The renderer uses a multi-pass approach:
79/// 1. **Background Pass**: Renders animated starfield using `StarRenderer`
80/// 2. **Main Pass**: Renders maze geometry (floors and walls) with depth testing
81/// 3. **Debug Pass**: Optional overlay rendering for development tools
82///
83/// ## Memory Management
84///
85/// Vertex data for floors and walls is combined into a single buffer for optimal
86/// GPU performance. The depth texture is dynamically resized based on the surface
87/// dimensions and automatically managed to prevent memory leaks.
88///
89/// ## Coordinate System
90///
91/// The renderer uses a right-handed coordinate system with:
92/// - X-axis pointing right
93/// - Y-axis pointing up
94/// - Z-axis pointing toward the viewer
95/// - Maze positioned at world origin
96///
97/// ## Shader Integration
98///
99/// Works with `shader.wgsl` which expects:
100/// ```wgsl
101/// struct Uniforms {
102///     matrix: mat4x4<f32>,
103/// }
104/// @group(0) @binding(0) var<uniform> uniforms: Uniforms;
105/// ```
106///
107/// # Fields
108///
109/// - `pipeline` - Main render pipeline for maze geometry with depth testing and alpha blending
110/// - `vertex_buffer` - Combined vertex buffer containing both floor and wall geometry data
111/// - `vertex_count` - Total number of vertices to render from the combined buffer
112/// - `uniform_buffer` - GPU buffer storing model-view-projection matrix for vertex transformations
113/// - `uniform_bind_group` - WebGPU bind group linking uniform buffer to shader binding point 0
114/// - `depth_texture` - Optional depth buffer for proper 3D occlusion (recreated on resize)
115/// - `star_renderer` - Background renderer for animated starfield effects
116/// - `debug_renderer` - Development tools for rendering bounding boxes and debug overlays
117pub struct GameRenderer {
118    pub pipeline: wgpu::RenderPipeline,
119    /// Combined vertex buffer containing both floor and wall geometry data.
120    pub vertex_buffer: wgpu::Buffer,
121    /// Total number of vertices to render from the combined buffer.
122    pub vertex_count: u32,
123    /// GPU buffer storing model-view-projection matrix for vertex transformations.
124    pub uniform_buffer: wgpu::Buffer,
125    /// WebGPU bind group linking uniform buffer to shader binding point 0.
126    pub uniform_bind_group: wgpu::BindGroup,
127    /// Optional depth buffer for proper 3D occlusion (recreated on resize).
128    pub depth_texture: Option<wgpu::Texture>,
129    /// Background renderer for animated starfield effects.
130    pub star_renderer: StarRenderer,
131    /// Development tools for rendering bounding boxes and debug overlays.
132    pub debug_renderer: DebugRenderer,
133    /// Renderer for compass
134    pub compass_renderer: CompassRenderer,
135    pub exit_position: Option<(f32, f32)>,
136    pub enemy_renderer: EnemyRenderer,
137}

140    pub fn new(
141        device: &wgpu::Device,
142        queue: &wgpu::Queue,
143        surface_config: &wgpu::SurfaceConfiguration,
144    ) -> Self {
145        let uniforms = Uniforms::new();
146        let uniform_buffer = uniforms.create_buffer(device);
147        let (uniform_bind_group, uniform_bind_group_layout) =
148            uniforms.create_bind_group(&uniform_buffer, device);
149
150        let pipeline = PipelineBuilder::new(device, surface_config.format)
151            .with_label("Main Pipeline")
152            .with_shader(include_str!("./shaders/main-shader.wgsl"))
153            .with_vertex_buffer(Vertex::desc())
154            .with_bind_group_layout(&uniform_bind_group_layout)
155            .with_blend_state(wgpu::BlendState {
156                color: wgpu::BlendComponent {
157                    src_factor: wgpu::BlendFactor::SrcAlpha,
158                    dst_factor: wgpu::BlendFactor::OneMinusSrcAlpha,
159                    operation: wgpu::BlendOperation::Add,
160                },
161                alpha: wgpu::BlendComponent::OVER,
162            })
163            .with_no_culling()
164            .with_depth_stencil(wgpu::DepthStencilState {
165                depth_write_enabled: true,
166                depth_compare: wgpu::CompareFunction::Less,
167                format: wgpu::TextureFormat::Depth24Plus,
168                stencil: wgpu::StencilState::default(),
169                bias: wgpu::DepthBiasState::default(),
170            })
171            .build();
172
173        // Load wall grid from file
174        let (maze_grid, exit_cell) = parse_maze_file("src/maze/saved-mazes/test.mz");
175
176        let (mut floor_vertices, _exit_position) =
177            Vertex::create_floor_vertices(&maze_grid, exit_cell);
178
179        // Generate wall geometry
180        let mut wall_vertices = Vertex::create_wall_vertices(&maze_grid);
181
182        // Append wall vertices to floor
183        floor_vertices.append(&mut wall_vertices);
184
185        let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
186            label: Some("Combined Vertex Buffer"),
187            contents: bytemuck::cast_slice(&floor_vertices),
188            usage: wgpu::BufferUsages::VERTEX,
189        });
190
191        let star_renderer = stars::create_star_renderer(device, surface_config, 100);
192
193        let debug_renderer = DebugRenderer {
194            debug_render_bounding_boxes: false,
195            debug_vertex_buffer: None,
196            debug_vertex_count: 0,
197        };
198
199        let compass_renderer = CompassRenderer::new(device, queue, surface_config);
200        let enemy = Enemy::new([-1370.0, 50.0, 1370.0]);
201        let enemy_renderer = EnemyRenderer::new(enemy, device, queue, surface_config);
202        Self {
203            pipeline,
204            vertex_buffer,
205            vertex_count: floor_vertices.len() as u32,
206            uniform_buffer,
207            uniform_bind_group,
208            depth_texture: None,
209            star_renderer,
210            debug_renderer,
211            compass_renderer,
212            exit_position: None,
213            enemy_renderer,
214        }
215    }

217    pub fn update_depth_texture(
218        &mut self,
219        device: &wgpu::Device,
220        width: u32,
221        height: u32,
222    ) -> wgpu::TextureView {
223        if self.depth_texture.is_none()
224            || self.depth_texture.as_ref().unwrap().width() != width
225            || self.depth_texture.as_ref().unwrap().height() != height
226        {
227            if let Some(depth_texture) = self.depth_texture.take() {
228                // Manually drop the texture to free up resources
229                drop(depth_texture);
230            }
231
232            self.depth_texture = Some(device.create_texture(&wgpu::TextureDescriptor {
233                label: Some("Depth Texture"),
234                size: wgpu::Extent3d {
235                    width,
236                    height,
237                    depth_or_array_layers: 1,
238                },
239                mip_level_count: 1,
240                sample_count: 1,
241                dimension: wgpu::TextureDimension::D2,
242                format: wgpu::TextureFormat::Depth24Plus,
243                usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
244                view_formats: &[],
245            }));
246        }
247        self.depth_texture
248            .as_ref()
249            .unwrap()
250            .create_view(&wgpu::TextureViewDescriptor::default())
251    }

253    pub fn render_game(
254        &mut self,
255        queue: &wgpu::Queue,
256        game_state: &GameState,
257        pass: &mut wgpu::RenderPass,
258        aspect: f32,
259    ) {
260        // Calculate view and projection matrices once
261        let view_matrix = game_state.player.get_view_matrix();
262        let projection_matrix = Mat4::perspective(
263            deg_to_rad(game_state.player.fov),
264            aspect,
265            0.1,    // zNear
266            2000.0, // zFar
267        );
268        let view_proj_matrix = projection_matrix.multiply(&view_matrix);
269
270        // ==============================================
271        // 1. RENDER MAZE/FLOOR FIRST
272        // ==============================================
273        {
274            // Model Matrix for floor - identity since floor is at world origin
275            let model_matrix = Mat4::identity();
276
277            // Combine matrices: Projection * View * Model
278            let final_mvp_matrix = view_proj_matrix.multiply(&model_matrix);
279
280            let uniforms = Uniforms {
281                matrix: final_mvp_matrix.into(),
282            };
283
284            // Upload uniform values for the maze/floor
285            queue.write_buffer(&self.uniform_buffer, 0, uniforms.as_bytes());
286
287            // Render the maze/floor
288            pass.set_pipeline(&self.pipeline);
289            pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
290            pass.set_bind_group(0, &self.uniform_bind_group, &[]);
291            pass.draw(0..self.vertex_count, 0..1);
292
293            // Debug rendering for maze/floor
294            if self.debug_renderer.debug_render_bounding_boxes
295                && self.debug_renderer.debug_vertex_count > 0
296            {
297                if let Some(debug_buffer) = &self.debug_renderer.debug_vertex_buffer {
298                    pass.set_vertex_buffer(0, debug_buffer.slice(..));
299                    pass.draw(0..self.debug_renderer.debug_vertex_count as u32, 0..1);
300                }
301            }
302        }
303
304        // ==============================================
305        // 2. RENDER ENEMIES
306        // ==============================================
307        {
308            // Update enemy transform with the combined view-projection matrix
309            self.enemy_renderer.update(
310                queue,
311                game_state,
312                view_proj_matrix.0, // Pass the view-projection matrix
313            );
314
315            // Actually render the enemy
316            self.enemy_renderer.render(pass); // You might need to pass the actual window here
317        }
318    }

380    ///
381    /// # Arguments

383    /// - `maze_data`: The new maze image data (RGBA, row-major).

393        queue.write_texture(

402                offset: 0,
403                bytes_per_row: Some(4 * width as u32),
404                rows_per_image: Some(height as u32),
405            },
406            wgpu::Extent3d {
407                width: width as u32,
408                height: height as u32,
409                depth_or_array_layers: 1,
410            },
411        );
412    }

499/// # Fields

502/// - `_padding` - Ensures proper WGSL alignment (16 bytes total)

506    pub progress: f32,
507    pub _padding: [f32; 3],
508}

545/// Renders a maze texture to the screen with proper scaling and filtering.
546///
547/// This renderer takes a pre-generated maze texture and displays it fullscreen
548/// or in a specified viewport. It handles texture sampling and can apply
549/// various filtering modes.
550///
551/// ## Rendering Pipeline
552///
553/// 1. Uses a fullscreen quad vertex buffer
554/// 2. Samples the maze texture with the provided sampler
555/// 3. Outputs directly to the framebuffer
556///
557/// ## Texture Requirements
558///
559/// - Format: `Rgba8UnormSrgb` (or compatible)
560/// - Usage: Must include `TEXTURE_BINDING`
561/// - The texture should contain the pre-rendered maze data
562///
563/// ## Performance Notes
564///
565/// This renderer is very efficient as it only performs texture sampling
566/// without complex computations. It's suitable for real-time rendering
567/// of static or infrequently updated maze textures.
568///
569/// ## Example Usage
570///
571/// ```rust,no_run
572/// # use crate::renderer::maze_renderer::MazeRenderer;
573/// # let device: egui_wgpu::wgpu::Device = unimplemented!();
574/// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
575/// # let texture_view: egui_wgpu::wgpu::TextureView = unimplemented!();
576/// # let sampler: egui_wgpu::wgpu::Sampler = unimplemented!();
577/// # let mut render_pass: egui_wgpu::wgpu::RenderPass = unimplemented!();
578///
579/// let renderer = MazeRenderer::new(&device, &surface_config, &texture_view, &sampler);
580///
581/// // In render loop:
582/// renderer.render(&mut render_pass);
583/// ```
584pub struct MazeRenderer {
585    /// The render pipeline configured for maze texture rendering
586    pub pipeline: wgpu::RenderPipeline,
587    /// Vertex buffer containing fullscreen quad vertices
588    pub vertex_buffer: wgpu::Buffer,
589    /// Bind group containing the maze texture and sampler
590    pub bind_group: wgpu::BindGroup,
591}

593impl MazeRenderer {
594    /// Create a new maze renderer.
595    ///
596    /// This sets up the complete rendering pipeline for displaying a maze texture,
597    /// including the render pipeline, vertex buffer, and bind group.
598    ///
599    /// ## Shader Requirements
600    ///
601    /// The maze shader should be located at `../maze/2D-maze-shader.wgsl` relative
602    /// to this module and should expect:
603    /// - Vertex input: `@location(0) position: vec2<f32>`
604    /// - Texture binding: `@group(0) @binding(0) var maze_texture: texture_2d<f32>;`
605    /// - Sampler binding: `@group(0) @binding(1) var maze_sampler: sampler;`
606    ///
607    /// # Parameters
608    ///
609    /// - `device` - WGPU device for creating GPU resources
610    /// - `surface_config` - Surface configuration (provides target format)
611    /// - `texture_view` - View of the maze texture to render
612    /// - `sampler` - Sampler for texture filtering (typically nearest neighbor for pixel art)
613    ///
614    /// # Returns
615    ///
616    /// A configured `MazeRenderer` ready for rendering.
617    ///
618    /// # Example
619    ///
620    /// ```rust,no_run
621    /// # use crate::renderer::maze_renderer::{MazeRenderer, MazeRenderConfig};
622    /// # let device: egui_wgpu::wgpu::Device = unimplemented!();
623    /// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
624    ///
625    /// // Create maze texture
626    /// let config = MazeRenderConfig::new(25, 25);
627    /// let (texture, texture_view, sampler) = config.create_maze_texture(&device);
628    ///
629    /// let renderer = MazeRenderer::new(&device, &surface_config, &texture_view, &sampler);
630    /// ```
631    pub fn new(
632        device: &wgpu::Device,
633        surface_config: &wgpu::SurfaceConfiguration,
634        texture_view: &wgpu::TextureView,
635        sampler: &wgpu::Sampler,
636    ) -> Self {
637        let bind_group_layout = BindGroupLayoutBuilder::new(device)
638            .with_label("Maze Texture Bind Group Layout")
639            .with_texture(0, wgpu::ShaderStages::FRAGMENT)
640            .with_sampler(1, wgpu::ShaderStages::FRAGMENT)
641            .build();
642
643        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
644            layout: &bind_group_layout,
645            entries: &[
646                wgpu::BindGroupEntry {
647                    binding: 0,
648                    resource: wgpu::BindingResource::TextureView(texture_view),
649                },
650                wgpu::BindGroupEntry {
651                    binding: 1,
652                    resource: wgpu::BindingResource::Sampler(sampler),
653                },
654            ],
655            label: Some("Maze Texture Bind Group"),

1017    pub fn render_to_cell(
1018        &self,
1019        render_pass: &mut wgpu::RenderPass,
1020        window: &winit::window::Window,
1021        exit_cell: (usize, usize),
1022    ) {
1023        // Calculate scissor rectangle for the exit cell
1024        let grid_size = 25.0;
1025        let shrink_factor = 0.845;
1026        let border = 3.0;
1027
1028        let window_size = window.inner_size();
1029        let total_width = window_size.width as f32;
1030        let total_height = window_size.height as f32;
1031
1032        let usable_width = total_width - 2.0 * border;
1033        let usable_height = total_height - 2.0 * border;
1034
1035        let cell_width = usable_width / grid_size;
1036        let cell_height = usable_height / grid_size;
1037
1038        let full_x = border + exit_cell.0 as f32 * cell_width;
1039        let full_y = border + exit_cell.1 as f32 * cell_height;
1040
1041        let shrunk_width = cell_width * shrink_factor;
1042        let shrunk_height = cell_height * shrink_factor;
1043
1044        let offset_x = (cell_width - shrunk_width) / 2.0;
1045        let offset_y = (cell_height - shrunk_height) / 2.0;
1046
1047        let scissor_x = (full_x + offset_x).round() as u32;
1048        let scissor_y = (full_y + offset_y).round() as u32;
1049        let scissor_width = shrunk_width.round() as u32;
1050        let scissor_height = shrunk_height.round() as u32;
1051
1052        // Set scissor rect and render
1053        render_pass.set_scissor_rect(scissor_x, scissor_y, scissor_width, scissor_height);
1054        render_pass.set_pipeline(&self.pipeline);
1055        render_pass.set_bind_group(0, &self.bind_group, &[]);
1056        render_pass.draw(0..3, 0..1); // Full-screen triangle
1057    }

1592    /// Calculate which needle image to show based on player and exit positions
1593    pub fn update_compass_direction(&mut self, player_pos: (f32, f32), exit_pos: (f32, f32)) {
1594        // Vector from player to exit
1595        let direction_vector = (player_pos.0 - exit_pos.0, player_pos.1 - exit_pos.1);
1596
1597        // Skip if too close (avoid jitter when on top of exit)
1598        let distance_sq =
1599            direction_vector.0 * direction_vector.0 + direction_vector.1 * direction_vector.1;
1600        if distance_sq < 0.0001 {
1601            return;
1602        }
1603
1604        // Calculate angle to exit in world space
1605        let mut target_angle = direction_vector.1.atan2(direction_vector.0); // [-, ]
1606
1607        // Normalize to [0, 2]
1608        if target_angle < 0.0 {
1609            target_angle += 2.0 * std::f32::consts::PI;
1610        }
1611
1612        // Smooth angle update (exponential smoothing)
1613        let alpha = self.smoothing_factor; // Lower = slower/smoother
1614        let mut delta = target_angle - self.smoothed_compass_angle;
1615
1616        // Wrap to [-, ] for shortest rotation
1617        if delta > std::f32::consts::PI {
1618            delta -= 2.0 * std::f32::consts::PI;
1619        } else if delta < -std::f32::consts::PI {
1620            delta += 2.0 * std::f32::consts::PI;
1621        }
1622
1623        self.smoothed_compass_angle += alpha * delta;
1624
1625        // Re-wrap smoothed angle to [0, 2]
1626        while self.smoothed_compass_angle < 0.0 {
1627            self.smoothed_compass_angle += 2.0 * std::f32::consts::PI;
1628        }
1629        while self.smoothed_compass_angle >= 2.0 * std::f32::consts::PI {
1630            self.smoothed_compass_angle -= 2.0 * std::f32::consts::PI;
1631        }
1632
1633        // Map to needle frame (011, since we have 12 needles indexed 1-12)
1634        let new_index = ((self.smoothed_compass_angle / (2.0 * std::f32::consts::PI)) * 12.0)
1635            .floor() as usize
1636            % 12;
1637
1638        self.current_needle_index = new_index;
1639    }

1641    /// Updates the compass to point toward the exit from the player's current position.
1642    ///
1643    /// This function calculates the direction from the player to the exit cell and
1644    /// adjusts for the player's current orientation (yaw) so that the compass always
1645    /// indicates the direction the player should move to reach the exit.
1646    ///
1647    /// # Arguments
1648    ///
1649    /// * `player_pos` - The player's position as (x, z) coordinates
1650    /// * `exit_pos` - The exit's position as (x, z) coordinates
1651    /// * `player_yaw_degrees` - The player's current yaw angle in degrees
1652    pub fn update_compass_with_yaw(
1653        &mut self,
1654        player_pos: (f32, f32), // (x, z) coordinates
1655        exit_pos: (f32, f32),   // (x, z) coordinates
1656        player_yaw_degrees: f32,
1657    ) {
1658        // Calculate vector from player to exit
1659        let dx = exit_pos.0 - player_pos.0; // Change in X
1660        let dz = exit_pos.1 - player_pos.1; // Change in Z
1661
1662        let distance_sq = dx * dx + dz * dz;
1663
1664        // Skip if too close to exit
1665        if distance_sq < 0.0001 {
1666            return;
1667        }
1668
1669        // Calculate direction to exit using the same trig approach as player movement
1670        // First, get forward vector based on player's yaw (same as in move_forward)
1671        let forward_x = player_yaw_degrees.to_radians().sin();
1672        let forward_z = player_yaw_degrees.to_radians().cos();
1673
1674        // Get right vector (same as in move_right)
1675        let right_x = player_yaw_degrees.to_radians().cos();
1676        let right_z = player_yaw_degrees.to_radians().sin();
1677
1678        // Normalize the direction vector to the exit
1679        let length = distance_sq.sqrt();
1680        let dir_x = dx / length;
1681        let dir_z = dz / length;
1682
1683        // Calculate dot products to determine the angle
1684        let forward_dot = -forward_x * dir_x - forward_z * dir_z; // Dot product with forward vector
1685        let right_dot = right_x * dir_x - right_z * dir_z; // Dot product with right vector
1686
1687        // Calculate angle using atan2
1688        let mut target_compass_angle = right_dot.atan2(forward_dot);
1689
1690        // Normalize to [-, ]
1691        target_compass_angle = self.normalize_angle(target_compass_angle);
1692
1693        // Initialize smoothed angle on first update
1694        if self.smoothed_compass_angle.is_nan() {
1695            self.smoothed_compass_angle = target_compass_angle;
1696        }
1697
1698        // Calculate the shortest angular distance for smooth interpolation
1699        let angle_diff =
1700            self.shortest_angle_diff(target_compass_angle, self.smoothed_compass_angle);
1701
1702        // Apply smoothing
1703        self.smoothed_compass_angle += angle_diff * self.smoothing_factor;
1704
1705        // Normalize the smoothed angle
1706        self.smoothed_compass_angle = self.normalize_angle(self.smoothed_compass_angle);
1707
1708        // Convert to needle index (0-11 for 12 needle sprites)
1709        // Convert from [-, ] to [0, 2] for easier indexing
1710        let angle_for_index = if self.smoothed_compass_angle < 0.0 {
1711            self.smoothed_compass_angle + 2.0 * std::f32::consts::PI
1712        } else {
1713            self.smoothed_compass_angle
1714        };
1715
1716        // Convert to 12-segment index (each segment is 30 = /6 radians)
1717        // Add half a segment (/12) for proper rounding to nearest segment
1718        let needle_index = ((angle_for_index + std::f32::consts::PI / 12.0)
1719            / (std::f32::consts::PI / 6.0))
1720            .floor() as usize
1721            % 12;
1722
1723        self.current_needle_index = needle_index;
1724    }

1726    /// Normalize angle to [-, ]
1727    fn normalize_angle(&self, mut angle: f32) -> f32 {
1728        while angle > std::f32::consts::PI {
1729            angle -= 2.0 * std::f32::consts::PI;
1730        }
1731        while angle < -std::f32::consts::PI {
1732            angle += 2.0 * std::f32::consts::PI;
1733        }
1734        angle
1735    }

1737    /// Calculate shortest angular difference between two angles
1738    fn shortest_angle_diff(&self, target: f32, current: f32) -> f32 {
1739        let mut diff = target - current;
1740
1741        // Wrap to shortest path
1742        if diff > std::f32::consts::PI {
1743            diff -= 2.0 * std::f32::consts::PI;
1744        } else if diff < -std::f32::consts::PI {
1745            diff += 2.0 * std::f32::consts::PI;
1746        }
1747
1748        diff
1749    }

1752    pub fn update_compass_with_smoothing(
1753        &mut self,
1754        player_pos: (f32, f32),
1755        exit_pos: (f32, f32),
1756        player_yaw_degrees: f32,

1762        self.update_compass_with_yaw(player_pos, exit_pos, player_yaw_degrees);
1763
1764        self.smoothing_factor = old_smoothing;
1765    }

1981    /// Update enemy position and rotation to face player
1982    pub fn update(
1983        &mut self,
1984        queue: &wgpu::Queue,
1985        game_state: &GameState,
1986        view_proj_matrix: [[f32; 4]; 4],
1987    ) {
1988        // Calculate rotation to face player
1989        let dx = game_state.player.position[0] - game_state.enemy.position[0];
1990        let dz = game_state.player.position[2] - game_state.enemy.position[2];
1991
1992        // Calculate target rotation using the same coordinate system as your compass
1993        // Your compass uses dx.atan2(dz) pattern, so use that here
1994        let target_rotation = dx.atan2(dz);
1995
1996        // Smooth rotation interpolation
1997        let mut rotation_diff = target_rotation - self.smoothed_rotation;
1998
1999        // Wrap to shortest path
2000        if rotation_diff > std::f32::consts::PI {
2001            rotation_diff -= 2.0 * std::f32::consts::PI;
2002        } else if rotation_diff < -std::f32::consts::PI {
2003            rotation_diff += 2.0 * std::f32::consts::PI;
2004        }
2005
2006        self.smoothed_rotation += rotation_diff * self.smoothing_factor;
2007
2008        // Update uniform buffer
2009        let uniforms = EnemyUniforms {
2010            view_proj_matrix,
2011            enemy_position: game_state.enemy.position,
2012            enemy_size: game_state.enemy.size,
2013            player_position: game_state.player.position,
2014            _padding: 0.0,
2015        };
2016
2017        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::cast_slice(&[uniforms]));
2018    }
