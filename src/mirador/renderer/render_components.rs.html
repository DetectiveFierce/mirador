1//! # Maze Generation Animation Rendering System
2//!
3//! This module provides specialized renderers for different aspects of a maze visualization
4//! application. It includes renderers for the maze itself, loading progress indicators,
5//! and animated exit effects. As well as the main 3D maze renderer.
6//!
7//! ## Architecture Overview
8//!
9//! The rendering system is built on top of the pipeline builder utilities and provides
10//! three main renderer types:
11//!
12//! - [`MazeRenderer`] - Renders the maze texture to screen
13//! - [`LoadingBarRenderer`] - Shows loading progress with a progress bar
14//! - [`ExitShaderRenderer`] - Creates animated effects for the maze exit
15//! - [`GameRenderer`] - Renders the maze in 3D with the background defined in stars.rs
16//!
17//! ## Coordinate System
18//!
19//! The maze uses a grid-based coordinate system where (0,0) is typically the top-left
20//! corner. The renderers handle conversion between grid coordinates and screen space.
21//!
22//! ## Usage Pattern
23//!
24//! 1. Create renderers during initialization with device and surface configuration
25//! 2. Update renderer state each frame (progress, time, etc.)
26//! 3. Render during the render pass
27//!
28//! ## Example Setup
29//!
30//! ```rust,no_run
31//! use crate::renderer::maze_renderer::{MazeRenderer, LoadingBarRenderer, MazeRenderConfig};
32//! # let device: egui_wgpu::wgpu::Device = unimplemented!();
33//! # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
34//!
35//! // Create maze texture
36//! let config = MazeRenderConfig::new(50, 50); // 50x50 maze
37//! let (texture, texture_view, sampler) = config.create_maze_texture(&device);
38//!
39//! // Create renderers
40//! let maze_renderer = MazeRenderer::new(&device, &surface_config, &texture_view, &sampler);
41//! let loading_renderer = LoadingBarRenderer::new(&device, &surface_config);
42//! ```

59/// Main renderer for the 3D maze game.
60///
61/// The `GameRenderer` is responsible for rendering the complete 3D maze environment,
62/// including floors, walls, starfield backgrounds, and debug overlays. It manages
63/// all rendering pipelines, vertex data, and GPU resources required for the game's
64/// visual presentation.
65///
66/// ## Rendering Pipeline
67///
68/// The renderer uses a multi-pass approach:
69/// 1. **Background Pass**: Renders animated starfield using `StarRenderer`
70/// 2. **Main Pass**: Renders maze geometry (floors and walls) with depth testing
71/// 3. **Debug Pass**: Optional overlay rendering for development tools
72///
73/// ## Memory Management
74///
75/// Vertex data for floors and walls is combined into a single buffer for optimal
76/// GPU performance. The depth texture is dynamically resized based on the surface
77/// dimensions and automatically managed to prevent memory leaks.
78///
79/// ## Coordinate System
80///
81/// The renderer uses a right-handed coordinate system with:
82/// - X-axis pointing right
83/// - Y-axis pointing up
84/// - Z-axis pointing toward the viewer
85/// - Maze positioned at world origin
86///
87/// ## Shader Integration
88///
89/// Works with `shader.wgsl` which expects:
90/// ```wgsl
91/// struct Uniforms {
92///     matrix: mat4x4<f32>,
93/// }
94/// @group(0) @binding(0) var<uniform> uniforms: Uniforms;
95/// ```
96///
97/// # Fields
98///
99/// - `pipeline` - Main render pipeline for maze geometry with depth testing and alpha blending
100/// - `vertex_buffer` - Combined vertex buffer containing both floor and wall geometry data
101/// - `vertex_count` - Total number of vertices to render from the combined buffer
102/// - `uniform_buffer` - GPU buffer storing model-view-projection matrix for vertex transformations
103/// - `uniform_bind_group` - WebGPU bind group linking uniform buffer to shader binding point 0
104/// - `depth_texture` - Optional depth buffer for proper 3D occlusion (recreated on resize)
105/// - `star_renderer` - Background renderer for animated starfield effects
106/// - `debug_renderer` - Development tools for rendering bounding boxes and debug overlays
107pub struct GameRenderer {
108    pub pipeline: wgpu::RenderPipeline,
109    /// Combined vertex buffer containing both floor and wall geometry data.
110    pub vertex_buffer: wgpu::Buffer,
111    /// Total number of vertices to render from the combined buffer.
112    pub vertex_count: u32,
113    /// GPU buffer storing model-view-projection matrix for vertex transformations.
114    pub uniform_buffer: wgpu::Buffer,
115    /// WebGPU bind group linking uniform buffer to shader binding point 0.
116    pub uniform_bind_group: wgpu::BindGroup,
117    /// Optional depth buffer for proper 3D occlusion (recreated on resize).
118    pub depth_texture: Option<wgpu::Texture>,
119    /// Background renderer for animated starfield effects.
120    pub star_renderer: StarRenderer,
121    /// Development tools for rendering bounding boxes and debug overlays.
122    pub debug_renderer: DebugRenderer,
123}

125impl GameRenderer {
126    pub fn new(device: &wgpu::Device, surface_config: &wgpu::SurfaceConfiguration) -> Self {
127        let uniforms = Uniforms::new();
128        let uniform_buffer = uniforms.create_buffer(device);
129        let (uniform_bind_group, uniform_bind_group_layout) =
130            uniforms.create_bind_group(&uniform_buffer, device);
131
132        let pipeline = PipelineBuilder::new(device, surface_config.format)
133            .with_label("Main Pipeline")
134            .with_shader(include_str!("shader.wgsl"))
135            .with_vertex_buffer(Vertex::desc())
136            .with_bind_group_layout(&uniform_bind_group_layout)
137            .with_blend_state(wgpu::BlendState {
138                color: wgpu::BlendComponent {
139                    src_factor: wgpu::BlendFactor::SrcAlpha,
140                    dst_factor: wgpu::BlendFactor::OneMinusSrcAlpha,
141                    operation: wgpu::BlendOperation::Add,
142                },
143                alpha: wgpu::BlendComponent::OVER,
144            })
145            .with_no_culling()
146            .with_depth_stencil(wgpu::DepthStencilState {
147                depth_write_enabled: true,
148                depth_compare: wgpu::CompareFunction::Less,
149                format: wgpu::TextureFormat::Depth24Plus,
150                stencil: wgpu::StencilState::default(),
151                bias: wgpu::DepthBiasState::default(),
152            })
153            .build();
154
155        // Load wall grid from file
156        let (maze_grid, exit_cell) = parse_maze_file("src/maze/saved-mazes/test.mz");
157
158        let mut floor_vertices = Vertex::create_floor_vertices(&maze_grid, exit_cell);
159
160        // Generate wall geometry
161        let mut wall_vertices = Vertex::create_wall_vertices(&maze_grid);
162
163        // Append wall vertices to floor
164        floor_vertices.append(&mut wall_vertices);
165
166        let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
167            label: Some("Combined Vertex Buffer"),
168            contents: bytemuck::cast_slice(&floor_vertices),
169            usage: wgpu::BufferUsages::VERTEX,
170        });
171
172        let star_renderer = stars::create_star_renderer(device, surface_config, 100);
173
174        let debug_renderer = DebugRenderer {
175            debug_render_bounding_boxes: false,
176            debug_vertex_buffer: None,
177            debug_vertex_count: 0,
178        };
179
180        Self {
181            pipeline,
182            vertex_buffer,
183            vertex_count: floor_vertices.len() as u32,
184            uniform_buffer,
185            uniform_bind_group,
186            depth_texture: None,
187            star_renderer,
188            debug_renderer,
189        }
190    }

192    pub fn update_depth_texture(
193        &mut self,
194        device: &wgpu::Device,
195        width: u32,
196        height: u32,
197    ) -> wgpu::TextureView {
198        if self.depth_texture.is_none()
199            || self.depth_texture.as_ref().unwrap().width() != width
200            || self.depth_texture.as_ref().unwrap().height() != height
201        {
202            if let Some(depth_texture) = self.depth_texture.take() {
203                // Manually drop the texture to free up resources
204                drop(depth_texture);
205            }
206
207            self.depth_texture = Some(device.create_texture(&wgpu::TextureDescriptor {
208                label: Some("Depth Texture"),
209                size: wgpu::Extent3d {
210                    width,
211                    height,
212                    depth_or_array_layers: 1,
213                },
214                mip_level_count: 1,
215                sample_count: 1,
216                dimension: wgpu::TextureDimension::D2,
217                format: wgpu::TextureFormat::Depth24Plus,
218                usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
219                view_formats: &[],
220            }));
221        }
222        self.depth_texture
223            .as_ref()
224            .unwrap()
225            .create_view(&wgpu::TextureViewDescriptor::default())
226    }

228    pub fn render_game(
229        &mut self,
230        queue: &wgpu::Queue,
231        player: &Player,
232        pass: &mut wgpu::RenderPass,
233        aspect: f32,
234    ) {
235        // Step 1: Model Matrix - Just identity since the floor is at world origin
236        let model_matrix = Mat4::identity();
237
238        // Step 2: View Matrix - Based on player's camera position and orientation
239        let view_matrix = player.get_view_matrix();
240
241        // Step 3: Projection Matrix - Using FOV from UI state
242        let projection_matrix = Mat4::perspective(
243            deg_to_rad(player.fov),
244            aspect,
245            0.1,    // zNear
246            2000.0, // zFar
247        );
248
249        // Step 4: Combine matrices: Projection * View * Model
250        let final_mvp_matrix = projection_matrix
251            .multiply(&view_matrix)
252            .multiply(&model_matrix);
253
254        let uniforms = Uniforms {
255            matrix: final_mvp_matrix.into(), // Access the inner `[[f32; 4]; 4]` array
256        };
257        // upload the uniform values to the uniform buffer
258        queue.write_buffer(&self.uniform_buffer, 0, uniforms.as_bytes());
259
260        pass.set_pipeline(&self.pipeline);
261        pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
262        pass.set_bind_group(0, &self.uniform_bind_group, &[]);
263        pass.draw(0..self.vertex_count, 0..1);
264
265        // Inside your render method:
266        if self.debug_renderer.debug_render_bounding_boxes
267            && self.debug_renderer.debug_vertex_count > 0
268        {
269            if let Some(debug_buffer) = &self.debug_renderer.debug_vertex_buffer {
270                pass.set_vertex_buffer(0, debug_buffer.slice(..));
271                pass.draw(0..self.debug_renderer.debug_vertex_count as u32, 0..1);
272            }
273        }
274    }

277/// Uniform data structure for the loading bar shader.
278///
279/// This structure is uploaded to the GPU and used by the loading bar fragment shader
280/// to determine how much of the progress bar to fill.
281///
282/// ## Memory Layout
283///
284/// The structure uses `#[repr(C)]` to ensure consistent memory layout between
285/// Rust and WGSL. The padding ensures 16-byte alignment as required by WGSL
286/// uniform buffer rules.
287///
288/// ## Shader Usage
289///
290/// In WGSL, this corresponds to:
291/// ```wgsl
292/// struct LoadingBarUniforms {
293///     progress: f32,
294///     // Implicit padding in WGSL
295/// }
296/// @group(0) @binding(0) var<uniform> uniforms: LoadingBarUniforms;
297/// ```
298///
299/// # Fields
300///
301/// - `progress` - Loading progress from 0.0 (0%) to 1.0 (100%)
302/// - `_padding` - Ensures proper WGSL alignment (16 bytes total)
303#[repr(C)]
304#[derive(Debug, Copy, Clone, bytemuck::Pod, bytemuck::Zeroable)]
305pub struct LoadingBarUniforms {
306    pub progress: f32,
307    pub _padding: [f32; 3],
308}

310/// Uniform data structure for the exit shader effects.
311///
312/// This structure provides time and screen resolution information to the
313/// exit shader for creating animated procedural effects.
314///
315/// ## Memory Layout
316///
317/// Uses `#[repr(C)]` for consistent layout. The resolution is stored as a 2-component
318/// vector for easy use in shader calculations.
319///
320/// ## Shader Usage
321///
322/// In WGSL, this corresponds to:
323/// ```wgsl
324/// struct ExitShaderUniforms {
325///     time: f32,
326///     resolution: vec2<f32>,
327///     // Implicit padding in WGSL
328/// }
329/// @group(0) @binding(0) var<uniform> uniforms: ExitShaderUniforms;
330/// ```
331///
332/// # Fields
333///
334/// - `time` - Elapsed time in seconds since shader start (for animation)
335/// - `resolution` - Screen/viewport resolution as [width, height]
336/// - `_padding` - Ensures proper WGSL alignment
337#[repr(C)]
338#[derive(Debug, Copy, Clone, bytemuck::Pod, bytemuck::Zeroable)]
339pub struct ExitShaderUniforms {
340    pub time: f32,
341    pub resolution: [f32; 2],
342    pub _padding: [f32; 3],
343}

345/// Renders a maze texture to the screen with proper scaling and filtering.
346///
347/// This renderer takes a pre-generated maze texture and displays it fullscreen
348/// or in a specified viewport. It handles texture sampling and can apply
349/// various filtering modes.
350///
351/// ## Rendering Pipeline
352///
353/// 1. Uses a fullscreen quad vertex buffer
354/// 2. Samples the maze texture with the provided sampler
355/// 3. Outputs directly to the framebuffer
356///
357/// ## Texture Requirements
358///
359/// - Format: `Rgba8UnormSrgb` (or compatible)
360/// - Usage: Must include `TEXTURE_BINDING`
361/// - The texture should contain the pre-rendered maze data
362///
363/// ## Performance Notes
364///
365/// This renderer is very efficient as it only performs texture sampling
366/// without complex computations. It's suitable for real-time rendering
367/// of static or infrequently updated maze textures.
368///
369/// ## Example Usage
370///
371/// ```rust,no_run
372/// # use crate::renderer::maze_renderer::MazeRenderer;
373/// # let device: egui_wgpu::wgpu::Device = unimplemented!();
374/// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
375/// # let texture_view: egui_wgpu::wgpu::TextureView = unimplemented!();
376/// # let sampler: egui_wgpu::wgpu::Sampler = unimplemented!();
377/// # let mut render_pass: egui_wgpu::wgpu::RenderPass = unimplemented!();
378///
379/// let renderer = MazeRenderer::new(&device, &surface_config, &texture_view, &sampler);
380///
381/// // In render loop:
382/// renderer.render(&mut render_pass);
383/// ```
384pub struct MazeRenderer {
385    /// The render pipeline configured for maze texture rendering
386    pub pipeline: wgpu::RenderPipeline,
387    /// Vertex buffer containing fullscreen quad vertices
388    pub vertex_buffer: wgpu::Buffer,
389    /// Bind group containing the maze texture and sampler
390    pub bind_group: wgpu::BindGroup,
391}

393impl MazeRenderer {
394    /// Create a new maze renderer.
395    ///
396    /// This sets up the complete rendering pipeline for displaying a maze texture,
397    /// including the render pipeline, vertex buffer, and bind group.
398    ///
399    /// ## Shader Requirements
400    ///
401    /// The maze shader should be located at `../maze/2D-maze-shader.wgsl` relative
402    /// to this module and should expect:
403    /// - Vertex input: `@location(0) position: vec2<f32>`
404    /// - Texture binding: `@group(0) @binding(0) var maze_texture: texture_2d<f32>;`
405    /// - Sampler binding: `@group(0) @binding(1) var maze_sampler: sampler;`
406    ///
407    /// # Parameters
408    ///
409    /// - `device` - WGPU device for creating GPU resources
410    /// - `surface_config` - Surface configuration (provides target format)
411    /// - `texture_view` - View of the maze texture to render
412    /// - `sampler` - Sampler for texture filtering (typically nearest neighbor for pixel art)
413    ///
414    /// # Returns
415    ///
416    /// A configured `MazeRenderer` ready for rendering.
417    ///
418    /// # Example
419    ///
420    /// ```rust,no_run
421    /// # use crate::renderer::maze_renderer::{MazeRenderer, MazeRenderConfig};
422    /// # let device: egui_wgpu::wgpu::Device = unimplemented!();
423    /// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
424    ///
425    /// // Create maze texture
426    /// let config = MazeRenderConfig::new(25, 25);
427    /// let (texture, texture_view, sampler) = config.create_maze_texture(&device);
428    ///
429    /// let renderer = MazeRenderer::new(&device, &surface_config, &texture_view, &sampler);
430    /// ```
431    pub fn new(
432        device: &wgpu::Device,
433        surface_config: &wgpu::SurfaceConfiguration,
434        texture_view: &wgpu::TextureView,
435        sampler: &wgpu::Sampler,
436    ) -> Self {
437        let bind_group_layout = BindGroupLayoutBuilder::new(device)
438            .with_label("Maze Texture Bind Group Layout")
439            .with_texture(0, wgpu::ShaderStages::FRAGMENT)
440            .with_sampler(1, wgpu::ShaderStages::FRAGMENT)
441            .build();
442
443        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
444            layout: &bind_group_layout,
445            entries: &[
446                wgpu::BindGroupEntry {
447                    binding: 0,
448                    resource: wgpu::BindingResource::TextureView(texture_view),
449                },
450                wgpu::BindGroupEntry {
451                    binding: 1,
452                    resource: wgpu::BindingResource::Sampler(sampler),
453                },
454            ],
455            label: Some("Maze Texture Bind Group"),
456        });
457
458        let pipeline = PipelineBuilder::new(device, surface_config.format)
459            .with_label("Maze Render Pipeline")
460            .with_shader(include_str!("../maze/2D-maze-shader.wgsl"))
461            .with_vertex_buffer(create_vertex_2d_layout())
462            .with_bind_group_layout(&bind_group_layout)
463            .build();
464
465        let vertex_buffer = create_fullscreen_vertices(device);
466
467        Self {
468            pipeline,
469            vertex_buffer,
470            bind_group,
471        }
472    }

474    /// Render the maze to the current render pass.
475    ///
476    /// This method performs the actual rendering by setting up the pipeline,
477    /// bind groups, and vertex buffer, then issuing a draw call for the
478    /// fullscreen quad.
479    ///
480    /// ## Render State
481    ///
482    /// This method assumes:
483    /// - A render pass is active
484    /// - The render pass targets are compatible with the pipeline
485    /// - No scissor test or viewport restrictions (renders fullscreen)
486    ///
487    /// ## Performance
488    ///
489    /// Very fast - only 6 vertices and simple texture sampling.
490    /// Suitable for real-time rendering at high frame rates.
491    ///
492    /// # Parameters
493    ///
494    /// - `render_pass` - Active render pass to render into
495    ///
496    /// # Example
497    ///
498    /// ```rust,no_run
499    /// # use crate::renderer::maze_renderer::MazeRenderer;
500    /// # let renderer: MazeRenderer = unimplemented!();
501    /// # let mut render_pass: egui_wgpu::wgpu::RenderPass = unimplemented!();
502    ///
503    /// // In render loop:
504    /// renderer.render(&mut render_pass);
505    /// ```
506    pub fn render(&self, render_pass: &mut wgpu::RenderPass) {
507        render_pass.set_pipeline(&self.pipeline);
508        render_pass.set_bind_group(0, &self.bind_group, &[]);
509        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
510        render_pass.draw(0..6, 0..1);
511    }
512}

514/// Renders an animated progress bar for loading operations.
515///
516/// This renderer displays a customizable loading bar that shows progress
517/// from 0% to 100%. It uses alpha blending to overlay on top of other
518/// content and can be styled through the shader.
519///
520/// ## Visual Design
521///
522/// The loading bar appearance is defined in the shader and typically includes:
523/// - Background bar (usually semi-transparent)
524/// - Filled portion indicating progress (usually opaque, colored)
525/// - Smooth progress animation
526/// - Optional decorative elements (borders, gradients, etc.)
527///
528/// ## Blending
529///
530/// Uses alpha blending to overlay on existing content. This allows the
531/// loading bar to appear on top of other UI elements or the maze background.
532///
533/// ## Performance
534///
535/// Very lightweight - only updates uniform buffer when progress changes
536/// and renders a simple fullscreen quad.
537///
538/// ## Example Usage
539///
540/// ```rust,no_run
541/// # use crate::renderer::maze_renderer::LoadingBarRenderer;
542/// # let device: egui_wgpu::wgpu::Device = unimplemented!();
543/// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
544/// # let queue: egui_wgpu::wgpu::Queue = unimplemented!();
545/// # let mut render_pass: egui_wgpu::wgpu::RenderPass = unimplemented!();
546///
547/// let mut renderer = LoadingBarRenderer::new(&device, &surface_config);
548///
549/// // Update progress (0.0 to 1.0)
550/// renderer.update_progress(&queue, 0.5); // 50% complete
551///
552/// // Render the loading bar
553/// renderer.render(&mut render_pass);
554/// ```
555pub struct LoadingBarRenderer {
556    /// The render pipeline configured for loading bar rendering with alpha blending
557    pub pipeline: wgpu::RenderPipeline,
558    /// Vertex buffer containing fullscreen quad vertices
559    pub vertex_buffer: wgpu::Buffer,
560    /// Uniform buffer storing the current progress value
561    pub uniform_buffer: wgpu::Buffer,
562    /// Bind group containing the uniform buffer
563    pub bind_group: wgpu::BindGroup,
564}

566impl LoadingBarRenderer {
567    /// Create a new loading bar renderer.
568    ///
569    /// This sets up the complete rendering pipeline for displaying a progress bar,
570    /// including alpha blending support for overlay rendering.
571    ///
572    /// ## Shader Requirements
573    ///
574    /// The loading bar shader should be located at `../maze/loading-bar-shader.wgsl`
575    /// and should expect:
576    /// - Vertex input: `@location(0) position: vec2<f32>`
577    /// - Uniform binding: `@group(0) @binding(0) var<uniform> uniforms: LoadingBarUniforms;`
578    ///
579    /// ## Initial State
580    ///
581    /// The renderer starts with 0% progress. Use [`update_progress()`](LoadingBarRenderer::update_progress)
582    /// to change the progress value.
583    ///
584    /// # Parameters
585    ///
586    /// - `device` - WGPU device for creating GPU resources
587    /// - `surface_config` - Surface configuration (provides target format)
588    ///
589    /// # Returns
590    ///
591    /// A configured `LoadingBarRenderer` ready for rendering.
592    ///
593    /// # Example
594    ///
595    /// ```rust,no_run
596    /// # use crate::renderer::maze_renderer::LoadingBarRenderer;
597    /// # let device: egui_wgpu::wgpu::Device = unimplemented!();
598    /// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
599    ///
600    /// let renderer = LoadingBarRenderer::new(&device, &surface_config);
601    /// ```
602    pub fn new(device: &wgpu::Device, surface_config: &wgpu::SurfaceConfiguration) -> Self {
603        let uniforms = LoadingBarUniforms {
604            progress: 0.0,
605            _padding: [0.0; 3],
606        };
607
608        let uniform_buffer = create_uniform_buffer(device, &uniforms, "Loading Bar Uniform Buffer");
609
610        let bind_group_layout = BindGroupLayoutBuilder::new(device)
611            .with_label("Loading Bar Bind Group Layout")
612            .with_uniform_buffer(0, wgpu::ShaderStages::FRAGMENT)
613            .build();
614
615        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
616            layout: &bind_group_layout,
617            entries: &[wgpu::BindGroupEntry {
618                binding: 0,
619                resource: uniform_buffer.as_entire_binding(),
620            }],
621            label: Some("Loading Bar Bind Group"),
622        });
623
624        let pipeline = PipelineBuilder::new(device, surface_config.format)
625            .with_label("Loading Bar Pipeline")
626            .with_shader(include_str!("../maze/loading-bar-shader.wgsl"))
627            .with_vertex_buffer(create_vertex_2d_layout())
628            .with_bind_group_layout(&bind_group_layout)
629            .with_alpha_blending()
630            .build();
631
632        let vertex_buffer = create_fullscreen_vertices(device);
633
634        Self {
635            pipeline,
636            vertex_buffer,
637            uniform_buffer,
638            bind_group,
639        }

642    /// Update the loading progress.
643    ///
644    /// This method uploads new progress data to the GPU uniform buffer.
645    /// The progress value is automatically clamped to the valid range [0.0, 1.0].
646    ///
647    /// ## Performance Notes
648    ///
649    /// This operation involves a GPU buffer write, so avoid calling it
650    /// unnecessarily frequently (e.g., every frame with the same value).
651    /// It's designed for periodic updates as loading progresses.
652    ///
653    /// # Parameters
654    ///
655    /// - `queue` - WGPU queue for buffer uploads
656    /// - `progress` - Progress value from 0.0 (0%) to 1.0 (100%)
657    ///
658    /// # Example
659    ///
660    /// ```rust,no_run
661    /// # use crate::renderer::maze_renderer::LoadingBarRenderer;
662    /// # let renderer: LoadingBarRenderer = unimplemented!();
663    /// # let queue: egui_wgpu::wgpu::Queue = unimplemented!();
664    ///
665    /// // Update progress to 75%
666    /// renderer.update_progress(&queue, 0.75);
667    ///
668    /// // Values outside [0,1] are automatically clamped
669    /// renderer.update_progress(&queue, 1.5); // Becomes 1.0
670    /// renderer.update_progress(&queue, -0.1); // Becomes 0.0
671    /// ```
672    pub fn update_progress(&self, queue: &wgpu::Queue, progress: f32) {
673        let uniforms = LoadingBarUniforms {
674            progress: progress.clamp(0.0, 1.0),
675            _padding: [0.0; 3],
676        };
677        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::cast_slice(&[uniforms]));
678    }

717/// Renders animated procedural effects for the maze exit cell.
718///
719/// This renderer creates dynamic visual effects specifically for the exit cell
720/// of the maze. It uses a full-screen triangle technique for efficiency and
721/// applies scissor testing to limit rendering to only the exit cell area.
722///
723/// ## Rendering Technique
724///
725/// Uses the "full-screen triangle" technique instead of a quad:
726/// - Single triangle with vertices outside the viewport
727/// - GPU clips to viewport automatically
728/// - More efficient than quad (fewer vertices, no diagonal edge)
729/// - Combined with scissor test for precise cell targeting
730///
731/// ## Coordinate System
732///
733/// The renderer expects maze coordinates where:
734/// - (0,0) is the top-left cell
735/// - Coordinates increase right (x) and down (y)
736/// - Grid size is assumed to be 25x25 cells
737/// - Cell rendering includes shrinking for visual borders
738///
739/// ## Animation
740///
741/// The shader receives time and resolution uniforms for creating
742/// time-based animations. The internal timer starts when the renderer
743/// is created.
744///
745/// ## Example Usage
746///
747/// ```rust,no_run
748/// # use crate::renderer::maze_renderer::ExitShaderRenderer;
749/// # let device: egui_wgpu::wgpu::Device = unimplemented!();
750/// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
751/// # let queue: egui_wgpu::wgpu::Queue = unimplemented!();
752/// # let window: &winit::window::Window = unimplemented!();
753/// # let mut render_pass: egui_wgpu::wgpu::RenderPass = unimplemented!();
754///
755/// let mut renderer = ExitShaderRenderer::new(&device, &surface_config);
756///
757/// // In render loop:
758/// let elapsed = renderer.start_time.elapsed().as_secs_f32();
759/// renderer.update_uniforms(&queue, [800.0, 600.0], elapsed);
760/// renderer.render_to_cell(&mut render_pass, window, (24, 24)); // Bottom-right exit
761/// ```
762pub struct ExitShaderRenderer {
763    pub pipeline: wgpu::RenderPipeline,
764    pub uniform_buffer: wgpu::Buffer,
765    pub bind_group: wgpu::BindGroup,
766    pub start_time: Instant,
767}
768
769impl ExitShaderRenderer {
770    pub fn new(device: &wgpu::Device, surface_config: &wgpu::SurfaceConfiguration) -> Self {
771        let uniforms = ExitShaderUniforms {
772            time: 0.0,
773            resolution: [800.0, 600.0], // Default resolution, will be updated
774            _padding: [0.0; 3],
775        };
776
777        let uniform_buffer = create_uniform_buffer(device, &uniforms, "Exit Shader Uniform Buffer");
778
779        let bind_group_layout = BindGroupLayoutBuilder::new(device)
780            .with_label("Exit Shader Bind Group Layout")
781            .with_uniform_buffer(0, wgpu::ShaderStages::VERTEX_FRAGMENT)
782            .build();
783
784        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
785            layout: &bind_group_layout,
786            entries: &[wgpu::BindGroupEntry {
787                binding: 0,
788                resource: uniform_buffer.as_entire_binding(),
789            }],
790            label: Some("Exit Shader Bind Group"),
791        });
792
793        // Exit shader uses a full-screen triangle trick, so no vertex buffer needed
794        let pipeline = PipelineBuilder::new(device, surface_config.format)
795            .with_label("Exit Shader Pipeline")
796            .with_shader(include_str!("../maze/exit_shader.wgsl"))
797            .with_bind_group_layout(&bind_group_layout)
798            .build();
799
800        Self {
801            pipeline,
802            uniform_buffer,
803            bind_group,
804            start_time: Instant::now(),
805        }
806    }
807
808    pub fn update_uniforms(&self, queue: &wgpu::Queue, resolution: [f32; 2], time: f32) {
809        let uniforms = ExitShaderUniforms {
810            time,
811            resolution,
812            _padding: [0.0; 3],
813        };
814        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::cast_slice(&[uniforms]));
815    }
816
817    pub fn render_to_cell(
818        &self,
819        render_pass: &mut wgpu::RenderPass,
820        window: &winit::window::Window,
821        exit_cell: (usize, usize),
822    ) {
823        // Calculate scissor rectangle for the exit cell
824        let grid_size = 25.0;
825        let shrink_factor = 0.845;
826        let border = 3.0;
827
828        let window_size = window.inner_size();
829        let total_width = window_size.width as f32;
830        let total_height = window_size.height as f32;
831
832        let usable_width = total_width - 2.0 * border;
833        let usable_height = total_height - 2.0 * border;
834
835        let cell_width = usable_width / grid_size;
836        let cell_height = usable_height / grid_size;
837
838        let full_x = border + exit_cell.0 as f32 * cell_width;
839        let full_y = border + exit_cell.1 as f32 * cell_height;
840
841        let shrunk_width = cell_width * shrink_factor;
842        let shrunk_height = cell_height * shrink_factor;
843
844        let offset_x = (cell_width - shrunk_width) / 2.0;
845        let offset_y = (cell_height - shrunk_height) / 2.0;
846
847        let scissor_x = (full_x + offset_x).round() as u32;
848        let scissor_y = (full_y + offset_y).round() as u32;
849        let scissor_width = shrunk_width.round() as u32;
850        let scissor_height = shrunk_height.round() as u32;
851
852        // Set scissor rect and render
853        render_pass.set_scissor_rect(scissor_x, scissor_y, scissor_width, scissor_height);
854        render_pass.set_pipeline(&self.pipeline);
855        render_pass.set_bind_group(0, &self.bind_group, &[]);
856        render_pass.draw(0..3, 0..1); // Full-screen triangle
857    }
858}

872        let render_height = maze_height * 5 + 1;

876            maze_height,
877            render_width,

881
882    pub fn create_maze_texture(
883        &self,
884        device: &wgpu::Device,
885    ) -> (wgpu::Texture, wgpu::TextureView, wgpu::Sampler) {
886        let texture_size = wgpu::Extent3d {
887            width: self.render_width,
888            height: self.render_height,
889            depth_or_array_layers: 1,
890        };
891
892        let texture = device.create_texture(&wgpu::TextureDescriptor {
893            label: Some("Maze Texture"),
894            size: texture_size,
895            mip_level_count: 1,
896            sample_count: 1,
897            dimension: wgpu::TextureDimension::D2,
898            format: wgpu::TextureFormat::Rgba8UnormSrgb,
899            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
900            view_formats: &[],
901        });
902
903        let texture_view = texture.create_view(&wgpu::TextureViewDescriptor::default());
904
905        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
906            address_mode_u: wgpu::AddressMode::ClampToEdge,
907            address_mode_v: wgpu::AddressMode::ClampToEdge,
908            address_mode_w: wgpu::AddressMode::ClampToEdge,
909            mag_filter: wgpu::FilterMode::Nearest,
910            min_filter: wgpu::FilterMode::Nearest,
911            mipmap_filter: wgpu::FilterMode::Nearest,
912            ..Default::default()
913        });
914
915        (texture, texture_view, sampler)
916    }
