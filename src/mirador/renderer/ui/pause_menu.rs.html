50    fn scaled_text_style(window_height: f32) -> crate::renderer::text::TextStyle {
51        // Virtual DPI scaling based on reference height
52        let reference_height = 1080.0;
53        let scale = (window_height / reference_height).clamp(0.7, 2.0);
54        let font_size = (32.0 * scale).clamp(16.0, 48.0); // 32px at 1080p, min 16, max 48
55        let line_height = (40.0 * scale).clamp(24.0, 60.0); // 40px at 1080p, min 24, max 60
56        crate::renderer::text::TextStyle {
57            font_family: "HankenGrotesk".to_string(),
58            font_size,
59            line_height,
60            color: crate::renderer::ui::button::create_primary_button_style()
61                .text_style
62                .color,
63            weight: glyphon::Weight::MEDIUM,
64            style: glyphon::Style::Normal,
65        }
66    }

68    fn create_menu_buttons(button_manager: &mut ButtonManager, window_size: PhysicalSize<u32>) {
69        let reference_height = 1080.0;
70        let scale = (window_size.height as f32 / reference_height).clamp(0.7, 2.0);
71        // Button sizing with DPI scaling
72        let button_width = (window_size.width as f32 * 0.38 * scale).clamp(180.0, 600.0);
73        let button_height = (window_size.height as f32 * 0.09 * scale).clamp(32.0, 140.0);
74        let button_spacing = (window_size.height as f32 * 0.015 * scale).clamp(2.0, 24.0);
75        let total_height = button_height * 5.0 + button_spacing * 4.0;
76        let center_x = window_size.width as f32 / 2.0;
77        let start_y = (window_size.height as f32 - total_height) / 2.0;
78        let text_style = Self::scaled_text_style(window_size.height as f32);
79
80        // Helper for y position
81        let y =
82            |i: usize| start_y + button_height / 2.0 + i as f32 * (button_height + button_spacing);
83
84        // Resume button
85        let mut resume_style = create_primary_button_style();
86        resume_style.text_style = text_style.clone();
87        let resume_button = Button::new("resume", "Resume Game")
88            .with_style(resume_style)
89            .with_text_align(TextAlign::Center)
90            .with_position(
91                ButtonPosition::new(center_x, y(0), button_width, button_height)
92                    .with_anchor(ButtonAnchor::Center),
93            );
94
95        // Restart Run button with goldenrod style
96        let mut restart_run_style = create_goldenrod_button_style();
97        restart_run_style.text_style = text_style.clone();
98        let restart_run_button = Button::new("restart_run", "Restart Run")
99            .with_style(restart_run_style)
100            .with_text_align(TextAlign::Center)
101            .with_position(
102                ButtonPosition::new(center_x, y(1), button_width, button_height)
103                    .with_anchor(ButtonAnchor::Center),
104            );
105
106        // Toggle Test Mode button with goldenrod style
107        let mut test_mode_style = create_goldenrod_button_style();
108        test_mode_style.text_style = text_style.clone();
109        let test_mode_button = Button::new("toggle_test_mode", "Toggle Test Mode")
110            .with_style(test_mode_style)
111            .with_text_align(TextAlign::Center)
112            .with_position(
113                ButtonPosition::new(center_x, y(2), button_width, button_height)
114                    .with_anchor(ButtonAnchor::Center),
115            );
116
117        // Quit to Lobby button with less saturated red style
118        let mut quit_lobby_style = create_lobby_button_style();
119        quit_lobby_style.text_style = text_style.clone();
120        let quit_lobby_button = Button::new("quit_lobby", "Quit to Lobby")
121            .with_style(quit_lobby_style)
122            .with_text_align(TextAlign::Center)
123            .with_position(
124                ButtonPosition::new(center_x, y(3), button_width, button_height)
125                    .with_anchor(ButtonAnchor::Center),
126            );
127
128        // Quit to Menu button - Center aligned for comparison, now darker red
129        let mut quit_style = create_danger_button_style();
130        quit_style.text_style = text_style.clone();
131        let quit_menu_button = Button::new("quit_menu", "Quit App")
132            .with_style(quit_style)
133            .with_text_align(TextAlign::Center)
134            .with_position(
135                ButtonPosition::new(center_x, y(4), button_width, button_height)
136                    .with_anchor(ButtonAnchor::Center),
137            );
138
139        // Add debug button in bottom left
140        let mut debug_style = create_warning_button_style();
141        debug_style.text_style.font_size = text_style.font_size * 0.5;
142        debug_style.text_style.line_height = text_style.line_height * 0.5;
143        debug_style.padding = (2.0 * scale, 6.0 * scale); // minimal horizontal, some vertical padding
144        debug_style.spacing = crate::renderer::ui::button::ButtonSpacing::Wrap;
145        // Measure the text width for three lines
146        let (_min_x, text_width, text_height) = button_manager
147            .text_renderer
148            .measure_text(" Show\nDebug\n  Info", &debug_style.text_style);
149        let debug_button_side = text_width.max(text_height) + 2.0 * debug_style.padding.1;
150        let debug_button = Button::new("debug", " Show\nDebug\n  Info")
151            .with_style(debug_style)
152            .with_text_align(TextAlign::Center)
153            .with_position(ButtonPosition {
154                x: 60.0,
155                y: window_size.height as f32 - debug_button_side - 16.0, // 16px from bottom
156                width: debug_button_side,
157                height: debug_button_side,
158                anchor: ButtonAnchor::TopLeft,
159            });
160
161        // Add buttons to manager
162        button_manager.add_button(resume_button);
163        button_manager.add_button(restart_run_button);
164        button_manager.add_button(test_mode_button);
165        button_manager.add_button(quit_lobby_button);
166        button_manager.add_button(quit_menu_button);
167        button_manager.add_button(debug_button);
168
169        // Update button positions to ensure text is properly centered
170        button_manager.update_button_positions();
171    }

173    pub fn show(&mut self, is_test_mode: bool) {
174        self.visible = true;
175        self.last_action = PauseMenuAction::None;
176
177        // Show all buttons
178        for button in self.button_manager.buttons.values_mut() {
179            button.set_visible(true);
180        }
181        // Ensure button text is made visible and styled immediately
182        self.button_manager.update_button_states();
183        // Update the test mode button text
184        self.update_test_mode_button_text(is_test_mode);
185    }

189        self.last_action = PauseMenuAction::None;

191        // Hide all buttons

209    pub fn handle_input(&mut self, event: &WindowEvent, audio_manager: &mut GameAudioManager) {
210        if !self.visible {
211            return;
212        }
213
214        self.button_manager.handle_input(event);
215
216        // Check for button clicks and play select sound
217        if self.button_manager.is_button_clicked("resume") {
218            self.last_action = PauseMenuAction::Resume;
219            let _ = audio_manager.play_select();
220        }
221
222        if self.button_manager.is_button_clicked("restart_run") {
223            self.last_action = PauseMenuAction::Restart;
224            let _ = audio_manager.play_select();
225        }
226
227        if self.button_manager.is_button_clicked("quit_lobby") {
228            self.last_action = PauseMenuAction::QuitToMenu;
229            let _ = audio_manager.play_select();
230        }
231
232        if self.button_manager.is_button_clicked("toggle_test_mode") {
233            self.last_action = PauseMenuAction::ToggleTestMode;
234            let _ = audio_manager.play_select();
235        }
236
237        if self.button_manager.is_button_clicked("quit_menu") {
238            self.last_action = PauseMenuAction::QuitApp;
239            let _ = audio_manager.play_select();
240        }
241        if self.button_manager.is_button_clicked("debug") {
242            self.show_debug_panel = !self.show_debug_panel;
243            let _ = audio_manager.play_select();
244        }
245    }

264    fn recreate_buttons_for_new_size(&mut self) {
265        let window_size = self.button_manager.window_size;
266        let reference_height = 1080.0;
267        let scale = (window_size.height as f32 / reference_height).clamp(0.7, 2.0);
268        let button_width = (window_size.width as f32 * 0.38 * scale).clamp(180.0, 600.0);
269        let button_height = (window_size.height as f32 * 0.09 * scale).clamp(32.0, 140.0);
270        let button_spacing = (window_size.height as f32 * 0.015 * scale).clamp(2.0, 24.0);
271        let total_height = button_height * 5.0 + button_spacing * 4.0;
272        let center_x = window_size.width as f32 / 2.0;
273        let start_y = (window_size.height as f32 - total_height) / 2.0;
274        let text_style = Self::scaled_text_style(window_size.height as f32);
275        let y =
276            |i: usize| start_y + button_height / 2.0 + i as f32 * (button_height + button_spacing);
277
278        // Update button positions and text style
279        if let Some(resume_button) = self.button_manager.get_button_mut("resume") {
280            resume_button.position.x = center_x;
281            resume_button.position.y = y(0);
282            resume_button.position.width = button_width;
283            resume_button.position.height = button_height;
284            resume_button.position.anchor = ButtonAnchor::Center;
285            resume_button.style.text_style = text_style.clone();
286        }
287
288        if let Some(restart_run_button) = self.button_manager.get_button_mut("restart_run") {
289            restart_run_button.text = "Restart Run".to_string();
290            restart_run_button.style = create_goldenrod_button_style();
291            restart_run_button.style.text_style = text_style.clone();
292            restart_run_button.position.x = center_x;
293            restart_run_button.position.y = y(1);
294            restart_run_button.position.width = button_width;
295            restart_run_button.position.height = button_height;
296            restart_run_button.position.anchor = ButtonAnchor::Center;
297        }
298
299        if let Some(test_mode_button) = self.button_manager.get_button_mut("toggle_test_mode") {
300            test_mode_button.text = "Toggle Test Mode".to_string();
301            test_mode_button.style = create_goldenrod_button_style();
302            test_mode_button.style.text_style = text_style.clone();
303            test_mode_button.position.x = center_x;
304            test_mode_button.position.y = y(2);
305            test_mode_button.position.width = button_width;
306            test_mode_button.position.height = button_height;
307            test_mode_button.position.anchor = ButtonAnchor::Center;
308        }
309
310        if let Some(quit_lobby_button) = self.button_manager.get_button_mut("quit_lobby") {
311            quit_lobby_button.text = "Quit to Lobby".to_string();
312            quit_lobby_button.style = create_lobby_button_style();
313            quit_lobby_button.style.text_style = text_style.clone();
314            quit_lobby_button.position.x = center_x;
315            quit_lobby_button.position.y = y(3);
316            quit_lobby_button.position.width = button_width;
317            quit_lobby_button.position.height = button_height;
318            quit_lobby_button.position.anchor = ButtonAnchor::Center;
319        }
320
321        if let Some(quit_menu_button) = self.button_manager.get_button_mut("quit_menu") {
322            quit_menu_button.style = create_danger_button_style();
323            quit_menu_button.style.text_style = text_style.clone();
324            quit_menu_button.position.x = center_x;
325            quit_menu_button.position.y = y(4);
326            quit_menu_button.position.width = button_width;
327            quit_menu_button.position.height = button_height;
328            quit_menu_button.position.anchor = ButtonAnchor::Center;
329        }
330
331        // Update debug button position for new window size
332        let (style, padding) =
333            if let Some(debug_button) = self.button_manager.get_button_mut("debug") {
334                debug_button.style.spacing = crate::renderer::ui::button::ButtonSpacing::Wrap;
335                (
336                    debug_button.style.text_style.clone(),
337                    debug_button.style.padding,
338                )
339            } else {
340                (create_warning_button_style().text_style, (2.0, 6.0))
341            };
342        let (_min_x, text_width, text_height) = self
343            .button_manager
344            .text_renderer
345            .measure_text("Show\nDebug\nInfo", &style);
346        let side = text_width.max(text_height) + 2.0 * padding.1;
347        if let Some(debug_button) = self.button_manager.get_button_mut("debug") {
348            debug_button.position.x = 60.0;
349            debug_button.position.y = window_size.height as f32 - side - 16.0;
350            debug_button.position.width = side;
351            debug_button.position.height = side;
352            debug_button.position.anchor = ButtonAnchor::TopLeft;
353        }
354
355        // Update text positions
356        self.button_manager.update_button_positions();
357    }
