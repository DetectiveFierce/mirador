3// Color manipulation helpers for glyphon::Color
4pub trait ColorExt {
5    fn darken(&self, factor: f32) -> Self;
6    fn brighten(&self, factor: f32) -> Self;
7    fn saturate(&self, factor: f32) -> Self;
8}
9
10impl ColorExt for Color {
11    fn darken(&self, factor: f32) -> Self {
12        let factor = factor.clamp(0.0, 1.0);
13        Color::rgba(
14            (self.r() as f32 * (1.0 - factor)) as u8,
15            (self.g() as f32 * (1.0 - factor)) as u8,
16            (self.b() as f32 * (1.0 - factor)) as u8,
17            self.a(),
18        )
19    }
20    fn brighten(&self, factor: f32) -> Self {
21        let factor = factor.clamp(0.0, 1.0);
22        Color::rgba(
23            (self.r() as f32 + (255.0 - self.r() as f32) * factor) as u8,
24            (self.g() as f32 + (255.0 - self.g() as f32) * factor) as u8,
25            (self.b() as f32 + (255.0 - self.b() as f32) * factor) as u8,
26            self.a(),
27        )
28    }
29    fn saturate(&self, factor: f32) -> Self {
30        // Convert RGB to HSL, increase saturation, then convert back
31        let r = self.r() as f32 / 255.0;
32        let g = self.g() as f32 / 255.0;
33        let b = self.b() as f32 / 255.0;
34        let max = r.max(g).max(b);
35        let min = r.min(g).min(b);
36        let l = (max + min) / 2.0;
37        let d = max - min;
38        let mut s = if d == 0.0 {
39            0.0
40        } else {
41            d / (1.0 - (2.0 * l - 1.0).abs())
42        };
43        s = (s + factor).min(1.0);
44        // Recompute RGB from HSL (approximate, since hue is not changed)
45        // We'll just scale the color channels away from the gray axis
46        let gray = l;
47        let scale = if s == 0.0 { 0.0 } else { s };
48        let new_r = gray + (r - gray) * (1.0 + scale);
49        let new_g = gray + (g - gray) * (1.0 + scale);
50        let new_b = gray + (b - gray) * (1.0 + scale);
51        Color::rgba(
52            (new_r.clamp(0.0, 1.0) * 255.0) as u8,
53            (new_g.clamp(0.0, 1.0) * 255.0) as u8,
54            (new_b.clamp(0.0, 1.0) * 255.0) as u8,
55            self.a(),
56        )
57    }
58}
