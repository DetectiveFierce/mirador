3/// Extension trait for color manipulation operations.
4///
5/// This trait provides methods for modifying colors by darkening, brightening,
6/// and saturating them for UI button states and effects.
7pub trait ColorExt {
8    /// Darkens the color by the specified factor.
9    ///
10    /// # Arguments
11    /// * `factor` - Darkening factor between 0.0 and 1.0
12    ///
13    /// # Returns
14    /// A darker version of the color
15    fn darken(&self, factor: f32) -> Self;
16    /// Brightens the color by the specified factor.
17    ///
18    /// # Arguments
19    /// * `factor` - Brightening factor between 0.0 and 1.0
20    ///
21    /// # Returns
22    /// A brighter version of the color
23    fn brighten(&self, factor: f32) -> Self;
24    /// Increases the saturation of the color by the specified factor.
25    ///
26    /// # Arguments
27    /// * `factor` - Saturation factor between 0.0 and 1.0
28    ///
29    /// # Returns
30    /// A more saturated version of the color
31    fn saturate(&self, factor: f32) -> Self;
32}
33
34impl ColorExt for Color {
35    fn darken(&self, factor: f32) -> Self {
36        let factor = factor.clamp(0.0, 1.0);
37        Color::rgba(
38            (self.r() as f32 * (1.0 - factor)) as u8,
39            (self.g() as f32 * (1.0 - factor)) as u8,
40            (self.b() as f32 * (1.0 - factor)) as u8,
41            self.a(),
42        )
43    }
44    fn brighten(&self, factor: f32) -> Self {
45        let factor = factor.clamp(0.0, 1.0);
46        Color::rgba(
47            (self.r() as f32 + (255.0 - self.r() as f32) * factor) as u8,
48            (self.g() as f32 + (255.0 - self.g() as f32) * factor) as u8,
49            (self.b() as f32 + (255.0 - self.b() as f32) * factor) as u8,
50            self.a(),
51        )
52    }
53    fn saturate(&self, factor: f32) -> Self {
54        // Convert RGB to HSL, increase saturation, then convert back
55        let r = self.r() as f32 / 255.0;
56        let g = self.g() as f32 / 255.0;
57        let b = self.b() as f32 / 255.0;
58        let max = r.max(g).max(b);
59        let min = r.min(g).min(b);
60        let l = (max + min) / 2.0;
61        let d = max - min;
62        let mut s = if d == 0.0 {
63            0.0
64        } else {
65            d / (1.0 - (2.0 * l - 1.0).abs())
66        };
67        s = (s + factor).min(1.0);
68        // Recompute RGB from HSL (approximate, since hue is not changed)
69        // We'll just scale the color channels away from the gray axis
70        let gray = l;
71        let scale = if s == 0.0 { 0.0 } else { s };
72        let new_r = gray + (r - gray) * (1.0 + scale);
73        let new_g = gray + (g - gray) * (1.0 + scale);
74        let new_b = gray + (b - gray) * (1.0 + scale);
75        Color::rgba(
76            (new_r.clamp(0.0, 1.0) * 255.0) as u8,
77            (new_g.clamp(0.0, 1.0) * 255.0) as u8,
78            (new_b.clamp(0.0, 1.0) * 255.0) as u8,
79            self.a(),
80        )
81    }
82}

84/// Calculates DPI scaling factor based on window height.
85///
86/// This function provides a simple DPI scaling calculation that scales UI elements
87/// based on the window height, with reasonable bounds to prevent extreme scaling.
88///
89/// # Arguments
90/// * `window_height` - The height of the window in pixels
91///
92/// # Returns
93/// A scaling factor between 0.7 and 2.0
94pub fn dpi_scale(window_height: f32) -> f32 {
95    (window_height / 1080.0).clamp(0.7, 2.0)
