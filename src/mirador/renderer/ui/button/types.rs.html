71pub enum ButtonAnchor {

73    #[default]
74    Center,
75}
76
77impl ButtonPosition {
78    pub fn new(x: f32, y: f32, width: f32, height: f32) -> Self {
79        Self {
80            x,
81            y,
82            width,
83            height,
84            anchor: ButtonAnchor::TopLeft,
85        }
86    }
87
88    pub fn with_anchor(mut self, anchor: ButtonAnchor) -> Self {
89        self.anchor = anchor;
90        self
91    }
92
93    pub fn calculate_actual_position(&self) -> (f32, f32) {
94        let actual_x = match self.anchor {
95            ButtonAnchor::TopLeft => self.x,
96            ButtonAnchor::Center => self.x - self.width / 2.0,
97        };
98
99        let actual_y = match self.anchor {
100            ButtonAnchor::TopLeft => self.y,
101            ButtonAnchor::Center => self.y - self.height / 2.0,
102        };
103
104        (actual_x, actual_y)
105    }
106}
