9pub fn create_primary_button_style() -> ButtonStyle {
10    let scale = dpi_scale(1080.0); // Assuming a default window height for default values
11    ButtonStyle {
12        background_color: Color::rgb(30, 110, 30), // Slightly less saturated, dark mint green
13        hover_color: Color::rgb(25, 85, 25),       // Even darker, maintaining hue
14        pressed_color: Color::rgb(20, 65, 20),     // Darkest mint for pressed state
15        disabled_color: Color::rgb(110, 140, 110), // Muted, lighter mint for disabled state
16        border_color: Color::rgb(25, 85, 25),      // Matches hover color
17        border_width: 1.0,
18        corner_radius: 8.0,
19        padding: (16.0, 10.0),
20        text_style: TextStyle {
21            font_family: "HankenGrotesk".to_string(),
22            font_size: 18.0 * scale,
23            line_height: 20.0 * scale,
24            color: Color::rgb(255, 255, 255), // white
25            weight: Weight::MEDIUM,
26            style: Style::Normal,
27        },
28        text_align: TextAlign::Center,
29        spacing: ButtonSpacing::Hbar(0.3),
30    }
31}
