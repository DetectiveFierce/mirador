1//! # Maze Generation Animation Rendering System
2//!
3//! This module provides specialized renderers for different aspects of a maze visualization
4//! application. It includes renderers for the maze itself, loading progress indicators,
5//! and animated exit effects. As well as the main 3D maze renderer.
6//!
7//! ## Architecture Overview
8//!
9//! The rendering system is built on top of the pipeline builder utilities and provides
10//! three main renderer types:
11//!
12//! - [`MazeRenderer`] - Renders the maze texture to screen
13//! - [`LoadingBarRenderer`] - Shows loading progress with a progress bar
14//! - [`ExitShaderRenderer`] - Creates animated effects for the maze exit
15//! - [`GameRenderer`] - Renders the maze in 3D with the background defined in stars.rs
16//!
17//! ## Coordinate System
18//!
19//! The maze uses a grid-based coordinate system where (0,0) is typically the top-left
20//! corner. The renderers handle conversion between grid coordinates and screen space.
21//!
22//! ## Usage Pattern
23//!
24//! 1. Create renderers during initialization with device and surface configuration
25//! 2. Update renderer state each frame (progress, time, etc.)
26//! 3. Render during the render pass
27//!
28//! ## Example Setup
29//!
30//! ```rust,no_run
31//! use crate::renderer::maze_renderer::{MazeRenderer, LoadingBarRenderer, MazeRenderConfig};
32//! # let device: egui_wgpu::wgpu::Device = unimplemented!();
33//! # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
34//!
35//! // Create maze texture
36//! let config = MazeRenderConfig::new(50, 50); // 50x50 maze
37//! let (texture, texture_view, sampler) = config.create_maze_texture(&device);
38//!
39//! // Create renderers
40//! let maze_renderer = MazeRenderer::new(&device, &surface_config, &texture_view, &sampler);
41//! let loading_renderer = LoadingBarRenderer::new(&device, &surface_config);
42//! ```

56/// Handles rendering of the maze and loading bar on the maze generation animation screen.
57///
58/// This struct manages GPU resources for the maze texture, pipelines, vertex buffers, and loading bar.
59/// It also holds a [`MazeGenerator`] and a shared [`Maze`] instance for generating and displaying the maze.
60///
61/// # Fields
62/// - `generator`: Maze generator for producing new mazes.
63/// - `maze`: Shared, thread-safe reference to the current maze.
64/// - `maze_renderer`: Maze renderer for displaying the maze.
65/// - `loading_bar_renderer`: Loading bar renderer for displaying the loading progress.
66/// - `exit_shader_renderer`: Exit shader renderer for displaying the exit shader.
67/// - `texture`: Texture containing the maze image.
68/// - `last_update`: Timestamp of the last update (for animation/timing).
69pub struct LoadingRenderer {
70    /// Maze generator for producing new mazes.
71    pub generator: MazeGenerator,
72    /// Shared, thread-safe reference to the current maze.
73    pub maze: Arc<Mutex<Maze>>,
74
75    // Rendering components
76    pub maze_renderer: MazeRenderer,
77    pub loading_bar_renderer: LoadingBarRenderer,
78    pub exit_shader_renderer: ExitShaderRenderer,
79
80    /// Texture containing the maze image.
81    pub texture: wgpu::Texture,
82    /// Timestamp of the last update (for animation/timing).
83    pub last_update: Instant,

86impl LoadingRenderer {
87    /// Creates a new simplified maze generation animation screen renderer.
88    pub fn new(device: &wgpu::Device, surface_config: &wgpu::SurfaceConfiguration) -> Self {
89        // Initialize maze generation
90        let maze_width = 25;
91        let maze_height = 25;
92        let (generator, maze) = MazeGenerator::new(maze_width, maze_height);
93
94        // Get render dimensions
95        let config = MazeRenderConfig::new(maze_width as u32, maze_height as u32);
96        let (texture, texture_view, sampler) = config.create_maze_texture(device);
97
98        // Create rendering components
99        let maze_renderer = MazeRenderer::new(device, surface_config, &texture_view, &sampler);
100        let loading_bar_renderer = LoadingBarRenderer::new(device, surface_config);
101        let exit_shader_renderer = ExitShaderRenderer::new(device, surface_config);
102
103        Self {
104            generator,
105            maze,
106            maze_renderer,
107            loading_bar_renderer,
108            exit_shader_renderer,
109            texture,
110            last_update: Instant::now(),
111        }
112    }

114    /// Updates the maze texture with new pixel data.
115    ///
116    /// # Arguments
117    /// - `queue`: The wgpu queue to write to the texture.
118    /// - `maze_data`: The new maze image data (RGBA, row-major).
119    /// - `width`: Width of the maze image.
120    /// - `height`: Height of the maze image.
121    pub fn update_texture(
122        &self,
123        queue: &wgpu::Queue,
124        maze_data: &[u8],
125        width: usize,
126        height: usize,
127    ) {
128        queue.write_texture(
129            wgpu::TexelCopyTextureInfo {
130                texture: &self.texture,
131                mip_level: 0,
132                origin: wgpu::Origin3d::ZERO,
133                aspect: wgpu::TextureAspect::All,
134            },
135            maze_data,
136            wgpu::TexelCopyBufferLayout {
137                offset: 0,
138                bytes_per_row: Some(4 * width as u32),
139                rows_per_image: Some(height as u32),
140            },
141            wgpu::Extent3d {
142                width: width as u32,
143                height: height as u32,
144                depth_or_array_layers: 1,
145            },
146        );
147    }

149    /// Updates the loading bar progress.
150    pub fn update_loading_bar(&self, queue: &wgpu::Queue, progress: f32) {
151        self.loading_bar_renderer.update_progress(queue, progress);

154    /// Updates the exit shader animation.
155    pub fn update_exit_shader(&self, queue: &wgpu::Queue, window: &Window) {
156        let window_size = window.inner_size();
157        let resolution = [window_size.width as f32, window_size.height as f32];
158        let time = self.exit_shader_renderer.start_time.elapsed().as_secs_f32();
159        self.exit_shader_renderer
160            .update_uniforms(queue, resolution, time);
161    }

163    /// Renders the exit cell with the special shader effect.
164    ///
165    /// # Arguments
166    /// - `render_pass`: The render pass to draw into.
167    /// - `exit_cell`: The exit cell coordinates.
168    /// - `cell_size`: Size of each cell in pixels.
169    /// - `screen_size`: Screen dimensions.
170    ///
171    /// Renders all components for the maze generation animation.
172    pub fn render(&self, render_pass: &mut wgpu::RenderPass, window: &Window) {
173        // Render maze background
174        self.maze_renderer.render(render_pass);
175
176        // Render loading bar overlay
177        self.loading_bar_renderer.render(render_pass);
178
179        // Render exit cell effect if maze has an exit
180        if let Ok(maze_guard) = self.maze.lock() {
181            if let Some(exit_cell) = maze_guard.exit_cell {
182                self.exit_shader_renderer.render_to_cell(
183                    render_pass,
184                    window,
185                    (exit_cell.col, exit_cell.row),
186                );
187            }
188        }
189    }

201    /// Get maze dimensions for texture updates.
202    pub fn get_maze_dimensions(&self) -> (u32, u32) {
203        if let Ok(maze_guard) = self.maze.lock() {
204            let (width, height) = maze_guard.get_dimensions();
205            (width as u32, height as u32)
206        } else {
207            (126, 126) // Default fallback
208        }
209    }

212/// Uniform data structure for the loading bar shader.
213///
214/// This structure is uploaded to the GPU and used by the loading bar fragment shader
215/// to determine how much of the progress bar to fill.
216///
217/// ## Memory Layout
218///
219/// The structure uses `#[repr(C)]` to ensure consistent memory layout between
220/// Rust and WGSL. The padding ensures 16-byte alignment as required by WGSL
221/// uniform buffer rules.
222///
223/// ## Shader Usage
224///
225/// In WGSL, this corresponds to:
226/// ```wgsl
227/// struct LoadingBarUniforms {
228///     progress: f32,
229///     // Implicit padding in WGSL
230/// }
231/// @group(0) @binding(0) var<uniform> uniforms: LoadingBarUniforms;
232/// ```
233///
234/// # Fields
235///
236/// - `progress` - Loading progress from 0.0 (0%) to 1.0 (100%)
237/// - `_padding` - Ensures proper WGSL alignment (16 bytes total)
238#[repr(C)]
239#[derive(Debug, Copy, Clone, bytemuck::Pod, bytemuck::Zeroable)]
240pub struct LoadingBarUniforms {
241    pub progress: f32,
242    pub _padding: [f32; 3],
243}

245/// Uniform data structure for the exit shader effects.
246///
247/// This structure provides time and screen resolution information to the
248/// exit shader for creating animated procedural effects.
249///
250/// ## Memory Layout
251///
252/// Uses `#[repr(C)]` for consistent layout. The resolution is stored as a 2-component
253/// vector for easy use in shader calculations.
254///
255/// ## Shader Usage
256///
257/// In WGSL, this corresponds to:
258/// ```wgsl
259/// struct ExitShaderUniforms {
260///     time: f32,
261///     resolution: vec2<f32>,
262///     // Implicit padding in WGSL
263/// }
264/// @group(0) @binding(0) var<uniform> uniforms: ExitShaderUniforms;
265/// ```
266///
267/// # Fields
268///
269/// - `time` - Elapsed time in seconds since shader start (for animation)
270/// - `resolution` - Screen/viewport resolution as [width, height]
271/// - `_padding` - Ensures proper WGSL alignment
272#[repr(C)]
273#[derive(Debug, Copy, Clone, bytemuck::Pod, bytemuck::Zeroable)]
274pub struct ExitShaderUniforms {
275    pub time: f32,
276    pub resolution: [f32; 2],
277    pub _padding: [f32; 3],
278}

280/// Renders a maze texture to the screen with proper scaling and filtering.
281///
282/// This renderer takes a pre-generated maze texture and displays it fullscreen
283/// or in a specified viewport. It handles texture sampling and can apply
284/// various filtering modes.
285///
286/// ## Rendering Pipeline
287///
288/// 1. Uses a fullscreen quad vertex buffer
289/// 2. Samples the maze texture with the provided sampler
290/// 3. Outputs directly to the framebuffer
291///
292/// ## Texture Requirements
293///
294/// - Format: `Rgba8UnormSrgb` (or compatible)
295/// - Usage: Must include `TEXTURE_BINDING`
296/// - The texture should contain the pre-rendered maze data
297///
298/// ## Performance Notes
299///
300/// This renderer is very efficient as it only performs texture sampling
301/// without complex computations. It's suitable for real-time rendering
302/// of static or infrequently updated maze textures.
303///
304/// ## Example Usage
305///
306/// ```rust,no_run
307/// # use crate::renderer::maze_renderer::MazeRenderer;
308/// # let device: egui_wgpu::wgpu::Device = unimplemented!();
309/// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
310/// # let texture_view: egui_wgpu::wgpu::TextureView = unimplemented!();
311/// # let sampler: egui_wgpu::wgpu::Sampler = unimplemented!();
312/// # let mut render_pass: egui_wgpu::wgpu::RenderPass = unimplemented!();
313///
314/// let renderer = MazeRenderer::new(&device, &surface_config, &texture_view, &sampler);
315///
316/// // In render loop:
317/// renderer.render(&mut render_pass);
318/// ```
319pub struct MazeRenderer {
320    /// The render pipeline configured for maze texture rendering
321    pub pipeline: wgpu::RenderPipeline,
322    /// Vertex buffer containing fullscreen quad vertices
323    pub vertex_buffer: wgpu::Buffer,
324    /// Bind group containing the maze texture and sampler
325    pub bind_group: wgpu::BindGroup,
326}

328impl MazeRenderer {
329    /// Create a new maze renderer.
330    ///
331    /// This sets up the complete rendering pipeline for displaying a maze texture,
332    /// including the render pipeline, vertex buffer, and bind group.
333    ///
334    /// ## Shader Requirements
335    ///
336    /// The maze shader should be located at `../maze/2D-maze-shader.wgsl` relative
337    /// to this module and should expect:
338    /// - Vertex input: `@location(0) position: vec2<f32>`
339    /// - Texture binding: `@group(0) @binding(0) var maze_texture: texture_2d<f32>;`
340    /// - Sampler binding: `@group(0) @binding(1) var maze_sampler: sampler;`
341    ///
342    /// # Parameters
343    ///
344    /// - `device` - WGPU device for creating GPU resources
345    /// - `surface_config` - Surface configuration (provides target format)
346    /// - `texture_view` - View of the maze texture to render
347    /// - `sampler` - Sampler for texture filtering (typically nearest neighbor for pixel art)
348    ///
349    /// # Returns
350    ///
351    /// A configured `MazeRenderer` ready for rendering.
352    ///
353    /// # Example
354    ///
355    /// ```rust,no_run
356    /// # use crate::renderer::maze_renderer::{MazeRenderer, MazeRenderConfig};
357    /// # let device: egui_wgpu::wgpu::Device = unimplemented!();
358    /// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
359    ///
360    /// // Create maze texture
361    /// let config = MazeRenderConfig::new(25, 25);
362    /// let (texture, texture_view, sampler) = config.create_maze_texture(&device);
363    ///
364    /// let renderer = MazeRenderer::new(&device, &surface_config, &texture_view, &sampler);
365    /// ```
366    pub fn new(
367        device: &wgpu::Device,
368        surface_config: &wgpu::SurfaceConfiguration,
369        texture_view: &wgpu::TextureView,
370        sampler: &wgpu::Sampler,
371    ) -> Self {
372        let bind_group_layout = BindGroupLayoutBuilder::new(device)
373            .with_label("Maze Texture Bind Group Layout")
374            .with_texture(0, wgpu::ShaderStages::FRAGMENT)
375            .with_sampler(1, wgpu::ShaderStages::FRAGMENT)
376            .build();
377
378        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
379            layout: &bind_group_layout,
380            entries: &[
381                wgpu::BindGroupEntry {
382                    binding: 0,
383                    resource: wgpu::BindingResource::TextureView(texture_view),
384                },
385                wgpu::BindGroupEntry {
386                    binding: 1,
387                    resource: wgpu::BindingResource::Sampler(sampler),
388                },
389            ],
390            label: Some("Maze Texture Bind Group"),
391        });
392
393        let pipeline = PipelineBuilder::new(device, surface_config.format)
394            .with_label("Maze Render Pipeline")
395            .with_shader(include_str!("./shaders/2D-maze-shader.wgsl"))
396            .with_vertex_buffer(create_vertex_2d_layout())
397            .with_bind_group_layout(&bind_group_layout)
398            .build();
399
400        let vertex_buffer = create_fullscreen_vertices(device);
401
402        Self {
403            pipeline,
404            vertex_buffer,
405            bind_group,
406        }
407    }
408
409    /// Render the maze to the current render pass.
410    ///
411    /// This method performs the actual rendering by setting up the pipeline,
412    /// bind groups, and vertex buffer, then issuing a draw call for the
413    /// fullscreen quad.
414    ///
415    /// ## Render State
416    ///
417    /// This method assumes:
418    /// - A render pass is active
419    /// - The render pass targets are compatible with the pipeline
420    /// - No scissor test or viewport restrictions (renders fullscreen)
421    ///
422    /// ## Performance
423    ///
424    /// Very fast - only 6 vertices and simple texture sampling.
425    /// Suitable for real-time rendering at high frame rates.
426    ///
427    /// # Parameters
428    ///
429    /// - `render_pass` - Active render pass to render into
430    ///
431    /// # Example
432    ///
433    /// ```rust,no_run
434    /// # use crate::renderer::maze_renderer::MazeRenderer;
435    /// # let renderer: MazeRenderer = unimplemented!();
436    /// # let mut render_pass: egui_wgpu::wgpu::RenderPass = unimplemented!();
437    ///
438    /// // In render loop:
439    /// renderer.render(&mut render_pass);
440    /// ```
441    pub fn render(&self, render_pass: &mut wgpu::RenderPass) {
442        render_pass.set_pipeline(&self.pipeline);
443        render_pass.set_bind_group(0, &self.bind_group, &[]);
444        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
445        render_pass.draw(0..6, 0..1);
446    }
447}

449/// Renders an animated progress bar for loading operations.
450///
451/// This renderer displays a customizable loading bar that shows progress
452/// from 0% to 100%. It uses alpha blending to overlay on top of other
453/// content and can be styled through the shader.
454///
455/// ## Visual Design
456///
457/// The loading bar appearance is defined in the shader and typically includes:
458/// - Background bar (usually semi-transparent)
459/// - Filled portion indicating progress (usually opaque, colored)
460/// - Smooth progress animation
461/// - Optional decorative elements (borders, gradients, etc.)
462///
463/// ## Blending
464///
465/// Uses alpha blending to overlay on existing content. This allows the
466/// loading bar to appear on top of other UI elements or the maze background.
467///
468/// ## Performance
469///
470/// Very lightweight - only updates uniform buffer when progress changes
471/// and renders a simple fullscreen quad.
472///
473/// ## Example Usage
474///
475/// ```rust,no_run
476/// # use crate::renderer::maze_renderer::LoadingBarRenderer;
477/// # let device: egui_wgpu::wgpu::Device = unimplemented!();
478/// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
479/// # let queue: egui_wgpu::wgpu::Queue = unimplemented!();
480/// # let mut render_pass: egui_wgpu::wgpu::RenderPass = unimplemented!();
481///
482/// let mut renderer = LoadingBarRenderer::new(&device, &surface_config);
483///
484/// // Update progress (0.0 to 1.0)
485/// renderer.update_progress(&queue, 0.5); // 50% complete
486///
487/// // Render the loading bar
488/// renderer.render(&mut render_pass);
489/// ```
490pub struct LoadingBarRenderer {
491    /// The render pipeline configured for loading bar rendering with alpha blending
492    pub pipeline: wgpu::RenderPipeline,
493    /// Vertex buffer containing fullscreen quad vertices
494    pub vertex_buffer: wgpu::Buffer,
495    /// Uniform buffer storing the current progress value
496    pub uniform_buffer: wgpu::Buffer,
497    /// Bind group containing the uniform buffer
498    pub bind_group: wgpu::BindGroup,
499}

501impl LoadingBarRenderer {
502    /// Create a new loading bar renderer.
503    ///
504    /// This sets up the complete rendering pipeline for displaying a progress bar,
505    /// including alpha blending support for overlay rendering.
506    ///
507    /// ## Shader Requirements
508    ///
509    /// The loading bar shader should be located at `../maze/loading-bar-shader.wgsl`
510    /// and should expect:
511    /// - Vertex input: `@location(0) position: vec2<f32>`
512    /// - Uniform binding: `@group(0) @binding(0) var<uniform> uniforms: LoadingBarUniforms;`
513    ///
514    /// ## Initial State
515    ///
516    /// The renderer starts with 0% progress. Use [`update_progress()`](LoadingBarRenderer::update_progress)
517    /// to change the progress value.
518    ///
519    /// # Parameters
520    ///
521    /// - `device` - WGPU device for creating GPU resources
522    /// - `surface_config` - Surface configuration (provides target format)
523    ///
524    /// # Returns
525    ///
526    /// A configured `LoadingBarRenderer` ready for rendering.
527    ///
528    /// # Example
529    ///
530    /// ```rust,no_run
531    /// # use crate::renderer::maze_renderer::LoadingBarRenderer;
532    /// # let device: egui_wgpu::wgpu::Device = unimplemented!();
533    /// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
534    ///
535    /// let renderer = LoadingBarRenderer::new(&device, &surface_config);
536    /// ```
537    pub fn new(device: &wgpu::Device, surface_config: &wgpu::SurfaceConfiguration) -> Self {
538        let uniforms = LoadingBarUniforms {
539            progress: 0.0,
540            _padding: [0.0; 3],
541        };
542
543        let uniform_buffer = create_uniform_buffer(device, &uniforms, "Loading Bar Uniform Buffer");
544
545        let bind_group_layout = BindGroupLayoutBuilder::new(device)
546            .with_label("Loading Bar Bind Group Layout")
547            .with_uniform_buffer(0, wgpu::ShaderStages::FRAGMENT)
548            .build();
549
550        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
551            layout: &bind_group_layout,
552            entries: &[wgpu::BindGroupEntry {
553                binding: 0,
554                resource: uniform_buffer.as_entire_binding(),
555            }],
556            label: Some("Loading Bar Bind Group"),
557        });
558
559        let pipeline = PipelineBuilder::new(device, surface_config.format)
560            .with_label("Loading Bar Pipeline")
561            .with_shader(include_str!("./shaders/loading-bar-shader.wgsl"))
562            .with_vertex_buffer(create_vertex_2d_layout())
563            .with_bind_group_layout(&bind_group_layout)
564            .with_alpha_blending()
565            .build();
566
567        let vertex_buffer = create_fullscreen_vertices(device);
568
569        Self {
570            pipeline,
571            vertex_buffer,
572            uniform_buffer,
573            bind_group,
574        }
575    }

577    /// Update the loading progress.
578    ///
579    /// This method uploads new progress data to the GPU uniform buffer.
580    /// The progress value is automatically clamped to the valid range [0.0, 1.0].
581    ///
582    /// ## Performance Notes
583    ///
584    /// This operation involves a GPU buffer write, so avoid calling it
585    /// unnecessarily frequently (e.g., every frame with the same value).
586    /// It's designed for periodic updates as loading progresses.
587    ///
588    /// # Parameters
589    ///
590    /// - `queue` - WGPU queue for buffer uploads
591    /// - `progress` - Progress value from 0.0 (0%) to 1.0 (100%)
592    ///
593    /// # Example
594    ///
595    /// ```rust,no_run
596    /// # use crate::renderer::maze_renderer::LoadingBarRenderer;
597    /// # let renderer: LoadingBarRenderer = unimplemented!();
598    /// # let queue: egui_wgpu::wgpu::Queue = unimplemented!();
599    ///
600    /// // Update progress to 75%
601    /// renderer.update_progress(&queue, 0.75);
602    ///
603    /// // Values outside [0,1] are automatically clamped
604    /// renderer.update_progress(&queue, 1.5); // Becomes 1.0
605    /// renderer.update_progress(&queue, -0.1); // Becomes 0.0
606    /// ```
607    pub fn update_progress(&self, queue: &wgpu::Queue, progress: f32) {
608        let uniforms = LoadingBarUniforms {
609            progress: progress.clamp(0.0, 1.0),
610            _padding: [0.0; 3],
611        };
612        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::cast_slice(&[uniforms]));
613    }

615    /// Render the loading bar to the current render pass.
616    ///
617    /// This method renders the loading bar as an overlay using alpha blending.
618    /// The bar will appear on top of any previously rendered content.
619    ///
620    /// ## Render State
621    ///
622    /// This method assumes:
623    /// - A render pass is active
624    /// - Alpha blending is desired (the pipeline enables it)
625    /// - The loading bar should cover the full viewport
626    ///
627    /// # Parameters
628    ///
629    /// - `render_pass` - Active render pass to render into
630    ///
631    /// # Example
632    ///
633    /// ```rust,no_run
634    /// # use crate::renderer::maze_renderer::LoadingBarRenderer;
635    /// # let renderer: LoadingBarRenderer = unimplemented!();
636    /// # let mut render_pass: egui_wgpu::wgpu::RenderPass = unimplemented!();
637    ///
638    /// // Render background content first
639    /// // ... other rendering ...
640    ///
641    /// // Render loading bar on top
642    /// renderer.render(&mut render_pass);
643    /// ```
644    pub fn render(&self, render_pass: &mut wgpu::RenderPass) {
645        render_pass.set_pipeline(&self.pipeline);
646        render_pass.set_bind_group(0, &self.bind_group, &[]);
647        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
648        render_pass.draw(0..6, 0..1);
649    }

652/// Renders animated procedural effects for the maze exit cell.
653///
654/// This renderer creates dynamic visual effects specifically for the exit cell
655/// of the maze. It uses a full-screen triangle technique for efficiency and
656/// applies scissor testing to limit rendering to only the exit cell area.
657///
658/// ## Rendering Technique
659///
660/// Uses the "full-screen triangle" technique instead of a quad:
661/// - Single triangle with vertices outside the viewport
662/// - GPU clips to viewport automatically
663/// - More efficient than quad (fewer vertices, no diagonal edge)
664/// - Combined with scissor test for precise cell targeting
665///
666/// ## Coordinate System
667///
668/// The renderer expects maze coordinates where:
669/// - (0,0) is the top-left cell
670/// - Coordinates increase right (x) and down (y)
671/// - Grid size is assumed to be 25x25 cells
672/// - Cell rendering includes shrinking for visual borders
673///
674/// ## Animation
675///
676/// The shader receives time and resolution uniforms for creating
677/// time-based animations. The internal timer starts when the renderer
678/// is created.
679///
680/// ## Example Usage
681///
682/// ```rust,no_run
683/// # use crate::renderer::maze_renderer::ExitShaderRenderer;
684/// # let device: egui_wgpu::wgpu::Device = unimplemented!();
685/// # let surface_config: egui_wgpu::wgpu::SurfaceConfiguration = unimplemented!();
686/// # let queue: egui_wgpu::wgpu::Queue = unimplemented!();
687/// # let window: &winit::window::Window = unimplemented!();
688/// # let mut render_pass: egui_wgpu::wgpu::RenderPass = unimplemented!();
689///
690/// let mut renderer = ExitShaderRenderer::new(&device, &surface_config);
691///
692/// // In render loop:
693/// let elapsed = renderer.start_time.elapsed().as_secs_f32();
694/// renderer.update_uniforms(&queue, [800.0, 600.0], elapsed);
695/// renderer.render_to_cell(&mut render_pass, window, (24, 24)); // Bottom-right exit
696/// ```
697pub struct ExitShaderRenderer {
698    pub pipeline: wgpu::RenderPipeline,
699    pub uniform_buffer: wgpu::Buffer,
700    pub bind_group: wgpu::BindGroup,
701    pub start_time: Instant,
702}
703
704impl ExitShaderRenderer {
705    pub fn new(device: &wgpu::Device, surface_config: &wgpu::SurfaceConfiguration) -> Self {
706        let uniforms = ExitShaderUniforms {
707            time: 0.0,
708            resolution: [800.0, 600.0], // Default resolution, will be updated
709            _padding: [0.0; 3],
710        };
711
712        let uniform_buffer = create_uniform_buffer(device, &uniforms, "Exit Shader Uniform Buffer");
713
714        let bind_group_layout = BindGroupLayoutBuilder::new(device)
715            .with_label("Exit Shader Bind Group Layout")
716            .with_uniform_buffer(0, wgpu::ShaderStages::VERTEX_FRAGMENT)
717            .build();
718
719        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
720            layout: &bind_group_layout,
721            entries: &[wgpu::BindGroupEntry {
722                binding: 0,
723                resource: uniform_buffer.as_entire_binding(),
724            }],
725            label: Some("Exit Shader Bind Group"),
726        });
727
728        // Exit shader uses a full-screen triangle trick, so no vertex buffer needed
729        let pipeline = PipelineBuilder::new(device, surface_config.format)
730            .with_label("Exit Shader Pipeline")
731            .with_shader(include_str!("./shaders/exit_shader.wgsl"))
732            .with_bind_group_layout(&bind_group_layout)
733            .build();
734
735        Self {
736            pipeline,
737            uniform_buffer,
738            bind_group,
739            start_time: Instant::now(),
740        }
741    }
742
743    pub fn update_uniforms(&self, queue: &wgpu::Queue, resolution: [f32; 2], time: f32) {
744        let uniforms = ExitShaderUniforms {
745            time,
746            resolution,
747            _padding: [0.0; 3],
748        };
749        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::cast_slice(&[uniforms]));
750    }
751
752    pub fn render_to_cell(
753        &self,
754        render_pass: &mut wgpu::RenderPass,
755        window: &winit::window::Window,
756        exit_cell: (usize, usize),
757    ) {
758        // Calculate scissor rectangle for the exit cell
759        let grid_size = 25.0;
760        let shrink_factor = 0.845;
761        let border = 3.0;
762
763        let window_size = window.inner_size();
764        let total_width = window_size.width as f32;
765        let total_height = window_size.height as f32;
766
767        let usable_width = total_width - 2.0 * border;
768        let usable_height = total_height - 2.0 * border;
769
770        let cell_width = usable_width / grid_size;
771        let cell_height = usable_height / grid_size;
772
773        let full_x = border + exit_cell.0 as f32 * cell_width;
774        let full_y = border + exit_cell.1 as f32 * cell_height;
775
776        let shrunk_width = cell_width * shrink_factor;
777        let shrunk_height = cell_height * shrink_factor;
778
779        let offset_x = (cell_width - shrunk_width) / 2.0;
780        let offset_y = (cell_height - shrunk_height) / 2.0;
781
782        let scissor_x = (full_x + offset_x).round() as u32;
783        let scissor_y = (full_y + offset_y).round() as u32;
784        let scissor_width = shrunk_width.round() as u32;
785        let scissor_height = shrunk_height.round() as u32;
786
787        // Set scissor rect and render
788        render_pass.set_scissor_rect(scissor_x, scissor_y, scissor_width, scissor_height);
789        render_pass.set_pipeline(&self.pipeline);
790        render_pass.set_bind_group(0, &self.bind_group, &[]);
791        render_pass.draw(0..3, 0..1); // Full-screen triangle
792    }
793}

795/// Configuration for maze rendering setup
796pub struct MazeRenderConfig {
797    pub maze_width: u32,
798    pub maze_height: u32,
799    pub render_width: u32,
800    pub render_height: u32,
801}
802
803impl MazeRenderConfig {
804    pub fn new(maze_width: u32, maze_height: u32) -> Self {
805        // Calculate render dimensions (assuming 5x scale factor + 1 for borders)
806        let render_width = maze_width * 5 + 1;
807        let render_height = maze_height * 5 + 1;
808
809        Self {
810            maze_width,
811            maze_height,
812            render_width,
813            render_height,
814        }
815    }
816
817    pub fn create_maze_texture(
818        &self,
819        device: &wgpu::Device,
820    ) -> (wgpu::Texture, wgpu::TextureView, wgpu::Sampler) {
821        let texture_size = wgpu::Extent3d {
822            width: self.render_width,
823            height: self.render_height,
824            depth_or_array_layers: 1,
825        };
826
827        let texture = device.create_texture(&wgpu::TextureDescriptor {
828            label: Some("Maze Texture"),
829            size: texture_size,
830            mip_level_count: 1,
831            sample_count: 1,
832            dimension: wgpu::TextureDimension::D2,
833            format: wgpu::TextureFormat::Rgba8UnormSrgb,
834            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
835            view_formats: &[],
836        });
837
838        let texture_view = texture.create_view(&wgpu::TextureViewDescriptor::default());
839
840        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
841            address_mode_u: wgpu::AddressMode::ClampToEdge,
842            address_mode_v: wgpu::AddressMode::ClampToEdge,
843            address_mode_w: wgpu::AddressMode::ClampToEdge,
844            mag_filter: wgpu::FilterMode::Nearest,
845            min_filter: wgpu::FilterMode::Nearest,
846            mipmap_filter: wgpu::FilterMode::Nearest,
847            ..Default::default()
848        });
849
850        (texture, texture_view, sampler)
851    }
