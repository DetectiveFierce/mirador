1//! Compass rendering system for directional navigation.
2//!
3//! This module provides a complete compass rendering system that displays
4//! a directional indicator pointing toward the maze exit. The compass uses
5//! a base texture with an animated needle that rotates to show the correct
6//! direction relative to the player's orientation.
7//!
8//! # Features
9//!
10//! - **Directional Navigation**: Points toward the maze exit from player position
11//! - **Player Orientation**: Accounts for player's current facing direction
12//! - **Smooth Animation**: Interpolated needle rotation for fluid movement
13//! - **Multiple Needle Frames**: 12-directional needle sprites for precise indication
14//! - **Screen Positioning**: Configurable position and size via uniforms
15//!
16//! # Usage
17//!
18//! ```rust
19//! // Create compass renderer
20//! let compass = CompassRenderer::new(device, queue, surface_config);
21//!
22//! // Update direction based on player and exit positions
23//! compass.update_compass_with_yaw(player_pos, exit_pos, player_yaw);
24//!
25//! // Render compass overlay
26//! compass.render(&mut render_pass, window);
27//! ```
28//!
29//! # Texture Requirements
30//!
31//! The compass system expects the following texture files:
32//! - `assets/compass/gold-compass.png` - Base compass background
33//! - `assets/compass/needle/needle-0.png` through `needle-11.png` - Needle sprites
34//!
35//! # Coordinate System
36//!
37//! The compass uses normalized screen coordinates (0.0 to 1.0) for positioning
38//! and world coordinates for direction calculations. The needle rotation is
39//! calculated relative to the player's forward direction.

47/// Uniform data for compass positioning and sizing.
48///
49/// This struct contains the data sent to the GPU shader to control
50/// the compass position and size on screen. The data is packed to
51/// match GPU memory alignment requirements.
52///
53/// # Memory Layout
54///
55/// - `screen_position`: Normalized screen coordinates [x, y] (0.0 to 1.0)
56/// - `compass_size`: Size as fraction of screen [width, height] (0.0 to 1.0)
57/// - `_padding`: Ensures proper GPU memory alignment
58///
59/// # Default Values
60///
61/// - Position: Bottom-right corner (0.85, 0.85)
62/// - Size: 12% of screen (0.12, 0.12)
63#[repr(C)]
64#[derive(Debug, Copy, Clone, bytemuck::Pod, bytemuck::Zeroable)]
65struct CompassUniforms {
66    /// Screen position in normalized coordinates (0.0 to 1.0).
67    /// Origin is at bottom-left, with (1.0, 1.0) at top-right.
68    screen_position: [f32; 2],
69
70    /// Compass size as fraction of screen dimensions.
71    /// Values of 0.12 mean 12% of screen width/height.
72    compass_size: [f32; 2],
73
74    /// Padding for GPU memory alignment requirements.
75    _padding: [f32; 4],
76}

78/// Compass renderer for directional navigation overlay.
79///
80/// This struct manages the complete compass rendering system, including
81/// texture loading, shader pipeline setup, and directional calculations.
82/// The compass provides visual feedback to help players navigate toward
83/// the maze exit.
84///
85/// # Rendering Pipeline
86///
87/// The compass uses a two-pass rendering approach:
88/// 1. **Base Pass**: Renders the compass background texture
89/// 2. **Needle Pass**: Renders the directional needle on top
90///
91/// # Texture Management
92///
93/// - **Base Texture**: Static compass background (gold-compass.png)
94/// - **Needle Textures**: 12 directional needle sprites (needle-0.png to needle-11.png)
95/// - **Bind Groups**: Separate bind groups for base and each needle direction
96///
97/// # Smoothing System
98///
99/// The compass uses exponential smoothing to prevent jarring needle movements:
100/// - `smoothing_factor`: Controls responsiveness (0.01 = very smooth, 1.0 = instant)
101/// - `smoothed_compass_angle`: Current interpolated angle for needle selection
102///
103/// # Performance Characteristics
104///
105/// - **GPU Memory**: Pre-allocated textures and bind groups
106/// - **CPU Usage**: Minimal per-frame calculations
107/// - **Rendering**: Two draw calls per frame (base + needle)
108///
109/// # Thread Safety
110///
111/// This struct is not thread-safe and should only be accessed from the
112/// main rendering thread.
113///
114/// # Example
115///
116/// ```rust
117/// # use crate::renderer::game_renderer::compass::CompassRenderer;
118/// # let device: wgpu::Device = unimplemented!();
119/// # let queue: wgpu::Queue = unimplemented!();
120/// # let surface_config: wgpu::SurfaceConfiguration = unimplemented!();
121///
122/// let compass = CompassRenderer::new(&device, &queue, &surface_config);
123///
124/// // Update direction
125/// compass.update_compass_with_yaw(player_pos, exit_pos, player_yaw);
126///
127/// // Render
128/// compass.render(&mut render_pass, window);
129/// ```
130pub struct CompassRenderer {
131    /// WGPU render pipeline for compass rendering.
132    ///
133    /// Handles vertex processing, fragment shading, and blending for
134    /// the compass overlay. Uses alpha blending for semitransparent rendering.
135    pipeline: wgpu::RenderPipeline,
136
137    /// Vertex buffer containing compass quad geometry.
138    ///
139    /// Contains a simple quad (-1 to 1) that gets positioned and scaled
140    /// via uniforms in the vertex shader.
141    vertex_buffer: wgpu::Buffer,
142
143    /// Uniform buffer for compass position and size data.
144    ///
145    /// Contains `CompassUniforms` struct that controls where and how
146    /// large the compass appears on screen.
147    uniform_buffer: wgpu::Buffer,
148
149    /// Bind group for the compass base texture.
150    ///
151    /// Contains the background texture, sampler, and uniform buffer
152    /// for rendering the compass base.
153    base_bind_group: wgpu::BindGroup,
154
155    /// Bind groups for each needle direction texture.
156    ///
157    /// Array of 12 bind groups, one for each needle sprite (0-11).
158    /// Each bind group contains the needle texture, sampler, and uniform buffer.
159    needle_bind_groups: Vec<wgpu::BindGroup>,
160
161    /// Current needle sprite index (0-11).
162    ///
163    /// Determines which needle texture is currently rendered.
164    /// Updated based on calculated direction to exit.
165    current_needle_index: usize,
166
167    /// Current smoothed compass angle in radians.
168    ///
169    /// This is the interpolated angle used for needle selection.
170    /// Ranges from 0 to 2Ï€ and is updated with smoothing applied.
171    smoothed_compass_angle: f32,
172
173    /// Smoothing factor for compass movement (0.01 to 1.0).
174    ///
175    /// Controls how quickly the needle responds to direction changes.
176    /// Lower values = smoother but slower response.
177    /// Higher values = faster but potentially jittery response.
178    smoothing_factor: f32,
179}

182    /// Creates a new `CompassRenderer` instance and initializes all GPU resources required for compass rendering.
183    ///
184    /// This function loads the compass base texture and all 12 needle textures, creates the uniform buffer,
185    /// sets up the bind group layout for textures, samplers, and uniforms, and builds the render pipeline.
186    pub fn new(
187        device: &wgpu::Device,
188        queue: &wgpu::Queue,
189        surface_config: &wgpu::SurfaceConfiguration,
190    ) -> Self {
191        // Load compass base texture
192        let base_texture = Self::load_base_texture(device, queue);
193
194        // Load all needle textures
195        let needle_textures = Self::load_needle_textures(device, queue);
196
197        let uniforms = CompassUniforms {
198            screen_position: [0.85, 0.85], // Bottom-right corner (normalized coordinates)
199            compass_size: [0.12, 0.12],    // 12% of screen size
200            _padding: [0.0; 4],
201        };
202
203        let uniform_buffer = create_uniform_buffer(device, &uniforms, "Compass Uniform Buffer");
204
205        // Create bind group layout for texture + sampler + uniforms
206        let bind_group_layout = BindGroupLayoutBuilder::new(device)
207            .with_label("Compass Bind Group Layout")
208            .with_uniform_buffer(0, wgpu::ShaderStages::VERTEX | wgpu::ShaderStages::FRAGMENT)
209            .with_texture(1, wgpu::ShaderStages::FRAGMENT)
210            .with_sampler(2, wgpu::ShaderStages::FRAGMENT)
211            .build();
212
213        // Create sampler for all textures
214        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
215            address_mode_u: wgpu::AddressMode::ClampToEdge,
216            address_mode_v: wgpu::AddressMode::ClampToEdge,
217            address_mode_w: wgpu::AddressMode::ClampToEdge,
218            mag_filter: wgpu::FilterMode::Linear,
219            min_filter: wgpu::FilterMode::Linear,
220            mipmap_filter: wgpu::FilterMode::Nearest,
221            ..Default::default()
222        });
223
224        // Create bind group for base texture
225        let base_texture_view = base_texture.create_view(&wgpu::TextureViewDescriptor::default());
226        let base_bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
227            layout: &bind_group_layout,
228            entries: &[
229                wgpu::BindGroupEntry {
230                    binding: 0,
231                    resource: uniform_buffer.as_entire_binding(),
232                },
233                wgpu::BindGroupEntry {
234                    binding: 1,
235                    resource: wgpu::BindingResource::TextureView(&base_texture_view),
236                },
237                wgpu::BindGroupEntry {
238                    binding: 2,
239                    resource: wgpu::BindingResource::Sampler(&sampler),
240                },
241            ],
242            label: Some("Compass Base Bind Group"),
243        });
244
245        // Create bind groups for each needle texture
246        let needle_bind_groups: Vec<wgpu::BindGroup> = needle_textures
247            .iter()
248            .enumerate()
249            .map(|(i, texture)| {
250                let texture_view = texture.create_view(&wgpu::TextureViewDescriptor::default());
251
252                device.create_bind_group(&wgpu::BindGroupDescriptor {
253                    layout: &bind_group_layout,
254                    entries: &[
255                        wgpu::BindGroupEntry {
256                            binding: 0,
257                            resource: uniform_buffer.as_entire_binding(),
258                        },
259                        wgpu::BindGroupEntry {
260                            binding: 1,
261                            resource: wgpu::BindingResource::TextureView(&texture_view),
262                        },
263                        wgpu::BindGroupEntry {
264                            binding: 2,
265                            resource: wgpu::BindingResource::Sampler(&sampler),
266                        },
267                    ],
268                    label: Some(&format!("Compass Needle Bind Group {}", i)),
269                })
270            })
271            .collect();
272
273        // Create vertex buffer layout for position + tex_coords
274        let vertex_buffer_layout = wgpu::VertexBufferLayout {
275            array_stride: 4 * 4, // 4 floats * 4 bytes each
276            step_mode: wgpu::VertexStepMode::Vertex,
277            attributes: &[
278                wgpu::VertexAttribute {
279                    offset: 0,
280                    shader_location: 0,
281                    format: wgpu::VertexFormat::Float32x2, // position
282                },
283                wgpu::VertexAttribute {
284                    offset: 2 * 4, // 2 floats * 4 bytes
285                    shader_location: 1,
286                    format: wgpu::VertexFormat::Float32x2, // tex_coords
287                },
288            ],
289        };
290
291        let pipeline = PipelineBuilder::new(device, surface_config.format)
292            .with_label("Compass Pipeline")
293            .with_shader(include_str!("../shaders/compass.wgsl"))
294            .with_vertex_buffer(vertex_buffer_layout)
295            .with_bind_group_layout(&bind_group_layout)
296            .with_alpha_blending()
297            .build();
298
299        let vertex_buffer = Self::create_compass_vertices(device);
300
301        Self {
302            pipeline,
303            vertex_buffer,
304            uniform_buffer,
305            base_bind_group,
306            needle_bind_groups,
307            current_needle_index: 0,
308
309            smoothed_compass_angle: 0.0,
310            smoothing_factor: 0.8, // Higher = more responsive, lower = smoother
311        }
312    }

314    fn load_base_texture(device: &wgpu::Device, queue: &wgpu::Queue) -> wgpu::Texture {
315        let path = "assets/compass/gold-compass.png";
316
317        // Load image using image crate
318        let img = match image::open(path) {
319            Ok(img) => img.to_rgba8(),
320            Err(e) => {
321                eprintln!("Failed to load compass base texture {}: {}", path, e);
322                // Create a fallback texture (solid color or default compass)
323                image::RgbaImage::new(64, 64)
324            }
325        };
326
327        let dimensions = img.dimensions();
328        let texture_size = wgpu::Extent3d {
329            width: dimensions.0,
330            height: dimensions.1,
331            depth_or_array_layers: 1,
332        };
333
334        let texture = device.create_texture(&wgpu::TextureDescriptor {
335            label: Some("Compass Base Texture"),
336            size: texture_size,
337            mip_level_count: 1,
338            sample_count: 1,
339            dimension: wgpu::TextureDimension::D2,
340            format: wgpu::TextureFormat::Rgba8UnormSrgb,
341            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
342            view_formats: &[],
343        });
344
345        queue.write_texture(
346            wgpu::TexelCopyTextureInfo {
347                aspect: wgpu::TextureAspect::All,
348                texture: &texture,
349                mip_level: 0,
350                origin: wgpu::Origin3d::ZERO,
351            },
352            &img,
353            wgpu::TexelCopyBufferLayout {
354                offset: 0,
355                bytes_per_row: Some(4 * dimensions.0),
356                rows_per_image: Some(dimensions.1),
357            },
358            texture_size,
359        );
360
361        texture
362    }

493    /// Renders the compass overlay to the current render pass.
494    ///
495    /// This method performs a two-pass rendering approach:
496    /// 1. Renders the compass base texture (background)
497    /// 2. Renders the current needle texture on top
498    ///
499    /// The compass is rendered as an overlay using alpha blending,
500    /// allowing it to appear on top of the game scene while maintaining
501    /// transparency for visual integration.
502    ///
503    /// # Render State Requirements
504    ///
505    /// This method assumes:
506    /// - A render pass is active and configured for alpha blending
507    /// - The game scene has already been rendered
508    /// - The viewport is set to the full screen dimensions
509    ///
510    /// # Performance Characteristics
511    ///
512    /// - **Draw Calls**: 2 draw calls per frame (base + needle)
513    /// - **GPU Memory**: Minimal - only vertex buffer and bind group switches
514    /// - **CPU Usage**: Negligible - only pipeline and bind group setup
515    ///
516    /// # Parameters
517    ///
518    /// - `render_pass` - Active render pass to render into
519    /// - `window` - Window reference (currently unused, reserved for future use)
520    ///
521    /// # Example
522    ///
523    /// ```rust
524    /// # use crate::renderer::game_renderer::compass::CompassRenderer;
525    /// # let compass: CompassRenderer = unimplemented!();
526    /// # let mut render_pass: wgpu::RenderPass = unimplemented!();
527    /// # let window: &winit::window::Window = unimplemented!();
528    ///
529    /// // Render game scene first
530    /// // ... render background, maze, player, etc ...
531    ///
532    /// // Render compass overlay on top
533    /// compass.render(&mut render_pass, window);
534    /// ```
535    pub fn render(&self, render_pass: &mut wgpu::RenderPass, _window: &winit::window::Window) {
536        render_pass.set_pipeline(&self.pipeline);
537        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
538
539        // First render the compass base
540        render_pass.set_bind_group(0, &self.base_bind_group, &[]);
541        render_pass.draw(0..6, 0..1);
542
543        // Then render the needle on top
544        render_pass.set_bind_group(0, &self.needle_bind_groups[self.current_needle_index], &[]);
545        render_pass.draw(0..6, 0..1);
546    }

548    /// Calculate which needle image to show based on player and exit positions
549    pub fn update_compass_direction(&mut self, player_pos: (f32, f32), exit_pos: (f32, f32)) {
550        // Vector from player to exit
551        let direction_vector = (player_pos.0 - exit_pos.0, player_pos.1 - exit_pos.1);
552
553        // Skip if too close (avoid jitter when on top of exit)
554        let distance_sq =
555            direction_vector.0 * direction_vector.0 + direction_vector.1 * direction_vector.1;
556        if distance_sq < 0.0001 {
557            return;
558        }
559
560        // Calculate angle to exit in world space
561        let mut target_angle = direction_vector.1.atan2(direction_vector.0); // [-Ï€, Ï€]
562
563        // Normalize to [0, 2Ï€]
564        if target_angle < 0.0 {
565            target_angle += 2.0 * std::f32::consts::PI;
566        }
567
568        // Smooth angle update (exponential smoothing)
569        let alpha = self.smoothing_factor; // Lower = slower/smoother
570        let mut delta = target_angle - self.smoothed_compass_angle;
571
572        // Wrap to [-Ï€, Ï€] for shortest rotation
573        if delta > std::f32::consts::PI {
574            delta -= 2.0 * std::f32::consts::PI;
575        } else if delta < -std::f32::consts::PI {
576            delta += 2.0 * std::f32::consts::PI;
577        }
578
579        self.smoothed_compass_angle += alpha * delta;
580
581        // Re-wrap smoothed angle to [0, 2Ï€]
582        while self.smoothed_compass_angle < 0.0 {
583            self.smoothed_compass_angle += 2.0 * std::f32::consts::PI;
584        }
585        while self.smoothed_compass_angle >= 2.0 * std::f32::consts::PI {
586            self.smoothed_compass_angle -= 2.0 * std::f32::consts::PI;
587        }
588
589        // Map to needle frame (0â€“11, since we have 12 needles indexed 1-12)
590        let new_index = ((self.smoothed_compass_angle / (2.0 * std::f32::consts::PI)) * 12.0)
591            .floor() as usize
592            % 12;
593
594        self.current_needle_index = new_index;
595    }

597    /// Updates the compass to point toward the exit from the player's current position.
598    ///
599    /// This function calculates the direction from the player to the exit cell and
600    /// adjusts for the player's current orientation (yaw) so that the compass always
601    /// indicates the direction the player should move to reach the exit.
602    ///
603    /// # Direction Calculation
604    ///
605    /// The method uses a sophisticated approach to calculate the compass direction:
606    /// 1. **Vector Calculation**: Computes direction vector from player to exit
607    /// 2. **Player Orientation**: Accounts for player's current facing direction
608    /// 3. **Coordinate Transformation**: Converts world direction to player-relative direction
609    /// 4. **Smoothing**: Applies exponential smoothing to prevent jarring movements
610    /// 5. **Needle Selection**: Maps smoothed angle to appropriate needle sprite (0-11)
611    ///
612    /// # Coordinate Systems
613    ///
614    /// - **World Coordinates**: Player and exit positions in maze space
615    /// - **Player Coordinates**: Direction relative to player's forward vector
616    /// - **Compass Coordinates**: Angle mapped to 12-directional needle sprites
617    ///
618    /// # Smoothing Behavior
619    ///
620    /// The compass uses exponential smoothing to create fluid needle movement:
621    /// - **Shortest Path**: Always takes the shortest angular distance
622    /// - **Configurable Response**: Smoothing factor controls responsiveness
623    /// - **Wrapping**: Properly handles angle wrapping around 0Â°/360Â°
624    ///
625    /// # Performance Notes
626    ///
627    /// This method performs minimal calculations and is safe to call every frame.
628    /// The smoothing calculations are CPU-efficient and don't involve GPU operations.
629    ///
630    /// # Parameters
631    ///
632    /// * `player_pos` - The player's position as (x, z) coordinates in world space
633    /// * `exit_pos` - The exit's position as (x, z) coordinates in world space
634    /// * `player_yaw_degrees` - The player's current yaw angle in degrees (0-360)
635    ///
636    /// # Example
637    ///
638    /// ```rust
639    /// # use crate::renderer::game_renderer::compass::CompassRenderer;
640    /// # let mut compass: CompassRenderer = unimplemented!();
641    ///
642    /// // Update compass direction
643    /// compass.update_compass_with_yaw(
644    ///     (player.x, player.z),      // Player position
645    ///     (exit.x, exit.z),          // Exit position
646    ///     player.yaw_degrees         // Player facing direction
647    /// );
648    /// ```
649    pub fn update_compass_with_yaw(
650        &mut self,
651        player_pos: (f32, f32), // (x, z) coordinates
652        exit_pos: (f32, f32),   // (x, z) coordinates
653        player_yaw_degrees: f32,
654    ) {
655        // Calculate vector from player to exit
656        let dx = exit_pos.0 - player_pos.0; // Change in X
657        let dz = exit_pos.1 - player_pos.1; // Change in Z
658
659        let distance_sq = dx * dx + dz * dz;
660
661        // Skip if too close to exit
662        if distance_sq < 0.0001 {
663            return;
664        }
665
666        // Calculate direction to exit using the same trig approach as player movement
667        // First, get forward vector based on player's yaw (same as in move_forward)
668        let forward_x = player_yaw_degrees.to_radians().sin();
669        let forward_z = player_yaw_degrees.to_radians().cos();
670
671        // Get right vector (same as in move_right)
672        let right_x = player_yaw_degrees.to_radians().cos();
673        let right_z = player_yaw_degrees.to_radians().sin();
674
675        // Normalize the direction vector to the exit
676        let length = distance_sq.sqrt();
677        let dir_x = dx / length;
678        let dir_z = dz / length;
679
680        // Calculate dot products to determine the angle
681        let forward_dot = -forward_x * dir_x - forward_z * dir_z; // Dot product with forward vector
682        let right_dot = right_x * dir_x - right_z * dir_z; // Dot product with right vector
683
684        // Calculate angle using atan2
685        let mut target_compass_angle = right_dot.atan2(forward_dot);
686
687        // Normalize to [-Ï€, Ï€]
688        target_compass_angle = self.normalize_angle(target_compass_angle);
689
690        // Initialize smoothed angle on first update
691        if self.smoothed_compass_angle.is_nan() {
692            self.smoothed_compass_angle = target_compass_angle;
693        }
694
695        // Calculate the shortest angular distance for smooth interpolation
696        let angle_diff =
697            self.shortest_angle_diff(target_compass_angle, self.smoothed_compass_angle);
698
699        // Apply smoothing
700        self.smoothed_compass_angle += angle_diff * self.smoothing_factor;
701
702        // Normalize the smoothed angle
703        self.smoothed_compass_angle = self.normalize_angle(self.smoothed_compass_angle);
704
705        // Convert to needle index (0-11 for 12 needle sprites)
706        // Convert from [-Ï€, Ï€] to [0, 2Ï€] for easier indexing
707        let angle_for_index = if self.smoothed_compass_angle < 0.0 {
708            self.smoothed_compass_angle + 2.0 * std::f32::consts::PI
709        } else {
710            self.smoothed_compass_angle
711        };
712
713        // Convert to 12-segment index (each segment is 30Â° = Ï€/6 radians)
714        // Add half a segment (Ï€/12) for proper rounding to nearest segment
715        let needle_index = ((angle_for_index + std::f32::consts::PI / 12.0)
716            / (std::f32::consts::PI / 6.0))
717            .floor() as usize
718            % 12;
719
720        self.current_needle_index = needle_index;
721    }
722
723    /// Normalize angle to [-Ï€, Ï€]
724    fn normalize_angle(&self, mut angle: f32) -> f32 {
725        while angle > std::f32::consts::PI {
726            angle -= 2.0 * std::f32::consts::PI;
727        }
728        while angle < -std::f32::consts::PI {
729            angle += 2.0 * std::f32::consts::PI;
730        }
731        angle
732    }
733
734    /// Calculate shortest angular difference between two angles
735    fn shortest_angle_diff(&self, target: f32, current: f32) -> f32 {
736        let mut diff = target - current;
737
738        // Wrap to shortest path
739        if diff > std::f32::consts::PI {
740            diff -= 2.0 * std::f32::consts::PI;
741        } else if diff < -std::f32::consts::PI {
742            diff += 2.0 * std::f32::consts::PI;
743        }
744
745        diff
746    }

748    /// Alternative update with configurable smoothing
749    pub fn update_compass_with_smoothing(
750        &mut self,
751        player_pos: (f32, f32),
752        exit_pos: (f32, f32),
753        player_yaw_degrees: f32,
754        smoothing: f32, // 0.0 = very smooth, 1.0 = instant response
755    ) {
756        let old_smoothing = self.smoothing_factor;
757        self.smoothing_factor = smoothing.clamp(0.01, 1.0);
758
759        self.update_compass_with_yaw(player_pos, exit_pos, player_yaw_degrees);
760
761        self.smoothing_factor = old_smoothing;
762    }
