1//! Stamina Bar Renderer Module
2//!
3//! This module provides a GPU-accelerated stamina bar renderer using WebGPU.
4//! The stamina bar is rendered as a full-screen quad using a vertex shader that
5//! generates geometry procedurally, eliminating the need for vertex buffers.
6
7use crate::renderer::pipeline_builder::{
8    BindGroupLayoutBuilder, PipelineBuilder, create_uniform_buffer,
9};
10use std::time::Instant;
11use wgpu;
12
13/// Uniform data structure passed to the stamina bar shader.
14///
15/// This struct is laid out in memory according to WebGPU's uniform buffer
16/// alignment requirements. The `_padding` field ensures proper 16-byte alignment.
17///
18/// # Memory Layout
19/// - `progress`: Current stamina level (0.0 = empty, 1.0 = full)
20/// - `time`: Elapsed time in seconds for animations
21/// - `resolution`: Screen resolution [width, height] for aspect ratio correction
22/// - `_padding`: Padding bytes to maintain 16-byte alignment
23#[repr(C)]
24#[derive(Debug, Copy, Clone, bytemuck::Pod, bytemuck::Zeroable)]
25pub struct StaminaBarUniforms {
26    /// Current stamina progress from 0.0 (empty) to 1.0 (full)
27    pub progress: f32,
28    /// Elapsed time in seconds since renderer creation, used for animations
29    pub time: f32,
30    /// Screen resolution as [width, height] for proper aspect ratio
31    pub resolution: [f32; 2],
32    /// Padding to ensure proper GPU memory alignment (unused)
33    pub _padding: [f32; 2],
34}
35
36/// GPU-accelerated stamina bar renderer.
37///
38/// This renderer displays a stamina bar using a custom shader pipeline.
39/// It uses a full-screen triangle rendering technique where the vertex shader
40/// generates geometry procedurally, requiring no vertex buffers.
41///
42/// # Features
43/// - Hardware-accelerated rendering
44/// - Alpha blending support for transparency effects
45/// - Real-time uniform updates for smooth animations
46/// - Automatic aspect ratio correction
47///
48/// # Shader Requirements
49/// The renderer expects a shader file at "../shaders/loading-bar.wgsl" that
50/// implements the stamina bar visual effects.
51pub struct StaminaBarRenderer {
52    /// The WebGPU render pipeline for stamina bar rendering
53    pub pipeline: wgpu::RenderPipeline,
54    /// GPU buffer containing uniform data (progress, time, resolution)
55    pub uniform_buffer: wgpu::Buffer,
56    /// Bind group that associates the uniform buffer with shader bindings
57    pub bind_group: wgpu::BindGroup,
58    /// Creation timestamp for calculating elapsed time
59    pub start_time: Instant,
60}
61
62impl StaminaBarRenderer {
63    /// Creates a new stamina bar renderer.
64    ///
65    /// Initializes the complete rendering pipeline including shaders, uniforms,
66    /// and GPU resources. The renderer starts with a full stamina bar (progress = 1.0).
67    ///
68    /// # Arguments
69    /// * `device` - WebGPU device for creating GPU resources
70    /// * `surface_config` - Surface configuration containing the target pixel format
71    ///
72    /// # Returns
73    /// A fully initialized `StaminaBarRenderer` ready for rendering
74    ///
75    /// # Example
76    /// ```rust
77    /// let renderer = StaminaBarRenderer::new(&device, &surface_config);
78    /// ```
79    pub fn new(device: &wgpu::Device, surface_config: &wgpu::SurfaceConfiguration) -> Self {
80        // Initialize uniform data with default values
81        let uniforms = StaminaBarUniforms {
82            progress: 1.0,              // Start with full stamina
83            time: 0.0,                  // Start time at zero
84            resolution: [800.0, 600.0], // Default resolution
85            _padding: [0.0; 2],         // Padding for alignment
86        };
87
88        // Create GPU uniform buffer
89        let uniform_buffer = create_uniform_buffer(device, &uniforms, "Stamina Bar Uniform Buffer");
90
91        // Build bind group layout for shader uniform access
92        let bind_group_layout = BindGroupLayoutBuilder::new(device)
93            .with_label("Stamina Bar Bind Group Layout")
94            .with_uniform_buffer(0, wgpu::ShaderStages::VERTEX_FRAGMENT)
95            .build();
96
97        // Create bind group linking uniform buffer to shader binding point 0
98        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
99            layout: &bind_group_layout,
100            entries: &[wgpu::BindGroupEntry {
101                binding: 0,
102                resource: uniform_buffer.as_entire_binding(),
103            }],
104            label: Some("Stamina Bar Bind Group"),
105        });
106
107        // Build render pipeline with shader and alpha blending
108        let pipeline = PipelineBuilder::new(device, surface_config.format)
109            .with_label("Stamina Bar Pipeline")
110            .with_shader(include_str!("../shaders/loading-bar.wgsl"))
111            .with_bind_group_layout(&bind_group_layout)
112            .with_alpha_blending() // Enable transparency support
113            .build();
114
115        Self {
116            pipeline,
117            uniform_buffer,
118            bind_group,
119            start_time: Instant::now(),
120        }
121    }
122
123    /// Updates the stamina bar's uniform data on the GPU.
124    ///
125    /// This method should be called each frame to update the stamina bar's
126    /// visual state. The progress value is automatically clamped to ensure
127    /// it stays within valid bounds.
128    ///
129    /// # Arguments
130    /// * `queue` - WebGPU command queue for GPU operations
131    /// * `progress` - Stamina level from 0.0 (empty) to 1.0 (full)
132    /// * `resolution` - Current screen resolution as [width, height]
133    /// * `time` - Current time in seconds for shader animations
134    ///
135    /// # Note
136    /// The progress value is clamped to [0.0, 1.0] range and is NOT inverted,
137    /// meaning the bar shrinks as stamina decreases (0.0 = empty bar).
138    ///
139    /// # Example
140    /// ```rust
141    /// renderer.update_uniforms(&queue, 0.75, [1920.0, 1080.0], elapsed_time);
142    /// ```
143    pub fn update_uniforms(
144        &self,
145        queue: &wgpu::Queue,
146        progress: f32,
147        resolution: [f32; 2],
148        time: f32,
149    ) {
150        let uniforms = StaminaBarUniforms {
151            progress: progress.clamp(0.0, 1.0), // Ensure valid range
152            time,
153            resolution,
154            _padding: [0.0; 2],
155        };
156
157        // Upload uniform data to GPU buffer
158        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::cast_slice(&[uniforms]));
159    }
160
161    /// Renders the stamina bar to the current render pass.
162    ///
163    /// This method draws the stamina bar using a procedural full-screen triangle
164    /// technique. The vertex shader generates 3 vertices that cover the entire
165    /// screen, and the fragment shader handles the stamina bar visualization.
166    ///
167    /// # Arguments
168    /// * `render_pass` - Active WebGPU render pass to draw into
169    ///
170    /// # Rendering Details
171    /// - Uses 3 vertices (full-screen triangle) with no vertex buffer
172    /// - Single instance rendering (0..1)
173    /// - Relies on vertex shader to generate screen-covering geometry
174    ///
175    /// # Example
176    /// ```rust
177    /// let mut render_pass = encoder.begin_render_pass(&render_pass_desc);
178    /// renderer.render(&mut render_pass);
179    /// ```
180    pub fn render(&self, render_pass: &mut wgpu::RenderPass) {
181        // Set the stamina bar rendering pipeline
182        render_pass.set_pipeline(&self.pipeline);
183
184        // Bind uniform data at binding point 0
185        render_pass.set_bind_group(0, &self.bind_group, &[]);
186
187        // Draw full-screen triangle (3 vertices, 1 instance)
188        // The vertex shader generates screen-covering geometry procedurally
189        render_pass.draw(0..3, 0..1);
190    }
191}
