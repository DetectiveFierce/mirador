1//! Debug rendering utilities for visualizing collision systems.
2//!
3//! This module provides tools for rendering debug elements like bounding boxes,
4//! which are useful for understanding and debugging the collision system.
5
6use crate::game::collision::CollisionSystem;
7use crate::game::collision::{AABB, BVH, BVHNode};
8use crate::renderer::primitives::Vertex;
9use wgpu;
10use wgpu::util::DeviceExt;
11/// Material ID for debug bounding boxes
12pub const BOUNDING_BOX_MATERIAL: u32 = 2;
13
14pub struct DebugRenderer {
15    /// Whether to render bounding boxes for debugging.
16    pub debug_render_bounding_boxes: bool,
17    /// Vertex buffer for the debug renderer
18    pub debug_vertex_buffer: Option<wgpu::Buffer>,
19    pub debug_vertex_count: usize,
20}
21
22impl DebugRenderer {
23    pub fn update_debug_vertices(
24        &mut self,
25        device: &wgpu::Device,
26        collision_system: &CollisionSystem,
27    ) {
28        // Skip if debug rendering is disabled
29        if !self.debug_render_bounding_boxes {
30            self.debug_vertex_count = 0;
31            return;
32        }
33
34        // Collect only wall face AABBs, not the entire BVH hierarchy
35        let debug_vertices = collect_wall_face_debug_vertices(&collision_system.bvh);
36
37        // Create or update the debug vertex buffer
38        self.debug_vertex_count = debug_vertices.len();
39        if self.debug_vertex_count > 0 {
40            let debug_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
41                label: Some("Debug Vertex Buffer"),
42                contents: bytemuck::cast_slice(&debug_vertices),
43                usage: wgpu::BufferUsages::VERTEX,
44            });
45            self.debug_vertex_buffer = Some(debug_buffer);
46        }
47    }
48}

273/// Collects all wall face AABB debug vertices from the BVH tree.
274///
275/// This traverses the BVH to find all leaf nodes and generates visualization
276/// vertices only for the actual wall faces, not for the BVH structure itself.
277///
278/// # Arguments
279///
280/// * `bvh` - Reference to the collision system BVH
281///
282/// # Returns
283///
284/// A vector of vertices representing the wall face bounding boxes
285pub fn collect_wall_face_debug_vertices(bvh: &BVH) -> Vec<Vertex> {
286    let mut vertices = Vec::new();
287
288    if let Some(root) = &bvh.root {
289        collect_wall_faces_recursive(root, &mut vertices);
290    }
291
292    vertices
293}
294
295/// Helper function to recursively collect wall face AABBs from BVH nodes
296fn collect_wall_faces_recursive(node: &BVHNode, vertices: &mut Vec<Vertex>) {
297    match node {
298        BVHNode::Internal { left, right, .. } => {
299            // Just traverse the tree, don't visualize internal nodes
300            collect_wall_faces_recursive(left, vertices);
301            collect_wall_faces_recursive(right, vertices);
302        }
303        BVHNode::Leaf { faces, .. } => {
304            // Visualize only the actual wall faces
305            for face in faces {
306                vertices.extend(create_aabb_box_vertices(&face.aabb));
307            }
308        }
309    }
