1//! Enemy rendering module for the Mirador game.
2//!
3//! This module handles the rendering of enemy entities in the game world.
4//! It provides billboard-based rendering with smooth rotation towards the player,
5//! texture support, and depth-aware rendering.
6
7use crate::game::GameState;
8use crate::game::enemy::Enemy;
9use crate::renderer::pipeline_builder::{
10    BindGroupLayoutBuilder, PipelineBuilder, create_uniform_buffer,
11};
12use wgpu::{self, util::DeviceExt};

14/// Uniform data structure for enemy rendering shader.
15///
16/// Contains view-projection matrix, enemy position, size, player position,
17/// and padding for proper memory alignment.
18#[repr(C)]
19#[derive(Copy, Clone, Debug, bytemuck::Pod, bytemuck::Zeroable)]
20struct EnemyUniforms {
21    /// View-projection matrix for transforming vertices to screen space
22    view_proj_matrix: [[f32; 4]; 4],
23    /// Current position of the enemy in world space (x, y, z)
24    enemy_position: [f32; 3],
25    /// Size/scale of the enemy sprite
26    enemy_size: f32,
27    /// Current position of the player in world space (x, y, z)
28    player_position: [f32; 3],
29    /// Padding for proper memory alignment
30    _padding: f32,
31}
32
33/// Renders enemy entities as billboard sprites that face the player.
34///
35/// The enemy renderer creates textured billboards that automatically rotate
36/// to face the player with smooth interpolation. It supports depth testing
37/// and alpha blending for proper integration with the game world.
38pub struct EnemyRenderer {
39    /// The render pipeline for enemy rendering
40    pipeline: wgpu::RenderPipeline,
41    /// Vertex buffer containing billboard quad vertices
42    vertex_buffer: wgpu::Buffer,
43    /// Uniform buffer containing shader uniforms
44    uniform_buffer: wgpu::Buffer,
45    /// Bind group containing uniforms, texture, and sampler
46    bind_group: wgpu::BindGroup,
47
48    /// Current smoothed rotation angle in radians
49    smoothed_rotation: f32,
50    /// Smoothing factor for rotation interpolation (0.0 = very smooth, 1.0 = instant)
51    smoothing_factor: f32,
52}
53
54impl EnemyRenderer {
55    /// Creates a new enemy renderer for the specified enemy.
56    ///
57    /// # Arguments
58    ///
59    /// * `enemy` - The enemy entity to render
60    /// * `device` - WGPU device for creating GPU resources
61    /// * `queue` - WGPU queue for uploading data to GPU
62    /// * `surface_config` - Surface configuration for pipeline creation
63    ///
64    /// # Returns
65    ///
66    /// A new `EnemyRenderer` instance configured for the given enemy.
67    pub fn new(
68        enemy: Enemy,
69        device: &wgpu::Device,
70        queue: &wgpu::Queue,
71        surface_config: &wgpu::SurfaceConfiguration,
72    ) -> Self {
73        // Load jeffree texture
74        let jeffree_texture = Self::load_slime_texture(device, queue);
75
76        let uniforms = EnemyUniforms {
77            view_proj_matrix: [[0.0; 4]; 4],
78            enemy_position: enemy.pathfinder.position,
79            enemy_size: enemy.size,
80            player_position: [0.0; 3],
81            _padding: 0.0,
82        };
83
84        let uniform_buffer = create_uniform_buffer(device, &uniforms, "Enemy Uniform Buffer");
85
86        // Create bind group layout for texture + sampler + uniforms
87        let bind_group_layout = BindGroupLayoutBuilder::new(device)
88            .with_label("Enemy Bind Group Layout")
89            .with_uniform_buffer(0, wgpu::ShaderStages::VERTEX | wgpu::ShaderStages::FRAGMENT)
90            .with_texture(1, wgpu::ShaderStages::FRAGMENT)
91            .with_sampler(2, wgpu::ShaderStages::FRAGMENT)
92            .build();
93
94        // Create sampler
95        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
96            address_mode_u: wgpu::AddressMode::ClampToEdge,
97            address_mode_v: wgpu::AddressMode::ClampToEdge,
98            address_mode_w: wgpu::AddressMode::ClampToEdge,
99            mag_filter: wgpu::FilterMode::Linear,
100            min_filter: wgpu::FilterMode::Linear,
101            mipmap_filter: wgpu::FilterMode::Nearest,
102            ..Default::default()
103        });
104
105        // Create bind group for jeffree texture
106        let jeffree_texture_view =
107            jeffree_texture.create_view(&wgpu::TextureViewDescriptor::default());
108        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
109            layout: &bind_group_layout,
110            entries: &[
111                wgpu::BindGroupEntry {
112                    binding: 0,
113                    resource: uniform_buffer.as_entire_binding(),
114                },
115                wgpu::BindGroupEntry {
116                    binding: 1,
117                    resource: wgpu::BindingResource::TextureView(&jeffree_texture_view),
118                },
119                wgpu::BindGroupEntry {
120                    binding: 2,
121                    resource: wgpu::BindingResource::Sampler(&sampler),
122                },
123            ],
124            label: Some("Enemy Bind Group"),
125        });
126
127        // Create vertex buffer layout for position + tex_coords
128        let vertex_buffer_layout = wgpu::VertexBufferLayout {
129            array_stride: 5 * 4, // 5 floats * 4 bytes each
130            step_mode: wgpu::VertexStepMode::Vertex,
131            attributes: &[
132                wgpu::VertexAttribute {
133                    offset: 0,
134                    shader_location: 0,
135                    format: wgpu::VertexFormat::Float32x3, // position (x, y, z)
136                },
137                wgpu::VertexAttribute {
138                    offset: 3 * 4, // 3 floats * 4 bytes
139                    shader_location: 1,
140                    format: wgpu::VertexFormat::Float32x2, // tex_coords
141                },
142            ],
143        };
144
145        let pipeline = PipelineBuilder::new(device, surface_config.format)
146            .with_label("Enemy Pipeline")
147            .with_shader(include_str!("../shaders/enemy.wgsl"))
148            .with_vertex_buffer(vertex_buffer_layout)
149            .with_bind_group_layout(&bind_group_layout)
150            .with_alpha_blending()
151            .with_depth_stencil(wgpu::DepthStencilState {
152                depth_write_enabled: true,
153                depth_compare: wgpu::CompareFunction::LessEqual,
154                format: wgpu::TextureFormat::Depth24Plus,
155                stencil: wgpu::StencilState::default(),
156                bias: wgpu::DepthBiasState::default(),
157            })
158            .build();
159
160        let vertex_buffer = Self::create_billboard_vertices(device);
161
162        Self {
163            pipeline,
164            vertex_buffer,
165            uniform_buffer,
166            bind_group,
167            smoothed_rotation: 0.0,
168            smoothing_factor: 0.85, // Smooth rotation
169        }
170    }
171
172    /// Loads the slime texture from the assets directory.
173    ///
174    /// Attempts to load the slime texture from "assets/Slime.png". If loading fails,
175    /// creates a fallback red texture for debugging purposes.
176    ///
177    /// # Arguments
178    ///
179    /// * `device` - WGPU device for creating the texture
180    /// * `queue` - WGPU queue for uploading texture data
181    ///
182    /// # Returns
183    ///
184    /// A WGPU texture containing the slime image or a fallback texture.
185    fn load_slime_texture(device: &wgpu::Device, queue: &wgpu::Queue) -> wgpu::Texture {
186        let path = "assets/Slime.png";
187
188        // Load image using image crate
189        let img = match image::open(path) {
190            Ok(img) => img.to_rgba8(),
191            Err(e) => {
192                eprintln!("Failed to load slime texture {}: {}", path, e);
193                // Create a fallback texture (solid red square)
194                let mut fallback = image::RgbaImage::new(64, 64);
195                for pixel in fallback.pixels_mut() {
196                    *pixel = image::Rgba([255, 0, 0, 255]); // Red
197                }
198                fallback
199            }
200        };
201
202        let dimensions = img.dimensions();
203        let texture_size = wgpu::Extent3d {
204            width: dimensions.0,
205            height: dimensions.1,
206            depth_or_array_layers: 1,
207        };
208
209        let texture = device.create_texture(&wgpu::TextureDescriptor {
210            label: Some("jeffree Texture"),
211            size: texture_size,
212            mip_level_count: 1,
213            sample_count: 1,
214            dimension: wgpu::TextureDimension::D2,
215            format: wgpu::TextureFormat::Rgba8UnormSrgb,
216            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
217            view_formats: &[],
218        });
219
220        queue.write_texture(
221            wgpu::TexelCopyTextureInfo {
222                aspect: wgpu::TextureAspect::All,
223                texture: &texture,
224                mip_level: 0,
225                origin: wgpu::Origin3d::ZERO,
226            },
227            &img,
228            wgpu::TexelCopyBufferLayout {
229                offset: 0,
230                bytes_per_row: Some(4 * dimensions.0),
231                rows_per_image: Some(dimensions.1),
232            },
233            texture_size,
234        );
235
236        texture
237    }

239    /// Creates vertex buffer containing billboard quad vertices.
240    ///
241    /// Creates a quad centered at the origin that will be transformed by the shader
242    /// to create a billboard effect. The quad includes both position and texture coordinates.
243    ///
244    /// # Arguments
245    ///
246    /// * `device` - WGPU device for creating the vertex buffer
247    ///
248    /// # Returns
249    ///
250    /// A WGPU buffer containing the billboard vertices.
251    fn create_billboard_vertices(device: &wgpu::Device) -> wgpu::Buffer {
252        // Create a quad centered at origin that will be positioned and rotated by the shader
253        // The quad is in local space and will be transformed to world space
254        let vertices: &[f32] = &[
255            // Position (x, y, z)    // Texture coords (u, v)
256            // Triangle 1
257            -0.5, -0.5, 0.0, 0.0, 1.0, // Bottom-left
258            0.5, -0.5, 0.0, 1.0, 1.0, // Bottom-right
259            -0.5, 0.5, 0.0, 0.0, 0.0, // Top-left
260            // Triangle 2
261            0.5, -0.5, 0.0, 1.0, 1.0, // Bottom-right
262            0.5, 0.5, 0.0, 1.0, 0.0, // Top-right
263            -0.5, 0.5, 0.0, 0.0, 0.0, // Top-left
264        ];
265
266        device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
267            label: Some("Enemy Billboard Vertex Buffer"),
268            contents: bytemuck::cast_slice(vertices),
269            usage: wgpu::BufferUsages::VERTEX,
270        })
271    }
272
273    /// Updates enemy position and rotation to face the player.
274    ///
275    /// Calculates the direction to the player and smoothly interpolates the enemy's
276    /// rotation to face that direction. Updates the uniform buffer with current
277    /// game state data.
278    ///
279    /// # Arguments
280    ///
281    /// * `queue` - WGPU queue for uploading uniform data
282    /// * `game_state` - Current game state containing player and enemy positions
283    /// * `view_proj_matrix` - Current view-projection matrix for rendering
284    pub fn update(
285        &mut self,
286        queue: &wgpu::Queue,
287        game_state: &GameState,
288        view_proj_matrix: [[f32; 4]; 4],
289    ) {
290        // Calculate rotation to face player
291        let dx = game_state.player.position[0] - game_state.enemy.pathfinder.position[0];
292        let dz = game_state.player.position[2] - game_state.enemy.pathfinder.position[2];
293
294        // Calculate target rotation using the same coordinate system as your compass
295        // Your compass uses dx.atan2(dz) pattern, so use that here
296        let target_rotation = dx.atan2(dz);
297
298        // Smooth rotation interpolation
299        let mut rotation_diff = target_rotation - self.smoothed_rotation;
300
301        // Wrap to shortest path
302        if rotation_diff > std::f32::consts::PI {
303            rotation_diff -= 2.0 * std::f32::consts::PI;
304        } else if rotation_diff < -std::f32::consts::PI {
305            rotation_diff += 2.0 * std::f32::consts::PI;
306        }
307
308        self.smoothed_rotation += rotation_diff * self.smoothing_factor;
309
310        // Update uniform buffer
311        let uniforms = EnemyUniforms {
312            view_proj_matrix,
313            enemy_position: game_state.enemy.pathfinder.position,
314            enemy_size: game_state.enemy.size,
315            player_position: game_state.player.position,
316            _padding: 0.0,
317        };
318
319        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::cast_slice(&[uniforms]));
320    }
321
322    /// Renders the enemy to the specified render pass.
323    ///
324    /// Sets up the render pipeline, vertex buffer, and bind group, then draws
325    /// the enemy billboard using 6 vertices (2 triangles).
326    ///
327    /// # Arguments
328    ///
329    /// * `render_pass` - The render pass to draw to
330    pub fn render(&self, render_pass: &mut wgpu::RenderPass) {
331        render_pass.set_pipeline(&self.pipeline);
332        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
333        render_pass.set_bind_group(0, &self.bind_group, &[]);
334        render_pass.draw(0..6, 0..1);
335    }

337    /// Gets the current rotation angle of the enemy.
338    ///
339    /// # Returns
340    ///
341    /// The current smoothed rotation angle in radians.
342    pub fn get_rotation(&self) -> f32 {
343        self.smoothed_rotation
344    }

346    /// Sets the smoothing factor for rotation interpolation.
347    ///
348    /// Controls how quickly the enemy rotates to face the player.
349    /// Lower values create smoother, slower rotation while higher values
350    /// create more responsive, faster rotation.
351    ///
352    /// # Arguments
353    ///
354    /// * `factor` - Smoothing factor between 0.01 and 1.0
355    ///   - 0.01: Very smooth, slow rotation
356    ///   - 1.0: Instant rotation with no smoothing
357    pub fn set_smoothing_factor(&mut self, factor: f32) {
358        self.smoothing_factor = factor.clamp(0.01, 1.0);
359    }
