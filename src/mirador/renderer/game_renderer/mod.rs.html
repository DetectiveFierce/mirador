1//! Game renderer module for the Mirador 3D maze game.
2//!
3//! This module provides the main rendering system for the 3D maze game, handling
4//! all visual elements including the maze geometry, enemies, UI elements, and
5//! special effects. The renderer uses WebGPU for hardware-accelerated graphics
6//! and implements a multi-pass rendering pipeline for optimal performance.
7//!
8//! # Overview
9//!
10//! The game renderer consists of several specialized renderers:
11//! - `GameRenderer`: Main renderer coordinating all visual elements
12//! - `CompassRenderer`: Renders the directional compass overlay
13//! - `EnemyRenderer`: Handles enemy visualization and animation
14//! - `StarRenderer`: Creates animated starfield background effects
15//! - `TimerBarRenderer`: Renders the time remaining indicator
16//! - `StaminaBarRenderer`: Displays player stamina levels
17//! - `DebugRenderer`: Development tools for debugging and visualization
18//!
19//! # Rendering Pipeline
20//!
21//! The renderer uses a structured multi-pass approach:
22//! 1. **Background Pass**: Animated starfield using `StarRenderer`
23//! 2. **Geometry Pass**: Maze floors and walls with depth testing
24//! 3. **Entity Pass**: Enemies and interactive elements
25//! 4. **UI Pass**: Compass, timer, stamina bars, and overlays
26//! 5. **Debug Pass**: Optional development overlays
27//!
28//! # Coordinate System
29//!
30//! Uses a right-handed coordinate system:
31//! - X-axis: Right direction
32//! - Y-axis: Up direction  
33//! - Z-axis: Toward viewer (negative Z is forward)
34//! - Maze positioned at world origin (0, 0, 0)
35//!
36//! # Usage
37//!
38//! ```rust
39//! use mirador::renderer::game_renderer::GameRenderer;
40//! use wgpu::{Device, Queue, SurfaceConfiguration};
41//!
42//! // Create renderer
43//! let mut renderer = GameRenderer::new(&device, &queue, &surface_config);
44//!
45//! // Load textures
46//! renderer.load_ceiling_texture(&device, &queue)?;
47//!
48//! // Update depth buffer on resize
49//! let depth_view = renderer.update_depth_texture(&device, width, height);
50//!
51//! // Render frame
52//! renderer.render_game(&queue, &game_state, &mut pass, aspect_ratio);
53//! ```

80/// Main renderer for the 3D maze game.
81///
82/// The `GameRenderer` is responsible for rendering the complete 3D maze environment,
83/// including floors, walls, starfield backgrounds, and debug overlays. It manages
84/// all rendering pipelines, vertex data, and GPU resources required for the game's
85/// visual presentation.
86///
87/// ## Rendering Pipeline
88///
89/// The renderer uses a multi-pass approach:
90/// 1. **Background Pass**: Renders animated starfield using `StarRenderer`
91/// 2. **Main Pass**: Renders maze geometry (floors and walls) with depth testing
92/// 3. **Debug Pass**: Optional overlay rendering for development tools
93///
94/// ## Memory Management
95///
96/// Vertex data for floors and walls is combined into a single buffer for optimal
97/// GPU performance. The depth texture is dynamically resized based on the surface
98/// dimensions and automatically managed to prevent memory leaks.
99///
100/// ## Coordinate System
101///
102/// The renderer uses a right-handed coordinate system with:
103/// - X-axis pointing right
104/// - Y-axis pointing up
105/// - Z-axis pointing toward the viewer
106/// - Maze positioned at world origin
107///
108/// ## Shader Integration
109///
110/// Works with `shader.wgsl` which expects:
111/// ```wgsl
112/// struct Uniforms {
113///     matrix: mat4x4<f32>,
114/// }
115/// @group(0) @binding(0) var<uniform> uniforms: Uniforms;
116/// ```
117///
118/// # Fields
119///
120/// - `pipeline` - Main render pipeline for maze geometry with depth testing and alpha blending
121/// - `vertex_buffer` - Combined vertex buffer containing both floor and wall geometry data
122/// - `vertex_count` - Total number of vertices to render from the combined buffer
123/// - `uniform_buffer` - GPU buffer storing model-view-projection matrix for vertex transformations
124/// - `uniform_bind_group` - WebGPU bind group linking uniform buffer to shader binding point 0
125/// - `depth_texture` - Optional depth buffer for proper 3D occlusion (recreated on resize)
126/// - `star_renderer` - Background renderer for animated starfield effects
127/// - `debug_renderer` - Development tools for rendering bounding boxes and debug overlays
128/// - `compass_renderer` - Renders the directional compass overlay
129/// - `exit_position` - Optional coordinates of the maze exit for special rendering
130/// - `enemy_renderer` - Handles enemy visualization and animation
131/// - `start_time` - Tracks animation start time for time-based effects
132/// - `timer_bar_renderer` - Renders the time remaining indicator
133/// - `stamina_bar_renderer` - Displays player stamina levels
134/// - `ceiling_texture` - Optional texture for ceiling rendering
135/// - `ceiling_texture_view` - Texture view for ceiling rendering
136/// - `ceiling_sampler` - Sampler for ceiling texture filtering
137/// - `ceiling_bind_group` - Bind group for ceiling texture resources

365    /// Loads the ceiling texture and creates the bind group for texturing.
366    ///
367    /// This method loads the ceiling texture from the assets directory and sets up
368    /// all necessary GPU resources for texture rendering, including the texture
369    /// itself, texture view, sampler, and bind group.
370    ///
371    /// # Arguments
372    ///
373    /// * `device` - WebGPU device for creating GPU resources
374    /// * `queue` - WebGPU queue for uploading texture data
375    ///
376    /// # Returns
377    ///
378    /// `Result<(), Box<dyn std::error::Error>>` - Success or error from texture loading
379    ///
380    /// # Examples
381    ///
382    /// ```rust
383    /// use mirador::renderer::game_renderer::GameRenderer;
384    /// use wgpu::{Device, Queue};
385    ///
386    /// let mut renderer = GameRenderer::new(&device, &queue, &surface_config);
387    /// renderer.load_ceiling_texture(&device, &queue)?;
388    /// ```
389    ///
390    /// # Texture Details
391    ///
392    /// - Loads texture from `assets/tiles.jpg`
393    /// - Creates RGBA8 texture with sRGB format
394    /// - Uses repeat addressing for seamless tiling
395    /// - Linear filtering for smooth texture interpolation
396    /// - Creates bind group with uniform buffer, texture, and sampler
397    pub fn load_ceiling_texture(
398        &mut self,
399        device: &wgpu::Device,
400        queue: &wgpu::Queue,
401    ) -> Result<(), Box<dyn std::error::Error>> {
402        // Load the tiles texture
403        let img = image::open("assets/tiles.jpg")?;
404        let rgba = img.to_rgba8();
405        let dimensions = rgba.dimensions();
406
407        let texture_size = wgpu::Extent3d {
408            width: dimensions.0,
409            height: dimensions.1,
410            depth_or_array_layers: 1,
411        };
412
413        let texture = device.create_texture(&wgpu::TextureDescriptor {
414            label: Some("Ceiling Texture"),
415            size: texture_size,
416            mip_level_count: 1,
417            sample_count: 1,
418            dimension: wgpu::TextureDimension::D2,
419            format: wgpu::TextureFormat::Rgba8UnormSrgb,
420            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
421            view_formats: &[],
422        });
423
424        queue.write_texture(
425            wgpu::TexelCopyTextureInfo {
426                aspect: wgpu::TextureAspect::All,
427                texture: &texture,
428                mip_level: 0,
429                origin: wgpu::Origin3d::ZERO,
430            },
431            &rgba,
432            wgpu::TexelCopyBufferLayout {
433                offset: 0,
434                bytes_per_row: Some(4 * dimensions.0),
435                rows_per_image: Some(dimensions.1),
436            },
437            texture_size,
438        );
439
440        let texture_view = texture.create_view(&wgpu::TextureViewDescriptor::default());
441
442        // Create sampler with repeat addressing for tiling
443        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
444            address_mode_u: wgpu::AddressMode::Repeat,
445            address_mode_v: wgpu::AddressMode::Repeat,
446            address_mode_w: wgpu::AddressMode::Repeat,
447            mag_filter: wgpu::FilterMode::Linear,
448            min_filter: wgpu::FilterMode::Linear,
449            mipmap_filter: wgpu::FilterMode::Nearest,
450            ..Default::default()
451        });
452
453        // Create bind group layout for texture + sampler + uniforms
454        let bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
455            label: Some("Ceiling Texture Bind Group Layout"),
456            entries: &[
457                // Uniform buffer (binding 0)
458                wgpu::BindGroupLayoutEntry {
459                    binding: 0,
460                    visibility: wgpu::ShaderStages::VERTEX | wgpu::ShaderStages::FRAGMENT,
461                    ty: wgpu::BindingType::Buffer {
462                        ty: wgpu::BufferBindingType::Uniform,
463                        has_dynamic_offset: false,
464                        min_binding_size: None,
465                    },
466                    count: None,
467                },
468                // Texture (binding 1)
469                wgpu::BindGroupLayoutEntry {
470                    binding: 1,
471                    visibility: wgpu::ShaderStages::FRAGMENT,
472                    ty: wgpu::BindingType::Texture {
473                        sample_type: wgpu::TextureSampleType::Float { filterable: true },
474                        view_dimension: wgpu::TextureViewDimension::D2,
475                        multisampled: false,
476                    },
477                    count: None,
478                },
479                // Sampler (binding 2)
480                wgpu::BindGroupLayoutEntry {
481                    binding: 2,
482                    visibility: wgpu::ShaderStages::FRAGMENT,
483                    ty: wgpu::BindingType::Sampler(wgpu::SamplerBindingType::Filtering),
484                    count: None,
485                },
486            ],
487        });
488
489        // Create bind group
490        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
491            label: Some("Ceiling Texture Bind Group"),
492            layout: &bind_group_layout,
493            entries: &[
494                wgpu::BindGroupEntry {
495                    binding: 0,
496                    resource: self.uniform_buffer.as_entire_binding(),
497                },
498                wgpu::BindGroupEntry {
499                    binding: 1,
500                    resource: wgpu::BindingResource::TextureView(&texture_view),
501                },
502                wgpu::BindGroupEntry {
503                    binding: 2,
504                    resource: wgpu::BindingResource::Sampler(&sampler),
505                },
506            ],
507        });
508
509        // Store the resources
510        self.ceiling_texture = Some(texture);
511        self.ceiling_texture_view = Some(texture_view);
512        self.ceiling_sampler = Some(sampler);
513        self.ceiling_bind_group = Some(bind_group);
514
515        Ok(())
516    }

518    /// Updates or creates the depth texture for proper 3D occlusion.
519    ///
520    /// This method manages the depth buffer, creating a new one when the surface
521    /// dimensions change or when no depth texture exists. The depth texture is
522    /// essential for proper 3D rendering with depth testing.
523    ///
524    /// # Arguments
525    ///
526    /// * `device` - WebGPU device for creating GPU resources
527    /// * `width` - New width of the surface
528    /// * `height` - New height of the surface
529    ///
530    /// # Returns
531    ///
532    /// A `wgpu::TextureView` of the depth texture for use in render passes.
533    ///
534    /// # Examples
535    ///
536    /// ```rust
537    /// use mirador::renderer::game_renderer::GameRenderer;
538    /// use wgpu::Device;
539    ///
540    /// let mut renderer = GameRenderer::new(&device, &queue, &surface_config);
541    /// let depth_view = renderer.update_depth_texture(&device, 1920, 1080);
542    /// ```
543    ///
544    /// # Memory Management
545    ///
546    /// - Automatically drops old depth texture when recreating
547    /// - Only recreates when dimensions actually change
548    /// - Uses Depth24Plus format for optimal precision and performance
549    pub fn update_depth_texture(
550        &mut self,
551        device: &wgpu::Device,
552        width: u32,
553        height: u32,
554    ) -> wgpu::TextureView {
555        if self.depth_texture.is_none()
556            || self
557                .depth_texture
558                .as_ref()
559                .expect("Depth texture should exist")
560                .width()
561                != width
562            || self
563                .depth_texture
564                .as_ref()
565                .expect("Depth texture should exist")
566                .height()
567                != height
568        {
569            if let Some(depth_texture) = self.depth_texture.take() {
570                // Manually drop the texture to free up resources
571                drop(depth_texture);
572            }
573
574            self.depth_texture = Some(device.create_texture(&wgpu::TextureDescriptor {
575                label: Some("Depth Texture"),
576                size: wgpu::Extent3d {
577                    width,
578                    height,
579                    depth_or_array_layers: 1,
580                },
581                mip_level_count: 1,
582                sample_count: 1,
583                dimension: wgpu::TextureDimension::D2,
584                format: wgpu::TextureFormat::Depth24Plus,
585                usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
586                view_formats: &[],
587            }));
588        }
589        self.depth_texture
590            .as_ref()
591            .expect("Depth texture should exist")
592            .create_view(&wgpu::TextureViewDescriptor::default())
593    }

595    /// Renders the complete game scene including maze, enemies, and UI elements.
596    ///
597    /// This is the main rendering method that coordinates all visual elements
598    /// in the correct order. It calculates view and projection matrices once
599    /// and applies them to all rendered objects for consistency.
600    ///
601    /// # Arguments
602    ///
603    /// * `queue` - WebGPU queue for command submission
604    /// * `game_state` - Current game state containing player and enemy information
605    /// * `pass` - Render pass to record drawing commands
606    /// * `aspect` - Aspect ratio of the surface for projection calculations
607    ///
608    /// # Examples
609    ///
610    /// ```rust
611    /// use mirador::renderer::game_renderer::GameRenderer;
612    /// use mirador::game::GameState;
613    /// use wgpu::{Queue, RenderPass};
614    ///
615    /// let mut renderer = GameRenderer::new(&device, &queue, &surface_config);
616    /// renderer.render_game(&queue, &game_state, &mut pass, 16.0 / 9.0);
617    /// ```
618    ///
619    /// # Rendering Order
620    ///
621    /// The method renders elements in this order:
622    /// 1. **Maze/Floor**: Main geometry with depth testing
623    /// 2. **Enemies**: Animated enemy entities
624    /// 3. **UI Elements**: Compass, timer, stamina bars (handled separately)
625    ///
626    /// # Matrix Calculations
627    ///
628    /// - View matrix from player camera position and orientation
629    /// - Projection matrix with configurable FOV and aspect ratio
630    /// - Combined view-projection matrix for efficient rendering
631    /// - Model matrix for floor (identity) and individual enemy transforms
632    pub fn render_game(
633        &mut self,
634        queue: &wgpu::Queue,
635        game_state: &GameState,
636        pass: &mut wgpu::RenderPass,
637        aspect: f32,
638    ) {
639        // Calculate view and projection matrices once
640        let view_matrix = game_state.player.get_view_matrix();
641        let projection_matrix = Mat4::perspective(
642            deg_to_rad(game_state.player.fov),
643            aspect,
644            0.1,    // zNear
645            2000.0, // zFar
646        );
647        let view_proj_matrix = view_matrix.multiply(&projection_matrix);
648
649        // ==============================================
650        // 1. RENDER MAZE/FLOOR FIRST
651        // ==============================================
652        {
653            // Model Matrix for floor - identity since floor is at world origin
654            let model_matrix = Mat4::identity();
655
656            // Combine matrices: Projection * View * Model
657            let final_mvp_matrix = model_matrix.multiply(&view_proj_matrix);
658
659            // Calculate elapsed time for animation
660            let elapsed = self.start_time.elapsed().as_secs_f32();
661
662            let uniforms = Uniforms {
663                matrix: final_mvp_matrix.into(),
664                time: elapsed,
665                _padding: [0.0; 7],
666            };
667
668            // Upload uniform values for the maze/floor
669            queue.write_buffer(&self.uniform_buffer, 0, uniforms.as_bytes());
670
671            // Render the maze/floor only if we have vertices to render
672            if self.vertex_count > 0 {
673                pass.set_pipeline(&self.pipeline);
674                pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
675
676                // Use ceiling texture bind group if available, otherwise fall back to uniform bind group
677                if let Some(ceiling_bind_group) = &self.ceiling_bind_group {
678                    pass.set_bind_group(0, ceiling_bind_group, &[]);
679                } else {
680                    pass.set_bind_group(0, &self.uniform_bind_group, &[]);
681                }
682
683                pass.draw(0..self.vertex_count, 0..1);
684            }
685
686            // Debug rendering for maze/floor
687            if self.debug_renderer.debug_render_bounding_boxes
688                && self.debug_renderer.debug_vertex_count > 0
689            {
690                if let Some(debug_buffer) = &self.debug_renderer.debug_vertex_buffer {
691                    pass.set_vertex_buffer(0, debug_buffer.slice(..));
692                    pass.draw(0..self.debug_renderer.debug_vertex_count as u32, 0..1);
693                }
694            }
695        }
696
697        // ==============================================
698        // 2. RENDER ENEMIES
699        // ==============================================
700        {
701            // Update enemy transform with the combined view-projection matrix
702            self.enemy_renderer.update(
703                queue,
704                game_state,
705                view_proj_matrix.0, // Pass the view-projection matrix
706            );
707
708            // Actually render the enemy
709            self.enemy_renderer.render(pass);
710        }
711    }
712}
