8#[derive(Debug, Copy, Clone, bytemuck::Pod, bytemuck::Zeroable)]
9struct GameOverUniforms {
10    time: f32,
11    _padding: [f32; 3],
12}

14pub struct GameOverRenderer {

17    uniform_buffer: wgpu::Buffer,
18    bind_group: wgpu::BindGroup,

21impl GameOverRenderer {
22    /// Create a new game over renderer.
23    ///
24    /// This sets up the complete rendering pipeline for displaying a game over overlay,
25    /// including alpha blending support for semitransparent rendering over the game scene.
26    ///
27    /// ## Shader Requirements
28    ///
29    /// The game over shader should be located at `../maze/game-over.wgsl`
30    /// and should expect:
31    /// - Vertex input: `@location(0) position: vec2<f32>`
32    /// - Uniform binding: `@group(0) @binding(0) var<uniform> uniforms: GameOverUniforms;`
33    ///
34    /// ## Initial State
35    ///
36    /// The renderer starts with time = 0.0. Use [`update_time()`](GameOverRenderer::update_time)
37    /// to animate the overlay effect.
38    ///
39    /// # Parameters
40    ///
41    /// - `device` - WGPU device for creating GPU resources
42    /// - `surface_config` - Surface configuration (provides target format)
43    ///
44    /// # Returns
45    ///
46    /// A configured `GameOverRenderer` ready for rendering.
47    ///
48    /// # Example
49    ///
50    /// ```rust,no_run
51    /// # use crate::renderer::maze_renderer::GameOverRenderer;
52    /// # let device: wgpu::Device = unimplemented!();
53    /// # let surface_config: wgpu::SurfaceConfiguration = unimplemented!();
54    ///
55    /// let renderer = GameOverRenderer::new(&device, &surface_config);
56    /// ```
57    pub fn new(device: &wgpu::Device, surface_config: &wgpu::SurfaceConfiguration) -> Self {
58        let uniforms = GameOverUniforms {
59            time: 0.0,
60            _padding: [0.0; 3],
61        };
62
63        let uniform_buffer = create_uniform_buffer(device, &uniforms, "Game Over Uniform Buffer");
64
65        let bind_group_layout = BindGroupLayoutBuilder::new(device)
66            .with_label("Game Over Bind Group Layout")
67            .with_uniform_buffer(0, wgpu::ShaderStages::FRAGMENT)
68            .build();
69
70        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
71            layout: &bind_group_layout,
72            entries: &[wgpu::BindGroupEntry {
73                binding: 0,
74                resource: uniform_buffer.as_entire_binding(),
75            }],
76            label: Some("Game Over Bind Group"),
77        });
78
79        let pipeline = PipelineBuilder::new(device, surface_config.format)
80            .with_label("Game Over Pipeline")
81            .with_shader(include_str!("../shaders/game-over.wgsl"))
82            .with_vertex_buffer(create_vertex_2d_layout())
83            .with_bind_group_layout(&bind_group_layout)
84            .with_alpha_blending()
85            .build();
86
87        let vertex_buffer = create_fullscreen_vertices(device);
88
89        Self {
90            pipeline,
91            vertex_buffer,
92            uniform_buffer,
93            bind_group,
94        }
95    }

97    /// Update the animation time for the game over overlay.
98    ///
99    /// This method uploads new time data to the GPU uniform buffer to enable
100    /// animated effects like pulsing, fading, or other time-based animations
101    /// in the game over shader.
102    ///
103    /// ## Performance Notes
104    ///
105    /// This operation involves a GPU buffer write. It's typically called once
106    /// per frame during the game over state to maintain smooth animations.
107    ///
108    /// # Parameters
109    ///
110    /// - `queue` - WGPU queue for buffer uploads
111    /// - `time` - Time value in seconds (typically elapsed time since game over)
112    ///
113    /// # Example
114    ///
115    /// ```rust,no_run
116    /// # use crate::renderer::maze_renderer::GameOverRenderer;
117    /// # let renderer: GameOverRenderer = unimplemented!();
118    /// # let queue: wgpu::Queue = unimplemented!();
119    ///
120    /// // Update with elapsed time for animations
121    /// let elapsed = start_time.elapsed().as_secs_f32();
122    /// renderer.update_time(&queue, elapsed);
123    /// ```
124    pub fn update_time(&self, queue: &wgpu::Queue, time: f32) {
125        let uniforms = GameOverUniforms {
126            time,
127            _padding: [0.0; 3],
128        };
129        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::cast_slice(&[uniforms]));
130    }
131
132    /// Render the game over overlay to the current render pass.
133    ///
134    /// This method renders a semitransparent red overlay that covers the entire
135    /// screen using alpha blending. The overlay appears on top of the frozen
136    /// game scene underneath.
137    ///
138    /// ## Render State
139    ///
140    /// This method assumes:
141    /// - A render pass is active
142    /// - Alpha blending is desired (the pipeline enables it)
143    /// - The game scene has already been rendered
144    /// - The overlay should cover the full viewport
145    ///
146    /// # Parameters
147    ///
148    /// - `render_pass` - Active render pass to render into
149    /// - `window` - Window reference for potential window-specific adjustments
150    ///
151    /// # Example
152    ///
153    /// ```rust,no_run
154    /// # use crate::renderer::maze_renderer::GameOverRenderer;
155    /// # let renderer: GameOverRenderer = unimplemented!();
156    /// # let mut render_pass: wgpu::RenderPass = unimplemented!();
157    /// # let window: &winit::window::Window = unimplemented!();
158    ///
159    /// // Render game scene first
160    /// // ... render background, stars, game objects ...
161    ///
162    /// // Render game over overlay on top
163    /// renderer.render(&mut render_pass, window);
164    /// ```
165    pub fn render(&self, render_pass: &mut wgpu::RenderPass, _window: &winit::window::Window) {
166        render_pass.set_pipeline(&self.pipeline);
167        render_pass.set_bind_group(0, &self.bind_group, &[]);
168        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
169        render_pass.draw(0..6, 0..1);
170    }
