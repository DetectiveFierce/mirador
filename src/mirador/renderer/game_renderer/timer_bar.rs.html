1//! Timer Bar Renderer Module
2//!
3//! This module provides a GPU-accelerated timer bar renderer using wgpu.
4//! The timer bar can display progress with animated effects and is designed
5//! for real-time rendering applications.
6
7use crate::renderer::pipeline_builder::{
8    BindGroupLayoutBuilder, PipelineBuilder, create_uniform_buffer,
9};
10use std::time::Instant;
11use wgpu;
12
13/// Uniform buffer data structure for the timer bar shader.
14///
15/// This structure is passed to the GPU shader to control the appearance
16/// and animation of the timer bar. The layout uses explicit padding to
17/// ensure proper GPU memory alignment.
18///
19/// # Memory Layout
20/// The struct uses `#[repr(C)]` to ensure consistent memory layout across
21/// platforms, which is required for GPU buffer uploads.
22#[repr(C)]
23#[derive(Debug, Copy, Clone, bytemuck::Pod, bytemuck::Zeroable)]
24pub struct TimerBarUniforms {
25    /// Progress value from 0.0 to 1.0 representing the completion percentage
26    pub progress: f32,
27
28    /// Current time in seconds since the timer started, used for animations
29    pub time: f32,
30
31    /// Screen resolution as [width, height] in pixels for proper scaling
32    pub resolution: [f32; 2],
33
34    /// Padding to ensure proper GPU memory alignment (16-byte alignment)
35    pub _padding: [f32; 2],
36}
37
38/// GPU-accelerated timer bar renderer.
39///
40/// This renderer creates and manages a fullscreen timer bar that can display
41/// progress with smooth animations. It uses a dedicated render pipeline with
42/// custom shaders for efficient GPU rendering.
43///
44/// # Usage
45/// ```rust,no_run
46/// # use wgpu;
47/// # let device: wgpu::Device = todo!();
48/// # let surface_config: wgpu::SurfaceConfiguration = todo!();
49/// let timer_bar = TimerBarRenderer::new(&device, &surface_config);
50///
51/// // In your render loop:
52/// # let queue: wgpu::Queue = todo!();
53/// # let mut render_pass: wgpu::RenderPass = todo!();
54/// timer_bar.update_uniforms(&queue, 0.5, [800.0, 600.0], 1.0);
55/// timer_bar.render(&mut render_pass);
56/// ```
57pub struct TimerBarRenderer {
58    /// The GPU render pipeline configured for timer bar rendering
59    pub pipeline: wgpu::RenderPipeline,
60
61    /// GPU buffer containing the uniform data (TimerBarUniforms)
62    pub uniform_buffer: wgpu::Buffer,
63
64    /// Bind group that associates the uniform buffer with the shader
65    pub bind_group: wgpu::BindGroup,
66
67    /// Start time for calculating elapsed time in animations
68    pub start_time: Instant,
69}
70
71impl TimerBarRenderer {
72    /// Creates a new timer bar renderer.
73    ///
74    /// This function initializes all GPU resources needed for timer bar rendering,
75    /// including the render pipeline, uniform buffer, and bind groups.
76    ///
77    /// # Arguments
78    /// * `device` - The wgpu device for creating GPU resources
79    /// * `surface_config` - Surface configuration containing the target format
80    ///
81    /// # Returns
82    /// A new `TimerBarRenderer` instance ready for rendering
83    ///
84    /// # Example
85    /// ```rust,no_run
86    /// # use wgpu;
87    /// # let device: wgpu::Device = todo!();
88    /// # let surface_config: wgpu::SurfaceConfiguration = todo!();
89    /// let timer_bar = TimerBarRenderer::new(&device, &surface_config);
90    /// ```
91    pub fn new(device: &wgpu::Device, surface_config: &wgpu::SurfaceConfiguration) -> Self {
92        // Initialize uniform data with default values
93        let uniforms = TimerBarUniforms {
94            progress: 1.0,              // Start at full progress
95            time: 0.0,                  // Start time at zero
96            resolution: [800.0, 600.0], // Default resolution
97            _padding: [0.0; 2],         // Zero padding
98        };
99
100        // Create the uniform buffer on the GPU
101        let uniform_buffer = create_uniform_buffer(device, &uniforms, "Timer Bar Uniform Buffer");
102
103        // Create bind group layout defining how uniforms are accessed in shaders
104        let bind_group_layout = BindGroupLayoutBuilder::new(device)
105            .with_label("Timer Bar Bind Group Layout")
106            .with_uniform_buffer(0, wgpu::ShaderStages::VERTEX_FRAGMENT)
107            .build();
108
109        // Create bind group that connects the uniform buffer to the layout
110        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
111            layout: &bind_group_layout,
112            entries: &[wgpu::BindGroupEntry {
113                binding: 0,
114                resource: uniform_buffer.as_entire_binding(),
115            }],
116            label: Some("Timer Bar Bind Group"),
117        });
118
119        // Create the render pipeline with the timer bar shader
120        let pipeline = PipelineBuilder::new(device, surface_config.format)
121            .with_label("Timer Bar Pipeline")
122            .with_shader(include_str!("../shaders/timer-bar.wgsl"))
123            .with_bind_group_layout(&bind_group_layout)
124            .with_alpha_blending() // Enable transparency for smooth edges
125            .build();
126
127        Self {
128            pipeline,
129            uniform_buffer,
130            bind_group,
131            start_time: Instant::now(),
132        }
133    }
134
135    /// Updates the uniform buffer with new timer bar parameters.
136    ///
137    /// This method should be called each frame to update the timer bar's
138    /// appearance and animation state. The progress value is automatically
139    /// clamped to the valid range [0.0, 1.0].
140    ///
141    /// # Arguments
142    /// * `queue` - The wgpu command queue for buffer uploads
143    /// * `progress` - Progress value (0.0 = empty, 1.0 = full), will be clamped
144    /// * `resolution` - Current screen resolution as [width, height]
145    /// * `time` - Current time in seconds for animations
146    ///
147    /// # Example
148    /// ```rust,no_run
149    /// # use wgpu;
150    /// # let timer_bar: TimerBarRenderer = todo!();
151    /// # let queue: wgpu::Queue = todo!();
152    /// // Update with 75% progress at 1920x1080 resolution
153    /// timer_bar.update_uniforms(&queue, 0.75, [1920.0, 1080.0], 2.5);
154    /// ```
155    pub fn update_uniforms(
156        &self,
157        queue: &wgpu::Queue,
158        progress: f32,
159        resolution: [f32; 2],
160        time: f32,
161    ) {
162        let uniforms = TimerBarUniforms {
163            progress: progress.clamp(0.0, 1.0), // Ensure progress stays in valid range
164            time,
165            resolution,
166            _padding: [0.0; 2],
167        };
168
169        // Upload the new uniform data to the GPU
170        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::cast_slice(&[uniforms]));
171    }
172
173    /// Renders the timer bar to the current render pass.
174    ///
175    /// This method draws the timer bar using a fullscreen triangle technique
176    /// (3 vertices) which is more efficient than a quad for fullscreen effects.
177    /// The actual timer bar shape and positioning is handled in the shader.
178    ///
179    /// # Arguments
180    /// * `render_pass` - Active render pass to draw into
181    ///
182    /// # Note
183    /// Make sure to call `update_uniforms()` before rendering to ensure
184    /// the timer bar displays the current state.
185    ///
186    /// # Example
187    /// ```rust,no_run
188    /// # use wgpu;
189    /// # let timer_bar: TimerBarRenderer = todo!();
190    /// # let mut render_pass: wgpu::RenderPass = todo!();
191    /// timer_bar.render(&mut render_pass);
192    /// ```
193    pub fn render(&self, render_pass: &mut wgpu::RenderPass) {
194        // Set the timer bar pipeline as active
195        render_pass.set_pipeline(&self.pipeline);
196
197        // Bind the uniform buffer to the shader
198        render_pass.set_bind_group(0, &self.bind_group, &[]);
199
200        // Draw using fullscreen triangle technique (3 vertices, 1 instance)
201        render_pass.draw(0..3, 0..1);
202    }
203}
