12/// Vertex data structure for rendering icon quads.
13///
14/// Each vertex contains position coordinates in normalized device coordinates (-1 to 1)
15/// and UV texture coordinates (0 to 1) for texture sampling.
16#[repr(C)]
17#[derive(Copy, Clone, Debug, bytemuck::Pod, bytemuck::Zeroable)]
18struct IconVertex {
19    /// Position in normalized device coordinates [x, y]
20    position: [f32; 2],
21    /// Texture coordinates [u, v] where (0,0) is top-left and (1,1) is bottom-right
22    uv: [f32; 2],
23}
24
25impl IconVertex {
26    /// Returns the vertex buffer layout descriptor for the graphics pipeline.
27    ///
28    /// Defines how vertex data is laid out in memory and which shader locations
29    /// correspond to which vertex attributes.
30    fn desc<'a>() -> VertexBufferLayout<'a> {
31        VertexBufferLayout {
32            array_stride: mem::size_of::<IconVertex>() as wgpu::BufferAddress,
33            step_mode: wgpu::VertexStepMode::Vertex,
34            attributes: &[
35                // Position attribute (location 0)
36                VertexAttribute {
37                    offset: 0,
38                    shader_location: 0,
39                    format: VertexFormat::Float32x2,
40                },
41                // UV coordinate attribute (location 1)
42                VertexAttribute {
43                    offset: mem::size_of::<[f32; 2]>() as wgpu::BufferAddress,
44                    shader_location: 1,
45                    format: VertexFormat::Float32x2,
46                },
47            ],
48        }
49    }
50}

52/// Represents a 2D icon with position, size, and texture information.
53///
54/// Icons are positioned using screen coordinates where (0,0) is the top-left corner.
55/// The texture_id corresponds to a loaded texture that will be used for rendering.

63    pub width: f32,
64    /// Height in screen coordinates (pixels)

68}

93/// High-performance batch renderer for 2D icons using WGPU.
94///
95/// This renderer optimizes performance through several techniques:
96/// - **Texture Batching**: Groups icons by texture to minimize bind group changes
97/// - **Buffer Caching**: Reuses vertex and index buffers when icon counts haven't changed
98/// - **Instanced Rendering**: Renders multiple icons with the same texture in a single draw call
99///
100/// The renderer converts screen coordinates to normalized device coordinates automatically
101/// and handles texture loading and management.
102pub struct IconRenderer {
103    /// The graphics pipeline used for rendering icons
104    render_pipeline: RenderPipeline,
105    /// Layout descriptor for texture and sampler bind groups
106    bind_group_layout: BindGroupLayout,
107    /// Collection of icons to be rendered
108    icons: Vec<Icon>,
109    /// Cache of loaded textures, views, and bind groups keyed by texture ID
110    textures: HashMap<String, (Texture, TextureView, BindGroup)>,
111    /// Current window width in pixels (used for coordinate conversion)
112    window_width: f32,
113    /// Current window height in pixels (used for coordinate conversion)
114    window_height: f32,
115    /// Cached vertex buffers per texture to avoid recreation
116    cached_vertex_buffers: HashMap<String, wgpu::Buffer>,
117    /// Cached index buffers per texture to avoid recreation
118    cached_index_buffers: HashMap<String, wgpu::Buffer>,
119    /// Number of icons for each texture in the cached buffers
120    cached_icon_counts: HashMap<String, usize>,
121}
122
123impl IconRenderer {
124    /// Creates a new IconRenderer with the specified device and surface format.
125    ///
126    /// This method sets up the complete graphics pipeline including:
127    /// - Shader module loading
128    /// - Bind group layout for textures and samplers
129    /// - Render pipeline with alpha blending enabled
130    /// - Default window dimensions
131    ///
132    /// # Arguments
133    /// * `device` - The WGPU device for creating graphics resources
134    /// * `surface_format` - The texture format of the render target
135    ///
136    /// # Returns
137    /// A new `IconRenderer` instance ready for use
138    pub fn new(device: &Device, surface_format: TextureFormat) -> Self {
139        // Load the icon shader from an embedded WGSL file
140        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
141            label: Some("Icon Shader"),
142            source: wgpu::ShaderSource::Wgsl(include_str!("shaders/icon.wgsl").into()),
143        });
144
145        // Create bind group layout for texture and sampler
146        let bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
147            label: Some("Icon Bind Group Layout"),
148            entries: &[
149                // Texture binding (binding 0)
150                wgpu::BindGroupLayoutEntry {
151                    binding: 0,
152                    visibility: ShaderStages::FRAGMENT,
153                    ty: wgpu::BindingType::Texture {
154                        sample_type: wgpu::TextureSampleType::Float { filterable: true },
155                        view_dimension: wgpu::TextureViewDimension::D2,
156                        multisampled: false,
157                    },
158                    count: None,
159                },
160                // Sampler binding (binding 1)
161                wgpu::BindGroupLayoutEntry {
162                    binding: 1,
163                    visibility: ShaderStages::FRAGMENT,
164                    ty: wgpu::BindingType::Sampler(SamplerBindingType::Filtering),
165                    count: None,
166                },
167            ],
168        });
169
170        // Create the pipeline layout
171        let render_pipeline_layout =
172            device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
173                label: Some("Icon Pipeline Layout"),
174                bind_group_layouts: &[&bind_group_layout],
175                push_constant_ranges: &[],
176            });
177
178        // Create the render pipeline with alpha blending
179        let render_pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
180            label: Some("Icon Pipeline"),
181            layout: Some(&render_pipeline_layout),
182            vertex: VertexState {
183                module: &shader,
184                entry_point: Some("vs_main"),
185                buffers: &[IconVertex::desc()],
186                compilation_options: Default::default(),
187            },
188            fragment: Some(FragmentState {
189                module: &shader,
190                entry_point: Some("fs_main"),
191                targets: &[Some(ColorTargetState {
192                    format: surface_format,
193                    // Enable alpha blending for transparent icons
194                    blend: Some(wgpu::BlendState {
195                        color: wgpu::BlendComponent {
196                            src_factor: wgpu::BlendFactor::SrcAlpha,
197                            dst_factor: wgpu::BlendFactor::OneMinusSrcAlpha,
198                            operation: wgpu::BlendOperation::Add,
199                        },
200                        alpha: wgpu::BlendComponent {
201                            src_factor: wgpu::BlendFactor::One,
202                            dst_factor: wgpu::BlendFactor::OneMinusSrcAlpha,
203                            operation: wgpu::BlendOperation::Add,
204                        },
205                    }),
206                    write_mask: ColorWrites::ALL,
207                })],
208                compilation_options: Default::default(),
209            }),
210            primitive: PrimitiveState {
211                topology: wgpu::PrimitiveTopology::TriangleList,
212                strip_index_format: None,
213                front_face: wgpu::FrontFace::Ccw,
214                cull_mode: Some(wgpu::Face::Back),
215                polygon_mode: wgpu::PolygonMode::Fill,
216                unclipped_depth: false,
217                conservative: false,
218            },
219            depth_stencil: None,
220            multisample: MultisampleState {
221                count: 1,
222                mask: !0,
223                alpha_to_coverage_enabled: false,
224            },
225            multiview: None,
226            cache: None,
227        });
228
229        Self {
230            render_pipeline,
231            bind_group_layout,
232            icons: Vec::new(),
233            textures: HashMap::new(),
234            window_width: 1360.0,
235            window_height: 768.0,
236            cached_vertex_buffers: HashMap::new(),
237            cached_index_buffers: HashMap::new(),
238            cached_icon_counts: HashMap::new(),
239        }
240    }

242    /// Loads a texture from a file path and creates associated GPU resources.
243    ///
244    /// This method:
245    /// 1. Loads an image from the specified file path
246    /// 2. Converts it to RGBA8 format
247    /// 3. Creates a WGPU texture and uploads the image data
248    /// 4. Creates a texture view and sampler
249    /// 5. Creates a bind group for use in rendering
250    /// 6. Caches all resources for later use
251    ///
252    /// # Arguments
253    /// * `device` - The WGPU device for creating resources
254    /// * `queue` - The WGPU queue for uploading texture data
255    /// * `path` - File system path to the image file
256    /// * `texture_id` - Unique identifier for this texture
257    ///
258    /// # Returns
259    /// `Ok(())` on success, or an error if the image cannot be loaded
260    ///
261    /// # Errors
262    /// Returns an error if:
263    /// - The image file cannot be found or read
264    /// - The image format is not supported
265    /// - GPU resource creation fails
266    pub fn load_texture(
267        &mut self,
268        device: &Device,
269        queue: &wgpu::Queue,
270        path: &str,
271        texture_id: &str,
272    ) -> Result<(), Box<dyn std::error::Error>> {
273        // Load and convert image to RGBA8
274        let img = image::open(Path::new(path))?;
275        let rgba = img.to_rgba8();
276        let dimensions = rgba.dimensions();
277
278        let size = wgpu::Extent3d {
279            width: dimensions.0,
280            height: dimensions.1,
281            depth_or_array_layers: 1,
282        };
283
284        // Create GPU texture
285        let texture = device.create_texture(&wgpu::TextureDescriptor {
286            label: Some(&format!("Icon texture: {}", texture_id)),
287            size,
288            mip_level_count: 1,
289            sample_count: 1,
290            dimension: wgpu::TextureDimension::D2,
291            format: wgpu::TextureFormat::Rgba8Unorm,
292            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
293            view_formats: &[],
294        });
295
296        // Upload image data to GPU
297        queue.write_texture(
298            wgpu::TexelCopyTextureInfo {
299                aspect: wgpu::TextureAspect::All,
300                texture: &texture,
301                mip_level: 0,
302                origin: wgpu::Origin3d::ZERO,
303            },
304            &rgba,
305            wgpu::TexelCopyBufferLayout {
306                offset: 0,
307                bytes_per_row: Some(4 * dimensions.0),
308                rows_per_image: Some(dimensions.1),
309            },
310            size,
311        );
312
313        // Create texture view and sampler
314        let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
315        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
316            address_mode_u: wgpu::AddressMode::ClampToEdge,
317            address_mode_v: wgpu::AddressMode::ClampToEdge,
318            address_mode_w: wgpu::AddressMode::ClampToEdge,
319            mag_filter: wgpu::FilterMode::Linear,
320            min_filter: wgpu::FilterMode::Linear,
321            mipmap_filter: wgpu::FilterMode::Nearest,
322            ..Default::default()
323        });
324
325        // Create bind group for this texture
326        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
327            label: Some(&format!("Icon bind group: {}", texture_id)),
328            layout: &self.bind_group_layout,
329            entries: &[
330                wgpu::BindGroupEntry {
331                    binding: 0,
332                    resource: wgpu::BindingResource::TextureView(&view),
333                },
334                wgpu::BindGroupEntry {
335                    binding: 1,
336                    resource: wgpu::BindingResource::Sampler(&sampler),
337                },
338            ],
339        });
340
341        // Cache all resources
342        self.textures
343            .insert(texture_id.to_string(), (texture, view, bind_group));
344
345        Ok(())
346    }

389    /// Renders all queued icons to the specified render pass.
390    ///
391    /// This method implements several optimizations:
392    ///
393    /// ## Texture Batching
394    /// Icons are grouped by texture ID to minimize the number of bind group changes,
395    /// which is an expensive GPU operation.
396    ///
397    /// ## Buffer Caching
398    /// Vertex and index buffers are cached and reused when the number of icons
399    /// per texture hasn't changed since the last render.
400    ///
401    /// ## Coordinate Conversion
402    /// Screen coordinates are automatically converted to normalized device coordinates
403    /// (-1 to 1 range) required by the vertex shader.
404    ///
405    /// # Arguments
406    /// * `device` - The WGPU device for creating new buffers if needed
407    /// * `render_pass` - The active render pass to submit draw commands to
408    ///
409    /// # Performance Notes
410    /// - Best performance is achieved when icons using the same texture are rendered together
411    /// - Buffer recreation only occurs when icon counts change
412    /// - Empty icon lists result in immediate return with no GPU work
413    pub fn render(&mut self, device: &Device, render_pass: &mut RenderPass) {
414        if self.icons.is_empty() {
415            return;
416        }
417
418        render_pass.set_pipeline(&self.render_pipeline);
419
420        // Group icons by texture to minimize bind group changes
421        let mut icons_by_texture: HashMap<String, Vec<&Icon>> = HashMap::new();
422        for icon in &self.icons {
423            icons_by_texture
424                .entry(icon.texture_id.clone())
425                .or_insert_with(Vec::new)
426                .push(icon);
427        }
428
429        // Render each texture batch
430        for (texture_id, icons) in icons_by_texture {
431            if let Some((_, _, bind_group)) = self.textures.get(&texture_id) {
432                render_pass.set_bind_group(0, bind_group, &[]);
433
434                // Check if we can reuse cached buffers for this texture
435                let cached_count = self.cached_icon_counts.get(&texture_id).unwrap_or(&0);
436                let need_new_buffers = *cached_count != icons.len();
437
438                if need_new_buffers {
439                    // Create vertices for all icons using this texture
440                    let mut all_vertices = Vec::new();
441                    let mut all_indices = Vec::new();
442
443                    for (icon_index, icon) in icons.iter().enumerate() {
444                        // Convert screen coordinates to normalized device coordinates
445                        let x = (icon.x / self.window_width) * 2.0 - 1.0;
446                        let y = (icon.y / self.window_height) * 2.0 - 1.0;
447                        let width = (icon.width / self.window_width) * 2.0;
448                        let height = (icon.height / self.window_height) * 2.0;
449
450                        // Create quad vertices for this icon
451                        let vertices = [
452                            // Top-left
453                            IconVertex {
454                                position: [x, y],
455                                uv: [0.0, 1.0], // Flip V coordinate for correct texture orientation
456                            },
457                            // Top-right
458                            IconVertex {
459                                position: [x + width, y],
460                                uv: [1.0, 1.0],
461                            },
462                            // Bottom-right
463                            IconVertex {
464                                position: [x + width, y + height],
465                                uv: [1.0, 0.0],
466                            },
467                            // Bottom-left
468                            IconVertex {
469                                position: [x, y + height],
470                                uv: [0.0, 0.0],
471                            },
472                        ];
473
474                        // Add vertices to the batch
475                        all_vertices.extend_from_slice(&vertices);
476
477                        // Create triangle indices for this quad (two triangles)
478                        let base_index = (icon_index * 4) as u16;
479                        let indices = [
480                            // First triangle (top-left, top-right, bottom-right)
481                            base_index,
482                            base_index + 1,
483                            base_index + 2,
484                            // Second triangle (top-left, bottom-right, bottom-left)
485                            base_index,
486                            base_index + 2,
487                            base_index + 3,
488                        ];
489                        all_indices.extend_from_slice(&indices);
490                    }
491
492                    // Create new vertex buffer for all icons using this texture
493                    let vertex_buffer =
494                        device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
495                            label: Some("Icon Vertex Buffer"),
496                            contents: bytemuck::cast_slice(&all_vertices),
497                            usage: BufferUsages::VERTEX,
498                        });
499
500                    // Create new index buffer for all icons using this texture
501                    let index_buffer =
502                        device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
503                            label: Some("Icon Index Buffer"),
504                            contents: bytemuck::cast_slice(&all_indices),
505                            usage: BufferUsages::INDEX,
506                        });
507
508                    // Cache the new buffers
509                    self.cached_vertex_buffers
510                        .insert(texture_id.clone(), vertex_buffer);
511                    self.cached_index_buffers
512                        .insert(texture_id.clone(), index_buffer);
513                    self.cached_icon_counts
514                        .insert(texture_id.clone(), icons.len());
515                }
516
517                // Use cached buffers for rendering
518                if let (Some(vertex_buffer), Some(index_buffer)) = (
519                    self.cached_vertex_buffers.get(&texture_id),
520                    self.cached_index_buffers.get(&texture_id),
521                ) {
522                    render_pass.set_vertex_buffer(0, vertex_buffer.slice(..));
523                    render_pass.set_index_buffer(index_buffer.slice(..), wgpu::IndexFormat::Uint16);
524                    // Draw all icons for this texture (6 indices per icon: 2 triangles × 3 vertices each)
525                    render_pass.draw_indexed(0..(icons.len() * 6) as u32, 0, 0..1);
526                }
527            }
528        }
529    }
