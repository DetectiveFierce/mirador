19impl IconVertex {
20    fn desc<'a>() -> VertexBufferLayout<'a> {
21        VertexBufferLayout {
22            array_stride: mem::size_of::<IconVertex>() as wgpu::BufferAddress,
23            step_mode: wgpu::VertexStepMode::Vertex,
24            attributes: &[
25                VertexAttribute {
26                    offset: 0,
27                    shader_location: 0,
28                    format: VertexFormat::Float32x2,
29                },
30                VertexAttribute {
31                    offset: mem::size_of::<[f32; 2]>() as wgpu::BufferAddress,
32                    shader_location: 1,
33                    format: VertexFormat::Float32x2,
34                },
35            ],
36        }
37    }
38}

73impl IconRenderer {
74    pub fn new(device: &Device, surface_format: TextureFormat) -> Self {
75        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
76            label: Some("Icon Shader"),
77            source: wgpu::ShaderSource::Wgsl(include_str!("shaders/icon.wgsl").into()),
78        });
79
80        let bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
81            label: Some("Icon Bind Group Layout"),
82            entries: &[
83                // Texture
84                wgpu::BindGroupLayoutEntry {
85                    binding: 0,
86                    visibility: ShaderStages::FRAGMENT,
87                    ty: wgpu::BindingType::Texture {
88                        sample_type: wgpu::TextureSampleType::Float { filterable: true },
89                        view_dimension: wgpu::TextureViewDimension::D2,
90                        multisampled: false,
91                    },
92                    count: None,
93                },
94                // Sampler
95                wgpu::BindGroupLayoutEntry {
96                    binding: 1,
97                    visibility: ShaderStages::FRAGMENT,
98                    ty: wgpu::BindingType::Sampler(SamplerBindingType::Filtering),
99                    count: None,
100                },
101            ],
102        });
103
104        let render_pipeline_layout =
105            device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
106                label: Some("Icon Pipeline Layout"),
107                bind_group_layouts: &[&bind_group_layout],
108                push_constant_ranges: &[],
109            });
110
111        let render_pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
112            label: Some("Icon Pipeline"),
113            layout: Some(&render_pipeline_layout),
114            vertex: VertexState {
115                module: &shader,
116                entry_point: Some("vs_main"),
117                buffers: &[IconVertex::desc()],
118                compilation_options: Default::default(),
119            },
120            fragment: Some(FragmentState {
121                module: &shader,
122                entry_point: Some("fs_main"),
123                targets: &[Some(ColorTargetState {
124                    format: surface_format,
125                    blend: Some(wgpu::BlendState {
126                        color: wgpu::BlendComponent {
127                            src_factor: wgpu::BlendFactor::SrcAlpha,
128                            dst_factor: wgpu::BlendFactor::OneMinusSrcAlpha,
129                            operation: wgpu::BlendOperation::Add,
130                        },
131                        alpha: wgpu::BlendComponent {
132                            src_factor: wgpu::BlendFactor::One,
133                            dst_factor: wgpu::BlendFactor::OneMinusSrcAlpha,
134                            operation: wgpu::BlendOperation::Add,
135                        },
136                    }),
137                    write_mask: ColorWrites::ALL,
138                })],
139                compilation_options: Default::default(),
140            }),
141            primitive: PrimitiveState {
142                topology: wgpu::PrimitiveTopology::TriangleList,
143                strip_index_format: None,
144                front_face: wgpu::FrontFace::Ccw,
145                cull_mode: Some(wgpu::Face::Back),
146                polygon_mode: wgpu::PolygonMode::Fill,
147                unclipped_depth: false,
148                conservative: false,
149            },
150            depth_stencil: None,
151            multisample: MultisampleState {
152                count: 1,
153                mask: !0,
154                alpha_to_coverage_enabled: false,
155            },
156            multiview: None,
157            cache: None,
158        });
159
160        Self {
161            render_pipeline,
162            bind_group_layout,
163            icons: Vec::new(),
164            textures: HashMap::new(),
165            window_width: 1360.0,
166            window_height: 768.0,
167            cached_vertex_buffers: HashMap::new(),
168            cached_index_buffers: HashMap::new(),
169            cached_icon_counts: HashMap::new(),
170        }
171    }

173    pub fn load_texture(
174        &mut self,
175        device: &Device,
176        queue: &wgpu::Queue,
177        path: &str,
178        texture_id: &str,
179    ) -> Result<(), Box<dyn std::error::Error>> {
180        let img = image::open(Path::new(path))?;
181        let rgba = img.to_rgba8();
182        let dimensions = rgba.dimensions();
183
184        let size = wgpu::Extent3d {
185            width: dimensions.0,
186            height: dimensions.1,
187            depth_or_array_layers: 1,
188        };
189
190        let texture = device.create_texture(&wgpu::TextureDescriptor {
191            label: Some(&format!("Icon texture: {}", texture_id)),
192            size,
193            mip_level_count: 1,
194            sample_count: 1,
195            dimension: wgpu::TextureDimension::D2,
196            format: wgpu::TextureFormat::Rgba8Unorm,
197            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
198            view_formats: &[],
199        });
200
201        queue.write_texture(
202            wgpu::TexelCopyTextureInfo {
203                aspect: wgpu::TextureAspect::All,
204                texture: &texture,
205                mip_level: 0,
206                origin: wgpu::Origin3d::ZERO,
207            },
208            &rgba,
209            wgpu::TexelCopyBufferLayout {
210                offset: 0,
211                bytes_per_row: Some(4 * dimensions.0),
212                rows_per_image: Some(dimensions.1),
213            },
214            size,
215        );
216
217        let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
218        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
219            address_mode_u: wgpu::AddressMode::ClampToEdge,
220            address_mode_v: wgpu::AddressMode::ClampToEdge,
221            address_mode_w: wgpu::AddressMode::ClampToEdge,
222            mag_filter: wgpu::FilterMode::Linear,
223            min_filter: wgpu::FilterMode::Linear,
224            mipmap_filter: wgpu::FilterMode::Nearest,
225            ..Default::default()
226        });
227
228        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
229            label: Some(&format!("Icon bind group: {}", texture_id)),
230            layout: &self.bind_group_layout,
231            entries: &[
232                wgpu::BindGroupEntry {
233                    binding: 0,
234                    resource: wgpu::BindingResource::TextureView(&view),
235                },
236                wgpu::BindGroupEntry {
237                    binding: 1,
238                    resource: wgpu::BindingResource::Sampler(&sampler),
239                },
240            ],
241        });
242
243        self.textures
244            .insert(texture_id.to_string(), (texture, view, bind_group));
245
246        Ok(())
247    }

270    pub fn render(&mut self, device: &Device, render_pass: &mut RenderPass) {
271        if self.icons.is_empty() {
272            return;
273        }
274
275        render_pass.set_pipeline(&self.render_pipeline);
276
277        // Group icons by texture to minimize bind group changes
278        let mut icons_by_texture: HashMap<String, Vec<&Icon>> = HashMap::new();
279        for icon in &self.icons {
280            icons_by_texture
281                .entry(icon.texture_id.clone())
282                .or_insert_with(Vec::new)
283                .push(icon);
284        }
285
286        for (texture_id, icons) in icons_by_texture {
287            if let Some((_, _, bind_group)) = self.textures.get(&texture_id) {
288                render_pass.set_bind_group(0, bind_group, &[]);
289
290                // Check if we can reuse cached buffers for this texture
291                let cached_count = self.cached_icon_counts.get(&texture_id).unwrap_or(&0);
292                let need_new_buffers = *cached_count != icons.len();
293
294                if need_new_buffers {
295                    // Create vertices for all icons using this texture
296                    let mut all_vertices = Vec::new();
297                    let mut all_indices = Vec::new();
298
299                    for (icon_index, icon) in icons.iter().enumerate() {
300                        // Convert screen coordinates to normalized device coordinates
301                        let x = (icon.x / self.window_width) * 2.0 - 1.0;
302                        let y = (icon.y / self.window_height) * 2.0 - 1.0;
303                        let width = (icon.width / self.window_width) * 2.0;
304                        let height = (icon.height / self.window_height) * 2.0;
305
306                        // Create vertices for this icon
307                        let vertices = [
308                            // Top-left
309                            IconVertex {
310                                position: [x, y],
311                                uv: [0.0, 1.0], // Flip V coordinate
312                            },
313                            // Top-right
314                            IconVertex {
315                                position: [x + width, y],
316                                uv: [1.0, 1.0], // Flip V coordinate
317                            },
318                            // Bottom-right
319                            IconVertex {
320                                position: [x + width, y + height],
321                                uv: [1.0, 0.0], // Flip V coordinate
322                            },
323                            // Bottom-left
324                            IconVertex {
325                                position: [x, y + height],
326                                uv: [0.0, 0.0], // Flip V coordinate
327                            },
328                        ];
329
330                        // Add vertices to the batch
331                        all_vertices.extend_from_slice(&vertices);
332
333                        // Create indices for this icon (offset by the current vertex count)
334                        let base_index = (icon_index * 4) as u16;
335                        let indices = [
336                            base_index,
337                            base_index + 1,
338                            base_index + 2,
339                            base_index,
340                            base_index + 2,
341                            base_index + 3,
342                        ];
343                        all_indices.extend_from_slice(&indices);
344                    }
345
346                    // Create new vertex buffer for all icons using this texture
347                    let vertex_buffer =
348                        device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
349                            label: Some("Icon Vertex Buffer"),
350                            contents: bytemuck::cast_slice(&all_vertices),
351                            usage: BufferUsages::VERTEX,
352                        });
353
354                    // Create new index buffer for all icons using this texture
355                    let index_buffer =
356                        device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
357                            label: Some("Icon Index Buffer"),
358                            contents: bytemuck::cast_slice(&all_indices),
359                            usage: BufferUsages::INDEX,
360                        });
361
362                    // Cache the new buffers
363                    self.cached_vertex_buffers
364                        .insert(texture_id.clone(), vertex_buffer);
365                    self.cached_index_buffers
366                        .insert(texture_id.clone(), index_buffer);
367                    self.cached_icon_counts
368                        .insert(texture_id.clone(), icons.len());
369                }
370
371                // Use cached buffers
372                if let (Some(vertex_buffer), Some(index_buffer)) = (
373                    self.cached_vertex_buffers.get(&texture_id),
374                    self.cached_index_buffers.get(&texture_id),
375                ) {
376                    render_pass.set_vertex_buffer(0, vertex_buffer.slice(..));
377                    render_pass.set_index_buffer(index_buffer.slice(..), wgpu::IndexFormat::Uint16);
378                    render_pass.draw_indexed(0..(icons.len() * 6) as u32, 0, 0..1);
379                }
380            }
381        }
382    }
