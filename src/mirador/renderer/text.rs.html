102            window_size: size,
103            loaded_fonts: Vec::new(),

105
106        // Try to load the custom font, but don't fail if it doesn't exist
107        match renderer.load_font(
108            "fonts/HankenGrotesk/HankenGrotesk-Medium.ttf",

133    /// Create a new text buffer with the given ID, text, style, and position
134    pub fn create_text_buffer(
135        &mut self,
136        id: &str,
137        text: &str,
138        style: Option<TextStyle>,
139        position: Option<TextPosition>,
140    ) {
141        let mut style = style.unwrap_or_default();
142        let position = position.unwrap_or_default();
143
144        // If the requested font isn't loaded, fall back to a system font
145        if !self.loaded_fonts.contains(&style.font_family) && style.font_family == "HankenGrotesk" {
146            println!(
147                "Font '{}' not found, falling back to system font",
148                style.font_family
149            );
150            style.font_family = "DejaVu Sans".to_string();
151        }
152
153        let metrics = Metrics::new(style.font_size, style.line_height);
154        let mut buffer = Buffer::new(&mut self.font_system, metrics);
155
156        // Set buffer size based on position constraints or window size
157        let width = position.max_width.unwrap_or(self.window_size.width as f32);
158        let height = position
159            .max_height
160            .unwrap_or(self.window_size.height as f32);
161
162        println!(
163            "Creating text buffer '{}' with size: {}x{}, text: '{}'",
164            id, width, height, text
165        );
166        buffer.set_size(&mut self.font_system, Some(width), Some(height));
167
168        let attrs = Attrs::new()
169            .family(Family::Name(&style.font_family))
170            .weight(style.weight)
171            .style(style.style);
172
173        buffer.set_text(&mut self.font_system, text, attrs, Shaping::Advanced);
174        buffer.shape_until_scroll(&mut self.font_system, false);
175
176        // Debug: Check if buffer has any content
177        if buffer.lines.is_empty() {
178            println!("WARNING: Buffer '{}' has no lines after text setting", id);
179        } else {
180            println!("Buffer '{}' created with {} lines", id, buffer.lines.len());
181        }
182
183        let text_buffer = TextBuffer {
184            buffer,
185            style,
186            position,
187            scale: 1.0,
188            visible: true,
189            text_content: text.to_string(),
190        };
191
192        self.text_buffers.insert(id.to_string(), text_buffer);
193        println!("Text buffer '{}' added to collection", id);
194    }

198        let text_buffer = self

205            .weight(text_buffer.style.weight)
206            .style(text_buffer.style.style);
207
208        text_buffer
209            .buffer
210            .set_text(&mut self.font_system, text, attrs, Shaping::Advanced);
211        text_buffer
212            .buffer

214
215        text_buffer.text_content = text.to_string();

219    /// Update the style of an existing buffer
220    pub fn update_style(&mut self, id: &str, mut style: TextStyle) -> Result<(), String> {
221        let text_buffer = self
222            .text_buffers
223            .get_mut(id)
224            .ok_or_else(|| format!("Text buffer '{}' not found", id))?;
225
226        // If the requested font isn't loaded, fall back to a system font
227        if !self.loaded_fonts.contains(&style.font_family) && style.font_family == "HankenGrotesk" {
228            style.font_family = "DejaVu Sans".to_string();
229        }
230
231        // Update metrics if font size or line height changed
232        if text_buffer.style.font_size != style.font_size
233            || text_buffer.style.line_height != style.line_height
234        {
235            let metrics = Metrics::new(style.font_size, style.line_height);
236            text_buffer
237                .buffer
238                .set_metrics(&mut self.font_system, metrics);
239        }
240
241        text_buffer.style = style;
242
243        // Re-apply text with new attributes using stored content
244        let attrs = Attrs::new()
245            .family(Family::Name(&text_buffer.style.font_family))
246            .weight(text_buffer.style.weight)
247            .style(text_buffer.style.style);
248
249        text_buffer.buffer.set_text(
250            &mut self.font_system,
251            &text_buffer.text_content,
252            attrs,
253            Shaping::Advanced,
254        );
255        text_buffer
256            .buffer
257            .shape_until_scroll(&mut self.font_system, false);
258        Ok(())
259    }

290    /// Prepare text rendering for the current frame
291    pub fn prepare(
292        &mut self,
293        device: &Device,
294        queue: &Queue,
295        surface_config: &SurfaceConfiguration,
296    ) -> Result<(), glyphon::PrepareError> {
297        // Collect all visible text areas
298        let text_areas: Vec<TextArea> = self
299            .text_buffers
300            .iter()
301            .filter(|(_, buffer)| buffer.visible)
302            .map(|(_, buffer)| TextArea {
303                buffer: &buffer.buffer,
304                left: buffer.position.x,
305                top: buffer.position.y,
306                scale: buffer.scale * self.window_scale_factor,
307                bounds: TextBounds {
308                    left: buffer.position.x as i32,
309                    top: buffer.position.y as i32,
310                    right: (buffer.position.x
311                        + buffer
312                            .position
313                            .max_width
314                            .unwrap_or(surface_config.width as f32))
315                        as i32,
316                    bottom: (buffer.position.y
317                        + buffer
318                            .position
319                            .max_height
320                            .unwrap_or(surface_config.height as f32))
321                        as i32,
322                },
323                default_color: buffer.style.color,
324                custom_glyphs: &[],
325            })
326            .collect();
327
328        // Prepare the text renderer
329        self.text_renderer.prepare(
330            device,
331            queue,
332            &mut self.font_system,
333            &mut self.atlas,
334            &self.viewport,
335            text_areas,
336            &mut self.swash_cache,
337        )?;
338
339        Ok(())
340    }
