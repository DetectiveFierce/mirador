129    /// Create a new text buffer with the given ID, text, style, and position
130    pub fn create_text_buffer(
131        &mut self,
132        id: &str,
133        text: &str,
134        style: Option<TextStyle>,
135        position: Option<TextPosition>,
136    ) {
137        let mut style = style.unwrap_or_default();
138        let position = position.unwrap_or_default();
139
140        // If the requested font isn't loaded, fall back to a system font
141        if !self.loaded_fonts.contains(&style.font_family) && style.font_family == "HankenGrotesk" {
142            style.font_family = "DejaVu Sans".to_string();
143        }
144
145        let metrics = Metrics::new(style.font_size, style.line_height);
146        let mut buffer = Buffer::new(&mut self.font_system, metrics);
147
148        // Set buffer size based on position constraints or window size
149        let width = position.max_width.unwrap_or(self.window_size.width as f32);
150        let height = position
151            .max_height
152            .unwrap_or(self.window_size.height as f32);
153
154        buffer.set_size(&mut self.font_system, Some(width), Some(height));
155
156        let attrs = Attrs::new()
157            .family(Family::Name(&style.font_family))
158            .weight(style.weight)
159            .style(style.style);
160
161        buffer.set_text(&mut self.font_system, text, attrs, Shaping::Advanced);
162        buffer.shape_until_scroll(&mut self.font_system, false);
163
164        let text_buffer = TextBuffer {
165            buffer,
166            style,
167            position,
168            scale: 1.0,
169            visible: true,
170            text_content: text.to_string(),
171        };
172
173        self.text_buffers.insert(id.to_string(), text_buffer);
174    }

176    /// Update the style of an existing buffer
177    pub fn update_style(&mut self, id: &str, mut style: TextStyle) -> Result<(), String> {
178        let text_buffer = self
179            .text_buffers
180            .get_mut(id)
181            .ok_or_else(|| format!("Text buffer '{}' not found", id))?;
182
183        // If the requested font isn't loaded, fall back to a system font
184        if !self.loaded_fonts.contains(&style.font_family) && style.font_family == "HankenGrotesk" {
185            style.font_family = "DejaVu Sans".to_string();
186        }
187
188        // Update metrics if font size or line height changed
189        if text_buffer.style.font_size != style.font_size
190            || text_buffer.style.line_height != style.line_height
191        {
192            let metrics = Metrics::new(style.font_size, style.line_height);
193            text_buffer
194                .buffer
195                .set_metrics(&mut self.font_system, metrics);
196        }
197
198        text_buffer.style = style;
199
200        // Re-apply text with new attributes using stored content
201        let attrs = Attrs::new()
202            .family(Family::Name(&text_buffer.style.font_family))
203            .weight(text_buffer.style.weight)
204            .style(text_buffer.style.style);
205
206        text_buffer.buffer.set_text(
207            &mut self.font_system,
208            &text_buffer.text_content,
209            attrs,
210            Shaping::Advanced,
211        );
212        text_buffer
213            .buffer
214            .shape_until_scroll(&mut self.font_system, false);
215        Ok(())
216    }

246    pub fn prepare(
247        &mut self,
248        device: &Device,
249        queue: &Queue,
250        _surface_config: &SurfaceConfiguration,
251    ) -> Result<(), glyphon::PrepareError> {
252        let mut text_areas = Vec::new();
253
254        for text_buffer in self.text_buffers.values() {
255            if !text_buffer.visible {
256                continue;
257            }
258
259            let bounds = TextBounds {
260                left: text_buffer.position.x as i32,
261                top: text_buffer.position.y as i32,
262                right: (text_buffer.position.x
263                    + text_buffer
264                        .position
265                        .max_width
266                        .unwrap_or(self.window_size.width as f32)) as i32,
267                bottom: (text_buffer.position.y
268                    + text_buffer
269                        .position
270                        .max_height
271                        .unwrap_or(self.window_size.height as f32)) as i32,
272            };
273
274            let text_area = TextArea {
275                buffer: &text_buffer.buffer,
276                left: text_buffer.position.x,
277                top: text_buffer.position.y,
278                scale: text_buffer.scale,
279                bounds,
280                default_color: text_buffer.style.color,
281                custom_glyphs: &[],
282            };
283
284            text_areas.push(text_area);
285        }
286
287        self.glyph_renderer.prepare(
288            device,
289            queue,
290            &mut self.font_system,
291            &mut self.atlas,
292            &self.viewport,
293            text_areas,
294            &mut self.swash_cache,
295        )
296    }

303    pub fn measure_text(&mut self, text: &str, style: &TextStyle) -> (f32, f32, f32) {
304        let metrics = Metrics::new(style.font_size, style.line_height);
305        let mut buffer = Buffer::new(&mut self.font_system, metrics);
306
307        let attrs = Attrs::new()
308            .family(Family::Name(&style.font_family))
309            .weight(style.weight)
310            .style(style.style);
311
312        buffer.set_text(&mut self.font_system, text, attrs, Shaping::Advanced);
313        buffer.shape_until_scroll(&mut self.font_system, false);
314
315        // Calculate text dimensions from layout runs
316        let mut min_x = f32::MAX;
317        let mut max_x: f32 = 0.0;
318        let mut height: f32 = 0.0;
319
320        for run in buffer.layout_runs() {
321            if let Some(first_glyph) = run.glyphs.first() {
322                min_x = min_x.min(first_glyph.x);
323            }
324            if let Some(last_glyph) = run.glyphs.last() {
325                max_x = max_x.max(last_glyph.x + last_glyph.w);
326            }
327            height += run.line_height;
328        }
329
330        // If no runs, estimate based on text length and font size
331        if min_x == f32::MAX && !text.is_empty() {
332            min_x = 0.0;
333            max_x = text.len() as f32 * style.font_size * 0.6;
334            height = style.line_height;
335        }
336
337        let width = max_x - min_x;
338        (min_x, width, height)
339    }
