1//! # Text Rendering System
2//!
3//! A comprehensive text rendering system built on top of the Glyphon library for WGPU-based applications.
4//! This module provides a high-level interface for rendering text with custom fonts, styles, and positioning
5//! in games or graphical applications.
6//!
7//! ## Features
8//!
9//! - Custom font loading and management
10//! - Dynamic text styling (font size, color, weight, style)
11//! - Flexible positioning with constraints
12//! - Text measurement and layout calculation
13//! - Game-specific UI elements (game over screens, score displays)
14//! - DPI-aware scaling for different screen sizes
15//! - Multiple text buffer management with unique IDs
16//!
17//! ## Usage
18//!
19//! ```rust
20//! // Create a text renderer
21//! let mut text_renderer = TextRenderer::new(&device, &queue, surface_format, &window);
22//!
23//! // Create a text buffer
24//! text_renderer.create_text_buffer(
25//!     "my_text",
26//!     "Hello, World!",
27//!     Some(TextStyle::default()),
28//!     Some(TextPosition::default())
29//! );
30//!
31//! // Render in your main loop
32//! text_renderer.prepare(&device, &queue, &surface_config)?;
33//! text_renderer.render(&mut render_pass)?;
34//! ```

259    /// Creates a new text buffer with the specified content, style, and position.
260    ///
261    /// This method creates a new text buffer that can be rendered to the screen.
262    /// Each buffer is identified by a unique string ID that can be used to update
263    /// or reference the buffer later.
264    ///
265    /// # Arguments
266    ///
267    /// * `id` - Unique identifier for this text buffer
268    /// * `text` - The text content to display
269    /// * `style` - Optional text styling (uses default if None)
270    /// * `position` - Optional positioning (uses default if None)
271    ///
272    /// # Behavior
273    ///
274    /// - If the requested font family isn't loaded, falls back to "DejaVu Sans"
275    /// - Automatically shapes the text for proper rendering
276    /// - Sets buffer size based on position constraints or window size
277    ///
278    /// # Example
279    ///
280    /// ```rust
281    /// let style = TextStyle {
282    ///     font_size: 24.0,
283    ///     color: Color::rgb(255, 0, 0),
284    ///     ..Default::default()
285    /// };
286    /// let position = TextPosition {
287    ///     x: 100.0,
288    ///     y: 50.0,
289    ///     max_width: Some(300.0),
290    ///     ..Default::default()
291    /// };
292    /// renderer.create_text_buffer("title", "Hello World", Some(style), Some(position));
293    /// ```
294    pub fn create_text_buffer(
295        &mut self,
296        id: &str,
297        text: &str,
298        style: Option<TextStyle>,
299        position: Option<TextPosition>,
300    ) {
301        let mut style = style.unwrap_or_default();
302        let position = position.unwrap_or_default();
303
304        // If the requested font isn't loaded, fall back to a system font
305        if !self.loaded_fonts.contains(&style.font_family) && style.font_family == "Hanken Grotesk"
306        {
307            style.font_family = "DejaVu Sans".to_string();
308        }
309
310        let metrics = Metrics::new(style.font_size, style.line_height);
311        let mut buffer = Buffer::new(&mut self.font_system, metrics);
312
313        // Set buffer size based on position constraints or window size
314        let width = position.max_width.unwrap_or(self.window_size.width as f32);
315        let height = position
316            .max_height
317            .unwrap_or(self.window_size.height as f32);
318
319        buffer.set_size(&mut self.font_system, Some(width), Some(height));
320
321        let attrs = Attrs::new()
322            .family(Family::Name(&style.font_family))
323            .weight(style.weight)
324            .style(style.style);
325
326        buffer.set_text(&mut self.font_system, text, attrs, Shaping::Advanced);
327        buffer.shape_until_scroll(&mut self.font_system, false);
328
329        let text_buffer = TextBuffer {
330            buffer,
331            style,
332            position,
333            scale: 1.0,
334            visible: true,
335            text_content: text.to_string(),
336        };
337
338        self.text_buffers.insert(id.to_string(), text_buffer);
339    }

341    /// Updates the visual style of an existing text buffer.
342    ///
343    /// This method allows you to change the appearance of existing text without
344    /// recreating the entire buffer. The text content remains the same but is
345    /// re-shaped with the new styling attributes.
346    ///
347    /// # Arguments
348    ///
349    /// * `id` - The unique identifier of the text buffer to update
350    /// * `style` - The new text style to apply
351    ///
352    /// # Returns
353    ///
354    /// * `Ok(())` if the style was updated successfully
355    /// * `Err(String)` if the text buffer with the given ID was not found
356    ///
357    /// # Behavior
358    ///
359    /// - Falls back to system fonts if the requested font isn't loaded
360    /// - Updates metrics if font size or line height changed
361    /// - Re-shapes the text with new attributes
362    ///
363    /// # Example
364    ///
365    /// ```rust
366    /// let new_style = TextStyle {
367    ///     font_size: 32.0,
368    ///     color: Color::rgb(0, 255, 0),
369    ///     weight: Weight::BOLD,
370    ///     ..Default::default()
371    /// };
372    /// renderer.update_style("title", new_style)?;
373    /// ```
374    pub fn update_style(&mut self, id: &str, mut style: TextStyle) -> Result<(), String> {
375        let text_buffer = self
376            .text_buffers
377            .get_mut(id)
378            .ok_or_else(|| format!("Text buffer '{}' not found", id))?;
379
380        // If the requested font isn't loaded, fall back to a system font
381        if !self.loaded_fonts.contains(&style.font_family) && style.font_family == "Hanken Grotesk"
382        {
383            style.font_family = "DejaVu Sans".to_string();
384        }
385
386        // Update metrics if font size or line height changed
387        if text_buffer.style.font_size != style.font_size
388            || text_buffer.style.line_height != style.line_height
389        {
390            let metrics = Metrics::new(style.font_size, style.line_height);
391            text_buffer
392                .buffer
393                .set_metrics(&mut self.font_system, metrics);
394        }
395
396        text_buffer.style = style;
397
398        // Re-apply text with new attributes using stored content
399        let attrs = Attrs::new()
400            .family(Family::Name(&text_buffer.style.font_family))
401            .weight(text_buffer.style.weight)
402            .style(text_buffer.style.style);
403
404        text_buffer.buffer.set_text(
405            &mut self.font_system,
406            &text_buffer.text_content,
407            attrs,
408            Shaping::Advanced,
409        );
410        text_buffer
411            .buffer
412            .shape_until_scroll(&mut self.font_system, false);
413        Ok(())
414    }

497    /// Prepares all visible text buffers for rendering.
498    ///
499    /// This method must be called before rendering to update the texture atlas
500    /// with any changes to text content, styling, or positioning. It processes
501    /// all visible text buffers and prepares them for GPU rendering.
502    ///
503    /// # Arguments
504    ///
505    /// * `device` - WGPU device for GPU operations
506    /// * `queue` - WGPU command queue
507    /// * `_surface_config` - Surface configuration (currently unused)
508    ///
509    /// # Returns
510    ///
511    /// * `Ok(())` if preparation was successful
512    /// * `Err(glyphon::PrepareError)` if preparation failed
513    ///
514    /// # Behavior
515    ///
516    /// - Only processes visible text buffers
517    /// - Calculates text bounds based on position and constraints
518    /// - Updates the glyph texture atlas as needed
519    ///
520    /// # Example
521    ///
522    /// ```rust
523    /// // In your render loop
524    /// renderer.prepare(&device, &queue, &surface_config)?;
525    /// renderer.render(&mut render_pass)?;
526    /// ```
527    pub fn prepare(
528        &mut self,
529        device: &Device,
530        queue: &Queue,
531        _surface_config: &SurfaceConfiguration,
532    ) -> Result<(), glyphon::PrepareError> {
533        let mut text_areas = Vec::new();
534
535        for text_buffer in self.text_buffers.values() {
536            if !text_buffer.visible {
537                continue;
538            }
539
540            let bounds = TextBounds {
541                left: text_buffer.position.x as i32,
542                top: text_buffer.position.y as i32,
543                right: (text_buffer.position.x
544                    + text_buffer
545                        .position
546                        .max_width
547                        .unwrap_or(self.window_size.width as f32)) as i32,
548                bottom: (text_buffer.position.y
549                    + text_buffer
550                        .position
551                        .max_height
552                        .unwrap_or(self.window_size.height as f32)) as i32,
553            };
554
555            let text_area = TextArea {
556                buffer: &text_buffer.buffer,
557                left: text_buffer.position.x,
558                top: text_buffer.position.y,
559                scale: text_buffer.scale,
560                bounds,
561                default_color: text_buffer.style.color,
562                custom_glyphs: &[],
563            };
564
565            text_areas.push(text_area);
566        }
567
568        self.glyph_renderer.prepare(
569            device,
570            queue,
571            &mut self.font_system,
572            &mut self.atlas,
573            &self.viewport,
574            text_areas,
575            &mut self.swash_cache,
576        )
577    }

605    /// Measures the dimensions of text without creating a buffer.
606    ///
607    /// This utility method calculates how much space text will occupy when
608    /// rendered with the given style. Useful for layout calculations and
609    /// positioning decisions.
610    ///
611    /// # Arguments
612    ///
613    /// * `text` - The text content to measure
614    /// * `style` - The text style to use for measurement
615    ///
616    /// # Returns
617    ///
618    /// A tuple containing `(min_x, width, height)`:
619    /// - `min_x`: The leftmost x coordinate of the text
620    /// - `width`: The total width of the text
621    /// - `height`: The total height of the text
622    ///
623    /// # Behavior
624    ///
625    /// - Creates a temporary buffer for measurement
626    /// - Calculates dimensions from layout runs
627    /// - Provides fallback estimates for empty text
628    ///
629    /// # Example
630    ///
631    /// ```rust
632    /// let style = TextStyle::default();
633    /// let (min_x, width, height) = renderer.measure_text("Hello World", &style);
634    /// println!("Text dimensions: {}x{} at x={}", width, height, min_x);
635    /// ```
636    pub fn measure_text(&mut self, text: &str, style: &TextStyle) -> (f32, f32, f32) {
637        let metrics = Metrics::new(style.font_size, style.line_height);
638        let mut buffer = Buffer::new(&mut self.font_system, metrics);
639
640        let attrs = Attrs::new()
641            .family(Family::Name(&style.font_family))
642            .weight(style.weight)
643            .style(style.style);
644
645        buffer.set_text(&mut self.font_system, text, attrs, Shaping::Advanced);
646        buffer.shape_until_scroll(&mut self.font_system, false);
647
648        // Calculate text dimensions from layout runs
649        let mut min_x = f32::MAX;
650        let mut max_x: f32 = 0.0;
651        let mut height: f32 = 0.0;
652
653        for run in buffer.layout_runs() {
654            if let Some(first_glyph) = run.glyphs.first() {
655                min_x = min_x.min(first_glyph.x);
656            }
657            if let Some(last_glyph) = run.glyphs.last() {
658                max_x = max_x.max(last_glyph.x + last_glyph.w);
659            }
660            height += run.line_height;
661        }
662
663        // If no runs, estimate based on text length and font size
664        if min_x == f32::MAX && !text.is_empty() {
665            min_x = 0.0;
666            max_x = text.len() as f32 * style.font_size * 0.6;
667            height = style.line_height;
668        }
669
670        let width = max_x - min_x;
671        (min_x, width, height)
672    }

1357        text: &str,
1358        mut style: TextStyle,
1359        position: TextPosition,
1360    ) -> Result<(), String> {
1361        // Validate input parameters
1362        if id.is_empty() {
1363            return Err("Text buffer ID cannot be empty".to_string());
1364        }
1365
1366        if text.is_empty() {
1367            return Err("Text content cannot be empty".to_string());

1372            return Err("Font size must be greater than 0".to_string());

1376            return Err("Line height must be greater than 0".to_string());

1378

1380        if position.x < 0.0 {
1381            return Err("X position cannot be negative".to_string());
1382        }
1383
1384        if position.y < 0.0 {
1385            return Err("Y position cannot be negative".to_string());
1386        }
1387
1388        let text_buffer = self
1389            .text_buffers
1390            .get_mut(id)
1391            .ok_or_else(|| format!("Text buffer '{}' not found", id))?;
1392
1393        // If the requested font isn't loaded, fall back to a system font
1394        if !self.loaded_fonts.contains(&style.font_family) && style.font_family == "Hanken Grotesk"
1395        {
1396            style.font_family = "DejaVu Sans".to_string();
1397        }
1398
1399        // Update metrics if font size or line height changed
1400        if text_buffer.style.font_size != style.font_size
1401            || text_buffer.style.line_height != style.line_height
1402        {
1403            let metrics = Metrics::new(style.font_size, style.line_height);
1404            text_buffer
1405                .buffer
1406                .set_metrics(&mut self.font_system, metrics);
1407        }

1410        if text_buffer.position.max_width != position.max_width
1411            || text_buffer.position.max_height != position.max_height
1412        {
1413            let width = position.max_width.unwrap_or(self.window_size.width as f32);
1414            let height = position
1415                .max_height
1416                .unwrap_or(self.window_size.height as f32);
1417            text_buffer
1418                .buffer
1419                .set_size(&mut self.font_system, Some(width), Some(height));
1420        }
1421
1422        // Update all properties
1423        text_buffer.text_content = text.to_string();
1424        text_buffer.style = style;
1425        text_buffer.position = position;
1426
1427        // Re-apply text with new attributes
1428        let attrs = Attrs::new()
1429            .family(Family::Name(&text_buffer.style.font_family))
1430            .weight(text_buffer.style.weight)
1431            .style(text_buffer.style.style);

1434            .buffer

1436        text_buffer

1684            // Check font family

1686                return Err(format!("Buffer '{}' has empty font family", id));
1687            }
1688
1689            // Check font size
1690            if buffer.style.font_size <= 0.0 {
1691                return Err(format!(
1692                    "Buffer '{}' has invalid font size: {}",
1693                    id, buffer.style.font_size
1694                ));
1695            }
1696
1697            // Check line height
1698            if buffer.style.line_height <= 0.0 {
1699                return Err(format!(
1700                    "Buffer '{}' has invalid line height: {}",
1701                    id, buffer.style.line_height
1702                ));
1703            }
1704
1705            // Check position
1706            if buffer.position.x < 0.0 {
1707                return Err(format!(
1708                    "Buffer '{}' has negative X position: {}",
1709                    id, buffer.position.x
1710                ));
1711            }
1712
1713            if buffer.position.y < 0.0 {
1714                return Err(format!(
1715                    "Buffer '{}' has negative Y position: {}",
1716                    id, buffer.position.y
1717                ));
1718            }
1719
1720            // Check scale
1721            if buffer.scale <= 0.0 {
1722                return Err(format!(
1723                    "Buffer '{}' has invalid scale: {}",
1724                    id, buffer.scale
1725                ));
1726            }
1727
1728            // Check text content
1729            if buffer.text_content.is_empty() {
1730                return Err(format!("Buffer '{}' has empty text content", id));
1731            }
1732        }
1733
1734        Ok(())
