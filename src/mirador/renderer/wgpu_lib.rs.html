1//! WGPU-based renderer for the Mirador game.
2//!
3//! This module provides [`WgpuRenderer`], which manages all GPU resources, pipelines, and rendering
4//! logic for the main game scene, background, and title screen. It handles initialization of WGPU,
5//! creation of vertex/uniform buffers, pipelines, and orchestrates the rendering of the maze, player,
6//! animated background, and UI overlays.
7//!
8//! # Features
9//! - Loads maze geometry and floor/wall vertices
10//! - Renders a starfield background and animated title screen
11//! - Handles depth buffering and uniform updates for camera/player movement
12//! - Integrates with egui for UI overlays
13//!
14//! # Usage
15//! Create a [`WgpuRenderer`] via [`WgpuRenderer::new`] and call [`WgpuRenderer::update_canvas`]
16//! each frame to render the current game state.

33/// Main WGPU renderer for the Mirador game.
34///
35/// This struct manages all GPU resources, pipelines, and rendering logic for the game scene,
36/// including the maze, player, animated background, and title screen.
37///
38/// # Fields
39/// - `surface`: The WGPU surface for presenting rendered frames.
40/// - `device`: The WGPU device for resource creation.
41/// - `queue`: The WGPU queue for submitting commands.
42/// - `surface_config`: The surface configuration (format, size, etc.).
43/// - `pipeline`: Main render pipeline for the maze and floor.
44/// - `vertex_buffer`: Combined vertex buffer for floor and wall geometry.
45/// - `num_vertices`: Number of vertices to draw.
46/// - `uniform_buffer`: Uniform buffer for camera/view/projection matrices.
47/// - `uniform_bind_group`: Bind group for the uniform buffer.
48/// - `depth_texture`: Optional depth texture for depth testing.
49/// - `background`: StarRenderer for animated starfield background.
50/// - `animation_renderer`: Renderer for the title screen maze and loading bar.
51pub struct WgpuRenderer {
52    /// The WGPU surface for presenting rendered frames.
53    pub surface: wgpu::Surface<'static>,
54    /// The WGPU device for resource creation.
55    pub device: wgpu::Device,
56    /// The WGPU queue for submitting commands.
57    pub queue: wgpu::Queue,
58    /// The surface configuration (format, size, etc.).
59    pub surface_config: wgpu::SurfaceConfiguration,
60    /// Main render pipeline for the maze and floor.
61    pub pipeline: wgpu::RenderPipeline,
62    /// Combined vertex buffer for floor and wall geometry.
63    pub vertex_buffer: wgpu::Buffer,
64    /// Number of vertices to draw.
65    pub num_vertices: u32,
66    /// Uniform buffer for camera/view/projection matrices.
67    pub uniform_buffer: wgpu::Buffer,
68    /// Bind group for the uniform buffer.
69    pub uniform_bind_group: wgpu::BindGroup,
70    /// Optional depth texture for depth testing.
71    pub depth_texture: Option<wgpu::Texture>,
72    /// StarRenderer for animated starfield background.
73    pub background: StarRenderer,
74    /// Renderer for the title screen maze and loading bar.
75    pub animation_renderer: AnimationRenderer,
76    /// Renderer for debug information.
77    /// Whether to render bounding boxes for debugging.
78    pub debug_render_bounding_boxes: bool,
79    /// Vertex buffer for the debug renderer
80    pub debug_vertex_buffer: Option<wgpu::Buffer>,
81    pub debug_vertex_count: usize,
82}

84impl WgpuRenderer {
85    /// Initializes a new [`WgpuRenderer`] and all associated GPU resources.
86    ///
87    /// # Arguments
88    /// - `instance`: The WGPU instance.
89    /// - `surface`: The WGPU surface for presentation.
90    /// - `width`: Initial width of the surface.
91    /// - `height`: Initial height of the surface.
92    ///
93    /// # Returns
94    /// A fully initialized [`WgpuRenderer`] ready for rendering.
95    pub async fn new(
96        instance: &wgpu::Instance,
97        surface: wgpu::Surface<'static>,
98        width: u32,
99        height: u32,
100    ) -> Self {
101        let power_pref = wgpu::PowerPreference::default();
102        let adapter = instance
103            .request_adapter(&wgpu::RequestAdapterOptions {
104                power_preference: power_pref,
105                force_fallback_adapter: false,
106                compatible_surface: Some(&surface),
107            })
108            .await
109            .expect("Failed to find an appropriate adapter");
110
111        let features = wgpu::Features::empty();
112        let (device, queue) = adapter
113            .request_device(
114                &wgpu::DeviceDescriptor {
115                    label: None,
116                    required_features: features,
117                    required_limits: Default::default(),
118                    memory_hints: Default::default(),
119                },
120                None,
121            )
122            .await
123            .expect("Failed to create device");
124
125        let swapchain_capabilities = surface.get_capabilities(&adapter);
126        let selected_format = wgpu::TextureFormat::Bgra8UnormSrgb;
127        let swapchain_format = swapchain_capabilities
128            .formats
129            .iter()
130            .find(|d| **d == selected_format)
131            .expect("failed to select proper surface texture format!");
132
133        let surface_config = wgpu::SurfaceConfiguration {
134            usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
135            format: *swapchain_format,
136            width,
137            height,
138            present_mode: wgpu::PresentMode::AutoVsync,
139            desired_maximum_frame_latency: 0,
140            alpha_mode: swapchain_capabilities.alpha_modes[0],
141            view_formats: vec![],
142        };
143
144        surface.configure(&device, &surface_config);
145
146        let uniforms = Uniforms::new();
147        let uniform_buffer = uniforms.create_buffer(&device);
148        let (uniform_bind_group, uniform_bind_group_layout) =
149            uniforms.create_bind_group(&uniform_buffer, &device);
150
151        let pipeline = PipelineBuilder::new(&device, surface_config.format)
152            .with_label("Main Pipeline")
153            .with_shader(include_str!("shader.wgsl"))
154            .with_vertex_buffer(Vertex::desc())
155            .with_bind_group_layout(&uniform_bind_group_layout)
156            .with_blend_state(wgpu::BlendState {
157                color: wgpu::BlendComponent {
158                    src_factor: wgpu::BlendFactor::SrcAlpha,
159                    dst_factor: wgpu::BlendFactor::OneMinusSrcAlpha,
160                    operation: wgpu::BlendOperation::Add,
161                },
162                alpha: wgpu::BlendComponent::OVER,
163            })
164            .with_no_culling()
165            .with_depth_stencil(wgpu::DepthStencilState {
166                depth_write_enabled: true,
167                depth_compare: wgpu::CompareFunction::Less,
168                format: wgpu::TextureFormat::Depth24Plus,
169                stencil: wgpu::StencilState::default(),
170                bias: wgpu::DepthBiasState::default(),
171            })
172            .build();
173
174        // Load wall grid from file
175        let (maze_grid, exit_cell) = parse_maze_file("src/maze/saved-mazes/test.mz");
176
177        let mut floor_vertices = Vertex::create_floor_vertices(&maze_grid, exit_cell);
178
179        // Generate wall geometry
180        let mut wall_vertices = Vertex::create_wall_vertices(&maze_grid);
181
182        // Append wall vertices to floor
183        floor_vertices.append(&mut wall_vertices);
184
185        let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
186            label: Some("Combined Vertex Buffer"),
187            contents: bytemuck::cast_slice(&floor_vertices),
188            usage: wgpu::BufferUsages::VERTEX,
189        });
190
191        let star_renderer = stars::create_star_renderer(&device, &surface_config, 100);
192        let animation_renderer = maze_animation::AnimationRenderer::new(&device, &surface_config);
193
194        let debug_render_bounding_boxes = false;
195        let debug_vertex_buffer = None;
196        let debug_vertex_count = 0;
197        Self {
198            surface,
199            device,
200            queue,
201            surface_config,
202            uniform_buffer,
203            uniform_bind_group,
204            pipeline,
205            vertex_buffer,
206            num_vertices: floor_vertices.len() as u32,
207            depth_texture: None,
208            background: star_renderer,
209            animation_renderer,
210            debug_render_bounding_boxes,
211            debug_vertex_buffer,
212            debug_vertex_count,
213        }
214    }

216    /// Renders the current frame to the surface, including the maze, player, background, and UI.
217    ///
218    /// # Arguments
219    /// - `window`: The window for retrieving DPI scaling.
220    /// - `ui_state`: Current UI state (colors, etc.).
221    /// - `encoder`: Command encoder for submitting draw commands.
222    /// - `start_time`: Start time for animation timing.
223    /// - `player`: Reference to the current player state.
224    /// - `title`: If true, renders the title screen; otherwise renders the main game scene.
225    ///
226    /// # Returns
227    /// - `Ok((TextureView, ScreenDescriptor, SurfaceTexture))` on success.
228    /// - `Err(String)` if the surface is outdated or unavailable.
229    pub fn update_canvas(
230        &mut self,
231        window: &winit::window::Window,
232        ui_state: &UiState,
233        encoder: &mut wgpu::CommandEncoder,
234        start_time: std::time::Instant,
235        player: &Player,
236        title: bool,
237    ) -> Result<(TextureView, ScreenDescriptor, SurfaceTexture), String> {
238        let surface_texture_obj = self.surface.get_current_texture();
239
240        let surface_texture = match surface_texture_obj {
241            Err(wgpu::SurfaceError::Outdated) => {
242                // Ignoring outdated to allow resizing and minimization
243                println!("wgpu surface outdated");
244                return Err("wgpu surface outdated".to_string());
245            }
246            Err(_) => {
247                surface_texture_obj.expect("Failed to acquire next swap chain texture");
248                return Err("Failed to acquire next swap chain texture".to_string());
249            }
250            Ok(surface_texture) => surface_texture,
251        };
252
253        let surface_view = surface_texture
254            .texture
255            .create_view(&wgpu::TextureViewDescriptor::default());
256
257        let screen_descriptor = ScreenDescriptor {
258            size_in_pixels: [self.surface_config.width, self.surface_config.height],
259            pixels_per_point: window.scale_factor() as f32,
260        };
261        // If we don't have a depth texture OR if its size is different
262        // from the canvasTexture when make a new depth texture
263        let (width, height) = (self.surface_config.width, self.surface_config.height);
264
265        let depth_texture_view = {
266            if self.depth_texture.is_none()
267                || self.depth_texture.as_ref().unwrap().width() != width
268                || self.depth_texture.as_ref().unwrap().height() != height
269            {
270                if let Some(depth_texture) = self.depth_texture.take() {
271                    // Manually drop the texture to free up resources
272                    drop(depth_texture);
273                }
274
275                self.depth_texture = Some(self.device.create_texture(&wgpu::TextureDescriptor {
276                    label: Some("Depth Texture"),
277                    size: wgpu::Extent3d {
278                        width,
279                        height,
280                        depth_or_array_layers: 1,
281                    },
282                    mip_level_count: 1,
283                    sample_count: 1,
284                    dimension: wgpu::TextureDimension::D2,
285                    format: wgpu::TextureFormat::Depth24Plus,
286                    usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
287                    view_formats: &[],
288                }));
289            }
290            self.depth_texture
291                .as_ref()
292                .unwrap()
293                .create_view(&wgpu::TextureViewDescriptor::default())
294        };
295
296        if title {
297            {
298                let mut render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
299                    label: Some("Render Pass"),
300                    color_attachments: &[Some(wgpu::RenderPassColorAttachment {
301                        view: &surface_view,
302                        resolve_target: None,
303                        ops: wgpu::Operations {
304                            load: wgpu::LoadOp::Clear(wgpu::Color {
305                                r: 0.1,
306                                g: 0.2,
307                                b: 0.3,
308                                a: 1.0,
309                            }),
310                            store: wgpu::StoreOp::Store,
311                        },
312                    })],
313                    depth_stencil_attachment: None,
314                    occlusion_query_set: None,
315                    timestamp_writes: None,
316                });
317
318                // Render title screen using new component architecture
319                self.animation_renderer.render(&mut render_pass, window);
320            }
321
322            return Ok((surface_view, screen_descriptor, surface_texture)); // <- This return was already there
323        }
324        if !title {
325            let clear_pass_desc = wgpu::RenderPassDescriptor {
326                label: Some("Clear Pass"),
327                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
328                    view: &surface_view,
329                    resolve_target: None,
330                    ops: wgpu::Operations {
331                        load: wgpu::LoadOp::Clear(wgpu::Color {
332                            r: ui_state.r as f64,
333                            g: ui_state.g as f64,
334                            b: ui_state.b as f64,
335                            a: 1.0,
336                        }),
337                        store: wgpu::StoreOp::Store,
338                    },
339                })],
340                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
341                    view: &depth_texture_view,
342                    depth_ops: Some(wgpu::Operations {
343                        load: wgpu::LoadOp::Clear(1.0),
344                        store: wgpu::StoreOp::Store,
345                    }),
346                    stencil_ops: None,
347                }),
348                occlusion_query_set: None,
349                timestamp_writes: None,
350            };
351            let clear_pass = encoder.begin_render_pass(&clear_pass_desc);
352            // No draw calls needed - just clears
353
354            drop(clear_pass)
355        }
356
357        {
358            let elapsed_time = start_time.elapsed().as_secs_f32();
359            self.background
360                .update_background_color(&self.queue, [ui_state.r, ui_state.g, ui_state.b, 1.0]);
361            self.background.update_star_time(&self.queue, elapsed_time);
362            self.queue.write_buffer(
363                &self.background.time_buffer,
364                0,
365                bytemuck::cast_slice(&[elapsed_time]),
366            );
367
368            let star_pass_desc = wgpu::RenderPassDescriptor {
369                label: Some("Star Pass"),
370                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
371                    view: &surface_view,
372                    resolve_target: None,
373                    ops: wgpu::Operations {
374                        load: wgpu::LoadOp::Load, // Preserve cleared background
375                        store: wgpu::StoreOp::Store,
376                    },
377                })],
378                depth_stencil_attachment: None, // No depth testing for stars
379                occlusion_query_set: None,
380                timestamp_writes: None,
381            };
382            let mut star_pass = encoder.begin_render_pass(&star_pass_desc);
383            star_pass.set_pipeline(&self.background.pipeline);
384            star_pass.set_bind_group(0, &self.background.uniform_bind_group, &[]);
385            star_pass.set_vertex_buffer(0, self.background.vertex_buffer.slice(..));
386            star_pass.set_index_buffer(
387                self.background.index_buffer.slice(..),
388                wgpu::IndexFormat::Uint16,
389            );
390            star_pass.draw_indexed(0..self.background.num_indices, 0, 0..1);
391            drop(star_pass);
392        };
393
394        let aspect = width as f32 / height as f32;
395
396        // Step 1: Model Matrix - Just identity since the floor is at world origin
397        let model_matrix = Mat4::identity();
398
399        // Step 2: View Matrix - Based on player's camera position and orientation
400        let view_matrix = player.get_view_matrix();
401
402        // Step 3: Projection Matrix - Using FOV from UI state
403        let projection_matrix = Mat4::perspective(
404            deg_to_rad(player.fov),
405            aspect,
406            0.1,    // zNear
407            2000.0, // zFar
408        );
409
410        // Step 4: Combine matrices: Projection * View * Model
411        let final_mvp_matrix = projection_matrix
412            .multiply(&view_matrix)
413            .multiply(&model_matrix);
414
415        let uniforms = Uniforms {
416            matrix: final_mvp_matrix.into(), // Access the inner `[[f32; 4]; 4]` array
417        };
418        // upload the uniform values to the uniform buffer
419        self.queue
420            .write_buffer(&self.uniform_buffer, 0, uniforms.as_bytes());
421
422        let main_pass_desc = wgpu::RenderPassDescriptor {
423            label: Some("Main Render Pass"),
424            color_attachments: &[Some(wgpu::RenderPassColorAttachment {
425                view: &surface_view,
426                resolve_target: None,
427                ops: wgpu::Operations {
428                    load: wgpu::LoadOp::Load,
429                    store: wgpu::StoreOp::Store,
430                },
431            })],
432            depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
433                view: &depth_texture_view,
434                depth_ops: Some(wgpu::Operations {
435                    load: wgpu::LoadOp::Load,
436                    store: wgpu::StoreOp::Store,
437                }),
438                stencil_ops: None,
439            }),
440            occlusion_query_set: None,
441            timestamp_writes: None,
442        };
443        let mut main_pass = encoder.begin_render_pass(&main_pass_desc);
444        main_pass.set_pipeline(&self.pipeline);
445        main_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
446        main_pass.set_bind_group(0, &self.uniform_bind_group, &[]);
447        main_pass.draw(0..self.num_vertices, 0..1);
448
449        // Inside your render method:
450        if self.debug_render_bounding_boxes && self.debug_vertex_count > 0 {
451            if let Some(debug_buffer) = &self.debug_vertex_buffer {
452                main_pass.set_vertex_buffer(0, debug_buffer.slice(..));
453                main_pass.draw(0..self.debug_vertex_count as u32, 0..1);
454            }
455        }
456
457        Ok((surface_view, screen_descriptor, surface_texture))
458    }

460    pub fn update_debug_vertices(&mut self, collision_system: &CollisionSystem) {
461        // Skip if debug rendering is disabled
462        if !self.debug_render_bounding_boxes {
463            self.debug_vertex_count = 0;
464            return;
465        }
466
467        // Collect only wall face AABBs, not the entire BVH hierarchy
468        let debug_vertices = collect_wall_face_debug_vertices(&collision_system.bvh);
469
470        // Create or update the debug vertex buffer
471        self.debug_vertex_count = debug_vertices.len();
472        if self.debug_vertex_count > 0 {
473            let debug_buffer = self
474                .device
475                .create_buffer_init(&wgpu::util::BufferInitDescriptor {
476                    label: Some("Debug Vertex Buffer"),
477                    contents: bytemuck::cast_slice(&debug_vertices),
478                    usage: wgpu::BufferUsages::VERTEX,
479                });
480            self.debug_vertex_buffer = Some(debug_buffer);
481        }
482    }
