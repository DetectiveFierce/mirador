1//! WGPU-based renderer for the Mirador game.
2//!
3//! This module provides [`WgpuRenderer`], which manages all GPU resources, pipelines, and rendering
4//! logic for the main game scene, background, and title screen. It handles initialization of WGPU,
5//! creation of vertex/uniform buffers, pipelines, and orchestrates the rendering of the maze, player,
6//! animated background, and UI overlays.
7//!
8//! # Features
9//! - Loads maze geometry and floor/wall vertices
10//! - Renders a starfield background and animated title screen
11//! - Handles depth buffering and uniform updates for camera/player movement
12//! - Integrates with egui for UI overlays
13//!
14//! # Usage
15//! Create a [`WgpuRenderer`] via [`WgpuRenderer::new`] and call [`WgpuRenderer::update_canvas`]
16//! each frame to render the current game state.
17use crate::game::player::Player;
18use crate::maze::maze_animation::AnimationRenderer;
19use crate::renderer::render_components::GameRenderer;
20use crate::ui::ui_panel::UiState;
21use egui_wgpu::ScreenDescriptor;
22use egui_wgpu::wgpu;
23use egui_wgpu::wgpu::{SurfaceTexture, TextureView};

25/// Main WGPU renderer for the Mirador game.
26///
27/// This struct manages all GPU resources, pipelines, and rendering logic for the game scene,
28/// including the maze, player, animated background, and title screen.
29///
30/// # Fields
31/// - `surface`: The WGPU surface for presenting rendered frames.
32/// - `surface_config`: The surface configuration (format, size, etc.).
33/// - `device`: The WGPU device for resource creation.
34/// - `queue`: The WGPU queue for submitting commands.
35/// - `game_renderer`: Main render pipeline for the maze and floor.
36/// - `animation_renderer`: Renderer for the title screen maze and loading bar.
37pub struct WgpuRenderer {
38    /// The WGPU surface for presenting rendered frames.
39    pub surface: wgpu::Surface<'static>,
40    /// The surface configuration (format, size, etc.).
41    pub surface_config: wgpu::SurfaceConfiguration,
42    /// The WGPU device for resource creation.
43    pub device: wgpu::Device,
44    /// The WGPU queue for submitting commands.
45    pub queue: wgpu::Queue,
46    /// Main render pipeline for the maze and floor.
47    pub game_renderer: GameRenderer,
48    /// Renderer for the title screen maze and loading bar.
49    pub animation_renderer: AnimationRenderer,
50}
51
52impl WgpuRenderer {
53    /// Initializes a new [`WgpuRenderer`] and all associated GPU resources.
54    ///
55    /// # Arguments
56    /// - `instance`: The WGPU instance.
57    /// - `surface`: The WGPU surface for presentation.
58    /// - `width`: Initial width of the surface.
59    /// - `height`: Initial height of the surface.
60    ///
61    /// # Returns
62    /// A fully initialized [`WgpuRenderer`] ready for rendering.
63    pub async fn new(
64        instance: &wgpu::Instance,
65        surface: wgpu::Surface<'static>,
66        width: u32,
67        height: u32,
68    ) -> Self {
69        let power_pref = wgpu::PowerPreference::default();
70        let adapter = instance
71            .request_adapter(&wgpu::RequestAdapterOptions {
72                power_preference: power_pref,
73                force_fallback_adapter: false,
74                compatible_surface: Some(&surface),
75            })
76            .await
77            .expect("Failed to find an appropriate adapter");
78
79        let features = wgpu::Features::empty();
80        let (device, queue) = adapter
81            .request_device(
82                &wgpu::DeviceDescriptor {
83                    label: None,
84                    required_features: features,
85                    required_limits: Default::default(),
86                    memory_hints: Default::default(),
87                },
88                None,
89            )
90            .await
91            .expect("Failed to create device");
92
93        let swapchain_capabilities = surface.get_capabilities(&adapter);
94        let selected_format = wgpu::TextureFormat::Bgra8UnormSrgb;
95        let swapchain_format = swapchain_capabilities
96            .formats
97            .iter()
98            .find(|d| **d == selected_format)
99            .expect("failed to select proper surface texture format!");
100
101        let surface_config = wgpu::SurfaceConfiguration {
102            usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
103            format: *swapchain_format,
104            width,
105            height,
106            present_mode: wgpu::PresentMode::AutoVsync,
107            desired_maximum_frame_latency: 0,
108            alpha_mode: swapchain_capabilities.alpha_modes[0],
109            view_formats: vec![],
110        };
111
112        surface.configure(&device, &surface_config);
113
114        let game_renderer = GameRenderer::new(&device, &surface_config);
115        let animation_renderer = AnimationRenderer::new(&device, &surface_config);
116
117        Self {
118            surface,
119            surface_config,
120            device,
121            queue,
122            game_renderer,
123            animation_renderer,
124        }
125    }
126
127    /// Renders the current frame to the surface, including the maze, player, background, and UI.
128    ///
129    /// # Arguments
130    /// - `window`: The window for retrieving DPI scaling.
131    /// - `ui_state`: Current UI state (colors, etc.).
132    /// - `encoder`: Command encoder for submitting draw commands.
133    /// - `start_time`: Start time for animation timing.
134    /// - `player`: Reference to the current player state.
135    /// - `title`: If true, renders the title screen; otherwise renders the main game scene.
136    ///
137    /// # Returns
138    /// - `Ok((TextureView, ScreenDescriptor, SurfaceTexture))` on success.
139    /// - `Err(String)` if the surface is outdated or unavailable.
140    pub fn update_canvas(
141        &mut self,
142        window: &winit::window::Window,
143        ui_state: &UiState,
144        encoder: &mut wgpu::CommandEncoder,
145        start_time: std::time::Instant,
146        player: &Player,
147        title: bool,
148    ) -> Result<(TextureView, ScreenDescriptor, SurfaceTexture), String> {
149        let surface_texture_obj = self.surface.get_current_texture();
150
151        let surface_texture = match surface_texture_obj {
152            Err(wgpu::SurfaceError::Outdated) => {
153                // Ignoring outdated to allow resizing and minimization
154                println!("wgpu surface outdated");
155                return Err("wgpu surface outdated".to_string());
156            }
157            Err(_) => {
158                surface_texture_obj.expect("Failed to acquire next swap chain texture");
159                return Err("Failed to acquire next swap chain texture".to_string());
160            }
161            Ok(surface_texture) => surface_texture,
162        };
163
164        let surface_view = surface_texture
165            .texture
166            .create_view(&wgpu::TextureViewDescriptor::default());
167
168        let screen_descriptor = ScreenDescriptor {
169            size_in_pixels: [self.surface_config.width, self.surface_config.height],
170            pixels_per_point: window.scale_factor() as f32,
171        };
172        // If we don't have a depth texture OR if its size is different
173        // from the canvasTexture when make a new depth texture
174        let (width, height) = (self.surface_config.width, self.surface_config.height);
175        let aspect = width as f32 / height as f32;
176        let depth_texture_view =
177            self.game_renderer
178                .update_depth_texture(&self.device, width, height);
179
180        if title {
181            {
182                let mut render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
183                    label: Some("Render Pass"),
184                    color_attachments: &[Some(wgpu::RenderPassColorAttachment {
185                        view: &surface_view,
186                        resolve_target: None,
187                        ops: wgpu::Operations {
188                            load: wgpu::LoadOp::Clear(wgpu::Color {
189                                r: 0.1,
190                                g: 0.2,
191                                b: 0.3,
192                                a: 1.0,
193                            }),
194                            store: wgpu::StoreOp::Store,
195                        },
196                    })],
197                    depth_stencil_attachment: None,
198                    occlusion_query_set: None,
199                    timestamp_writes: None,
200                });
201
202                // Render title screen using new component architecture
203                self.animation_renderer.render(&mut render_pass, window);
204            }
205
206            return Ok((surface_view, screen_descriptor, surface_texture)); // <- This return was already there
207        }
208        if !title {
209            let clear_pass_desc = wgpu::RenderPassDescriptor {
210                label: Some("Clear Pass"),
211                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
212                    view: &surface_view,
213                    resolve_target: None,
214                    ops: wgpu::Operations {
215                        load: wgpu::LoadOp::Clear(wgpu::Color {
216                            r: ui_state.r as f64,
217                            g: ui_state.g as f64,
218                            b: ui_state.b as f64,
219                            a: 1.0,
220                        }),
221                        store: wgpu::StoreOp::Store,
222                    },
223                })],
224                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
225                    view: &depth_texture_view,
226                    depth_ops: Some(wgpu::Operations {
227                        load: wgpu::LoadOp::Clear(1.0),
228                        store: wgpu::StoreOp::Store,
229                    }),
230                    stencil_ops: None,
231                }),
232                occlusion_query_set: None,
233                timestamp_writes: None,
234            };
235            let clear_pass = encoder.begin_render_pass(&clear_pass_desc);
236            // No draw calls needed - just clears
237
238            drop(clear_pass)
239        }
240
241        {
242            let elapsed_time = start_time.elapsed().as_secs_f32();
243            self.game_renderer
244                .star_renderer
245                .update_background_color(&self.queue, [ui_state.r, ui_state.g, ui_state.b, 1.0]);
246            self.game_renderer
247                .star_renderer
248                .update_star_time(&self.queue, elapsed_time);
249            self.queue.write_buffer(
250                &self.game_renderer.star_renderer.time_buffer,
251                0,
252                bytemuck::cast_slice(&[elapsed_time]),
253            );
254
255            let star_pass_desc = wgpu::RenderPassDescriptor {
256                label: Some("Star Pass"),
257                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
258                    view: &surface_view,
259                    resolve_target: None,
260                    ops: wgpu::Operations {
261                        load: wgpu::LoadOp::Load, // Preserve cleared background
262                        store: wgpu::StoreOp::Store,
263                    },
264                })],
265                depth_stencil_attachment: None, // No depth testing for stars
266                occlusion_query_set: None,
267                timestamp_writes: None,
268            };
269            let mut star_pass = encoder.begin_render_pass(&star_pass_desc);
270            star_pass.set_pipeline(&self.game_renderer.star_renderer.pipeline);
271            star_pass.set_bind_group(0, &self.game_renderer.star_renderer.uniform_bind_group, &[]);
272            star_pass
273                .set_vertex_buffer(0, self.game_renderer.star_renderer.vertex_buffer.slice(..));
274            star_pass.set_index_buffer(
275                self.game_renderer.star_renderer.index_buffer.slice(..),
276                wgpu::IndexFormat::Uint16,
277            );
278            star_pass.draw_indexed(0..self.game_renderer.star_renderer.num_indices, 0, 0..1);
279            drop(star_pass);
280        };
281
282        {
283            let main_pass_desc = wgpu::RenderPassDescriptor {
284                label: Some("Main Render Pass"),
285                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
286                    view: &surface_view,
287                    resolve_target: None,
288                    ops: wgpu::Operations {
289                        load: wgpu::LoadOp::Load,
290                        store: wgpu::StoreOp::Store,
291                    },
292                })],
293                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
294                    view: &depth_texture_view,
295                    depth_ops: Some(wgpu::Operations {
296                        load: wgpu::LoadOp::Load,
297                        store: wgpu::StoreOp::Store,
298                    }),
299                    stencil_ops: None,
300                }),
301                occlusion_query_set: None,
302                timestamp_writes: None,
303            };
304            let mut main_pass = encoder.begin_render_pass(&main_pass_desc);
305            self.game_renderer
306                .render_game(&self.queue, player, &mut main_pass, aspect);
307        }
308
309        Ok((surface_view, screen_descriptor, surface_texture))
310    }
311}
