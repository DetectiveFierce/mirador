1//! WGPU-based renderer for the Mirador game.
2//!
3//! This module provides [`WgpuRenderer`], which manages all GPU resources, pipelines, and rendering
4//! logic for the main game scene, background, and title screen. It handles initialization of WGPU,
5//! creation of vertex/uniform buffers, pipelines, and orchestrates the rendering of the maze, player,
6//! animated background, and UI overlays.
7//!
8//! # Features
9//! - Loads maze geometry and floor/wall vertices
10//! - Renders a starfield background and animated title screen
11//! - Handles depth buffering and uniform updates for camera/player movement
12//! - Integrates with egui for UI overlays
13//!
14//! # Usage
15//! Create a [`WgpuRenderer`] via [`WgpuRenderer::new`] and call [`WgpuRenderer::update_canvas`]
16//! each frame to render the current game state.
17use crate::game::player::Player;
18use crate::maze::maze_animation::AnimationRenderer;
19use crate::renderer::render_components::GameRenderer;
20use crate::renderer::text::TextRenderer;
21use crate::ui::ui_panel::UiState;
22use egui_wgpu::ScreenDescriptor;
23use egui_wgpu::wgpu;
24use egui_wgpu::wgpu::{SurfaceTexture, TextureView};

26/// Main WGPU renderer for the Mirador game.
27///
28/// This struct manages all GPU resources, pipelines, and rendering logic for the game scene,
29/// including the maze, player, animated background, and title screen.
30///
31/// # Fields
32/// - `surface`: The WGPU surface for presenting rendered frames.
33/// - `surface_config`: The surface configuration (format, size, etc.).
34/// - `device`: The WGPU device for resource creation.
35/// - `queue`: The WGPU queue for submitting commands.
36/// - `game_renderer`: Main render pipeline for the maze and floor.
37/// - `animation_renderer`: Renderer for the title screen maze and loading bar.
38pub struct WgpuRenderer {
39    /// The WGPU surface for presenting rendered frames.
40    pub surface: wgpu::Surface<'static>,
41    /// The surface configuration (format, size, etc.).
42    pub surface_config: wgpu::SurfaceConfiguration,
43    /// The WGPU device for resource creation.
44    pub device: wgpu::Device,
45    /// The WGPU queue for submitting commands.
46    pub queue: wgpu::Queue,
47    /// Main render pipeline for the maze and floor.
48    pub game_renderer: GameRenderer,
49    /// Renderer for the title screen maze and loading bar.
50    pub animation_renderer: AnimationRenderer,
51}
52
53impl WgpuRenderer {
54    /// Initializes a new [`WgpuRenderer`] and all associated GPU resources.
55    ///
56    /// # Arguments
57    /// - `instance`: The WGPU instance.
58    /// - `surface`: The WGPU surface for presentation.
59    /// - `width`: Initial width of the surface.
60    /// - `height`: Initial height of the surface.
61    ///
62    /// # Returns
63    /// A fully initialized [`WgpuRenderer`] ready for rendering.
64    pub async fn new(
65        instance: &wgpu::Instance,
66        surface: wgpu::Surface<'static>,
67        width: u32,
68        height: u32,
69    ) -> Self {
70        let power_pref = wgpu::PowerPreference::default();
71        let adapter = instance
72            .request_adapter(&wgpu::RequestAdapterOptions {
73                power_preference: power_pref,
74                force_fallback_adapter: false,
75                compatible_surface: Some(&surface),
76            })
77            .await
78            .expect("Failed to find an appropriate adapter");
79
80        let features = wgpu::Features::empty();
81        let (device, queue) = adapter
82            .request_device(
83                &wgpu::DeviceDescriptor {
84                    label: None,
85                    required_features: features,
86                    required_limits: Default::default(),
87                    memory_hints: Default::default(),
88                },
89                None,
90            )
91            .await
92            .expect("Failed to create device");
93
94        let swapchain_capabilities = surface.get_capabilities(&adapter);
95        let selected_format = wgpu::TextureFormat::Bgra8UnormSrgb;
96        let swapchain_format = swapchain_capabilities
97            .formats
98            .iter()
99            .find(|d| **d == selected_format)
100            .expect("failed to select proper surface texture format!");
101
102        let surface_config = wgpu::SurfaceConfiguration {
103            usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
104            format: *swapchain_format,
105            width,
106            height,
107            present_mode: wgpu::PresentMode::AutoVsync,
108            desired_maximum_frame_latency: 0,
109            alpha_mode: swapchain_capabilities.alpha_modes[0],
110            view_formats: vec![],
111        };
112
113        surface.configure(&device, &surface_config);
114
115        let game_renderer = GameRenderer::new(&device, &surface_config);
116        let animation_renderer = AnimationRenderer::new(&device, &surface_config);
117
118        Self {
119            surface,
120            surface_config,
121            device,
122            queue,
123            game_renderer,
124            animation_renderer,
125        }
126    }
127
128    /// Renders the current frame to the surface, including the maze, player, background, and UI.
129    ///
130    /// # Arguments
131    /// - `window`: The window for retrieving DPI scaling.
132    /// - `ui_state`: Current UI state (colors, etc.).
133    /// - `encoder`: Command encoder for submitting draw commands.
134    /// - `start_time`: Start time for animation timing.
135    /// - `player`: Reference to the current player state.
136    /// - `title`: If true, renders the title screen; otherwise renders the main game scene.
137    ///
138    /// # Returns
139    /// - `Ok((TextureView, ScreenDescriptor, SurfaceTexture))` on success.
140    /// - `Err(String)` if the surface is outdated or unavailable.
141    #[allow(clippy::too_many_arguments)]
142    pub fn update_canvas(
143        &mut self,
144        window: &winit::window::Window,
145        ui_state: &UiState,
146        encoder: &mut wgpu::CommandEncoder,
147        start_time: std::time::Instant,
148        player: &Player,
149        title: bool,
150        text_renderer: &mut TextRenderer,
151    ) -> Result<(TextureView, ScreenDescriptor, SurfaceTexture), String> {
152        let surface_texture_obj = self.surface.get_current_texture();
153
154        let surface_texture = match surface_texture_obj {
155            Err(wgpu::SurfaceError::Outdated) => {
156                // Ignoring outdated to allow resizing and minimization
157                println!("wgpu surface outdated");
158                return Err("wgpu surface outdated".to_string());
159            }
160            Err(_) => {
161                surface_texture_obj.expect("Failed to acquire next swap chain texture");
162                return Err("Failed to acquire next swap chain texture".to_string());
163            }
164            Ok(surface_texture) => surface_texture,
165        };
166
167        let surface_view = surface_texture
168            .texture
169            .create_view(&wgpu::TextureViewDescriptor::default());
170
171        let screen_descriptor = ScreenDescriptor {
172            size_in_pixels: [self.surface_config.width, self.surface_config.height],
173            pixels_per_point: window.scale_factor() as f32,
174        };
175        // If we don't have a depth texture OR if its size is different
176        // from the canvasTexture when make a new depth texture
177        let (width, height) = (self.surface_config.width, self.surface_config.height);
178        let aspect = width as f32 / height as f32;
179        let depth_texture_view =
180            self.game_renderer
181                .update_depth_texture(&self.device, width, height);
182
183        if title {
184            {
185                let mut render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
186                    label: Some("Render Pass"),
187                    color_attachments: &[Some(wgpu::RenderPassColorAttachment {
188                        view: &surface_view,
189                        resolve_target: None,
190                        ops: wgpu::Operations {
191                            load: wgpu::LoadOp::Clear(wgpu::Color {
192                                r: 0.1,
193                                g: 0.2,
194                                b: 0.3,
195                                a: 1.0,
196                            }),
197                            store: wgpu::StoreOp::Store,
198                        },
199                    })],
200                    depth_stencil_attachment: None,
201                    occlusion_query_set: None,
202                    timestamp_writes: None,
203                });
204
205                // Render title screen using new component architecture
206                self.animation_renderer.render(&mut render_pass, window);
207            }
208
209            return Ok((surface_view, screen_descriptor, surface_texture)); // <- This return was already there
210        }
211        if !title {
212            let clear_pass_desc = wgpu::RenderPassDescriptor {
213                label: Some("Clear Pass"),
214                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
215                    view: &surface_view,
216                    resolve_target: None,
217                    ops: wgpu::Operations {
218                        load: wgpu::LoadOp::Clear(wgpu::Color {
219                            r: ui_state.r as f64,
220                            g: ui_state.g as f64,
221                            b: ui_state.b as f64,
222                            a: 1.0,
223                        }),
224                        store: wgpu::StoreOp::Store,
225                    },
226                })],
227                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
228                    view: &depth_texture_view,
229                    depth_ops: Some(wgpu::Operations {
230                        load: wgpu::LoadOp::Clear(1.0),
231                        store: wgpu::StoreOp::Store,
232                    }),
233                    stencil_ops: None,
234                }),
235                occlusion_query_set: None,
236                timestamp_writes: None,
237            };
238            let clear_pass = encoder.begin_render_pass(&clear_pass_desc);
239            // No draw calls needed - just clears
240
241            drop(clear_pass)
242        }
243
244        {
245            let elapsed_time = start_time.elapsed().as_secs_f32();
246            self.game_renderer
247                .star_renderer
248                .update_background_color(&self.queue, [ui_state.r, ui_state.g, ui_state.b, 1.0]);
249            self.game_renderer
250                .star_renderer
251                .update_star_time(&self.queue, elapsed_time);
252            self.queue.write_buffer(
253                &self.game_renderer.star_renderer.time_buffer,
254                0,
255                bytemuck::cast_slice(&[elapsed_time]),
256            );
257
258            let star_pass_desc = wgpu::RenderPassDescriptor {
259                label: Some("Star Pass"),
260                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
261                    view: &surface_view,
262                    resolve_target: None,
263                    ops: wgpu::Operations {
264                        load: wgpu::LoadOp::Load, // Preserve cleared background
265                        store: wgpu::StoreOp::Store,
266                    },
267                })],
268                depth_stencil_attachment: None, // No depth testing for stars
269                occlusion_query_set: None,
270                timestamp_writes: None,
271            };
272            let mut star_pass = encoder.begin_render_pass(&star_pass_desc);
273            star_pass.set_pipeline(&self.game_renderer.star_renderer.pipeline);
274            star_pass.set_bind_group(0, &self.game_renderer.star_renderer.uniform_bind_group, &[]);
275            star_pass
276                .set_vertex_buffer(0, self.game_renderer.star_renderer.vertex_buffer.slice(..));
277            star_pass.set_index_buffer(
278                self.game_renderer.star_renderer.index_buffer.slice(..),
279                wgpu::IndexFormat::Uint16,
280            );
281            star_pass.draw_indexed(0..self.game_renderer.star_renderer.num_indices, 0, 0..1);
282            drop(star_pass);
283        };
284
285        {
286            let main_pass_desc = wgpu::RenderPassDescriptor {
287                label: Some("Main Render Pass"),
288                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
289                    view: &surface_view,
290                    resolve_target: None,
291                    ops: wgpu::Operations {
292                        load: wgpu::LoadOp::Load,
293                        store: wgpu::StoreOp::Store,
294                    },
295                })],
296                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
297                    view: &depth_texture_view,
298                    depth_ops: Some(wgpu::Operations {
299                        load: wgpu::LoadOp::Load,
300                        store: wgpu::StoreOp::Store,
301                    }),
302                    stencil_ops: None,
303                }),
304                occlusion_query_set: None,
305                timestamp_writes: None,
306            };
307            let mut main_pass = encoder.begin_render_pass(&main_pass_desc);
308            self.game_renderer
309                .render_game(&self.queue, player, &mut main_pass, aspect);
310        }
311
312        {
313            text_renderer.resize(
314                &self.queue,
315                glyphon::Resolution {
316                    width: self.surface_config.width,
317                    height: self.surface_config.height,
318                },
319            );
320            match text_renderer.prepare(&self.device, &self.queue, &self.surface_config) {
321                Ok(_) => {}
322                Err(e) => {
323                    println!("Failed to prepare Glyphon: {:?}", e);
324                }
325            }
326
327            let mut text_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
328                label: Some("Main Render Pass"),
329                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
330                    view: &surface_view,
331                    resolve_target: None,
332                    ops: wgpu::Operations {
333                        load: wgpu::LoadOp::Load,
334                        store: wgpu::StoreOp::Store,
335                    },
336                })],
337                depth_stencil_attachment: None,
338                timestamp_writes: None,
339                occlusion_query_set: None,
340            });
341            match text_renderer.render(&mut text_pass) {
342                Ok(_) => {}
343                Err(e) => println!("Glyphon render failed: {:?}", e),
344            }
345        }
346
347        Ok((surface_view, screen_descriptor, surface_texture))
348    }
