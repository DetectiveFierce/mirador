17/// Holds all state required for a running Mirador game session.
18///
19/// This includes rendering backends, UI state, game logic, and input state.
20pub struct AppState {
21    /// The WGPU renderer for the main game and background.
22    pub wgpu_renderer: WgpuRenderer,
23    /// The main game state (player, timing, maze, etc.).
24    pub game_state: GameState,
25    /// The current input state (pressed keys, etc.).
26    pub key_state: KeyState,
27    /// The text renderer for all game UI text elements.
28    pub text_renderer: TextRenderer,
29    /// The timestamp when the game started.
30    pub start_time: Instant,
31    /// The total elapsed time since the game started.
32    pub elapsed_time: Duration,

34    pub pause_menu: crate::renderer::ui::pause_menu::PauseMenu,

36    pub upgrade_menu: crate::renderer::ui::upgrade_menu::UpgradeMenu,
37    /// Performance profiler for benchmarking
38    pub profiler: Profiler,

40    pub fps_counter: FrameRateCounter,
41}

45    ///
46    /// # Arguments

48    /// - `surface`: The WGPU surface for rendering.

53        instance: &wgpu::Instance,
54        surface: wgpu::Surface<'static>,
55        window: &Window,
56        width: u32,
57        height: u32,
58    ) -> Self {
59        use crate::benchmarks::{BenchmarkConfig, Profiler};

67            max_samples: 1000,
68        });
69
70        window.set_cursor_visible(false);
71
72        // Benchmark WgpuRenderer initialization (most taxing part)
73        init_profiler.start_section("wgpu_renderer_initialization");
74        let wgpu_renderer = WgpuRenderer::new(instance, surface, width, height).await;
75        init_profiler.end_section("wgpu_renderer_initialization");
76
77        // Benchmark TextRenderer initialization
78        init_profiler.start_section("text_renderer_initialization");
79        let mut text_renderer = TextRenderer::new(

85        init_profiler.end_section("text_renderer_initialization");
86
87        // Check if font loading was successful

89            println!("WARNING: No fonts loaded! Text may not render properly.");

104        // Benchmark pause menu creation
105        init_profiler.start_section("pause_menu_creation");

200    /// Resizes the WGPU surface and updates the configuration.
201    ///
202    /// # Arguments
203    /// - `width`: New width of the surface.
204    /// - `height`: New height of the surface.
205    pub fn resize_surface(&mut self, width: u32, height: u32) {
206        self.wgpu_renderer.surface_config.width = width;
207        self.wgpu_renderer.surface_config.height = height;
208        self.wgpu_renderer.surface.configure(
209            &self.wgpu_renderer.device,
210            &self.wgpu_renderer.surface_config,
211        );
212
213        self.wgpu_renderer
214            .game_renderer
215            .compass_renderer
216            .update_uniforms(&self.wgpu_renderer.queue, [0.75, 0.75], [4.75, 4.75]);
217        // Update game over display position for new window size
218        if let Err(e) = self.text_renderer.update_game_over_position(width, height) {
219            println!("Failed to update game over position: {}", e);
220        }
221    }

223    /// Updates the title screen maze and loading bar, and uploads new texture data.
224    pub fn handle_loading_screen(&mut self, window: &winit::window::Window) {
225        let progress = self
226            .wgpu_renderer
227            .loading_screen_renderer
228            .generator
229            .get_progress_ratio();
230
231        let (maze_width, maze_height) = match self.wgpu_renderer.loading_screen_renderer.maze.lock()
232        {
233            Ok(maze_lock) => maze_lock.get_dimensions(),
234            Err(err) => {
235                eprintln!("Failed to acquire maze lock for dimensions: {}", err);
236                return;
237            }
238        };
239
240        self.wgpu_renderer
241            .loading_screen_renderer
242            .update_loading_bar(&self.wgpu_renderer.queue, progress, window);
243
244        self.wgpu_renderer
245            .loading_screen_renderer
246            .update_exit_shader(&self.wgpu_renderer.queue, window);
247
248        let maze_data = match self.wgpu_renderer.loading_screen_renderer.maze.lock() {
249            Ok(maze_lock) => maze_lock.get_render_data(
250                &self
251                    .wgpu_renderer
252                    .loading_screen_renderer
253                    .generator
254                    .connected_cells,
255            ),
256            Err(err) => {
257                eprintln!("Failed to acquire maze lock: {}", err);
258                return;
259            }
260        };
261
262        self.wgpu_renderer.loading_screen_renderer.update_texture(
263            &self.wgpu_renderer.queue,
264            &maze_data,
265            maze_width,
266            maze_height,
267        );
268        self.wgpu_renderer.loading_screen_renderer.last_update = Instant::now();
269    }

271    /// Handles mouse capture and cursor visibility based on game state.
272    ///
273    /// Locks/unlocks the cursor and centers it if mouse capture is enabled.
274    pub fn triage_mouse(&mut self, window: &Window) {
275        if self.game_state.capture_mouse {
276            if let Err(e) = window.set_cursor_grab(winit::window::CursorGrabMode::Locked) {
277                eprintln!("Failed to lock cursor: {}, {}", e, e);
278            }
279            window.set_cursor_visible(false);
280            let window_size = window.inner_size().to_logical::<f64>(window.scale_factor());
281
282            let center_x = window_size.width / 2.0;
283            let center_y = window_size.height / 2.0;
284            if let Err(e) =
285                window.set_cursor_position(winit::dpi::LogicalPosition::new(center_x, center_y))
286            {
287                eprintln!("Failed to center cursor: {}, {}", e, e);
288            }
289        } else if !self.game_state.capture_mouse {
290            if let Err(e) = window.set_cursor_grab(winit::window::CursorGrabMode::None) {
291                eprintln!("Failed to unlock cursor: {}, {}", e, e);
292            }
293            window.set_cursor_visible(true);
294        }
295    }

297    /// Updates all game UI elements including timer, level, and score displays.
298    pub fn update_game_ui(&mut self, window: &winit::window::Window) {
299        // Start timer when game begins (not on title screen)
300        if self.game_state.current_screen == CurrentScreen::Game
301            && self.game_state.game_ui.timer.is_none()
302        {
303            // Configure timer with custom settings
304            let timer_config = TimerConfig {
305                duration: Duration::from_secs(30),
306                warning_threshold: Duration::from_secs(20),
307                critical_threshold: Duration::from_secs(10),
308                normal_color: Color::rgb(100, 255, 100),
309                warning_color: Color::rgb(255, 255, 100),
310                critical_color: Color::rgb(255, 100, 100),
311            };
312            self.game_state.start_game_timer(Some(timer_config));
313        }
314
315        // Hide game UI elements during loading screen or title screen
316        if self.game_state.current_screen == CurrentScreen::Loading
317            || self.game_state.current_screen == CurrentScreen::Title
318        {
319            // Hide timer, level, and score displays
320            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("main_timer") {
321                buffer.visible = false;
322            }
323            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("level") {
324                buffer.visible = false;
325            }
326            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("score") {
327                buffer.visible = false;
328            }
329        } else {
330            // Show game UI elements when not loading
331            // HIDE the timer text buffer always (replaced by bar)
332            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("main_timer") {
333                buffer.visible = false;
334            }
335            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("level") {
336                buffer.visible = true;
337            }
338            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("score") {
339                buffer.visible = true;
340            }
341        }
342
343        // Always update the text UI, but only update the timer if in Game
344        let timer_expired = game::update_game_ui(
345            &mut self.text_renderer,
346            &mut self.game_state.game_ui,
347            &self.game_state.current_screen,
348            window,
349        );
350
351        if timer_expired {
352            // Handle timer expiration - you can add game over logic here
353            println!("Timer expired! Game over.");
354            self.game_state.stop_game_timer();
355            self.game_state.current_screen = CurrentScreen::GameOver;
356        }
357
358        if self.game_state.enemy.pathfinder.reached_player {
359            self.game_state.stop_game_timer();
360            self.game_state.current_screen = CurrentScreen::GameOver;
361            self.game_state.enemy = Enemy::new([-0.5, 30.0, 0.0], 150.0);
362            self.game_state.enemy.pathfinder.reached_player = false;
363        }
364
365        // Show/hide game over display based on current screen
366        if self.game_state.current_screen == CurrentScreen::GameOver {
367            self.text_renderer.show_game_over_display();
368        } else {
369            self.text_renderer.hide_game_over_display();
370        }
371    }
