39    /// Asynchronously creates a new [`AppState`] with initialized renderers and game state.
40    ///
41    /// # Arguments
42    /// - `instance`: The WGPU instance.
43    /// - `surface`: The WGPU surface for rendering.
44    /// - `window`: The application window.
45    /// - `width`: Initial window width.
46    /// - `height`: Initial window height.
47    pub async fn new(
48        instance: &wgpu::Instance,
49        surface: wgpu::Surface<'static>,
50        window: &Window,
51        width: u32,
52        height: u32,
53    ) -> Self {
54        window.set_cursor_visible(false);
55        let wgpu_renderer = WgpuRenderer::new(instance, surface, width, height).await;
56        let egui_renderer = EguiRenderer::new(
57            &wgpu_renderer.device,
58            wgpu_renderer.surface_config.format,
59            None,
60            1,
61            window,
62        );
63
64        let mut text_renderer = TextRenderer::new(
65            &wgpu_renderer.device,
66            &wgpu_renderer.queue,
67            wgpu_renderer.surface_config.format,
68            window,
69        );
70
71        // Check if font loading was successful
72        if text_renderer.loaded_fonts.is_empty() {
73            println!("WARNING: No fonts loaded! Text may not render properly.");
74        } else {
75            println!("Loaded fonts: {:?}", text_renderer.loaded_fonts);
76        }
77
78        let game_state = GameState::new();
79        // Initialize all game UI elements
80        game::initialize_game_ui(&mut text_renderer, &game_state.game_ui, window);
81
82        // Create game over display
83        text_renderer.create_game_over_display(width, height);
84
85        let pause_menu = crate::ui::pause_menu::PauseMenu::new(
86            &wgpu_renderer.device,
87            &wgpu_renderer.queue,
88            wgpu_renderer.surface_config.format,
89            window,
90        );
91
92        // Add big boldMirador' text in the top right for the title screen
93        let width = wgpu_renderer.surface_config.width as f32;
94        let height = wgpu_renderer.surface_config.height as f32;
95        let mirador_style = crate::renderer::text::TextStyle {
96            font_family: "HankenGrotesk".to_string(),
97            font_size: 125.0,
98            line_height: 150.0,
99            color: Color::rgb(58, 53, 70), // #3
100            weight: glyphon::Weight::BOLD,
101            style: glyphon::Style::Normal,
102        };
103        // Estimate text width for right alignment
104        let text_width = 620.0; // Conservative estimate for large text
105        let text_height = 1500.0; // Let's assume a large height for the title
106        let mirador_position = crate::renderer::text::TextPosition {
107            x: width - text_width - 200.0, // 20px margin from right
108            y: 100.0,                      // 100px margin from top
109            max_width: Some(text_width),
110            max_height: Some(text_height),
111        };
112        text_renderer.create_text_buffer(
113            "title_mirador_overlay",
114            "Mirador",
115            Some(mirador_style),
116            Some(mirador_position.clone()),
117        );
118        // Add subtitle text at the same x, 60px from the bottom
119        let subtitle_style = crate::renderer::text::TextStyle {
120            font_family: "HankenGrotesk".to_string(),
121            font_size: 48.0,
122            line_height: 72.0,
123            color: Color::rgb(58, 53, 70), // #3
124            weight: glyphon::Weight::MEDIUM,
125            style: glyphon::Style::Normal,
126        };
127        let subtitle_text = "Click anywhere to get lost.";
128        let subtitle_text_height = 72.0;
129        let subtitle_position = crate::renderer::text::TextPosition {
130            x: mirador_position.x - 1200.0,
131            y: height + 1000.0,
132            max_width: Some(text_width),
133            max_height: Some(subtitle_text_height),
134        };
135        text_renderer.create_text_buffer(
136            "title_subtitle_overlay",
137            subtitle_text,
138            Some(subtitle_style),
139            Some(subtitle_position),
140        );
141
142        Self {
143            wgpu_renderer,
144            egui_renderer,
145            ui: UiState::new(),
146            game_state,
147            key_state: KeyState::new(),
148            text_renderer,
149            start_time: Instant::now(),
150            elapsed_time: Duration::ZERO,
151            pause_menu,
152        }
153    }

155    /// Resizes the WGPU surface and updates the configuration.
156    ///
157    /// # Arguments
158    /// - `width`: New width of the surface.
159    /// - `height`: New height of the surface.
160    pub fn resize_surface(&mut self, width: u32, height: u32) {
161        self.wgpu_renderer.surface_config.width = width;
162        self.wgpu_renderer.surface_config.height = height;
163        self.wgpu_renderer.surface.configure(
164            &self.wgpu_renderer.device,
165            &self.wgpu_renderer.surface_config,
166        );
167
168        self.wgpu_renderer
169            .game_renderer
170            .compass_renderer
171            .update_uniforms(&self.wgpu_renderer.queue, [0.75, 0.75], [4.75, 4.75]);
172        // Update game over display position for new window size
173        if let Err(e) = self.text_renderer.update_game_over_position(width, height) {
174            println!("Failed to update game over position: {}", e);
175        }
176    }

178    /// Updates the title screen maze and loading bar, and uploads new texture data.
179    pub fn handle_loading_screen(&mut self, window: &winit::window::Window) {
180        let progress = self
181            .wgpu_renderer
182            .loading_screen_renderer
183            .generator
184            .get_progress_ratio();
185
186        let (maze_width, maze_height) = match self.wgpu_renderer.loading_screen_renderer.maze.lock()
187        {
188            Ok(maze_lock) => maze_lock.get_dimensions(),
189            Err(err) => {
190                eprintln!("Failed to acquire maze lock for dimensions: {}", err);
191                return;
192            }
193        };
194
195        self.wgpu_renderer
196            .loading_screen_renderer
197            .update_loading_bar(&self.wgpu_renderer.queue, progress, window);
198
199        self.wgpu_renderer
200            .loading_screen_renderer
201            .update_exit_shader(&self.wgpu_renderer.queue, window);
202
203        let maze_data = match self.wgpu_renderer.loading_screen_renderer.maze.lock() {
204            Ok(maze_lock) => maze_lock.get_render_data(
205                &self
206                    .wgpu_renderer
207                    .loading_screen_renderer
208                    .generator
209                    .connected_cells,
210            ),
211            Err(err) => {
212                eprintln!("Failed to acquire maze lock: {}", err);
213                return;
214            }
215        };
216
217        self.wgpu_renderer.loading_screen_renderer.update_texture(
218            &self.wgpu_renderer.queue,
219            &maze_data,
220            maze_width,
221            maze_height,
222        );
223        self.wgpu_renderer.loading_screen_renderer.last_update = Instant::now();
224    }

226    /// Handles mouse capture and cursor visibility based on game state.
227    ///
228    /// Locks/unlocks the cursor and centers it if mouse capture is enabled.
229    pub fn triage_mouse(&mut self, window: &Window) {
230        if self.game_state.capture_mouse {
231            if let Err(e) = window.set_cursor_grab(winit::window::CursorGrabMode::Locked) {
232                eprintln!("Failed to lock cursor: {}, {}", e, e);
233            }
234            window.set_cursor_visible(false);
235            let window_size = window.inner_size().to_logical::<f64>(window.scale_factor());
236
237            let center_x = window_size.width / 2.0;
238            let center_y = window_size.height / 2.0;
239            if let Err(e) =
240                window.set_cursor_position(winit::dpi::LogicalPosition::new(center_x, center_y))
241            {
242                eprintln!("Failed to center cursor: {}, {}", e, e);
243            }
244        } else if !self.game_state.capture_mouse {
245            if let Err(e) = window.set_cursor_grab(winit::window::CursorGrabMode::None) {
246                eprintln!("Failed to unlock cursor: {}, {}", e, e);
247            }
248            window.set_cursor_visible(true);
249        }
250    }

252    /// Updates all game UI elements including timer, level, and score displays.
253    pub fn update_game_ui(&mut self, window: &winit::window::Window) {
254        // Start timer when game begins (not on title screen)
255        if self.game_state.current_screen == CurrentScreen::Game
256            && self.game_state.game_ui.timer.is_none()
257        {
258            // Configure timer with custom settings
259            let timer_config = TimerConfig {
260                duration: Duration::from_secs(30),
261                warning_threshold: Duration::from_secs(20),
262                critical_threshold: Duration::from_secs(10),
263                normal_color: Color::rgb(100, 255, 100),
264                warning_color: Color::rgb(255, 255, 100),
265                critical_color: Color::rgb(255, 100, 100),
266            };
267            self.game_state.start_game_timer(Some(timer_config));
268        }
269
270        // Hide game UI elements during loading screen or title screen
271        if self.game_state.current_screen == CurrentScreen::Loading
272            || self.game_state.current_screen == CurrentScreen::Title
273        {
274            // Hide timer, level, and score displays
275            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("main_timer") {
276                buffer.visible = false;
277            }
278            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("level") {
279                buffer.visible = false;
280            }
281            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("score") {
282                buffer.visible = false;
283            }
284        } else {
285            // Show game UI elements when not loading
286            // HIDE the timer text buffer always (replaced by bar)
287            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("main_timer") {
288                buffer.visible = false;
289            }
290            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("level") {
291                buffer.visible = true;
292            }
293            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("score") {
294                buffer.visible = true;
295            }
296        }
297
298        // Always update the text UI, but only update the timer if in Game
299        let timer_expired = game::update_game_ui(
300            &mut self.text_renderer,
301            &mut self.game_state.game_ui,
302            &self.game_state.current_screen,
303            window,
304        );
305
306        if timer_expired {
307            // Handle timer expiration - you can add game over logic here
308            println!("Timer expired! Game over.");
309            self.game_state.stop_game_timer();
310            self.game_state.current_screen = CurrentScreen::GameOver;
311        }
312
313        if self.game_state.enemy.pathfinder.reached_player {
314            self.game_state.stop_game_timer();
315            self.game_state.current_screen = CurrentScreen::GameOver;
316            self.game_state.enemy = Enemy::new([-0.5, 30.0, 0.0], 150.0);
317            self.game_state.enemy.pathfinder.reached_player = false;
318        }
319
320        // Show/hide game over display based on current screen
321        if self.game_state.current_screen == CurrentScreen::GameOver {
322            self.text_renderer.show_game_over_display();
323        } else {
324            self.text_renderer.hide_game_over_display();
325        }
326    }
