21    // handle_redraw
22    pub fn handle_redraw(&mut self) {
23        let window = self
24            .window
25            .as_ref()
26            .expect("Window must be initialized before use");
27        if window.is_minimized().unwrap_or(false) {
28            println!("Window is minimized");
29            return;
30        }
31
32        let state = self
33            .state
34            .as_mut()
35            .expect("State must be initialized before use");
36
37        if state.game_state.current_screen == CurrentScreen::Loading {
38            state
39                .game_state
40                .audio_manager
41                .pause_enemy_audio("enemy")
42                .expect("Failed to pause enemy audio");
43            state.handle_loading_screen(window);
44        } else if state.game_state.current_screen == CurrentScreen::Title {
45            crate::renderer::title::handle_title(state, window);
46            return;
47        } else {
48            state.game_state.player.update_cell(
49                &state
50                    .wgpu_renderer
51                    .loading_screen_renderer
52                    .maze
53                    .lock()
54                    .unwrap()
55                    .walls,
56                state.game_state.is_test_mode,
57            );
58        }
59
60        // Update game state and UI
61        state.key_state.update(&mut state.game_state);
62        state.update_game_ui(window);
63        state.update_ui(window);
64        state
65            .game_state
66            .audio_manager
67            .set_listener_position(state.game_state.player.position)
68            .expect("Failed to set listener position");
69        state
70            .game_state
71            .audio_manager
72            .update_enemy_position("enemy", state.game_state.enemy.pathfinder.position)
73            .expect("Failed to update enemy position");
74
75        // Prepare rendering commands
76        let mut encoder = state
77            .wgpu_renderer
78            .device
79            .create_command_encoder(&wgpu::CommandEncoderDescriptor { label: None });
80
81        // Update canvas surface
82        let (surface_view, surface_texture) = match state.wgpu_renderer.update_canvas(
83            window,
84            &mut encoder,
85            &state.ui,
86            &state.game_state,
87            &mut state.text_renderer,
88        ) {
89            Ok(result) => result,
90            Err(err) => {
91                eprintln!("Failed to update canvas: {}", err);
92                #[cfg(debug_assertions)]
93                eprintln!("Backtrace: {:?}", std::backtrace::Backtrace::capture());
94                return;
95            }
96        };
97
98        // --- Debug Info Panel ---
99        if state.pause_menu.is_debug_panel_visible() {
100            let window_size = &state.wgpu_renderer.surface_config;
101
102            // Add timer information to debug text
103            let timer_info = if let Some(timer) = &state.game_state.game_ui.timer {
104                let remaining = timer.get_remaining_time();
105                let total = timer.config.duration;
106                let remaining_secs = remaining.as_secs_f64();
107                let total_secs = total.as_secs_f64();
108                let progress = if total_secs > 0.0 {
109                    remaining_secs / total_secs
110                } else {
111                    0.0
112                };
113                format!(
114                    "Window Size: {} x {}\nTimer: {:.2}s / {:.2}s ({:.1}%)",
115                    window_size.width,
116                    window_size.height,
117                    remaining_secs,
118                    total_secs,
119                    progress * 100.0
120                )
121            } else {
122                format!(
123                    "Window Size: {} x {}\nTimer: Not active",
124                    window_size.width, window_size.height
125                )
126            };
127
128            let style = crate::renderer::text::TextStyle {
129                font_family: "HankenGrotesk".to_string(),
130                font_size: 22.0,
131                line_height: 26.0,
132                color: glyphon::Color::rgb(220, 40, 40),
133                weight: glyphon::Weight::BOLD,
134                style: glyphon::Style::Normal,
135            };
136            let pos = crate::renderer::text::TextPosition {
137                x: window_size.width as f32 - 320.0,
138                y: 20.0,
139                max_width: Some(300.0),
140                max_height: Some(80.0), // Increased height for two lines
141            };
142            state.text_renderer.create_text_buffer(
143                "debug_info",
144                &timer_info,
145                Some(style),
146                Some(pos),
147            );
148        } else {
149            // Hide debug info by making it transparent if it exists
150            if let Some(buf) = state.text_renderer.text_buffers.get_mut("debug_info") {
151                buf.visible = false;
152            }
153        }
154        // Prepare and render text BEFORE pause menu overlay
155        if let Err(e) = state.text_renderer.prepare(
156            &state.wgpu_renderer.device,
157            &state.wgpu_renderer.queue,
158            &state.wgpu_renderer.surface_config,
159        ) {
160            println!("Failed to prepare text renderer: {}", e);
161        }
162        {
163            let mut render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
164                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
165                    view: &surface_view,
166                    resolve_target: None,
167                    ops: wgpu::Operations {
168                        load: wgpu::LoadOp::Load,
169                        store: wgpu::StoreOp::Store,
170                    },
171                })],
172                depth_stencil_attachment: None,
173                timestamp_writes: None,
174                label: Some("text render pass"),
175                occlusion_query_set: None,
176            });
177            if let Err(e) = state.text_renderer.render(&mut render_pass) {
178                println!("Failed to render text: {}", e);
179            }
180        }
181        // --- End Game UI ---
182
183        // If paused, render the pause menu on top
184        if state.game_state.current_screen == CurrentScreen::Pause {
185            if !state.pause_menu.is_visible() {
186                state.pause_menu.show(state.game_state.is_test_mode);
187            }
188
189            // Create a render pass for the pause menu
190            let mut render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
191                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
192                    view: &surface_view,
193                    resolve_target: None,
194                    ops: wgpu::Operations {
195                        load: wgpu::LoadOp::Load,
196                        store: wgpu::StoreOp::Store,
197                    },
198                })],
199                depth_stencil_attachment: None,
200                timestamp_writes: None,
201                label: Some("pause menu render pass"),
202                occlusion_query_set: None,
203            });
204
205            // --- Add semi-transparent grey overlay ---
206            let overlay_color = [0.08, 0.09, 0.11, 0.88]; // darker, neutral semi-transparent grey
207            let (w, h) = (
208                state.wgpu_renderer.surface_config.width as f32,
209                state.wgpu_renderer.surface_config.height as f32,
210            );
211            state
212                .pause_menu
213                .button_manager
214                .rectangle_renderer
215                .add_rectangle(crate::renderer::rectangle::Rectangle::new(
216                    0.0,
217                    0.0,
218                    w,
219                    h,
220                    overlay_color,
221                ));
222            state
223                .pause_menu
224                .button_manager
225                .rectangle_renderer
226                .render(&state.wgpu_renderer.device, &mut render_pass);
227            // --- End overlay ---
228
229            // Prepare pause menu for rendering (text)
230            if let Err(e) = state.pause_menu.prepare(
231                &state.wgpu_renderer.device,
232                &state.wgpu_renderer.queue,
233                &state.wgpu_renderer.surface_config,
234            ) {
235                println!("Failed to prepare pause menu: {}", e);
236            }
237
238            // Render the pause menu (rectangles + text)
239            if let Err(e) = state
240                .pause_menu
241                .render(&state.wgpu_renderer.device, &mut render_pass)
242            {
243                println!("Failed to render pause menu: {}", e);
244            }
245        } else {
246            if state.pause_menu.is_visible() {
247                state.pause_menu.hide();
248            }
249            // Explicitly clear rectangles if menu is not visible
250            state
251                .pause_menu
252                .button_manager
253                .rectangle_renderer
254                .clear_rectangles();
255        }
256
257        window.request_redraw();
258
259        // Submit commands and present
260        state.wgpu_renderer.queue.submit(Some(encoder.finish()));
261
262        // Present the surface texture and ensure it's properly handled
263        surface_texture.present();
264
265        // Poll the device to process any pending operations
266        // This helps ensure resources are properly cleaned up and prevents
267        // the "SurfaceSemaphores still in use" error during cleanup
268        state.wgpu_renderer.device.poll(wgpu::Maintain::Poll);
269
270        // Update enemy pathfinding
271        state.game_state.enemy.update(
272            state.game_state.player.position,
273            state.game_state.delta_time,
274            state.game_state.game_ui.level as u32,
275            |from, to| {
276                state
277                    .game_state
278                    .collision_system
279                    .cylinder_intersects_geometry(from, to, 5.0)
280            },
281        );
282
283        // Handle title screen animation if needed
284        if state.game_state.current_screen == CurrentScreen::Loading {
285            state.game_state.game_ui.stop_timer();
286            self.handle_maze_generation();
287        } else if state.game_state.current_screen == CurrentScreen::NewGame {
288            state.text_renderer.hide_game_over_display();
289            self.new_level(true);
290        } else if state.game_state.current_screen == CurrentScreen::Game
291            && Some(state.game_state.player.current_cell) == state.game_state.exit_cell
292        {
293            state.game_state.enemy.pathfinder.position = [0.0, 30.0, 0.0];
294            state.game_state.enemy.pathfinder.locked = true;
295            self.new_level(false);
296        } else if state.game_state.current_screen == CurrentScreen::Game {
297            state
298                .game_state
299                .audio_manager
300                .resume_enemy_audio("enemy")
301                .expect("Failed to resume enemy audio");
302            if !state.game_state.is_test_mode {
303                state.game_state.enemy.pathfinder.locked = false;
304            }
305        }
306    }

308    // handle_frame_timing
309    pub fn handle_frame_timing(&mut self, current_time: Instant) {
310        if let Some(state) = self.state.as_mut() {
311            let duration = current_time.duration_since(state.game_state.last_fps_time);
312
313            state.elapsed_time = current_time.duration_since(state.start_time);
314            state.ui.elapsed_time += 1.0;
315            state.game_state.frame_count += 1;
316
317            if duration.as_secs_f32() >= 1.0 {
318                state.game_state.current_fps = state.game_state.frame_count;
319                state.game_state.frame_count = 0;
320                state.game_state.last_fps_time = current_time;
321            }
322
323            let delta_time = current_time
324                .duration_since(state.game_state.last_frame_time)
325                .as_secs_f32();
326
327            state.game_state.delta_time = delta_time;
328            state.game_state.last_frame_time = current_time;
329
330            if state
331                .wgpu_renderer
332                .game_renderer
333                .debug_renderer
334                .debug_render_bounding_boxes
335            {
336                state
337                    .wgpu_renderer
338                    .game_renderer
339                    .debug_renderer
340                    .update_debug_vertices(
341                        &state.wgpu_renderer.device,
342                        &state.game_state.collision_system,
343                    );
344            }
345        }
346    }

348    // handle_maze_generation
349    pub fn handle_maze_generation(&mut self) {
350        if let Some(state) = self.state.as_mut() {
351            // If in test mode, skip maze generation entirely and go directly to game
352            if state.game_state.is_test_mode {
353                println!("Test mode enabled - skipping maze generation and going to game");
354                // Mark generation as complete immediately
355                state
356                    .wgpu_renderer
357                    .loading_screen_renderer
358                    .generator
359                    .generation_complete = true;
360                // Set up test environment immediately
361                setup_test_environment(&mut state.game_state, &mut state.wgpu_renderer);
362                // Set a dummy maze path to prevent re-entry
363                state.game_state.maze_path = Some(std::path::PathBuf::from("test_mode"));
364                // Go directly to game screen
365                state.game_state.current_screen = CurrentScreen::Game;
366                return;
367            }
368
369            let renderer = &mut state.wgpu_renderer.loading_screen_renderer;
370
371            // Calculate update timing
372            let speed = if renderer.generator.fast_mode {
373                Duration::from_millis(10) / 20
374            } else {
375                Duration::from_millis(10)
376            };
377
378            // Skip if not time to update or already complete
379            if renderer.last_update.elapsed() < speed || renderer.generator.is_complete() {
380                return;
381            }
382
383            // Process animation steps
384            let steps = if renderer.generator.fast_mode {
385                300
386            } else {
387                100
388            };
389            for _ in 0..steps {
390                if !renderer.generator.step() {
391                    break;
392                }
393            }
394
395            // Report progress
396            let (current, total) = renderer.generator.get_progress();
397
398            if current % 50 == 0 || renderer.generator.is_complete() {
399                println!(
400                    "Progress: {}/{} ({:.1}%)",
401                    current,
402                    total,
403                    (current as f32 * 100.0 / total.max(1) as f32)
404                );
405            }
406            if renderer.generator.is_complete() && state.game_state.maze_path.is_none() {
407                println!("Maze generation complete! Saving to file...");
408
409                // Play completion sound
410                state
411                    .game_state
412                    .audio_manager
413                    .complete()
414                    .expect("Failed to play complete sound!");
415
416                // Handle completion
417                if renderer.generator.is_complete() {
418                    println!("Maze generation complete! Saving to file...");
419                    let maze_lock = renderer.maze.lock().unwrap();
420                    state.game_state.maze_path = maze_lock.save_to_file().map_or_else(
421                        |err| {
422                            eprintln!("Failed to save maze: {}", err);
423                            std::process::exit(1);
424                        },
425                        Some,
426                    );
427
428                    // Generate geometry if maze was saved successfully
429                    if let Some(maze_path) = &state.game_state.maze_path {
430                        let (maze_grid, exit_cell) = parse_maze_file(maze_path.to_str().unwrap());
431                        let (mut floor_vertices, exit_position) = Vertex::create_floor_vertices(
432                            &maze_grid,
433                            exit_cell,
434                            state.game_state.is_test_mode,
435                        );
436
437                        state.wgpu_renderer.game_renderer.exit_position = Some(exit_position);
438
439                        floor_vertices.append(&mut Vertex::create_wall_vertices(
440                            &maze_grid,
441                            state.game_state.is_test_mode,
442                        ));
443
444                        state.wgpu_renderer.game_renderer.vertex_buffer = state
445                            .wgpu_renderer
446                            .device
447                            .create_buffer_init(&wgpu::util::BufferInitDescriptor {
448                                label: Some("Combined Vertex Buffer"),
449                                contents: bytemuck::cast_slice(&floor_vertices),
450                                usage: wgpu::BufferUsages::VERTEX,
451                            });
452
453                        // Update vertex count so the renderer knows how many vertices to draw
454                        state.wgpu_renderer.game_renderer.vertex_count =
455                            floor_vertices.len() as u32;
456
457                        if let Some(exit_cell_position) = exit_cell {
458                            state.game_state.exit_cell = Some(exit_cell_position);
459                            state.game_state.enemy = place_enemy_standard(
460                                maze_to_world(
461                                    &exit_cell_position,
462                                    maze_lock.get_dimensions(),
463                                    30.0,
464                                    state.game_state.is_test_mode,
465                                ),
466                                state.game_state.player.position,
467                                state.game_state.game_ui.level,
468                                |from, to| {
469                                    state
470                                        .game_state
471                                        .collision_system
472                                        .cylinder_intersects_geometry(from, to, 5.0)
473                                },
474                            );
475                        }
476
477                        state
478                            .game_state
479                            .collision_system
480                            .build_from_maze(&maze_grid, state.game_state.is_test_mode);
481
482                        // Spawn the player at the bottom-left corner of the maze
483                        state
484                            .game_state
485                            .player
486                            .spawn_at_maze_entrance(&maze_grid, state.game_state.is_test_mode);
487                        // (No automatic transition to Game here)
488                    }
489                }
490            }
491        }
492    }

494    // new_level
495    pub fn new_level(&mut self, game_over: bool) {
496        let state = self
497            .state
498            .as_mut()
499            .expect("State must be initialized before use");
500        state.game_state.current_screen = CurrentScreen::Loading;
501        state.game_state.maze_path = None;
502        state.wgpu_renderer.loading_screen_renderer = LoadingRenderer::new(
503            &state.wgpu_renderer.device,
504            &state.wgpu_renderer.surface_config,
505        );
506
507        // Clear previous level state
508        state.game_state.player = Player::new();
509        state.game_state.enemy.pathfinder.position = [0.0, 30.0, 0.0];
510        state.game_state.enemy.pathfinder.locked = true;
511        state.game_state.exit_cell = None; // Clear exit cell to prevent accidental win condition
512
513        // Stop and reset timer
514        if let Some(timer) = &mut state.game_state.game_ui.timer {
515            timer.stop();
516            timer.reset();
517        }
518
519        if game_over {
520            state.game_state.set_level(1);
521            state.game_state.set_score(0);
522            state.game_state.game_ui.timer = Some(GameTimer::new(TimerConfig::default()));
523            game::update_game_ui(
524                &mut state.text_renderer,
525                &mut state.game_state.game_ui,
526                &state.game_state.current_screen,
527                self.window
528                    .as_ref()
529                    .expect("Window must be initialized before use"),
530            );
531            // Ensure clean state for new game
532            state.game_state.exit_cell = None;
533        } else {
534            let current_level = state.game_state.game_ui.level;
535
536            // Calculate completion time and performance metrics
537            let (completion_time, _) = if let Some(timer) = &state.game_state.game_ui.timer {
538                let remaining_time = timer.get_remaining_time().as_secs_f32();
539                let total_time = timer.config.duration.as_secs_f32();
540                let completion_time = total_time - remaining_time;
541
542                // Performance-based time bonus calculation
543                // Optimal time: 15 seconds, Average time: 25 seconds, Slow time: 35+ seconds
544                let time_bonus = if completion_time <= 15.0_f32 {
545                    // Exceptional performance: 15-25 seconds added
546                    let performance_ratio = (15.0 - completion_time).max(0.0_f32) / 15.0_f32;
547                    15.0_f32 + (performance_ratio * 10.0_f32)
548                } else if completion_time <= 25.0_f32 {
549                    // Good performance: 8-15 seconds added
550                    let performance_ratio = (25.0 - completion_time) / 10.0_f32;
551                    8.0_f32 + (performance_ratio * 7.0_f32)
552                } else if completion_time <= 35.0_f32 {
553                    // Average performance: 3-8 seconds added
554                    let performance_ratio = (35.0 - completion_time) / 10.0_f32;
555                    3.0_f32 + (performance_ratio * 5.0_f32)
556                } else {
557                    // Slow completion: 1-3 seconds added
558                    let performance_ratio = (45.0 - completion_time).max(0.0_f32) / 10.0_f32;
559                    1.0_f32 + (performance_ratio * 2.0_f32)
560                };
561
562                (completion_time, time_bonus)
563            } else {
564                (30.0_f32, 3.0_f32) // Fallback values
565            };
566
567            // Enhanced scoring system
568            let base_score = 150 * current_level as u32; // Increased base score
569
570            // Speed bonus calculation
571            let speed_bonus = if completion_time <= 15.0_f32 {
572                // Exceptional: 3x to 5x multiplier
573                let multiplier = 3.0_f32 + ((15.0_f32 - completion_time) / 15.0_f32) * 2.0_f32;
574                (base_score as f32 * multiplier) as u32
575            } else if completion_time <= 25.0_f32 {
576                // Good: 1.5x to 3x multiplier
577                let multiplier = 1.5_f32 + ((25.0_f32 - completion_time) / 10.0_f32) * 1.5_f32;
578                (base_score as f32 * multiplier) as u32
579            } else if completion_time <= 35.0_f32 {
580                // Average: 0.5x to 1.5x multiplier
581                let multiplier = 0.5_f32 + ((35.0_f32 - completion_time) / 10.0_f32) * 1.0_f32;
582                (base_score as f32 * multiplier) as u32
583            } else {
584                // Slow: 0.1x to 0.5x multiplier
585                let multiplier =
586                    0.1_f32 + ((45.0_f32 - completion_time).max(0.0_f32) / 10.0_f32) * 0.4_f32;
587                (base_score as f32 * multiplier) as u32
588            };
589
590            // Level progression bonus (small bonus for reaching higher levels)
591            let level_bonus = if current_level > 5 {
592                (current_level - 5) as u32 * 50
593            } else {
594                0
595            };
596
597            // Consecutive level bonus (reward for sustained performance)
598            let consecutive_bonus = if completion_time <= 20.0_f32 {
599                // Only give consecutive bonus for good performance
600                current_level as u32 * 25
601            } else {
602                0
603            };
604
605            let total_score = base_score + speed_bonus + level_bonus + consecutive_bonus;
606
607            // Update score and level
608            state
609                .game_state
610                .set_score(state.game_state.game_ui.score + total_score);
611            state.game_state.set_level(current_level + 1);
612
613            // Enhanced time management: Not supported in new timer, so skip add_time/subtract_time/prev_time
614        }
615    }
