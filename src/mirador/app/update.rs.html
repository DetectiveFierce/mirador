22    /// Handles the main rendering loop and game state updates.
23    ///
24    /// This method is called every frame and orchestrates the complete rendering pipeline.
25    /// It handles different game screens (loading, title, game, pause, upgrade menu),
26    /// updates game state, renders the scene, and manages UI overlays.
27    ///
28    /// # Rendering Pipeline
29    /// 1. **Screen-Specific Logic**: Handles different game screens appropriately
30    /// 2. **Game State Updates**: Updates player, enemy, audio, and UI systems
31    /// 3. **Rendering**: Creates command encoder and renders the scene
32    /// 4. **UI Overlays**: Renders pause menu, upgrade menu, and debug information
33    /// 5. **Frame Submission**: Submits commands and presents the frame
34    ///
35    /// # Screen Handling
36    /// - **Loading**: Handles maze generation and loading screen rendering
37    /// - **Title**: Renders title screen and handles transitions
38    /// - **Game**: Updates player movement, enemy AI, and game logic
39    /// - **Pause**: Renders pause menu overlay
40    /// - **UpgradeMenu**: Handles upgrade selection and menu interactions
41    ///
42    /// # Error Handling
43    /// - Logs errors for canvas update failures
44    /// - Continues execution even if some systems fail
45    /// - Provides debug backtraces in debug builds
46    ///
47    /// # Performance
48    /// - Skips rendering if window is minimized
49    /// - Uses efficient command encoding for GPU operations
50    /// - Manages GPU resource cleanup and polling

561    /// Updates frame timing and performance metrics.
562    ///
563    /// This method calculates delta time between frames, updates FPS counter,
564    /// and manages performance-related state. It's called every frame to ensure
565    /// smooth gameplay and accurate timing.
566    ///
567    /// # Arguments
568    /// - `current_time`: The current frame timestamp
569    ///
570    /// # Calculations
571    /// - **Delta Time**: Time elapsed since the last frame
572    /// - **FPS**: Frames per second, updated every second
573    /// - **Frame Count**: Total frames rendered since start
574    /// - **Elapsed Time**: Total time since application start
575    ///
576    /// # Performance Monitoring
577    /// - Updates debug renderer vertices if bounding box rendering is enabled
578    /// - Maintains accurate timing for game systems
579    /// - Provides timing data for performance analysis
580    ///
581    /// # Side Effects
582    /// - Updates `game_state.delta_time` for use by other systems
583    /// - Updates FPS counter and frame timing state
584    /// - Triggers debug renderer updates when needed
585    pub fn handle_frame_timing(&mut self, current_time: Instant) {
586        if let Some(state) = self.state.as_mut() {
587            let duration = current_time.duration_since(state.game_state.last_fps_time);
588
589            state.elapsed_time = current_time.duration_since(state.start_time);
590            state.game_state.frame_count += 1;
591
592            if duration.as_secs_f32() >= 1.0 {
593                state.game_state.current_fps = state.game_state.frame_count;
594                state.game_state.frame_count = 0;
595                state.game_state.last_fps_time = current_time;
596            }
597
598            let delta_time = current_time
599                .duration_since(state.game_state.last_frame_time)
600                .as_secs_f32();
601
602            state.game_state.delta_time = delta_time;
603            state.game_state.last_frame_time = current_time;
604
605            if state
606                .wgpu_renderer
607                .game_renderer
608                .debug_renderer
609                .debug_render_bounding_boxes
610            {
611                state
612                    .wgpu_renderer
613                    .game_renderer
614                    .debug_renderer
615                    .update_debug_vertices(
616                        &state.wgpu_renderer.device,
617                        &state.game_state.collision_system,
618                    );
619            }
620        }
621    }

623    /// Handles procedural maze generation and loading screen logic.
624    ///
625    /// This method manages the maze generation process, which can be either
626    /// animated (showing generation steps) or instant (fast mode). It handles
627    /// both normal maze generation and test mode maze creation.
628    ///
629    /// # Generation Process
630    /// 1. **Test Mode Check**: Skips generation and goes directly to game in test mode
631    /// 2. **Animation Control**: Manages generation speed and progress reporting
632    /// 3. **Completion Handling**: Saves maze to file and generates geometry
633    /// 4. **State Setup**: Initializes player, enemy, and collision systems
634    ///
635    /// # Animation Modes
636    /// - **Normal Mode**: Shows generation steps with configurable speed
637    /// - **Fast Mode**: Completes generation quickly for faster gameplay
638    /// - **Test Mode**: Uses predefined test maze instead of generation
639    ///
640    /// # Progress Reporting
641    /// - Logs generation progress every 50 steps
642    /// - Shows completion percentage
643    /// - Automatically completes generation when 70%+ complete
644    ///
645    /// # Maze Completion
646    /// - Saves maze to file with timestamp
647    /// - Generates floor, wall, and ceiling geometry
648    /// - Places player at maze entrance
649    /// - Places enemy based on exit position
650    /// - Builds collision system from maze data
651    ///
652    /// # Audio Integration
653    /// - Plays completion sound when generation finishes
654    /// - Manages audio state during generation process
655    pub fn handle_maze_generation(&mut self) {
656        if let Some(state) = self.state.as_mut() {
657            // Start timing maze generation
658            state.profiler.start_section("maze_generation");
659            // If in test mode, skip maze generation entirely and go directly to game
660            if state.game_state.is_test_mode {
661                println!("Test mode enabled - skipping maze generation and going to game");
662                // Mark generation as complete immediately
663                state
664                    .wgpu_renderer
665                    .loading_screen_renderer
666                    .generator
667                    .generation_complete = true;
668                // Set up test environment immediately
669                setup_test_environment(&mut state.game_state, &mut state.wgpu_renderer);
670                // Set a dummy maze path to prevent re-entry
671                state.game_state.maze_path = Some(std::path::PathBuf::from("test_mode"));
672                // Go directly to game screen
673                state.game_state.current_screen = CurrentScreen::Game;
674                return;
675            }
676
677            let renderer = &mut state.wgpu_renderer.loading_screen_renderer;
678
679            // Calculate update timing
680            let speed = if renderer.generator.fast_mode {
681                Duration::from_millis(10) / 20
682            } else {
683                Duration::from_millis(10)
684            };
685
686            // Skip if not time to update or already complete
687            if renderer.last_update.elapsed() < speed || renderer.generator.is_complete() {
688                return;
689            }
690
691            // Process animation steps
692            let steps = if renderer.generator.fast_mode {
693                2000
694            } else {
695                100
696            };
697            state.profiler.start_section("maze_generation_steps");
698            for _ in 0..steps {
699                if !renderer.generator.step() {
700                    break;
701                }
702            }
703            state.profiler.end_section("maze_generation_steps");
704
705            // Report progress
706            let (current, total) = renderer.generator.get_progress();
707
708            if current % 50 == 0 || renderer.generator.is_complete() {
709                println!(
710                    "Progress: {}/{} ({:.1}%)",
711                    current,
712                    total,
713                    (current as f32 * 100.0 / total.max(1) as f32)
714                );
715            }
716
717            // Complete generation all at once if less than 10% remains
718            let progress_ratio = current as f32 / total.max(1) as f32;
719            if progress_ratio > 0.7 && !renderer.generator.is_complete() {
720                while !renderer.generator.is_complete() {
721                    renderer.generator.step();
722                }
723            }
724
725            if renderer.generator.is_complete() && state.game_state.maze_path.is_none() {
726                println!("Maze generation complete! Saving to file...");
727
728                // Play completion sound
729                state
730                    .game_state
731                    .audio_manager
732                    .complete()
733                    .expect("Failed to play complete sound!");
734
735                // --- DEBUG PRINT: Applied Upgrades and Stats (print once, with complete sound) ---
736                println!("\n=== [DEBUG] Applied Upgrades and Influenced Stats ===");
737                use crate::game::upgrades::AvailableUpgrade;
738                let mgr = &state.upgrade_menu.upgrade_manager;
739                let player = &state.game_state.player;
740                for upgrade in [
741                    AvailableUpgrade::SpeedUp,
742                    AvailableUpgrade::Dash,
743                    AvailableUpgrade::TallBoots,
744                    AvailableUpgrade::SlowTime,
745                    AvailableUpgrade::SilentStep,
746                    AvailableUpgrade::HeadStart,
747                    AvailableUpgrade::Unknown,
748                ] {
749                    let count = mgr.get_upgrade_count(&upgrade);
750                    if count > 0 {
751                        let name = upgrade.to_upgrade().name;
752                        let stat = match upgrade {
753                            AvailableUpgrade::SpeedUp => {
754                                format!("base_speed: {:.2}", player.base_speed)
755                            }
756                            AvailableUpgrade::Dash => {
757                                format!("max_stamina: {:.2}", player.max_stamina)
758                            }
759                            AvailableUpgrade::TallBoots => {
760                                format!("height: {:.2}", player.position[1])
761                            }
762                            AvailableUpgrade::SlowTime => format!("timer: {}s", player.max_stamina), // Timer is in game_state, but not directly accessible here; placeholder
763                            AvailableUpgrade::SilentStep => "enemy pathfinding penalty".to_string(),
764                            AvailableUpgrade::HeadStart => "enemy lock time".to_string(),
765                            AvailableUpgrade::Unknown => "???".to_string(),
766                        };
767                        println!("- {} (x{}): {}", name, count, stat);
768                    }
769                }
770                println!("===============================================\n");
771                // Handle completion
772                if renderer.generator.is_complete() {
773                    println!("Maze generation complete! Saving to file...");
774                    state.profiler.start_section("maze_completion_processing");
775
776                    let maze_lock = renderer.maze.lock().expect("Failed to lock maze");
777                    state.game_state.maze_path = maze_lock.save_to_file().map_or_else(
778                        |err| {
779                            eprintln!("Failed to save maze: {}", err);
780                            std::process::exit(1);
781                        },
782                        Some,
783                    );
784
785                    // Generate geometry if maze was saved successfully
786                    if let Some(maze_path) = &state.game_state.maze_path {
787                        state.profiler.start_section("maze_geometry_generation");
788                        let (maze_grid, exit_cell) = parse_maze_file(
789                            maze_path
790                                .to_str()
791                                .expect("Failed to convert path to string"),
792                        );
793                        let (mut floor_vertices, exit_position) = Vertex::create_floor_vertices(
794                            &maze_grid,
795                            exit_cell,
796                            state.game_state.is_test_mode,
797                        );
798
799                        state.wgpu_renderer.game_renderer.exit_position = Some(exit_position);
800
801                        floor_vertices.append(&mut Vertex::create_wall_vertices(
802                            &maze_grid,
803                            state.game_state.is_test_mode,
804                        ));
805
806                        // Add ceiling vertices
807                        floor_vertices.append(&mut Vertex::create_ceiling_vertices(
808                            &maze_grid,
809                            state.game_state.is_test_mode,
810                        ));
811
812                        state.wgpu_renderer.game_renderer.vertex_buffer = state
813                            .wgpu_renderer
814                            .device
815                            .create_buffer_init(&wgpu::util::BufferInitDescriptor {
816                                label: Some("Combined Vertex Buffer"),
817                                contents: bytemuck::cast_slice(&floor_vertices),
818                                usage: wgpu::BufferUsages::VERTEX,
819                            });
820
821                        // Update vertex count so the renderer knows how many vertices to draw
822                        state.wgpu_renderer.game_renderer.vertex_count =
823                            floor_vertices.len() as u32;
824                        state.profiler.end_section("maze_geometry_generation");
825
826                        if let Some(exit_cell_position) = exit_cell {
827                            state.profiler.start_section("enemy_placement");
828                            state.game_state.exit_cell = Some(exit_cell_position);
829                            state.game_state.enemy = place_enemy_standard(
830                                maze_to_world(
831                                    &exit_cell_position,
832                                    maze_lock.get_dimensions(),
833                                    30.0,
834                                    state.game_state.is_test_mode,
835                                ),
836                                state.game_state.player.position,
837                                state.game_state.game_ui.level,
838                                |from, to| {
839                                    state
840                                        .game_state
841                                        .collision_system
842                                        .cylinder_intersects_geometry(from, to, 5.0)
843                                },
844                            );
845                            state.profiler.end_section("enemy_placement");
846                        }
847
848                        state.profiler.start_section("collision_system_build");
849                        state
850                            .game_state
851                            .collision_system
852                            .build_from_maze(&maze_grid, state.game_state.is_test_mode);
853                        state.profiler.end_section("collision_system_build");
854
855                        // Spawn the player at the bottom-left corner of the maze
856                        state
857                            .game_state
858                            .player
859                            .spawn_at_maze_entrance(&maze_grid, state.game_state.is_test_mode);
860                        // (No automatic transition to Game here)
861                    }
862
863                    state.profiler.end_section("maze_completion_processing");
864                }
865            }
866        }
867
868        // End timing maze generation
869        if let Some(state) = self.state.as_mut() {
870            state.profiler.end_section("maze_generation");
871        }
872    }

874    /// Initializes a new level or restarts the game.
875    ///
876    /// This method handles the transition to a new level or game restart. It manages
877    /// player state, enemy positioning, timer configuration, and scoring systems.
878    /// The method implements a sophisticated scoring system based on completion time
879    /// and performance metrics.
880    ///
881    /// # Arguments
882    /// - `game_over`: Whether this is a game restart (true) or level progression (false)
883    ///
884    /// # Game Restart (game_over = true)
885    /// - Resets player to starting position and stats
886    /// - Resets level to 1 and score to 0
887    /// - Initializes new timer with default configuration
888    /// - Restarts background music
889    /// - Clears all upgrade effects
890    ///
891    /// # Level Progression (game_over = false)
892    /// - Preserves player stats and upgrades
893    /// - Resets only position and orientation
894    /// - Calculates performance-based scoring
895    /// - Updates level counter
896    /// - Maintains upgrade effects
897    ///
898    /// # Scoring System
899    /// The scoring system rewards performance with multiple components:
900    /// - **Base Score**: 150 points per level
901    /// - **Speed Bonus**: Multiplier based on completion time
902    ///   - Exceptional (≤15s): 3x-5x multiplier
903    ///   - Good (≤25s): 1.5x-3x multiplier
904    ///   - Average (≤35s): 0.5x-1.5x multiplier
905    ///   - Slow (>35s): 0.1x-0.5x multiplier
906    /// - **Level Bonus**: Additional points for higher levels (level > 5)
907    /// - **Consecutive Bonus**: Reward for sustained good performance
908    ///
909    /// # State Management
910    /// - Resets maze path to trigger new generation
911    /// - Clears exit cell and timer state
912    /// - Resets enemy position and locks enemy
913    /// - Manages audio state transitions
914    ///
915    /// # Player State
916    /// - **Game Restart**: Complete reset of all stats
917    /// - **Level Progression**: Preserves upgrades, resets position only
918    /// - **Height Adjustment**: Accounts for TallBoots upgrades
919    /// - **Stamina Reset**: Refills stamina for new level
920    pub fn new_level(&mut self, game_over: bool) {
921        let state = self
922            .state
923            .as_mut()
924            .expect("State must be initialized before use");
925        state.game_state.current_screen = CurrentScreen::Loading;
926        state.game_state.maze_path = None;
927        state.wgpu_renderer.loading_screen_renderer = LoadingRenderer::new(
928            &state.wgpu_renderer.device,
929            &state.wgpu_renderer.surface_config,
930        );
931
932        // Clear previous level state
933        if game_over {
934            state.game_state.player = Player::new();
935        } else {
936            // Only reset position (x/z), orientation, and cell, not stats or height
937            let player = &mut state.game_state.player;
938            player.position[0] = 0.0;
939            player.position[2] = 0.0;
940            // Set height based on TallBoots upgrades
941            let tall_boots_count = state
942                .upgrade_menu
943                .upgrade_manager
944                .get_upgrade_count(&crate::game::upgrades::AvailableUpgrade::TallBoots);
945            player.position[1] = crate::math::coordinates::constants::PLAYER_HEIGHT
946                + 5.0 * (tall_boots_count as f32);
947            player.pitch = 3.0;
948            player.yaw = 316.0;
949            player.fov = 100.0;
950            player.current_cell = crate::game::maze::generator::Cell::default();
951            // Optionally, reset stamina to max for new level:
952            player.stamina = player.max_stamina;
953            // (Do not reset base_speed, max_stamina, regen rates, etc.)
954        }
955        state.game_state.enemy.pathfinder.position = [0.0, 30.0, 0.0];
956        state.game_state.enemy.pathfinder.locked = true;
957        state.game_state.exit_cell = None; // Clear exit cell to prevent accidental win condition
958        state.game_state.exit_reached_timer = 0.0; // Reset exit reached timer
959        state.game_state.beeper_rise_played = false; // Reset beeper rise played flag
960
961        // Stop and reset timer
962        if let Some(timer) = &mut state.game_state.game_ui.timer {
963            timer.stop();
964            timer.reset();
965        }
966
967        if game_over {
968            state.game_state.set_level(1);
969            state.game_state.set_score(0);
970            state.game_state.game_ui.timer = Some(GameTimer::new(TimerConfig::default()));
971
972            // Restart background music for new game
973            state
974                .game_state
975                .audio_manager
976                .restart_background_music()
977                .expect("Failed to restart background music");
978
979            // Set game audio volumes after restarting background music
980            state
981                .game_state
982                .audio_manager
983                .set_game_volumes()
984                .expect("Failed to set game volumes");
985
986            game::update_game_ui(
987                &mut state.text_renderer,
988                &mut state.game_state.game_ui,
989                &state.game_state.current_screen,
990                self.window
991                    .as_ref()
992                    .expect("Window must be initialized before use"),
993            );
994            // Ensure clean state for new game
995            state.game_state.exit_cell = None;
996        } else {
997            let current_level = state.game_state.game_ui.level;
998
999            // Calculate completion time and performance metrics
1000            let (completion_time, _) = if let Some(timer) = &state.game_state.game_ui.timer {
1001                let remaining_time = timer.get_remaining_time().as_secs_f32();
1002                let total_time = timer.config.duration.as_secs_f32();
1003                let completion_time = total_time - remaining_time;
1004
1005                // Performance-based time bonus calculation
1006                // Optimal time: 15 seconds, Average time: 25 seconds, Slow time: 35+ seconds
1007                let time_bonus = if completion_time <= 15.0_f32 {
1008                    // Exceptional performance: 15-25 seconds added
1009                    let performance_ratio = (15.0 - completion_time).max(0.0_f32) / 15.0_f32;
1010                    15.0_f32 + (performance_ratio * 10.0_f32)
1011                } else if completion_time <= 25.0_f32 {
1012                    // Good performance: 8-15 seconds added
1013                    let performance_ratio = (25.0 - completion_time) / 10.0_f32;
1014                    8.0_f32 + (performance_ratio * 7.0_f32)
1015                } else if completion_time <= 35.0_f32 {
1016                    // Average performance: 3-8 seconds added
1017                    let performance_ratio = (35.0 - completion_time) / 10.0_f32;
1018                    3.0_f32 + (performance_ratio * 5.0_f32)
1019                } else {
1020                    // Slow completion: 1-3 seconds added
1021                    let performance_ratio = (45.0 - completion_time).max(0.0_f32) / 10.0_f32;
1022                    1.0_f32 + (performance_ratio * 2.0_f32)
1023                };
1024
1025                (completion_time, time_bonus)
1026            } else {
1027                (30.0_f32, 3.0_f32) // Fallback values
1028            };
1029
1030            // Enhanced scoring system
1031            let base_score = 150 * current_level as u32; // Increased base score
1032
1033            // Speed bonus calculation
1034            let speed_bonus = if completion_time <= 15.0_f32 {
1035                // Exceptional: 3x to 5x multiplier
1036                let multiplier = 3.0_f32 + ((15.0_f32 - completion_time) / 15.0_f32) * 2.0_f32;
1037                (base_score as f32 * multiplier) as u32
1038            } else if completion_time <= 25.0_f32 {
1039                // Good: 1.5x to 3x multiplier
1040                let multiplier = 1.5_f32 + ((25.0_f32 - completion_time) / 10.0_f32) * 1.5_f32;
1041                (base_score as f32 * multiplier) as u32
1042            } else if completion_time <= 35.0_f32 {
1043                // Average: 0.5x to 1.5x multiplier
1044                let multiplier = 0.5_f32 + ((35.0_f32 - completion_time) / 10.0_f32) * 1.0_f32;
1045                (base_score as f32 * multiplier) as u32
1046            } else {
1047                // Slow: 0.1x to 0.5x multiplier
1048                let multiplier =
1049                    0.1_f32 + ((45.0_f32 - completion_time).max(0.0_f32) / 10.0_f32) * 0.4_f32;
1050                (base_score as f32 * multiplier) as u32
1051            };
1052
1053            // Level progression bonus (small bonus for reaching higher levels)
1054            let level_bonus = if current_level > 5 {
1055                (current_level - 5) as u32 * 50
1056            } else {
1057                0
1058            };
1059
1060            // Consecutive level bonus (reward for sustained performance)
1061            let consecutive_bonus = if completion_time <= 20.0_f32 {
1062                // Only give consecutive bonus for good performance
1063                current_level as u32 * 25
1064            } else {
1065                0
1066            };
1067
1068            let total_score = base_score + speed_bonus + level_bonus + consecutive_bonus;
1069
1070            // Update score and level
1071            state
1072                .game_state
1073                .set_score(state.game_state.game_ui.score + total_score);
1074            state.game_state.set_level(current_level + 1);
1075
1076            // Enhanced time management: Not supported in new timer, so skip add_time/subtract_time/prev_time
1077        }
1078    }
