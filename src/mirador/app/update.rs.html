23        let window = self
24            .window
25            .as_ref()
26            .expect("Window must be initialized before use");
27        if window.is_minimized().unwrap_or(false) {
28            println!("Window is minimized");
29            return;
30        }
31
32        let state = self
33            .state
34            .as_mut()
35            .expect("State must be initialized before use");
36
37        if state.game_state.current_screen == CurrentScreen::Loading {
38            state
39                .game_state
40                .audio_manager
41                .pause_enemy_audio("enemy")

43            state.handle_loading_screen(window);
44        } else if state.game_state.current_screen == CurrentScreen::Title {
45            crate::renderer::title::handle_title(state, window);
46            state.upgrade_menu.upgrade_manager.player_upgrades.clear();
47            state.game_state.player = crate::game::player::Player::new();
48            state.game_state.enemy = crate::game::enemy::Enemy::new([0.0, 30.0, 0.0], 150.0);
49            return;
50        } else if state.game_state.current_screen == CurrentScreen::UpgradeMenu {
51            // Handle upgrade menu - just update it, rendering is handled separately
52            state.upgrade_menu.update();
53            // Pass player and game_state to handle_input if needed (if input is handled here)
54
55            // Check if upgrade menu is no longer visible (upgrade was selected)
56            if !state.upgrade_menu.is_visible() {
57                // Continue to next level
58                state.game_state.current_screen = CurrentScreen::Loading;
59                // Recapture mouse when leaving upgrade menu
60                state.game_state.capture_mouse = true;
61                if let Some(window) = self.window.as_ref() {
62                    state.triage_mouse(window);
63                }
64                {
65                    // Limit the mutable borrow of self/state to this block
66                    self.new_level(false);
67                }
68                // Now that the mutable borrow is released, we can safely borrow state again
69                let state = self
70                    .state
71                    .as_mut()
72                    .expect("State must be initialized before use");
73                state
74                    .upgrade_menu
75                    .apply_upgrade_effects(&mut state.game_state);
76                return;
77            } else {
78                state.game_state.capture_mouse = false;
79            }
80            // Don't return early - let the normal rendering pipeline continue
81        } else {
82            state.game_state.player.update_cell(
83                &state
84                    .wgpu_renderer

103            .game_state

105            .update_enemy_position("enemy", state.game_state.enemy.pathfinder.position)
106            .expect("Failed to update enemy position");
107
108        // Update audio manager to process any pending audio operations

278                .pause_menu
279                .render(&state.wgpu_renderer.device, &mut render_pass)

281                println!("Failed to render pause menu: {}", e);

306            // Create a render pass for the upgrade menu
307            let mut render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
308                color_attachments: &[Some(wgpu::RenderPassColorAttachment {

391                }

396        }

398        // Update enemy pathfinding

400            state.game_state.player.position,

402            state.game_state.game_ui.level as u32,

405                    .game_state

409        );

411        // Handle title screen animation if needed
412        if state.game_state.current_screen == CurrentScreen::Loading {
413            state.game_state.game_ui.stop_timer();
414            self.handle_maze_generation();
415        } else if state.game_state.current_screen == CurrentScreen::NewGame {
416            state.text_renderer.hide_game_over_display();
417            state.upgrade_menu.upgrade_manager.player_upgrades.clear();

425            state.game_state.current_screen = CurrentScreen::ExitReached;
426            state.game_state.exit_reached_timer = 0.0;
427            state.game_state.enemy.pathfinder.position = [0.0, 30.0, 0.0];

429            if !state.game_state.beeper_rise_played {
430                let _ = state.game_state.audio_manager.play_beeper_rise();
431                state.game_state.beeper_rise_played = true;
432            }
433        } else if state.game_state.current_screen == CurrentScreen::ExitReached {
434            // Handle exit reached upward movement
435            state.game_state.exit_reached_timer += state.game_state.delta_time;
436
437            // Move player upward for 3 seconds
438            if state.game_state.exit_reached_timer < 1.0 {
439                state.game_state.player.move_up(state.game_state.delta_time);
440            } else {
441                // After 3 seconds, transition to appropriate next screen
442                let current_level = state.game_state.game_ui.level;
443                if current_level > 0 && current_level % 3 == 0 {
444                    // Show upgrade menu
445                    state.game_state.current_screen = CurrentScreen::UpgradeMenu;
446                    state.upgrade_menu.show();
447                } else {
448                    // Continue to next level
449                    self.new_level(false);
450                }
451            }
452        } else if state.game_state.current_screen == CurrentScreen::Game {
453            state
454                .game_state
455                .audio_manager
456                .resume_enemy_audio("enemy")
457                .expect("Failed to resume enemy audio");
458        }
459    }

461    // handle_frame_timing
462    pub fn handle_frame_timing(&mut self, current_time: Instant) {
463        if let Some(state) = self.state.as_mut() {
464            let duration = current_time.duration_since(state.game_state.last_fps_time);
465
466            state.elapsed_time = current_time.duration_since(state.start_time);
467            state.game_state.frame_count += 1;
468
469            if duration.as_secs_f32() >= 1.0 {
470                state.game_state.current_fps = state.game_state.frame_count;
471                state.game_state.frame_count = 0;
472                state.game_state.last_fps_time = current_time;
473            }
474
475            let delta_time = current_time
476                .duration_since(state.game_state.last_frame_time)
477                .as_secs_f32();
478
479            state.game_state.delta_time = delta_time;
480            state.game_state.last_frame_time = current_time;
481
482            if state
483                .wgpu_renderer
484                .game_renderer
485                .debug_renderer
486                .debug_render_bounding_boxes
487            {
488                state
489                    .wgpu_renderer
490                    .game_renderer
491                    .debug_renderer
492                    .update_debug_vertices(
493                        &state.wgpu_renderer.device,
494                        &state.game_state.collision_system,
495                    );
496            }
497        }
498    }

500    // handle_maze_generation
501    pub fn handle_maze_generation(&mut self) {
502        if let Some(state) = self.state.as_mut() {
503            // If in test mode, skip maze generation entirely and go directly to game
504            if state.game_state.is_test_mode {
505                println!("Test mode enabled - skipping maze generation and going to game");
506                // Mark generation as complete immediately
507                state
508                    .wgpu_renderer
509                    .loading_screen_renderer
510                    .generator
511                    .generation_complete = true;
512                // Set up test environment immediately
513                setup_test_environment(&mut state.game_state, &mut state.wgpu_renderer);
514                // Set a dummy maze path to prevent re-entry
515                state.game_state.maze_path = Some(std::path::PathBuf::from("test_mode"));
516                // Go directly to game screen
517                state.game_state.current_screen = CurrentScreen::Game;
518                return;
519            }
520
521            let renderer = &mut state.wgpu_renderer.loading_screen_renderer;
522
523            // Calculate update timing
524            let speed = if renderer.generator.fast_mode {
525                Duration::from_millis(10) / 20
526            } else {
527                Duration::from_millis(10)
528            };
529
530            // Skip if not time to update or already complete
531            if renderer.last_update.elapsed() < speed || renderer.generator.is_complete() {
532                return;
533            }
534
535            // Process animation steps
536            let steps = if renderer.generator.fast_mode {
537                2000
538            } else {
539                100
540            };
541            for _ in 0..steps {
542                if !renderer.generator.step() {
543                    break;
544                }
545            }
546
547            // Report progress
548            let (current, total) = renderer.generator.get_progress();
549
550            if current % 50 == 0 || renderer.generator.is_complete() {
551                println!(
552                    "Progress: {}/{} ({:.1}%)",
553                    current,
554                    total,
555                    (current as f32 * 100.0 / total.max(1) as f32)
556                );
557            }
558
559            // Complete generation all at once if less than 10% remains
560            let progress_ratio = current as f32 / total.max(1) as f32;
561            if progress_ratio > 0.7 && !renderer.generator.is_complete() {
562                while !renderer.generator.is_complete() {
563                    renderer.generator.step();
564                }
565            }
566
567            if renderer.generator.is_complete() && state.game_state.maze_path.is_none() {
568                println!("Maze generation complete! Saving to file...");
569
570                // Play completion sound
571                state
572                    .game_state
573                    .audio_manager
574                    .complete()
575                    .expect("Failed to play complete sound!");
576
577                // --- DEBUG PRINT: Applied Upgrades and Stats (print once, with complete sound) ---
578                println!("\n=== [DEBUG] Applied Upgrades and Influenced Stats ===");
579                use crate::game::upgrades::AvailableUpgrade;
580                let mgr = &state.upgrade_menu.upgrade_manager;
581                let player = &state.game_state.player;
582                for upgrade in [
583                    AvailableUpgrade::SpeedUp,
584                    AvailableUpgrade::Dash,
585                    AvailableUpgrade::TallBoots,
586                    AvailableUpgrade::SlowTime,
587                    AvailableUpgrade::SilentStep,
588                    AvailableUpgrade::HeadStart,
589                    AvailableUpgrade::Unknown,
590                ] {
591                    let count = mgr.get_upgrade_count(&upgrade);
592                    if count > 0 {
593                        let name = upgrade.to_upgrade().name;
594                        let stat = match upgrade {
595                            AvailableUpgrade::SpeedUp => {
596                                format!("base_speed: {:.2}", player.base_speed)
597                            }
598                            AvailableUpgrade::Dash => {
599                                format!("max_stamina: {:.2}", player.max_stamina)
600                            }
601                            AvailableUpgrade::TallBoots => {
602                                format!("height: {:.2}", player.position[1])
603                            }
604                            AvailableUpgrade::SlowTime => format!("timer: {}s", player.max_stamina), // Timer is in game_state, but not directly accessible here; placeholder
605                            AvailableUpgrade::SilentStep => "enemy pathfinding penalty".to_string(),
606                            AvailableUpgrade::HeadStart => "enemy lock time".to_string(),
607                            AvailableUpgrade::Unknown => "???".to_string(),
608                        };
609                        println!("- {} (x{}): {}", name, count, stat);
610                    }
611                }
612                println!("===============================================\n");
613                // Handle completion
614                if renderer.generator.is_complete() {
615                    println!("Maze generation complete! Saving to file...");
616                    let maze_lock = renderer.maze.lock().expect("Failed to lock maze");
617                    state.game_state.maze_path = maze_lock.save_to_file().map_or_else(
618                        |err| {
619                            eprintln!("Failed to save maze: {}", err);
620                            std::process::exit(1);
621                        },
622                        Some,
623                    );
624
625                    // Generate geometry if maze was saved successfully
626                    if let Some(maze_path) = &state.game_state.maze_path {
627                        let (maze_grid, exit_cell) = parse_maze_file(
628                            maze_path
629                                .to_str()
630                                .expect("Failed to convert path to string"),
631                        );
632                        let (mut floor_vertices, exit_position) = Vertex::create_floor_vertices(
633                            &maze_grid,
634                            exit_cell,
635                            state.game_state.is_test_mode,
636                        );
637
638                        state.wgpu_renderer.game_renderer.exit_position = Some(exit_position);
639
640                        floor_vertices.append(&mut Vertex::create_wall_vertices(
641                            &maze_grid,
642                            state.game_state.is_test_mode,
643                        ));
644
645                        // Add ceiling vertices
646                        floor_vertices.append(&mut Vertex::create_ceiling_vertices(
647                            &maze_grid,
648                            state.game_state.is_test_mode,
649                        ));
650
651                        state.wgpu_renderer.game_renderer.vertex_buffer = state
652                            .wgpu_renderer
653                            .device
654                            .create_buffer_init(&wgpu::util::BufferInitDescriptor {
655                                label: Some("Combined Vertex Buffer"),
656                                contents: bytemuck::cast_slice(&floor_vertices),
657                                usage: wgpu::BufferUsages::VERTEX,
658                            });
659
660                        // Update vertex count so the renderer knows how many vertices to draw
661                        state.wgpu_renderer.game_renderer.vertex_count =
662                            floor_vertices.len() as u32;
663
664                        if let Some(exit_cell_position) = exit_cell {
665                            state.game_state.exit_cell = Some(exit_cell_position);
666                            state.game_state.enemy = place_enemy_standard(
667                                maze_to_world(
668                                    &exit_cell_position,
669                                    maze_lock.get_dimensions(),
670                                    30.0,
671                                    state.game_state.is_test_mode,
672                                ),
673                                state.game_state.player.position,
674                                state.game_state.game_ui.level,
675                                |from, to| {
676                                    state
677                                        .game_state
678                                        .collision_system
679                                        .cylinder_intersects_geometry(from, to, 5.0)
680                                },
681                            );
682                        }
683
684                        state
685                            .game_state
686                            .collision_system
687                            .build_from_maze(&maze_grid, state.game_state.is_test_mode);
688
689                        // Spawn the player at the bottom-left corner of the maze
690                        state
691                            .game_state
692                            .player
693                            .spawn_at_maze_entrance(&maze_grid, state.game_state.is_test_mode);
694                        // (No automatic transition to Game here)
695                    }
696                }
697            }
698        }
699    }

701    // new_level
702    pub fn new_level(&mut self, game_over: bool) {
703        let state = self
704            .state
705            .as_mut()
706            .expect("State must be initialized before use");
707        state.game_state.current_screen = CurrentScreen::Loading;
708        state.game_state.maze_path = None;
709        state.wgpu_renderer.loading_screen_renderer = LoadingRenderer::new(
710            &state.wgpu_renderer.device,
711            &state.wgpu_renderer.surface_config,
712        );
713
714        // Clear previous level state
715        if game_over {
716            state.game_state.player = Player::new();
717        } else {
718            // Only reset position (x/z), orientation, and cell, not stats or height
719            let player = &mut state.game_state.player;
720            player.position[0] = 0.0;
721            player.position[2] = 0.0;
722            // Set height based on TallBoots upgrades
723            let tall_boots_count = state
724                .upgrade_menu
725                .upgrade_manager
726                .get_upgrade_count(&crate::game::upgrades::AvailableUpgrade::TallBoots);
727            player.position[1] = crate::math::coordinates::constants::PLAYER_HEIGHT
728                + 3.0 * (tall_boots_count as f32);
729            player.pitch = 3.0;
730            player.yaw = 316.0;
731            player.fov = 100.0;
732            player.current_cell = crate::game::maze::generator::Cell::default();
733            // Optionally, reset stamina to max for new level:
734            player.stamina = player.max_stamina;
735            // (Do not reset base_speed, max_stamina, regen rates, etc.)
736        }
737        state.game_state.enemy.pathfinder.position = [0.0, 30.0, 0.0];
738        state.game_state.enemy.pathfinder.locked = true;
739        state.game_state.exit_cell = None; // Clear exit cell to prevent accidental win condition
740        state.game_state.exit_reached_timer = 0.0; // Reset exit reached timer
741        state.game_state.beeper_rise_played = false; // Reset beeper rise played flag
742
743        // Stop and reset timer
744        if let Some(timer) = &mut state.game_state.game_ui.timer {
745            timer.stop();
746            timer.reset();
747        }
748
749        if game_over {
750            state.game_state.set_level(1);
751            state.game_state.set_score(0);
752            state.game_state.game_ui.timer = Some(GameTimer::new(TimerConfig::default()));
753
754            // Restart background music for new game
755            state
756                .game_state
757                .audio_manager
758                .restart_background_music()
759                .expect("Failed to restart background music");
760
761            // Set game audio volumes after restarting background music
762            state
763                .game_state
764                .audio_manager
765                .set_game_volumes()
766                .expect("Failed to set game volumes");
767
768            game::update_game_ui(
769                &mut state.text_renderer,
770                &mut state.game_state.game_ui,
771                &state.game_state.current_screen,
772                self.window
773                    .as_ref()
774                    .expect("Window must be initialized before use"),
775            );
776            // Ensure clean state for new game
777            state.game_state.exit_cell = None;
778        } else {
779            let current_level = state.game_state.game_ui.level;
780
781            // Calculate completion time and performance metrics
782            let (completion_time, _) = if let Some(timer) = &state.game_state.game_ui.timer {
783                let remaining_time = timer.get_remaining_time().as_secs_f32();
784                let total_time = timer.config.duration.as_secs_f32();
785                let completion_time = total_time - remaining_time;
786
787                // Performance-based time bonus calculation
788                // Optimal time: 15 seconds, Average time: 25 seconds, Slow time: 35+ seconds
789                let time_bonus = if completion_time <= 15.0_f32 {
790                    // Exceptional performance: 15-25 seconds added
791                    let performance_ratio = (15.0 - completion_time).max(0.0_f32) / 15.0_f32;
792                    15.0_f32 + (performance_ratio * 10.0_f32)
793                } else if completion_time <= 25.0_f32 {
794                    // Good performance: 8-15 seconds added
795                    let performance_ratio = (25.0 - completion_time) / 10.0_f32;
796                    8.0_f32 + (performance_ratio * 7.0_f32)
797                } else if completion_time <= 35.0_f32 {
798                    // Average performance: 3-8 seconds added
799                    let performance_ratio = (35.0 - completion_time) / 10.0_f32;
800                    3.0_f32 + (performance_ratio * 5.0_f32)
801                } else {
802                    // Slow completion: 1-3 seconds added
803                    let performance_ratio = (45.0 - completion_time).max(0.0_f32) / 10.0_f32;
804                    1.0_f32 + (performance_ratio * 2.0_f32)
805                };
806
807                (completion_time, time_bonus)
808            } else {
809                (30.0_f32, 3.0_f32) // Fallback values
810            };
811
812            // Enhanced scoring system
813            let base_score = 150 * current_level as u32; // Increased base score
814
815            // Speed bonus calculation
816            let speed_bonus = if completion_time <= 15.0_f32 {
817                // Exceptional: 3x to 5x multiplier
818                let multiplier = 3.0_f32 + ((15.0_f32 - completion_time) / 15.0_f32) * 2.0_f32;
819                (base_score as f32 * multiplier) as u32
820            } else if completion_time <= 25.0_f32 {
821                // Good: 1.5x to 3x multiplier
822                let multiplier = 1.5_f32 + ((25.0_f32 - completion_time) / 10.0_f32) * 1.5_f32;
823                (base_score as f32 * multiplier) as u32
824            } else if completion_time <= 35.0_f32 {
825                // Average: 0.5x to 1.5x multiplier
826                let multiplier = 0.5_f32 + ((35.0_f32 - completion_time) / 10.0_f32) * 1.0_f32;
827                (base_score as f32 * multiplier) as u32
828            } else {
829                // Slow: 0.1x to 0.5x multiplier
830                let multiplier =
831                    0.1_f32 + ((45.0_f32 - completion_time).max(0.0_f32) / 10.0_f32) * 0.4_f32;
832                (base_score as f32 * multiplier) as u32
833            };
834
835            // Level progression bonus (small bonus for reaching higher levels)
836            let level_bonus = if current_level > 5 {
837                (current_level - 5) as u32 * 50
838            } else {
839                0
840            };
841
842            // Consecutive level bonus (reward for sustained performance)
843            let consecutive_bonus = if completion_time <= 20.0_f32 {
844                // Only give consecutive bonus for good performance
845                current_level as u32 * 25
846            } else {
847                0
848            };
849
850            let total_score = base_score + speed_bonus + level_bonus + consecutive_bonus;
851
852            // Update score and level
853            state
854                .game_state
855                .set_score(state.game_state.game_ui.score + total_score);
856            state.game_state.set_level(current_level + 1);
857
858            // Enhanced time management: Not supported in new timer, so skip add_time/subtract_time/prev_time
859        }
860    }
