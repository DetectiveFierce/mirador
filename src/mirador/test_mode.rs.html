1//! Test mode functionality for Mirador.
2//!
3//! This module provides a simplified test environment when TEST_MODE is enabled.
4//! It creates a small maze with only perimeter walls, places the exit in the center,
5//! locks the enemy in a fixed position, and freezes the timer at 420.00 seconds.

18/// Test maze dimensions (smaller than normal maze)
19const TEST_MAZE_WIDTH: usize = 6;
20const TEST_MAZE_HEIGHT: usize = 6;
21
22/// Creates a test maze grid with only perimeter walls
23pub fn create_test_maze_grid() -> Vec<Vec<bool>> {
24    let mut maze_grid = vec![vec![false; TEST_MAZE_WIDTH * 2 + 1]; TEST_MAZE_HEIGHT * 2 + 1];
25
26    // Add perimeter walls
27    let width = maze_grid[0].len();
28    let height = maze_grid.len();
29
30    // Top and bottom walls
31    for col in 0..width {
32        maze_grid[0][col] = true; // Top wall
33        maze_grid[height - 1][col] = true; // Bottom wall
34    }
35
36    // Left and right walls
37    for row in 0..height {
38        maze_grid[row][0] = true; // Left wall
39        maze_grid[row][width - 1] = true; // Right wall
40    }
41
42    // Debug: Print the maze grid
43    println!("Test maze grid ({}x{}):", width, height);
44    for (i, row) in maze_grid.iter().enumerate() {
45        let row_str: String = row
46            .iter()
47            .map(|&cell| if cell { '#' } else { ' ' })
48            .collect();
49        println!("Row {:2}: {}", i, row_str);
50    }
51
52    maze_grid
53}

55/// Gets the center cell of the test maze for the exit
56pub fn get_test_exit_cell() -> Cell {
57    // For even-sized mazes, pick the cell closest to the world center
58    // (for 6x6, this is (2,2) or (3,3); we'll use (2,2) for logic)
59    Cell::new((TEST_MAZE_HEIGHT - 1) / 2, (TEST_MAZE_WIDTH - 1) / 2)
60}

62/// Returns the world coordinates of the center of the room for exit patch rendering
63pub fn get_world_center_for_exit_patch(maze_grid: &[Vec<bool>]) -> (f32, f32) {
64    let maze_width = maze_grid[0].len();
65    let maze_height = maze_grid.len();
66    let max_dimension = maze_width.max(maze_height) as f32;
67    let floor_size = crate::math::coordinates::constants::get_floor_size(true); // Test mode floor size
68    let cell_size = floor_size / max_dimension;
69    let origin_x = -(maze_width as f32 * cell_size) / 2.0;
70    let origin_z = -(maze_height as f32 * cell_size) / 2.0;
71    let center_x = origin_x + (maze_width as f32) * cell_size / 2.0;
72    let center_z = origin_z + (maze_height as f32) * cell_size / 2.0;
73    (center_x, center_z)
74}
75
76/// Converts maze cell to wall grid coordinates for exit placement
77pub fn maze_cell_to_wall_grid(cell: &Cell) -> (usize, usize) {
78    // Convert maze cell coordinates to wall grid coordinates
79    // Wall grid has walls at even indices, cells at odd indices
80    let wall_row = cell.row * 2 + 1;
81    let wall_col = cell.col * 2 + 1;
82    (wall_row, wall_col)
83}
84
85/// Places the exit marker in the maze grid
86pub fn place_exit_in_maze_grid(maze_grid: &mut Vec<Vec<bool>>, exit_cell: &Cell) {
87    let (wall_row, wall_col) = maze_cell_to_wall_grid(exit_cell);
88
89    // Ensure we're within bounds
90    if wall_row < maze_grid.len() && wall_col < maze_grid[0].len() {
91        // Mark the exit cell (we'll use a special value that parse_maze_file recognizes)
92        // Since parse_maze_file looks for '*' character, we need to handle this differently
93        // For now, we'll just mark it as a non-wall (false) and handle exit placement separately
94        maze_grid[wall_row][wall_col] = false;
95    }
96}

98/// Creates a locked enemy positioned further from the exit
99pub fn create_test_enemy(exit_patch_position: [f32; 3]) -> Enemy {
100    // Move the enemy 2 cells away in both X and Z directions
101    let floor_size = crate::math::coordinates::constants::get_floor_size(true); // Test mode floor size
102    let patch_size = floor_size / 13.0; // 13 is the wall grid size for 6x6 maze
103    let enemy_position = [
104        exit_patch_position[0] - 4.0 * patch_size, // 2 cells to the left
105        30.0,                                      // Same height
106        exit_patch_position[2] - 4.0 * patch_size, // 2 cells back
107    ];
108
109    let mut enemy = Enemy::new(enemy_position, 150.0);
110    // Lock the enemy in place
111    enemy.pathfinder.locked = true;
112
113    enemy
114}

116/// Creates a test timer configuration with frozen time
117pub fn create_test_timer_config() -> TimerConfig {
118    TimerConfig {
119        duration: Duration::from_secs_f32(420.0), // 420.00 seconds
120        warning_threshold: Duration::from_secs_f32(420.0), // Never warn
121        critical_threshold: Duration::from_secs_f32(420.0), // Never critical
122        normal_color: Color::rgb(100, 255, 100),
123        warning_color: Color::rgb(255, 255, 100),
124        critical_color: Color::rgb(255, 100, 100),
125    }
126}

128/// Sets up the test environment
129pub fn setup_test_environment(game_state: &mut GameState, wgpu_renderer: &mut WgpuRenderer) {
130    // Create test maze grid
131    let mut maze_grid = create_test_maze_grid();
132    let exit_cell = get_test_exit_cell();
133
134    // Place exit in the grid
135    place_exit_in_maze_grid(&mut maze_grid, &exit_cell);
136
137    // Generate geometry from the test maze
138    // Use the world center for the exit patch
139    let exit_patch_position = get_world_center_for_exit_patch(&maze_grid);
140    let (mut floor_vertices, _) = Vertex::create_floor_vertices(&maze_grid, None, true); // Test mode floor size
141    // Add a green exit patch at the world center
142    floor_vertices.extend(Vertex::create_exit_patch_at_world_position(
143        exit_patch_position,
144        true, // Test mode
145    ));
146
147    // Set exit position in renderer (as tuple)
148    wgpu_renderer.game_renderer.exit_position = Some(exit_patch_position);
149
150    // Add wall vertices (test mode always uses perimeter walls)
151    floor_vertices.append(&mut Vertex::create_wall_vertices(&maze_grid, true));
152
153    // Create vertex buffer
154    wgpu_renderer.game_renderer.vertex_buffer =
155        wgpu_renderer
156            .device
157            .create_buffer_init(&wgpu::util::BufferInitDescriptor {
158                label: Some("Test Maze Vertex Buffer"),
159                contents: bytemuck::cast_slice(&floor_vertices),
160                usage: wgpu::BufferUsages::VERTEX,
161            });
162
163    // Update vertex count so the renderer knows how many vertices to draw
164    wgpu_renderer.game_renderer.vertex_count = floor_vertices.len() as u32;
165
166    println!(
167        "Debug: Created {} vertices for test maze",
168        floor_vertices.len()
169    );
170    println!(
171        "Debug: Maze grid dimensions: {}x{}",
172        maze_grid.len(),
173        maze_grid[0].len()
174    );
175    println!(
176        "Debug: Wall count: {}",
177        maze_grid
178            .iter()
179            .flatten()
180            .filter(|&&is_wall| is_wall)
181            .count()
182    );
183
184    // Set exit cell in game state
185    game_state.exit_cell = Some(exit_cell);
186
187    // Create and place locked enemy (convert tuple to array for enemy position)
188    let enemy_position = [exit_patch_position.0, 30.0, exit_patch_position.1];
189    game_state.enemy = create_test_enemy(enemy_position);
190
191    // Build collision system from test maze (test mode always uses perimeter walls)
192    game_state
193        .collision_system
194        .build_from_maze(&maze_grid, true);
195
196    // Spawn player at the entrance (bottom-left corner)
197    game_state.player.spawn_at_maze_entrance(&maze_grid, true); // Test mode
198
199    // Set up frozen timer
200    let timer_config = create_test_timer_config();
201    game_state.start_game_timer(Some(timer_config));
202
203    // Immediately pause the timer to freeze it at 420.00
204    game_state.game_ui.pause_timer();
205
206    println!("Test mode initialized:");
207    println!("  - Maze size: {}x{}", TEST_MAZE_WIDTH, TEST_MAZE_HEIGHT);
208    println!("  - Exit at: {:?}", exit_cell);
209    println!("  - Timer frozen at 420.00 seconds");
210    println!(
211        "  - Enemy locked at position: {:?}",
212        game_state.enemy.pathfinder.position
213    );
