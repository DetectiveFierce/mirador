1//! Math utilities and types for 3D graphics and game logic.
2//!
3//! This module provides matrix and vector types and operations, as well as
4//! helper functions for angle conversions. All types are designed to be
5//! compatible with GPU memory layouts (e.g., for use with WGPU/WGSL).
6//!
7//! # Module Organization
8//!
9//! - [`vec`] module contains all vector operations (re-exported at root level)
10//! - [`mat`] module contains all matrix operations (re-exported at root level)
11//! - [`coordinates`] module contains coordinate system transformations
12//! - Utility functions like angle conversions are provided at root level
13
14pub mod coordinates;
15pub mod mat;
16pub mod vec;

18/// Converts degrees to radians.
19///
20/// This handles angle wrapping by first normalizing the input to the range [0, 360).
21///
22/// # Arguments
23///
24/// * `degrees` - The angle in degrees (can be any finite value)
25///
26/// # Returns
27///
28/// The angle in radians in range [0, 2π)
29///
30/// # Example
31/// ```
32/// use your_crate::math::deg_to_rad;
33///
34/// // Basic conversion
35/// assert_eq!(deg_to_rad(180.0), std::f32::consts::PI);
36///
37/// // Handles angle wrapping
38/// assert_eq!(deg_to_rad(540.0), std::f32::consts::PI);
39/// ```
40pub fn deg_to_rad(degrees: f32) -> f32 {
41    (degrees % 360.0) * (std::f32::consts::PI / 180.0)
42}

44/// Converts radians to degrees.
45///
46/// This handles angle wrapping by first normalizing the input to the range [0, 2π).
47///
48/// # Arguments
49///
50/// * `radians` - The angle in radians (can be any finite value)
51///
52/// # Returns
53///
54/// The angle in degrees in range [0, 360)
55///
56/// # Example
57/// ```
58/// use your_crate::math::rad_to_deg;
59///
60/// // Basic conversion
61/// assert_eq!(rad_to_deg(std::f32::consts::PI), 180.0);
62///
63/// // Handles angle wrapping
64/// assert_eq!(rad_to_deg(3.0 * std::f32::consts::PI), 180.0);
65/// ```
66#[allow(dead_code)]
67pub fn rad_to_deg(radians: f32) -> f32 {
68    (radians % (2.0 * std::f32::consts::PI)) * (180.0 / std::f32::consts::PI)
69}
