1//! Special positions and cell finding utilities.
2//!
3//! This module provides functions to find and work with special positions
4//! in the maze, such as the entrance, exit, and cardinal directions.

6use crate::maze::generator::Cell;

8/// Enum representing cardinal directions in the maze
9#[derive(Debug, Clone, Copy, PartialEq, Eq)]
10pub enum Direction {
11    North,
12    East,
13    South,
14    West,
15}

17/// Gets the bottom-left cell of the maze.
18///
19/// # Arguments
20/// * `maze_dimensions` - The dimensions of the maze (width, height) in cells
21///
22/// # Returns
23/// The cell representing the bottom-left corner
24pub fn get_bottom_left_cell(maze_dimensions: (usize, usize)) -> Cell {
25    let (_, maze_height) = maze_dimensions;
26    Cell::new(maze_height - 1, 0)
27}

64/// Gets the cell adjacent to the given cell in the specified direction.
65///
66/// # Arguments
67/// * `cell` - The current cell
68/// * `direction` - The direction to move
69/// * `maze_dimensions` - The dimensions of the maze (width, height) in cells
70///
71/// # Returns
72/// Option containing the adjacent cell, or None if it would be outside the maze
73pub fn get_adjacent_cell(
74    cell: &Cell,
75    direction: Direction,
76    maze_dimensions: (usize, usize),
77) -> Option<Cell> {
78    let (maze_width, maze_height) = maze_dimensions;
79    let Cell { row, col } = *cell;
80
81    match direction {
82        Direction::North if row > 0 => Some(Cell::new(row - 1, col)),
83        Direction::South if row < maze_height - 1 => Some(Cell::new(row + 1, col)),
84        Direction::East if col < maze_width - 1 => Some(Cell::new(row, col + 1)),
85        Direction::West if col > 0 => Some(Cell::new(row, col - 1)),
86        _ => None, // Outside the maze bounds
87    }
88}

90/// Converts a yaw angle (in degrees) to a cardinal direction.
91///
92/// # Arguments
93/// * `yaw` - The yaw angle in degrees
94///
95/// # Returns
96/// The closest cardinal direction
97pub fn yaw_to_direction(yaw: f32) -> Direction {
98    // Normalize angle to 0-360
99    let normalized_yaw = ((yaw % 360.0) + 360.0) % 360.0;
100
101    // Convert to cardinal direction (North = 0°, East = 90°, etc.)
102    match normalized_yaw as u32 {
103        315..=359 | 0..=45 => Direction::North,
104        46..=135 => Direction::East,
105        136..=225 => Direction::South,
106        226..=314 => Direction::West,
107        _ => Direction::North, // Should never happen due to normalization
108    }
109}

111/// Gets the relative direction from one cell to another.
112///
113/// # Arguments
114/// * `from` - The starting cell
115/// * `to` - The destination cell
116///
117/// # Returns
118/// The cardinal direction from the starting cell to the destination
119pub fn get_direction_between_cells(from: &Cell, to: &Cell) -> Option<Direction> {
120    let row_diff = to.row as isize - from.row as isize;
121    let col_diff = to.col as isize - from.col as isize;
122
123    // Determine primary direction based on which difference is larger
124    if row_diff.abs() > col_diff.abs() {
125        if row_diff < 0 {
126            Some(Direction::North)
127        } else {
128            Some(Direction::South)
129        }
130    } else if col_diff.abs() > row_diff.abs() {
131        if col_diff > 0 {
132            Some(Direction::East)
133        } else {
134            Some(Direction::West)
135        }
136    } else {
137        // Cells are diagonal or the same - no clear direction
138        None
139    }
140}

142/// Translates a direction into a yaw angle.
143///
144/// # Arguments
145/// * `direction` - The cardinal direction
146///
147/// # Returns
148/// The corresponding yaw angle in degrees
149pub fn direction_to_yaw(direction: Direction) -> f32 {
150    match direction {
151        Direction::North => 0.0,
152        Direction::East => 90.0,
153        Direction::South => 180.0,
154        Direction::West => 270.0,
155    }
