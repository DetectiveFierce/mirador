213    /// Updates the title screen maze and loading bar, and uploads new texture data.
214    pub fn handle_loading_screen(&mut self, window: &winit::window::Window) {
215        let progress = self
216            .wgpu_renderer
217            .loading_screen_renderer
218            .generator
219            .get_progress_ratio();
220
221        let (maze_width, maze_height) = match self.wgpu_renderer.loading_screen_renderer.maze.lock()
222        {
223            Ok(maze_lock) => maze_lock.get_dimensions(),
224
225            Err(err) => {
226                eprintln!("Failed to acquire maze lock for dimensions: {}", err);
227                return;
228            }
229        };
230
231        self.wgpu_renderer
232            .loading_screen_renderer
233            .update_loading_bar(&self.wgpu_renderer.queue, progress, window);
234
235        self.wgpu_renderer
236            .loading_screen_renderer
237            .update_exit_shader(&self.wgpu_renderer.queue, window);
238
239        let maze_data = match self.wgpu_renderer.loading_screen_renderer.maze.lock() {
240            Ok(maze_lock) => maze_lock.get_render_data(
241                &self
242                    .wgpu_renderer
243                    .loading_screen_renderer
244                    .generator
245                    .connected_cells,
246            ),
247
248            Err(err) => {
249                eprintln!("Failed to acquire maze lock: {}", err);
250                return;
251            }
252        };
253
254        self.wgpu_renderer.loading_screen_renderer.update_texture(
255            &self.wgpu_renderer.queue,
256            &maze_data,
257            maze_width,
258            maze_height,
259        );
260        self.wgpu_renderer.loading_screen_renderer.last_update = Instant::now();
261    }

263    /// Handles mouse capture and cursor visibility based on game state.
264    ///
265    /// Locks/unlocks the cursor and centers it if mouse capture is enabled.
266    pub fn triage_mouse(&mut self, window: &Window) {
267        if self.game_state.capture_mouse {
268            if let Err(e) = window.set_cursor_grab(winit::window::CursorGrabMode::Locked) {
269                eprintln!("Failed to lock cursor: {}", e);
270            }
271            window.set_cursor_visible(false);
272            let window_size = window.inner_size().to_logical::<f64>(window.scale_factor());
273
274            let center_x = window_size.width / 2.0;
275            let center_y = window_size.height / 2.0;
276            if let Err(e) =
277                window.set_cursor_position(winit::dpi::LogicalPosition::new(center_x, center_y))
278            {
279                eprintln!("Failed to center cursor: {}", e);
280            }
281        } else if !self.game_state.capture_mouse {
282            if let Err(e) = window.set_cursor_grab(winit::window::CursorGrabMode::None) {
283                eprintln!("Failed to unlock cursor: {}", e);
284            }
285            window.set_cursor_visible(true);
286        }
287    }

289    /// Updates all game UI elements including timer, level, and score displays.
290    pub fn update_game_ui(&mut self, window: &winit::window::Window) {
291        // Start timer when game begins (not on title screen)
292        if self.game_state.current_screen == CurrentScreen::Game
293            && self.game_state.game_ui.timer.is_none()
294        {
295            // Configure timer with custom settings
296            let timer_config = TimerConfig {
297                duration: Duration::from_secs(30),
298                warning_threshold: Duration::from_secs(20),
299                critical_threshold: Duration::from_secs(10),
300                normal_color: Color::rgb(100, 255, 100),
301                warning_color: Color::rgb(255, 255, 100),
302                critical_color: Color::rgb(255, 100, 100),
303            };
304            self.game_state.start_game_timer(Some(timer_config));
305        }
306
307        // Hide game UI elements during loading screen or title screen
308        if self.game_state.current_screen == CurrentScreen::Loading
309            || self.game_state.current_screen == CurrentScreen::Title
310        {
311            // Hide timer, level, and score displays
312            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("main_timer") {
313                buffer.visible = false;
314            }
315            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("level") {
316                buffer.visible = false;
317            }
318            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("score") {
319                buffer.visible = false;
320            }
321        } else {
322            // Show game UI elements when not loading
323            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("main_timer") {
324                buffer.visible = true;
325            }
326            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("level") {
327                buffer.visible = true;
328            }
329            if let Some(buffer) = self.text_renderer.text_buffers.get_mut("score") {
330                buffer.visible = true;
331            }
332        }
333
334        // Always update the text UI, but only update the timer if in Game
335        let timer_expired = game::update_game_ui(
336            &mut self.text_renderer,
337            &mut self.game_state.game_ui,
338            &self.game_state.current_screen,
339            window,
340        );
341
342        if timer_expired {
343            // Handle timer expiration - you can add game over logic here
344            println!("Timer expired! Game over.");
345            self.game_state.current_screen = CurrentScreen::GameOver;
346        }
347
348        if self.game_state.enemy.pathfinder.reached_player {
349            self.game_state.current_screen = CurrentScreen::GameOver;
350            self.game_state.enemy = Enemy::new([-0.5, 30.0, 0.0], 150.0);
351            self.game_state.enemy.pathfinder.reached_player = false
352        }
353
354        // Show/hide game over display based on current screen
355        if self.game_state.current_screen == CurrentScreen::GameOver {
356            self.text_renderer.show_game_over_display();
357        } else {
358            self.text_renderer.hide_game_over_display();
359        }
360    }

388    /// Asynchronously sets up the application window and initializes [`AppState`].
389    ///
390    /// # Arguments
391    /// - `window`: The application window to use.
392    pub async fn set_window(&mut self, window: Window) {
393        let window = Arc::new(window);
394        let initial_width = 1360;
395        let initial_height = 768;
396
397        let _ = window.request_inner_size(PhysicalSize::new(initial_width, initial_height));
398
399        let surface = self
400            .instance
401            .create_surface(window.clone())
402            .expect("Failed to create surface!");
403
404        let state = AppState::new(
405            &self.instance,
406            surface,
407            &window,
408            initial_width,
409            initial_height,
410        )
411        .await;
412
413        self.window.get_or_insert(window);
414        self.state.get_or_insert(state);
415    }

417    /// Handles window resize events and updates the rendering surface.
418    ///
419    /// # Arguments
420    /// - `width`: New width of the window.
421    /// - `height`: New height of the window.
422    pub fn handle_resized(&mut self, width: u32, height: u32) {
423        if width > 0 && height > 0 {
424            let state = match &mut self.state {
425                Some(state) => state,
426                None => {
427                    eprintln!("Cannot resize surface without state initialized!");
428                    #[cfg(debug_assertions)]
429                    eprintln!("Backtrace: {:?}", std::backtrace::Backtrace::capture());
430                    return;
431                }
432            };
433            state.resize_surface(width, height);
434            // Ensure pause menu resizes with the window
435            use glyphon::Resolution;
436            let resolution = Resolution { width, height };
437            state
438                .pause_menu
439                .resize(&state.wgpu_renderer.queue, resolution);
440        }
441    }

443    /// Handles redraw requests, updates game state, renders the frame, and manages the title screen.
444    pub fn handle_redraw(&mut self) {
445        let window = self
446            .window
447            .as_ref()
448            .expect("Window must be initialized before use");
449        if window.is_minimized().unwrap_or(false) {
450            println!("Window is minimized");
451            return;
452        }
453
454        let state = self
455            .state
456            .as_mut()
457            .expect("State must be initialized before use");
458
459        if state.game_state.current_screen == CurrentScreen::Loading {
460            state
461                .game_state
462                .audio_manager
463                .pause_enemy_audio("enemy")
464                .expect("Failed to pause enemy audio");
465            state.handle_loading_screen(window);
466        } else if state.game_state.current_screen == CurrentScreen::Title {
467            title::handle_title(state, window);
468            return;
469        } else {
470            state.game_state.player.update_cell(
471                &state
472                    .wgpu_renderer
473                    .loading_screen_renderer
474                    .maze
475                    .lock()
476                    .unwrap()
477                    .walls,
478                state.game_state.is_test_mode,
479            );
480        }
481
482        // Update game state and UI
483        state.key_state.update(&mut state.game_state);
484        state.update_game_ui(window);
485        state.update_ui(window);
486        state
487            .game_state
488            .audio_manager
489            .set_listener_position(state.game_state.player.position)
490            .expect("Failed to set listener position");
491        state
492            .game_state
493            .audio_manager
494            .update_enemy_position("enemy", state.game_state.enemy.pathfinder.position)
495            .expect("Failed to update enemy position");
496
497        // Prepare rendering commands
498        let mut encoder = state
499            .wgpu_renderer
500            .device
501            .create_command_encoder(&wgpu::CommandEncoderDescriptor { label: None });
502
503        // Update canvas surface
504        let (surface_view, surface_texture) = match state.wgpu_renderer.update_canvas(
505            window,
506            &mut encoder,
507            &state.ui,
508            &state.game_state,
509            &mut state.text_renderer,
510        ) {
511            Ok(result) => result,
512            Err(err) => {
513                eprintln!("Failed to update canvas: {}", err);
514                #[cfg(debug_assertions)]
515                eprintln!("Backtrace: {:?}", std::backtrace::Backtrace::capture());
516                return;
517            }
518        };
519
520        // --- Debug Info Panel ---
521        if state.pause_menu.is_debug_panel_visible() {
522            let window_size = &state.wgpu_renderer.surface_config;
523            let debug_text = format!(
524                "Window Size: {} x {}",
525                window_size.width, window_size.height
526            );
527            let style = crate::renderer::text::TextStyle {
528                font_family: "HankenGrotesk".to_string(),
529                font_size: 22.0,
530                line_height: 26.0,
531                color: Color::rgb(220, 40, 40),
532                weight: glyphon::Weight::BOLD,
533                style: glyphon::Style::Normal,
534            };
535            let pos = crate::renderer::text::TextPosition {
536                x: window_size.width as f32 - 320.0,
537                y: 20.0,
538                max_width: Some(300.0),
539                max_height: Some(40.0),
540            };
541            state.text_renderer.create_text_buffer(
542                "debug_info",
543                &debug_text,
544                Some(style),
545                Some(pos),
546            );
547        } else {
548            // Hide debug info by making it transparent if it exists
549            if let Some(buf) = state.text_renderer.text_buffers.get_mut("debug_info") {
550                buf.visible = false;
551            }
552        }
553        // Prepare and render text BEFORE pause menu overlay
554        if let Err(e) = state.text_renderer.prepare(
555            &state.wgpu_renderer.device,
556            &state.wgpu_renderer.queue,
557            &state.wgpu_renderer.surface_config,
558        ) {
559            println!("Failed to prepare text renderer: {}", e);
560        }
561        {
562            let mut render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
563                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
564                    view: &surface_view,
565                    resolve_target: None,
566                    ops: wgpu::Operations {
567                        load: wgpu::LoadOp::Load,
568                        store: wgpu::StoreOp::Store,
569                    },
570                })],
571                depth_stencil_attachment: None,
572                timestamp_writes: None,
573                label: Some("text render pass"),
574                occlusion_query_set: None,
575            });
576            if let Err(e) = state.text_renderer.render(&mut render_pass) {
577                println!("Failed to render text: {}", e);
578            }
579        }
580        // --- End Game UI ---
581
582        // If paused, render the pause menu on top
583        if state.game_state.current_screen == CurrentScreen::Pause {
584            if !state.pause_menu.is_visible() {
585                state.pause_menu.show(state.game_state.is_test_mode);
586            }
587
588            // Create a render pass for the pause menu
589            let mut render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
590                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
591                    view: &surface_view,
592                    resolve_target: None,
593                    ops: wgpu::Operations {
594                        load: wgpu::LoadOp::Load,
595                        store: wgpu::StoreOp::Store,
596                    },
597                })],
598                depth_stencil_attachment: None,
599                timestamp_writes: None,
600                label: Some("pause menu render pass"),
601                occlusion_query_set: None,
602            });
603
604            // --- Add semi-transparent grey overlay ---
605            let overlay_color = [0.08, 0.09, 0.11, 0.88]; // darker, neutral semi-transparent grey
606            let (w, h) = (
607                state.wgpu_renderer.surface_config.width as f32,
608                state.wgpu_renderer.surface_config.height as f32,
609            );
610            state
611                .pause_menu
612                .button_manager
613                .rectangle_renderer
614                .add_rectangle(crate::renderer::rectangle::Rectangle::new(
615                    0.0,
616                    0.0,
617                    w,
618                    h,
619                    overlay_color,
620                ));
621            state
622                .pause_menu
623                .button_manager
624                .rectangle_renderer
625                .render(&state.wgpu_renderer.device, &mut render_pass);
626            // --- End overlay ---
627
628            // Prepare pause menu for rendering (text)
629            if let Err(e) = state.pause_menu.prepare(
630                &state.wgpu_renderer.device,
631                &state.wgpu_renderer.queue,
632                &state.wgpu_renderer.surface_config,
633            ) {
634                println!("Failed to prepare pause menu: {}", e);
635            }
636
637            // Render the pause menu (rectangles + text)
638            if let Err(e) = state
639                .pause_menu
640                .render(&state.wgpu_renderer.device, &mut render_pass)
641            {
642                println!("Failed to render pause menu: {}", e);
643            }
644        } else {
645            if state.pause_menu.is_visible() {
646                state.pause_menu.hide();
647            }
648            // Explicitly clear rectangles if menu is not visible
649            state
650                .pause_menu
651                .button_manager
652                .rectangle_renderer
653                .clear_rectangles();
654        }
655
656        window.request_redraw();
657
658        // Submit commands and present
659        state.wgpu_renderer.queue.submit(Some(encoder.finish()));
660
661        // Present the surface texture and ensure it's properly handled
662        surface_texture.present();
663
664        // Poll the device to process any pending operations
665        // This helps ensure resources are properly cleaned up and prevents
666        // the "SurfaceSemaphores still in use" error during cleanup
667        state.wgpu_renderer.device.poll(wgpu::Maintain::Poll);
668
669        // Update enemy pathfinding
670        state.game_state.enemy.update(
671            state.game_state.player.position,

707    pub fn new_level(&mut self, game_over: bool) {
708        let state = self
709            .state
710            .as_mut()
711            .expect("State must be initialized before use");
712        state.game_state.current_screen = CurrentScreen::Loading;
713        state.game_state.maze_path = None;

719        // Clear previous level state
720        state.game_state.player = Player::new();
721        state.game_state.enemy.pathfinder.position = [0.0, 30.0, 0.0];
722        state.game_state.enemy.pathfinder.locked = true;
723        state.game_state.exit_cell = None; // Clear exit cell to prevent accidental win condition
724
725        // Stop and reset timer
726        if let Some(timer) = &mut state.game_state.game_ui.timer {
727            timer.stop();
728            timer.reset();
729        }
730
731        if game_over {
732            state.game_state.set_level(1);
733            state.game_state.set_score(0);
734            state.game_state.game_ui.timer = Some(GameTimer::new(TimerConfig::default()));
735            game::update_game_ui(
736                &mut state.text_renderer,
737                &mut state.game_state.game_ui,
738                &state.game_state.current_screen,
739                self.window
740                    .as_ref()
741                    .expect("Window must be initialized before use"),
742            );
743            // Ensure clean state for new game
744            state.game_state.exit_cell = None;
745        } else {
746            let current_level = state.game_state.game_ui.level;
747
748            // Calculate completion time and performance metrics
749            let (completion_time, _) = if let Some(timer) = &state.game_state.game_ui.timer {
750                let remaining_time = timer.get_remaining_time().as_secs_f32();
751                let total_time = timer.config.duration.as_secs_f32();
752                let completion_time = total_time - remaining_time;
753
754                // Performance-based time bonus calculation
755                // Optimal time: 15 seconds, Average time: 25 seconds, Slow time: 35+ seconds
756                let time_bonus = if completion_time <= 15.0 {
757                    // Exceptional performance: 15-25 seconds added
758                    let performance_ratio = (15.0 - completion_time).max(0.0) / 15.0;
759                    15.0 + (performance_ratio * 10.0)
760                } else if completion_time <= 25.0 {
761                    // Good performance: 8-15 seconds added
762                    let performance_ratio = (25.0 - completion_time) / 10.0;
763                    8.0 + (performance_ratio * 7.0)
764                } else if completion_time <= 35.0 {
765                    // Average performance: 3-8 seconds added
766                    let performance_ratio = (35.0 - completion_time) / 10.0;
767                    3.0 + (performance_ratio * 5.0)
768                } else {
769                    // Slow completion: 1-3 seconds added
770                    let performance_ratio = (45.0 - completion_time).max(0.0) / 10.0;
771                    1.0 + (performance_ratio * 2.0)
772                };
773
774                (completion_time, time_bonus)
775            } else {
776                (30.0, 3.0) // Fallback values
777            };
778
779            // Enhanced scoring system
780            let base_score = 150 * current_level as u32; // Increased base score
781
782            // Speed bonus calculation
783            let speed_bonus = if completion_time <= 15.0 {
784                // Exceptional: 3x to 5x multiplier
785                let multiplier = 3.0 + ((15.0 - completion_time) / 15.0) * 2.0;
786                (base_score as f32 * multiplier) as u32
787            } else if completion_time <= 25.0 {
788                // Good: 1.5x to 3x multiplier
789                let multiplier = 1.5 + ((25.0 - completion_time) / 10.0) * 1.5;
790                (base_score as f32 * multiplier) as u32
791            } else if completion_time <= 35.0 {
792                // Average: 0.5x to 1.5x multiplier
793                let multiplier = 0.5 + ((35.0 - completion_time) / 10.0) * 1.0;
794                (base_score as f32 * multiplier) as u32
795            } else {
796                // Slow: 0.1x to 0.5x multiplier
797                let multiplier = 0.1 + ((45.0 - completion_time).max(0.0) / 10.0) * 0.4;
798                (base_score as f32 * multiplier) as u32
799            };
800
801            // Level progression bonus (small bonus for reaching higher levels)
802            let level_bonus = if current_level > 5 {
803                (current_level - 5) as u32 * 50
804            } else {
805                0
806            };
807
808            // Consecutive level bonus (reward for sustained performance)
809            let consecutive_bonus = if completion_time <= 20.0 {
810                // Only give consecutive bonus for good performance
811                current_level as u32 * 25
812            } else {
813                0
814            };
815
816            let total_score = base_score + speed_bonus + level_bonus + consecutive_bonus;
817
818            // Update score and level
819            state
820                .game_state
821                .set_score(state.game_state.game_ui.score + total_score);
822            state.game_state.set_level(current_level + 1);
823
824            // Enhanced time management: Not supported in new timer, so skip add_time/subtract_time/prev_time
825        }
826    }
827    /// Updates frame timing, FPS, and delta time in the game state.

829    /// # Arguments

831    pub fn handle_frame_timing(&mut self, current_time: Instant) {
832        if let Some(state) = self.state.as_mut() {
833            let duration = current_time.duration_since(state.game_state.last_fps_time);
834
835            state.elapsed_time = current_time.duration_since(state.start_time);
836            state.ui.elapsed_time += 1.0;
837            state.game_state.frame_count += 1;
838
839            if duration.as_secs_f32() >= 1.0 {
840                state.game_state.current_fps = state.game_state.frame_count;
841                state.game_state.frame_count = 0;
842                state.game_state.last_fps_time = current_time;
843            }
844
845            let delta_time = current_time
846                .duration_since(state.game_state.last_frame_time)
847                .as_secs_f32();
848
849            state.game_state.delta_time = delta_time;
850            state.game_state.last_frame_time = current_time;
851
852            if state
853                .wgpu_renderer
854                .game_renderer
855                .debug_renderer
856                .debug_render_bounding_boxes
857            {
858                state
859                    .wgpu_renderer
860                    .game_renderer
861                    .debug_renderer
862                    .update_debug_vertices(
863                        &state.wgpu_renderer.device,
864                        &state.game_state.collision_system,
865                    );
866            }
867        }

870    /// Advances the maze generation animation and uploads new geometry when complete.
871    pub fn handle_maze_generation(&mut self) {
872        if let Some(state) = self.state.as_mut() {
873            // If in test mode, skip maze generation entirely and go directly to game
874            if state.game_state.is_test_mode {
875                println!("Test mode enabled - skipping maze generation and going to game");
876                // Mark generation as complete immediately
877                state
878                    .wgpu_renderer
879                    .loading_screen_renderer
880                    .generator
881                    .generation_complete = true;
882                // Set up test environment immediately
883                setup_test_environment(&mut state.game_state, &mut state.wgpu_renderer);
884                // Set a dummy maze path to prevent re-entry
885                state.game_state.maze_path = Some(std::path::PathBuf::from("test_mode"));
886                // Go directly to game screen
887                state.game_state.current_screen = CurrentScreen::Game;
888                return;
889            }
890
891            let renderer = &mut state.wgpu_renderer.loading_screen_renderer;
892
893            // Calculate update timing
894            let speed = if renderer.generator.fast_mode {
895                Duration::from_millis(10) / 20
896            } else {
897                Duration::from_millis(10)
898            };
899
900            // Skip if not time to update or already complete
901            if renderer.last_update.elapsed() < speed || renderer.generator.is_complete() {
902                return;
903            }
904
905            // Process animation steps
906            let steps = if renderer.generator.fast_mode {
907                300
908            } else {
909                100
910            };
911            for _ in 0..steps {
912                if !renderer.generator.step() {
913                    break;
914                }
915            }
916
917            // Report progress
918            let (current, total) = renderer.generator.get_progress();
919
920            if current % 50 == 0 || renderer.generator.is_complete() {
921                println!(
922                    "Progress: {}/{} ({:.1}%)",
923                    current,
924                    total,
925                    (current as f32 * 100.0 / total.max(1) as f32)
926                );
927            }
928            if renderer.generator.is_complete() && state.game_state.maze_path.is_none() {
929                println!("Maze generation complete! Saving to file...");
930
931                // Play completion sound
932                state
933                    .game_state
934                    .audio_manager
935                    .complete()
936                    .expect("Failed to play complete sound!");
937
938                // Handle completion
939                if renderer.generator.is_complete() {
940                    println!("Maze generation complete! Saving to file...");
941                    let maze_lock = renderer.maze.lock().unwrap();
942                    state.game_state.maze_path = maze_lock.save_to_file().map_or_else(
943                        |err| {
944                            eprintln!("Failed to save maze: {}", err);
945                            std::process::exit(1);
946                        },
947                        Some,
948                    );
949
950                    // Generate geometry if maze was saved successfully
951                    if let Some(maze_path) = &state.game_state.maze_path {
952                        let (maze_grid, exit_cell) = parse_maze_file(maze_path.to_str().unwrap());
953                        let (mut floor_vertices, exit_position) = Vertex::create_floor_vertices(
954                            &maze_grid,
955                            exit_cell,
956                            state.game_state.is_test_mode,
957                        );
958
959                        state.wgpu_renderer.game_renderer.exit_position = Some(exit_position);
960
961                        floor_vertices.append(&mut Vertex::create_wall_vertices(
962                            &maze_grid,
963                            state.game_state.is_test_mode,
964                        ));
965
966                        state.wgpu_renderer.game_renderer.vertex_buffer = state
967                            .wgpu_renderer
968                            .device
969                            .create_buffer_init(&wgpu::util::BufferInitDescriptor {
970                                label: Some("Combined Vertex Buffer"),
971                                contents: bytemuck::cast_slice(&floor_vertices),
972                                usage: wgpu::BufferUsages::VERTEX,
973                            });
974
975                        // Update vertex count so the renderer knows how many vertices to draw
976                        state.wgpu_renderer.game_renderer.vertex_count =
977                            floor_vertices.len() as u32;
978
979                        if let Some(exit_cell_position) = exit_cell {
980                            state.game_state.exit_cell = Some(exit_cell_position);
981                            state.game_state.enemy = place_enemy_standard(
982                                maze_to_world(
983                                    &exit_cell_position,
984                                    maze_lock.get_dimensions(),
985                                    30.0,
986                                    state.game_state.is_test_mode,
987                                ),
988                                state.game_state.player.position,
989                                state.game_state.game_ui.level,
990                                |from, to| {
991                                    state
992                                        .game_state
993                                        .collision_system
994                                        .cylinder_intersects_geometry(from, to, 5.0)
995                                },
996                            );
997                        }
998
999                        state
1000                            .game_state
1001                            .collision_system
1002                            .build_from_maze(&maze_grid, state.game_state.is_test_mode);
1003
1004                        // Spawn the player at the bottom-left corner of the maze
1005                        state
1006                            .game_state
1007                            .player
1008                            .spawn_at_maze_entrance(&maze_grid, state.game_state.is_test_mode);
1009                        // (No automatic transition to Game here)
1010                    }
1011                }
1012            }
1013        }
1014    }

1303                                            state.game_state.capture_mouse = false;
1304                                            // Show pause menu with current test mode state
1305                                            state.pause_menu.show(state.game_state.is_test_mode);
1306                                        }
1307                                        CurrentScreen::Pause => {
1308                                            // Return to previous screen
1309                                            if let Some(previous_screen) =
1310                                                state.game_state.previous_screen
1311                                            {
1312                                                state.game_state.current_screen = previous_screen;
1313                                                state.game_state.previous_screen = None;
1314
1315                                                match previous_screen {
1316                                                    CurrentScreen::Game => {
1317                                                        // Resume game
1318                                                        state.game_state.game_ui.resume_timer();
1319                                                        // Unlock enemy movement
1320                                                        state.game_state.enemy.pathfinder.locked =
1321                                                            false;
1322                                                        // Lock cursor
1323                                                        state.game_state.capture_mouse = true;
1324                                                    }
1325                                                    CurrentScreen::Title => {
1326                                                        // Return to title screen - cursor should be unlocked
1327                                                        state.game_state.capture_mouse = false;
1328                                                    }
1329                                                    _ => {
1330                                                        // For other screens, just unlock cursor
1331                                                        state.game_state.capture_mouse = false;
1332                                                    }
1333                                                }
1334                                            } else {
1335                                                // Fallback: return to title screen
1336                                                state.game_state.current_screen =
1337                                                    CurrentScreen::Title;
1338                                                state.game_state.capture_mouse = false;
1339                                            }
1340                                        }
1341                                        CurrentScreen::Title => {
1342                                            // Enter pause menu from title screen
1343                                            state.game_state.previous_screen =
1344                                                Some(CurrentScreen::Title);
1345                                            state.game_state.current_screen = CurrentScreen::Pause;
1346                                            // Unlock cursor to allow menu interaction
1347                                            state.game_state.capture_mouse = false;
1348                                            // Show pause menu with current test mode state
1349                                            state.pause_menu.show(state.game_state.is_test_mode);
1350                                        }
1351                                        _ => {
1352                                            // For all other screens, just toggle cursor lock
1353                                            state.game_state.capture_mouse =
1354                                                !state.game_state.capture_mouse;
1355                                        }
1356                                    }
1357                                }
1358
1359                                _ => {} // Movement keys are handled in process_movement
1360                            }
1361                        }
1362                        ElementState::Released => {
1363                            state.key_state.release_key(game_key);
1364                        }
1365                    }
1366                }
1367            }
1368
1369            WindowEvent::MouseInput {
1370                state: mouse_state,
1371                button,
1372                ..
1373            } => match mouse_state {
1374                ElementState::Pressed => {
1375                    if let Some(app_state) = self.state.as_mut() {
1376                        match button {
1377                            MouseButton::Left => {
1378                                // If on title screen, transition to loading
1379                                if app_state.game_state.current_screen == CurrentScreen::Title {
1380                                    app_state.game_state.current_screen = CurrentScreen::Loading;
1381                                    // Optionally, lock mouse here if needed
1382                                    app_state.game_state.capture_mouse = true;
1383                                    // Hide the overlay text
1384                                    if let Some(buf) = app_state
1385                                        .text_renderer
1386                                        .text_buffers
1387                                        .get_mut("title_mirador_overlay")
1388                                    {
1389                                        buf.visible = false;
1390                                    }
1391                                    // Hide the subtitle overlay
1392                                    if let Some(buf) = app_state
1393                                        .text_renderer
1394                                        .text_buffers
1395                                        .get_mut("title_subtitle_overlay")
1396                                    {
1397                                        buf.visible = false;
1398                                    }
1399                                    return;
1400                                }
1401                                app_state.key_state.press_key(GameKey::MouseButtonLeft);
1402                            }
1403                            MouseButton::Right => {
1404                                app_state.key_state.press_key(GameKey::MouseButtonRight);
1405                            }
1406                            _ => {}
1407                        }
1408                    }
1409                }
1410                ElementState::Released => {
1411                    if let Some(app_state) = self.state.as_mut() {
1412                        match button {
1413                            MouseButton::Left => {
1414                                app_state.key_state.release_key(GameKey::MouseButtonLeft);
1415                            }
1416                            MouseButton::Right => {
1417                                app_state.key_state.release_key(GameKey::MouseButtonRight);
1418                            }
1419                            _ => {}
1420                        }
1421                    }
1422                }
1423            },
1424
1425            WindowEvent::RedrawRequested => {
1426                let current_time = Instant::now();
1427                self.handle_frame_timing(current_time);
1428                self.handle_redraw();
1429            }
1430
1431            _ => {
1432                // Handle any other events that weren't caught above
1433            }
1434        }
1435    }
