1//! Mirador - A 3D Maze Runner Game
2//!
3//! This is the main entry point for the Mirador game application. Mirador is a first-person
4//! maze runner game built with Rust and WGPU, featuring procedurally generated mazes,
5//! enemy AI, audio systems, and a progression system with upgrades.
6//!
7//! # Features
8//! - **3D Graphics**: Real-time 3D rendering using WGPU
9//! - **Procedural Generation**: Dynamically generated mazes with increasing complexity
10//! - **Enemy AI**: Rotation-based pathfinding enemy that pursues the player
11//! - **Audio System**: 3D spatial audio with multiple sound sources
12//! - **Progression System**: Player upgrades and level-based difficulty scaling
13//! - **Test Mode**: Development mode with simplified gameplay for testing
14//!
15//! # Architecture
16//! The application follows a modular architecture:
17//! - `app/`: Application state management and event handling
18//! - `game/`: Core game logic, player, enemy, and maze systems
19//! - `renderer/`: Graphics rendering pipeline and UI components
20//! - `math/`: Mathematical utilities for 3D graphics
21//!
22//! # Usage
23//! Run the application with `cargo run`. The game supports both normal gameplay
24//! and test mode for development purposes.
25

43/// Main entry point for the Mirador game application.
44///
45/// This function initializes the application, sets up the event loop, and starts
46/// the game. It handles different compilation targets (native vs WASM) and
47/// optional memory profiling.
48///
49/// # Features
50/// - Memory profiling with dhat-heap feature
51/// - Cross-platform compatibility (native and WASM targets)
52/// - Graceful error handling for event loop creation
53///
54/// # Panics
55/// - If the event loop cannot be created
56/// - If the application fails to run
57fn main() {
58    #[cfg(feature = "dhat-heap")]
59    let _profiler = dhat::Profiler::new_heap();
60    #[cfg(not(target_arch = "wasm32"))]
61    {
62        pollster::block_on(run());
63    }
64}

66/// Asynchronously runs the main game loop.
67///
68/// This function creates the event loop, initializes the application state,
69/// and starts the game. It handles the complete lifecycle of the application
70/// from startup to shutdown.
71///
72/// # Returns
73/// This function runs indefinitely until the application is closed by the user.
74///
75/// # Errors
76/// - Returns early if event loop creation fails
77/// - Exits the process if the application fails to run
78async fn run() {
79    use crate::benchmarks::{BenchmarkConfig, Profiler};
80
81    // Initialize profiler for overall application initialization benchmarking
82    let mut init_profiler = Profiler::new(BenchmarkConfig {
83        enabled: true,
84        print_results: false, // Respect user's console output preference
85        write_to_file: false,
86        min_duration_threshold: std::time::Duration::from_micros(1),
87        max_samples: 1000,
88    });
89
90    // Benchmark complete application initialization
91    init_profiler.start_section("complete_application_initialization");
92
93    // Set up signal handler for graceful shutdown
94    let running = Arc::new(AtomicBool::new(true));
95    let r = running.clone();
96
97    #[cfg(not(target_arch = "wasm32"))]
98    {
99        ctrlc::set_handler(move || {
100            println!("\nReceived interrupt signal, saving benchmark results...");
101            // Save benchmark results before exiting
102            if let Err(e) = crate::benchmarks::utils::force_save_results() {
103                eprintln!("Failed to save benchmark results on exit: {}", e);
104            }
105            r.store(false, Ordering::SeqCst);
106            std::process::exit(0);
107        })
108        .expect("Error setting Ctrl-C handler");
109    }
110
111    let event_loop = match EventLoop::new() {
112        Ok(event_loop) => event_loop,
113        Err(err) => {
114            eprintln!("Error creating event loop: {}", err);
115            return;
116        }
117    };
118
119    event_loop.set_control_flow(ControlFlow::Poll);
120
121    let mut app = app::App::new();
122
123    event_loop.run_app(&mut app).expect("Failed to run app");
124
125    init_profiler.end_section("complete_application_initialization");
126}
